18  package org.newsclub.net.unix.demo;
19  
20  import java.io.DataInputStream;
21  import java.io.DataOutputStream;
22  import java.io.File;
23  import java.io.IOException;
24  import java.io.InputStream;
25  import java.io.OutputStream;
26  import java.net.SocketException;
27  
28  import org.newsclub.net.unix.AFUNIXSocket;
29  import org.newsclub.net.unix.AFUNIXSocketAddress;
30  
31  /**
32   * A simple demo client.
33   * 
34   * @author Christian Kohlschütter
35   * @see SimpleTestServer
36   */
37  public class SimpleTestClient {
38    public static void main(String[] args) throws IOException {
39      final File socketFile = new File(new File(System.getProperty("java.io.tmpdir")),
40          "junixsocket-test.sock");
41  
42      try (AFUNIXSocket sock = AFUNIXSocket.newInstance()) {
43        try {
44          sock.connect(AFUNIXSocketAddress.of(socketFile));
45        } catch (SocketException e) {
46          System.out.println("Cannot connect to server. Have you started it?");
47          System.out.println();
48          throw e;
49        }
50        System.out.println("Connected");
51  
52        try (InputStream is = sock.getInputStream(); //
53            OutputStream os = sock.getOutputStream();) {
54  
55          byte[] buf = new byte[128];
56  
57          int read = is.read(buf);
58          System.out.println("Server says: " + new String(buf, 0, read, "UTF-8"));
59  
60          System.out.println("Replying to server...");
61          os.write("Hello Server".getBytes("UTF-8"));
62          os.flush();
63  
64          System.out.println("Now reading numbers from the server...");
65          DataInputStream din = new DataInputStream(is);
66          DataOutputStream dout = new DataOutputStream(os);
67          while (!Thread.interrupted()) {
68            int number = din.readInt();
69            if (number == -123) {
70              // by convention of this demo, if the number is -123, we stop.
71              // If we don't do this, we'll get an EOFException upon the next unsuccessful read.
72              break;
73            }
74            System.out.println(number);
75  
76            int ourNumber = number * 2;
77  
78            System.out.println("Sending back " + ourNumber);
79            dout.writeInt(ourNumber);
80          }
81        }
82      }
83  
84      System.out.println("End of communication.");
85    }
86  }
