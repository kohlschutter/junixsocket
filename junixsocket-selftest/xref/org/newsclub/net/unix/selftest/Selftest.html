111   /**
112    * Run this from the command line to ensure junixsocket works correctly on the target system.
113    * 
114    * A zero error code indicates success.
115    * 
116    * @param args Ignored.
117    * @throws IOException on error.
118    */
119   @SuppressFBWarnings({
120       "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
121   public static void main(String[] args) throws Exception {
122     Selftest st = new Selftest();
123 
124     st.printExplanation();
125     st.dumpSystemProperties();
126     st.dumpOSReleaseFiles();
127     st.checkSupported();
128     st.checkCapabilities();
129 
130     SelftestProvider sp = new SelftestProvider();
131     Set<String> disabledModules = sp.modulesDisabledByDefault();
132 
133     List<String> messagesAtEnd = new ArrayList<>();
134     for (Entry<String, Class<?>[]> en : sp.tests().entrySet()) {
135       String module = en.getKey();
136       if (disabledModules.contains(module)) {
137         if (SystemPropertyUtil.getBooleanSystemProperty("selftest.enable-module." + module,
138             false)) {
139           // System.out.println("Enabled optional module: " + module);
140           st.modified = true;
141         } else {
142           messagesAtEnd.add("Skipping optional module: " + module
143               + "; enable by launching with -Dselftest.enable-module." + module + "=true");
144           continue;
145         }
146       } else if (SystemPropertyUtil.getBooleanSystemProperty("selftest.disable-module." + module,
147           false)) {
148         messagesAtEnd.add("Skipping required module: " + module + "; this taints the test");
149         st.withIssues = true;
150       }
151       st.runTests(module, en.getValue());
152     }
153 
154     if (!messagesAtEnd.isEmpty()) {
155       for (String m : messagesAtEnd) {
156         System.out.println(m);
157       }
158     }
159 
160     st.checkInitError();
161     st.dumpResults();
162 
163     int rc = st.isFail() ? 1 : 0;
164 
165     if (SystemPropertyUtil.getBooleanSystemProperty("selftest.wait.at-end", false)) {
166       System.gc(); // NOPMD
167       System.out.print("Press any key to end test. ");
168       System.out.flush();
169       System.in.read();
170       System.out.println("RC=" + rc);
171     }
172 
173     System.out.flush();
174     System.exit(rc); // NOPMD
175   }

190   public void dumpSystemProperties() {
191     out.println("System properties:");
192     out.println();
193     for (Map.Entry<Object, Object> en : new TreeMap<>(System.getProperties()).entrySet()) {
194       String key = String.valueOf(en.getKey());
195       String value = String.valueOf(en.getValue());
196       StringBuilder sb = new StringBuilder();
197       for (int i = 0; i < value.length(); i++) {
198         char c = value.charAt(i);
199         switch (c) {
200           case '\n':
201             sb.append("\\n");
202             break;
203           case '\r':
204             sb.append("\\r");
205             break;
206           case '\t':
207             sb.append("\\r");
208             break;
209           default:
210             if (c < 32 || c >= 127) {
211               sb.append(String.format(Locale.ENGLISH, "\\u%04x", (int) c));
212             }
213             sb.append(c);
214             break;
215         }
216       }
217       out.println(key + ": " + sb.toString());
218     }
219     out.println();
220   }

222   public void checkSupported() {
223     out.print("AFSocket.isSupported: ");
224     out.flush();
225 
226     boolean isSupported = AFSocket.isSupported();
227     out.println(isSupported);
228     out.println();
229     out.flush();
230 
231     if (!isSupported) {
232       out.println("FAIL: junixsocket is not supported on this platform");
233       out.println();
234       fail = true;
235     }
236 
237     out.print("AFUNIXSocket.isSupported: ");
238     out.flush();
239 
240     isSupportedAFUNIX = AFUNIXSocket.isSupported();
241     out.println(isSupportedAFUNIX);
242     out.println();
243     out.flush();
244 
245     if (!isSupportedAFUNIX) {
246       out.println("WARNING: AF_UNIX sockets are not supported on this platform");
247       out.println();
248       withIssues = true;
249     }

268   private void checkInitError() {
269     Throwable t = retrieveInitError();
270     if (t == null) {
271       return;
272     }
273 
274     important.add("The native library failed to load.");
275 
276     StringWriter sw = new StringWriter();
277     PrintWriter pw = new PrintWriter(sw);
278     t.printStackTrace(pw);
279     pw.flush();
280     String ts = sw.toString();
281     String tsLower = ts.toLowerCase(Locale.ENGLISH);
282 
283     if (tsLower.contains("not permitted") || ts.contains("permission")) {
284       important.add("It looks like there were some permission errors.");
285     }
286 
287     if (tsLower.contains("failed to map segment")) {
288       important.add("Your temporary directory is probably mounted with \"noexec\", "
289           + "which prevents the native library from loading.");
290       important.add("see: https://github.com/kohlschutter/junixsocket/issues/99");
291       Object tmpDir = retrieveTempDir();
292       if (tmpDir == null) {
293         tmpDir = System.getProperty("java.io.tmpdir");
294       }
295       if (tmpDir != null) {
296         important.add("Temp dir: " + tmpDir);
297       }
298       important.add(
299           "You can specify a different directory using -Dorg.newsclub.net.unix.library.tmpdir=");
300     }
301   }

307   public void dumpResults() { // NOPMD
308     if (modified) {
309       important.add("Selftest was modified, for example to exclude/include certain tests.");
310       inconclusive = true;
311     }
312     if (!isSupportedAFUNIX) {
313       important.add(
314           "Environment does not support UNIX sockets, which is an important part of junixsocket.");
315       // inconclusive = true;
316     }
317     if (inconclusive) {
318       important.add("Selftest results may be inconclusive.");
319     }
320 
321     if (withIssues) {
322       important.add("\"With issues\": "
323           + "Please carefully check the output above; the software may not be able to do what you want.");
324     }
325 
326     out.println();
327     out.println("Selftest results:");
328 
329     for (Map.Entry<String, ModuleResult> en : results.entrySet()) {
330       ModuleResult res = en.getValue();
331 
332       String result = res == null ? null : res.result.name();
333       String extra;
334       if (res == null) {
335         result = "SKIP";
336         extra = "(skipped by user request)";
337       } else if (res.summary == null) {
338         extra = res.throwable == null ? "(unknown error)" : res.throwable.toString();
339         fail = true;
340       } else {
341         TestExecutionSummary summary = res.summary;
342 
343         long nSucceeded = (summary.getTestsSucceededCount() + res.getNumAbortedNonIssues());
344         extra = nSucceeded + "/" + summary.getTestsFoundCount();
345         long nSkipped = summary.getTestsSkippedCount();
346         if (nSkipped > 0) {
347           extra += " (" + nSkipped + " skipped)";
348         }
349       }
350 
351       out.println(result + "\t" + en.getKey() + "\t" + extra);
352     }
353     out.println();
354 
355     if (!important.isEmpty()) {
356       for (String l : important) {
357         out.println("IMPORTANT: " + l);
358       }
359       out.println();
360     }
361 
362     out.println("Supported capabilities:   " + supportedCapabilites);
363     out.println("Unsupported capabilities: " + unsupportedCapabilites);
364     out.println();
365 
366     if (fail || withIssues) {
367       if (inconclusive || modified) {
368         out.println("Selftest INCONCLUSIVE");
369       } else if (fail) {
370         out.println("Selftest FAILED");
371       } else if (withIssues) {
372         out.println("Selftest PASSED WITH ISSUES");
373       }
374     } else {
375       out.println("Selftest PASSED");
376     }
377   }

389   private boolean checkIfCapabilitiesSupported(String className) {
390     if (CAP_ANNOTATION_CLASS != null) {
391       try {
392         Class<?> klass = Class.forName(className);
393         Annotation annotation = klass.getAnnotation(CAP_ANNOTATION_CLASS);
394         if (annotation != null) {
395           try {
396             AFSocketCapability[] caps = (AFSocketCapability[]) annotation.getClass().getMethod(
397                 "value").invoke(annotation);
398             if (caps != null) {
399               for (AFSocketCapability cap : caps) {
400                 if (!AFSocket.supports(cap)) {
401                   out.println("Skipping class " + className + "; unsupported capability: " + cap);
402                   return false;
403                 }
404               }
405             }
406           } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException
407               | NoSuchMethodException | SecurityException e) {
408             // ignore
409           }
410         }
411       } catch (ClassNotFoundException e) {
412         out.println("Class not found: " + className);
413         withIssues = true;
414       }
415     }
416 
417     return true;
418   }

420   /**
421    * Runs the given test classes for the specified module.
422    * 
423    * @param module The module name.
424    * @param testClasses The test classes.
425    */
426   @SuppressWarnings({"PMD.ExcessiveMethodLength", "PMD.NcssCount", "PMD.NPathComplexity"})
427   public void runTests(String module, Class<?>[] testClasses) {
428     String prefix = "Testing \"" + module + "\"... ";
429     out.markPosition();
430     out.update(prefix);
431     out.flush();
432 
433     String only = System.getProperty("selftest.only", "");
434     if (!only.isEmpty()) {
435       modified = true;
436     }
437 
438     final ModuleResult moduleResult;
439     if (Boolean.valueOf(System.getProperty("selftest.skip." + module, "false"))) {
440       out.println("Skipping module " + module + "; skipped by request");
441       withIssues = true;
442       modified = true;
443       moduleResult = new ModuleResult(Result.SKIP, null, null);
444     } else {
445       List<Class<?>> list = new ArrayList<>(testClasses.length);
446       for (Class<?> testClass : testClasses) {
447         if (testClass == null) {
448           // ignore
449           continue;
450         }
451         String className = testClass.getName();
452         String simpleName = testClass.getSimpleName();
453 
454         if (!only.isEmpty() && !only.equals(className) && !only.equals(simpleName)) {
455           continue;
456         }
457 
458         String skipFullyQualifiedProp = System.getProperty("selftest.skip." + className, "");
459         boolean skipFullyQualified = !skipFullyQualifiedProp.isEmpty() && Boolean.valueOf(
460             skipFullyQualifiedProp);
461 
462         if (skipFullyQualified || Boolean.valueOf(System.getProperty("selftest.skip." + simpleName,
463             "false"))) {
464           out.println("Skipping test class " + className + "; skipped by request");
465           modified = true;
466           withIssues = true;
467         } else if (checkIfCapabilitiesSupported(className)) {
468           list.add(testClass);
469         }
470       }
471 
472       TestExecutionSummary summary = null;
473       Exception exception = null;
474       long numAbortedNonIssues = 0;
475       try {
476         SelftestExecutor ex = new SelftestExecutor(list, prefix);
477         summary = ex.execute(out);
478 
479         for (Map.Entry<TestIdentifier, TestExecutionResult> en : ex.getTestsWithWarnings()
480             .entrySet()) {
481           TestIdentifier tid = en.getKey();
482           TestExecutionResult res = en.getValue();
483           Optional<Throwable> t = res.getThrowable();
484           if (t.isPresent() && t.get() instanceof TestAbortedWithImportantMessageException) {
485             String key = module + ": " + ex.getTestIdentifier(tid.getParentId().get())
486                 .getDisplayName() + "." + tid.getDisplayName();
487             TestAbortedWithImportantMessageException ime =
488                 (TestAbortedWithImportantMessageException) t.get();
489 
490             MessageType messageType = ime.messageType();
491             if (messageType.isIncludeTestInfo()) {
492               important.add(key + ": " + ime.getMessage());
493             } else {
494               important.add(ime.getMessage());
495             }
496             if (!messageType.isWithIssues()) {
497               numAbortedNonIssues++;
498             }
499           }
500         }
501       } catch (Exception e) {
502         e.printStackTrace(out);
503         exception = e;
504       }
505 
506       if (exception != null || summary == null) {
507         moduleResult = new ModuleResult(Result.FAIL, null, exception);
508         fail = true;
509       } else {
510         final Result result;
511         if (summary.getTestsFailedCount() > 0) {
512           result = Result.FAIL;
513           fail = true;
514         } else if (summary.getTestsFoundCount() == 0) {
515           result = Result.NONE;
516           fail = true;
517         } else if ((summary.getTestsSucceededCount() + summary.getTestsSkippedCount()
518             + numAbortedNonIssues) == summary.getTestsFoundCount()) {
519           result = Result.PASS;
520         } else if (summary.getTestsAbortedCount() > 0) {
521           result = Result.DONE;
522           withIssues = true;
523         } else {
524           result = Result.DONE;
525         }
526 
527         moduleResult = new ModuleResult(result, summary, null);
528         moduleResult.numAbortedNonIssues = numAbortedNonIssues;
529       }
530     }
531     results.put(module, moduleResult);
532   }

534   private void dumpContentsOfSystemConfigFile(File file) {
535     if (!file.exists()) {
536       return;
537     }
538     String p = file.getAbsolutePath();
539     System.out.println("BEGIN contents of file: " + p);
540 
541     final int maxToRead = 4096;
542     char[] buf = new char[4096];
543     int numRead = 0;
544     try (InputStreamReader isr = new InputStreamReader(new FileInputStream(file),
545         StandardCharsets.UTF_8);) {
546 
547       OutputStreamWriter outWriter = new OutputStreamWriter(System.out, Charset.defaultCharset());
548       int read = -1;
549       boolean lastWasNewline = false;
550       while (numRead < maxToRead && (read = isr.read(buf)) != -1) {
551         numRead += read;
552         outWriter.write(buf, 0, read);
553         outWriter.flush();
554         lastWasNewline = (read > 0 && buf[read - 1] == '\n');
555       }
556       if (!lastWasNewline) {
557         System.out.println();
558       }
559       if (read != -1) {
560         System.out.println("[...]");
561       }
562     } catch (Exception e) {
563       System.out.println("ERROR while reading contents of file: " + p + ": " + e);
564     }
565     System.out.println("=END= contents of file: " + p);
566     System.out.println();
567   }

569   public void dumpOSReleaseFiles() throws IOException {
570     Set<Path> canonicalPaths = new HashSet<>();
571     for (String f : new String[] {
572         "/etc/os-release", "/etc/lsb-release", "/etc/lsb_release", "/etc/system-release",
573         "/etc/system-release-cpe",
574         //
575         "/etc/debian_version", "/etc/fedora-release", "/etc/redhat-release", "/etc/centos-release",
576         "/etc/centos-release-upstream", "/etc/SuSE-release", "/etc/arch-release",
577         "/etc/gentoo-release", "/etc/ubuntu-release",}) {
578 
579       File file = new File(f);
580       if (!file.exists() || file.isDirectory()) {
581         continue;
582       }
583       Path p = file.toPath().toAbsolutePath();
584       for (int i = 0; i < 2; i++) {
585         if (Files.isSymbolicLink(p)) {
586           Path p2 = Files.readSymbolicLink(p);
587           if (!p2.isAbsolute()) {
588             p = new File(p.toFile().getParentFile(), p2.toString()).toPath().toAbsolutePath();
589           }
590         }
591       }
592 
593       if (!canonicalPaths.add(p)) {
594         continue;
595       }
596 
597       dumpContentsOfSystemConfigFile(file);
598     }
599   }
