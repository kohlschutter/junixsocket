60      org.newsclub.net.unix.MassiveParallelTest<AFUNIXSocketAddress> {
61    private static final int MAX_SERVER_THREADS = 32;
62  
63    protected MassiveParallelTest() {
64      super(AFUNIXAddressSpecifics.INSTANCE);
65    }

67    @Test
68    @SuppressWarnings({"PMD.CognitiveComplexity", "PMD.AvoidCatchingThrowable"})
69    public void testAcceptConnect() throws Exception {
70      if (!ThreadUtil.isVirtualThreadSupported()) {
71        throw new TestAbortedNotAnIssueException("Virtual Threads are not supported by this JVM");
72      }
73      // number of connections to perform
74      // final int numConnections = 1_000;
75      final int numConnections = 10_000;
76      // final int numConnections = 100_000;
77      // final int numConnections = 1_000_000;
78  
79      // limit the number of concurrently active servers/clients
80      // so we don't run out of file descriptors (the limit could be as low as 256)
81      final int nProc = Math.min(MAX_SERVER_THREADS, Runtime.getRuntime().availableProcessors());
82      final Semaphore concurrentClientPermits = new Semaphore(Math.min(100, nProc));
83  
84      AFUNIXSocketAddress listenAddr = AFUNIXSocketAddress.ofNewTempFile();
85  
86      long startTime;
87  
88      try (Server server = new Server(listenAddr, nProc, numConnections)) {
89        ExecutorService esClients = ThreadUtil.newVirtualThreadPerTaskExecutor();
90        // ExecutorService esClients = Executors.newWorkStealingPool();
91        final AtomicInteger connectAttempts = new AtomicInteger(0);
92        final AtomicInteger connected = new AtomicInteger(0);
93  
94        Runnable clientJob = new Runnable() {
95  
96          @Override
97          public void run() {
98            try (AFUNIXSocket socket = AFUNIXSocket.newInstance()) {
99              concurrentClientPermits.acquire();
100             try {
101               connectAttempts.incrementAndGet();
102               socket.connect(listenAddr);
103             } catch (SocketTimeoutException | SocketException e) {
104               // try again
105               return;
106             }
107 
108             connected.incrementAndGet();
109             try (InputStream in = socket.getInputStream();
110                 OutputStream out = socket.getOutputStream()) {
111               int rcv = in.read();
112               if (rcv != 0xAA) {
113                 if (server.isRunning()) {
114                   System.err.println("Wrong data: " + rcv);
115                 }
116               }
117 
118               byte[] otherBytes = new byte[5];
119               int pos = 0;
120               int remaining = otherBytes.length;
121               while (remaining > 0) {
122                 int read = in.read(otherBytes, pos, remaining);
123                 if (read == -1) {
124                   break;
125                 }
126                 remaining -= read;
127                 pos += read;
128               }
129               if (remaining != 0) {
130                 if (server.isRunning()) {
131                   System.err.println("Incomplete data; bytes missing " + remaining);
132                 }
133               }
134               if (server.isRunning()) {
135                 if (!Arrays.equals(new byte[] {(byte) 0xAB, (byte) 0xAC, 'X', 'Y', 'Z'},
136                     otherBytes)) {
137                   System.err.println("Wrong data received: " + Arrays.toString(otherBytes) + " vs "
138                       + Arrays.toString(new byte[] {(byte) 0xAB, (byte) 0xAC, 'X', 'Y', 'Z'}));
139                 }
140               }
141 
142               out.write(0xBB);
143               out.flush();
144             }
145 
146           } catch (Throwable e) {
147             if (server.isRunning()) {
148               e.printStackTrace();
149             }
150           } finally {
151             concurrentClientPermits.release();
152             if (server.isRunning()) {
153               // keep trying
154               esClients.submit(this);
155             }
156           }
157         }
158       };
159 
160       startTime = System.currentTimeMillis();
161 
162       for (int i = 0; i < numConnections; i++) {
163         esClients.submit(clientJob);
164       }
165 
166       boolean stopped = server.cl.await(10, TimeUnit.SECONDS);
167       server.stop();
168       if (!stopped) {
169         List<Runnable> remainingClients = esClients.shutdownNow();
170         List<Runnable> remainingServers = server.esServers.shutdownNow();
171         System.err.println("Not all connections were made; remaining: " + server.cl.getCount());
172         if (!remainingClients.isEmpty() || !remainingServers.isEmpty()) {
173           System.err.println("Remaining threads: servers=" + remainingServers.size() + "; clients="
174               + remainingClients.size());
175         }
176       }
177 
178       long elapsed = (System.currentTimeMillis() - startTime);
179       int completed = server.completed.intValue();
180       System.out.println("millis: " + elapsed);
181       System.out.println("completed: " + completed);
182       float timePerItem = elapsed / (float) completed;
183       System.out.println("time per completed connection: " + timePerItem + " ms");
184 
185       if (completed <= nProc && completed < numConnections / 10.0) {
186         fail("Not enough jobs were completed: " + completed + "; expected:" + numConnections);
187       }
188     }
189   }

191   private static final class Server implements Closeable {
192     final AtomicBoolean running = new AtomicBoolean(true);
193     final AtomicInteger accepted = new AtomicInteger(0);
194     final AtomicInteger serverThreads = new AtomicInteger(0);
195     final AtomicInteger completed = new AtomicInteger(0);
196 
197     final AtomicInteger acceptsInFlight = new AtomicInteger(0);
198     final AFUNIXServerSocket serverSocket;
199     final ExecutorService esServers;
200 
201     final CountDownLatch cl;
202 
203     public Server(AFUNIXSocketAddress listenAddr, int numServerThreads, int stopAfterNumConnections)
204         throws IOException {
205       cl = new CountDownLatch(stopAfterNumConnections);
206       serverSocket = AFUNIXServerSocket.newInstance();
207       serverSocket.bind(listenAddr); // allow for some default backlog
208 
209       // esServers = Executors.newCachedThreadPool();
210       // esServers = Executors.newWorkStealingPool();
211       esServers = ThreadUtil.newVirtualThreadPerTaskExecutor();
212 
213       for (int i = 0; i < numServerThreads; i++) {
214         esServers.submit(this::acceptJob);
215       }
216     }
217 
218     public boolean isRunning() {
219       return running.get() && !serverSocket.isClosed();
220     }
221 
222     @Override
223     public String toString() {
224       return super.toString() + "[closed=" + serverSocket.isClosed() + ";running=" + running.get()
225           + ";inAccept=" + acceptsInFlight + ";completed=" + completed + "]";
226     }
227 
228     @SuppressFBWarnings({"NP_LOAD_OF_KNOWN_NULL_VALUE", "RCN_REDUNDANT_NULLCHECK_OF_NULL_VALUE"})
229     private void acceptJob() {
230       serverThreads.incrementAndGet();
231       acceptsInFlight.incrementAndGet();
232       try {
233         ByteBuffer bbNonDirect = ByteBuffer.allocate(64);
234         ByteBuffer bbDirect = ByteBuffer.allocateDirect(64);
235 
236         while (isRunning()) {
237           try (AFUNIXSocket socket = serverSocket.accept()) {
238             accepted.incrementAndGet();
239             if (socket == null) {
240               continue;
241             }
242             cl.countDown();
243             if (cl.getCount() == 0) {
244               stop();
245             }
246 
247             try (AFUNIXSocketChannel channel = socket.getChannel();
248                 OutputStream out = socket.getOutputStream();) {
249               channel.configureBlocking(true);
250               ByteBuffer bb = bbDirect;
251               bb.clear();
252               bb.put((byte) 0xAA);
253               bb.put((byte) 0xAB);
254               bb.put((byte) 0xAC);
255               bb.flip();
256 
257               while (bb.hasRemaining()) {
258                 channel.write(bb);
259               }
260 
261               out.write(new byte[] {'X', 'Y', 'Z'});
262               out.flush();
263 
264               bb = bbNonDirect;
265 
266               bb.clear();
267               int read = channel.read(bb);
268               if (read != 1) {
269                 if (isRunning()) {
270                   System.err.println("Wrong response: " + read + " bytes");
271                 }
272               } else {
273                 int rcv = bb.get(0) & 0xFF;
274                 if (rcv != 0xbb) {
275                   if (isRunning()) {
276                     System.err.println("Wrong response: 0x" + Integer.toHexString(rcv));
277                   }
278                 } else {
279                   completed.incrementAndGet();
280                 }
281               }
282             }
283           } catch (InterruptedIOException | ConnectionResetSocketException | SocketClosedException
284               | ClosedChannelException e) {
285             // ignore
286           } catch (SocketException e) {
287             if (isRunning()) {
288               throw e;
289             } else {
290               // ignore
291             }
292           }
293         }
294       } catch (Throwable e) { // NOPMD.ExceptionAsFlowControl,AvoidCatchingThrowable
295         if (isRunning()) {
296           e.printStackTrace();
297         }
298       } finally {
299         acceptsInFlight.decrementAndGet();
300         if (isRunning()) {
301           System.err.println("Restarting failed server job");
302           try {
303             esServers.submit(this::acceptJob);
304           } catch (RejectedExecutionException e) {
305             if (isRunning()) {
306               e.printStackTrace();
307             }
308           } catch (Throwable t) { // NOPMD.AvoidCatchingThrowable
309             t.printStackTrace();
310           }
311         }
312       }
313     }
314 
315     public void stop() {
316       running.set(false);
317       esServers.shutdown();
318     }
319 
320     @Override
321     public void close() throws IOException {
322       stop();
323       serverSocket.close();
324       try {
325         if (!esServers.awaitTermination(5, TimeUnit.SECONDS)) {
326           esServers.shutdownNow();
327           if (!esServers.awaitTermination(5, TimeUnit.SECONDS)) {
328             throw new InterruptedIOException("did not terminate");
329           }
330         }
331       } catch (InterruptedException e) {
332         throw (InterruptedIOException) new InterruptedIOException("did not terminate").initCause(e);
333       }
334     }
335   }
