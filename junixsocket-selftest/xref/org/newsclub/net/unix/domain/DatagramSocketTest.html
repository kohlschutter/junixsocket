44  @AFSocketCapabilityRequirement({
45      AFSocketCapability.CAPABILITY_UNIX_DOMAIN, AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS})
46  @SuppressFBWarnings("NM_SAME_SIMPLE_NAME_AS_SUPERCLASS")
47  public final class DatagramSocketTest extends
48      org.newsclub.net.unix.DatagramSocketTest<AFUNIXSocketAddress> {
49  
50    public DatagramSocketTest() {
51      super(AFUNIXAddressSpecifics.INSTANCE);
52    }
53  
54    @Test
55    public void testSeqPacketPair() throws Exception {
56      AFUNIXSocketPair<AFUNIXDatagramChannel> pair;
57      try {
58        pair = AFUNIXSocketPair.openDatagram(AFSocketType.SOCK_SEQPACKET);
59      } catch (OperationNotSupportedSocketException e) {
60        throw new TestAbortedNotAnIssueException("SEQPACKET not supported", e);
61      }
62  
63      String msg = "Hello World";
64  
65      ByteBuffer buf = ByteBuffer.wrap(msg.getBytes(StandardCharsets.UTF_8));
66      ByteBuffer dst = ByteBuffer.allocate(64);
67      pair.getSocket1().send(buf, null);
68      int r = pair.getSocket2().read(dst);
69      assertEquals(buf.limit(), r);
70      dst.flip();
71      assertEquals(buf.limit(), dst.limit());
72  
73      assertEquals(msg, StandardCharsets.UTF_8.decode(dst).toString());
74    }
75  
76    @Test
77    public void testSeqPacket() throws Exception {
78      boolean gotInstance = false;
79      try (AFUNIXDatagramSocket s1 = AFUNIXDatagramSocket.newInstance(AFSocketType.SOCK_SEQPACKET)) {
80        gotInstance = true;
81        AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
82        s1.bind(addr);
83        s1.listen(0);
84  
85        CompletableFuture<Void> cf = CompletableFuture.runAsync(() -> {
86          try {
87            AFDatagramSocket<AFUNIXSocketAddress> s;
88            s = s1.accept();
89  
90            ByteBuffer bb = ByteBuffer.allocate(1024).order(ByteOrder.LITTLE_ENDIAN);
91            s.getChannel().receive(bb);
92            bb.flip();
93            assertEquals(4, bb.remaining());
94            assertEquals(0x04030201, bb.getInt());
95          } catch (IOException e) {
96            fail(e);
97          }
98        });
99  
100       try (AFUNIXDatagramSocket s2 = AFUNIXDatagramSocket.newInstance(
101           AFSocketType.SOCK_SEQPACKET)) {
102         s2.connect(addr);
103 
104         s2.getChannel().send(ByteBuffer.wrap(new byte[] {1, 2, 3, 4}), addr);
105       }
106 
107       cf.get(5, TimeUnit.SECONDS);
108 
109     } catch (OperationNotSupportedSocketException e) {
110       if (!gotInstance) {
111         throw new TestAbortedNotAnIssueException("SEQPACKET not supported", e);
112       }
113     }
114   }
