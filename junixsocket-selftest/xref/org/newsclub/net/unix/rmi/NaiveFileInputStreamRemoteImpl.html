18  package org.newsclub.net.unix.rmi;
19  
20  import java.io.File;
21  import java.io.FileDescriptor;
22  import java.io.FileInputStream;
23  import java.io.IOException;
24  
25  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
26  import com.kohlschutter.util.IOUtil;
27  
28  public final class NaiveFileInputStreamRemoteImpl extends FileInputStream implements
29      NaiveFileInputStreamRemote {
30    private final RemoteFileInput rfd;
31  
32    public NaiveFileInputStreamRemoteImpl(AFUNIXRMISocketFactory socketFactory, File file)
33        throws IOException {
34      super(file);
35      this.rfd = new RemoteFileInput(socketFactory, this);
36  
37      AFNaming.exportObject(this, socketFactory);
38    }
39  
40    public NaiveFileInputStreamRemoteImpl(AFUNIXRMISocketFactory socketFactory, FileDescriptor fd)
41        throws IOException {
42      super(fd);
43      this.rfd = new RemoteFileInput(socketFactory, this);
44  
45      AFNaming.exportObject(this, socketFactory);
46    }
47  
48    @Override
49    @SuppressFBWarnings("EI_EXPOSE_REP")
50    public RemoteFileInput getRemoteFileDescriptor() {
51      return rfd;
52    }
53  
54    @Override
55    public void close() throws IOException {
56      AFNaming.unexportObject(this);
57      super.close();
58    }
59  
60    @Override
61    public byte[] readAllBytes() throws IOException {
62      return IOUtil.readAllBytesNaively(this);
63    }
64  }
