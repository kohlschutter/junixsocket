182   public <T extends AutoCloseable> void testSocketInterruption(boolean delay, IOSupplier<T> socket,
183       IOConsumer<T> blockingOp, Class<?> expectedException, Predicate<T> closeCheck)
184       throws Throwable {
185     AtomicReference<Throwable> exceptionHolder = new AtomicReference<>();
186     CountDownLatch ready = new CountDownLatch(1);
187     Thread t = ThreadUtil.startNewDaemonThread(true, () -> exceptionHolder.set(runOperation(ready,
188         socket, blockingOp, expectedException, closeCheck)));
189 
190     ready.await();
191     if (delay) {
192       Thread.sleep(500);
193     }
194     t.interrupt();
195     t.join(Duration.of(1, ChronoUnit.SECONDS).toMillis());
196     if (t.isAlive()) {
197       // Thread.interrupt is not guaranteed to succeed
198       // observed with graalvm-jdk-17.0.9+11.1 when building with agent for junixsocket-native-graalvm
199       // What we need to do here is to close all socket-related resources and try again
200       closeAfterTest();
201       t.interrupt();
202       t.join(Duration.of(1, ChronoUnit.SECONDS).toMillis());
203       if (t.isAlive()) {
204         throw new RuntimeException("Thread failed to terminate after interrupt");
205       }
206     }
207     Throwable thrownException = exceptionHolder.get();
208     if (thrownException != null) {
209       throw thrownException;
210     }
211   }

218   private void withServer(boolean acceptConnections, ThrowingRunnable func) throws Throwable {
219     Semaphore done = new Semaphore(0);
220     try (ServerSocketChannel serverSocket = registerCloseable(newServerSocketChannel())) {
221       serverSocket.bind(address);
222       Thread serverThread = null;
223       if (acceptConnections) {
224         serverThread = ThreadUtil.startNewDaemonThread(false, () -> {
225           while (serverSocket.isOpen()) {
226             SocketChannel socket = null;
227             try {
228               socket = serverSocket.accept();
229             } catch (ClosedChannelException e) {
230               return;
231             } catch (IOException e) {
232               throw new RuntimeException("Unable to accept socket ", e);
233             } finally {
234               if (socket != null) {
235                 final SocketChannel socketToClose = socket;
236                 if (DELAY_CLOSE) {
237                   CompletableFuture.runAsync(() -> {
238                     try {
239                       if (!done.tryAcquire(1, TimeUnit.SECONDS)) {
240                         // ignore
241                       }
242                     } catch (InterruptedException e) {
243                       // ignore
244                     }
245                     try {
246                       socketToClose.close();
247                     } catch (IOException e) {
248                       // TODO Auto-generated catch block
249                       e.printStackTrace();
250                     }
251                   });
252                 } else {
253                   try {
254                     socketToClose.close();
255                   } catch (IOException e) {
256                     // TODO Auto-generated catch block
257                     e.printStackTrace();
258                   }
259                 }
260               }
261             }
262           }
263         });
264       }
265       try {
266         func.run();
267       } finally {
268         done.release();
269         serverSocket.close();
270         if (serverThread != null) {
271           serverThread.join();
272         }
273       }
274     }
275   }

277   <T extends AutoCloseable> Throwable runOperation(CountDownLatch ready, IOSupplier<T> socket,
278       IOConsumer<T> blockingOp, Class<?> expectedException, Predicate<T> closeCheck) {
279 
280     boolean supported = false;
281     Exception exc = null;
282     try {
283       @SuppressWarnings({"resource"})
284       T sock = registerCloseable(socket.get());
285       ready.countDown();
286 
287       supported = true;
288       try {
289         blockingOp.accept(sock);
290       } catch (Exception e) {
291         exc = e;
292         // These tests usually expect the "Thread interrupted" state to be set.
293         // However, when we accept any SocketException to be thrown, that state is not
294         // deterministic.
295         // Also, when we expect any kind of ClosedChannelException, it is only expected to be
296         // set when the actual exception thrown is from the ClosedByInterruptException subclass.
297         boolean ignoreInterruptState = SocketException.class.equals(expectedException);
298         boolean interruptStateOK = Thread.interrupted() || (ClosedChannelException.class
299             .isAssignableFrom(expectedException) && !(e instanceof ClosedByInterruptException));
300 
301         if (!expectedException.isAssignableFrom(e.getClass())) {
302           // log unexpected exception stack trace
303           e.printStackTrace();
304         }
305 
306         assertAll(() -> assertInstanceOf(expectedException, e, "Socket exception"),
307             () -> assertTrue(ignoreInterruptState || interruptStateOK, "Thread interrupted"),
308             () -> assertTrue(closeCheck.test(sock), "Socket closed"));
309       } finally {
310         ready.countDown();
311         if (sock != null) {
312           try {
313             sock.close();
314           } catch (Exception e) {
315             throw new RuntimeException("Unable to clean up socket", e);
316           }
317         }
318       }
319     } catch (TestAbortedNotAnIssueException e) {
320       return e;
321     } catch (Throwable e) {
322       e.printStackTrace();
323       return e;
324     } finally {
325       ready.countDown();
326       if (DEBUG) {
327         // print concise results for debugging:
328         if (DEBUG_VERBOSE) {
329           System.out.print(testInfo.getTestClass().get().getName() + "." + testInfo.getTestMethod()
330               .get().getName() + " " + testInfo.getDisplayName() + ": ");
331         }
332         System.out.println((supported ? (exc == null ? "no exception" : exc) : "unsupported"));
333       }
334     }
335     return null;
336   }

348   private SocketChannel connectSocketChannel() throws IOException {
349     SocketChannel socket = registerCloseable(newSocketChannel());
350     socket.connect(address);
351     return socket;
352   }
353 
354   private ServerSocketChannel bindServerSocketChannel() throws IOException {
355     ServerSocketChannel socket = registerCloseable(newServerSocketChannel());
356     try {
357       try {
358         socket.bind(address);
359       } catch (BindException e) {
360         // With Inet sockets, our reserved address may just have been taken by another process
361         // so let's try again
362         address = newAddress();
363         socket.bind(address);
364       }
365     } catch (BindException e) {
366       throw (BindException) new BindException(e.getMessage() + ": " + address).initCause(e);
367     }
368     return socket;
369   }
