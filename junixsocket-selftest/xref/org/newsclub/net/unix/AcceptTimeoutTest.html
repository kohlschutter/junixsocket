46  /**
47   * Verifies that accept properly times out when an soTimeout was specified.
48   *
49   * @author Christian Kohlschütter

51  @SuppressFBWarnings({
52      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
53  public abstract class AcceptTimeoutTest<A extends SocketAddress> extends SocketTestBase<A> {
54    private static final int TIMING_INACCURACY_MILLIS = 5000;
55  
56    protected AcceptTimeoutTest(AddressSpecifics<A> asp) {
57      super(asp);
58    }

60    @Test
61    public void testCatchTimeout() throws Exception {
62      final int timeoutMillis = 500;
63      assertTimeoutPreemptively(Duration.ofMillis(5 * timeoutMillis), () -> {
64        try (ServerSocket sock = startServer()) {
65          long time = System.currentTimeMillis();
66          sock.setSoTimeout(timeoutMillis);
67          long actualTimeout = sock.getSoTimeout();
68          if (actualTimeout == 0) {
69            // timeout not supported. So far we know this is only true for z/OS
70            if ("z/OS".equals(System.getProperty("os.name"))) {
71              return;
72            }
73          }
74          assertTrue(Math.abs(timeoutMillis - actualTimeout) <= TIMING_INACCURACY_MILLIS,
75              "We should roughly get the same timeout back that we set before, but was "
76                  + actualTimeout + " instead of " + timeoutMillis);
77          try (Socket socket = sock.accept()) {
78            fail("Did not receive " + SocketTimeoutException.class.getName() + "; socket=" + socket);
79          } catch (SocketException | SocketTimeoutException e) {
80            // expected
81            time = System.currentTimeMillis() - time;
82  
83            assertTrue(Math.abs(time - timeoutMillis) <= TIMING_INACCURACY_MILLIS,
84                "Timeout not properly honored. Exception thrown after " + time + "ms vs. expected "
85                    + timeoutMillis + "ms");
86          }
87        }
88      });
89    }

91    @Test
92    @SuppressWarnings({"PMD.CognitiveComplexity", "PMD.ExcessiveMethodLength"})
93    @SuppressFBWarnings("NP_NONNULL_PARAM_VIOLATION")
94    public void testTimeoutAfterDelay() throws Exception {
95      final int timeoutMillis = 5000;
96  
97      AtomicBoolean keepRunning = new AtomicBoolean(true);
98  
99      CompletableFuture<SocketAddress> serverAddressCF = new CompletableFuture<>();
100     try {
101       assertTimeoutPreemptively(Duration.ofMillis(2 * timeoutMillis), () -> {
102         try (ServerSocket serverSock = startServer()) {
103           final int connectDelayMillis = 50;
104           serverSock.setSoTimeout(timeoutMillis);
105 
106           long actualTimeout = serverSock.getSoTimeout();
107           if (actualTimeout == 0) {
108             // timeout not supported. So far we know this is only true for z/OS
109             if ("z/OS".equals(System.getProperty("os.name"))) {
110               return;
111             }
112           }
113           assertTrue(Math.abs(timeoutMillis - actualTimeout) <= 10,
114               "We should roughly get the same timeout back that we set before, but was "
115                   + actualTimeout + " instead of " + timeoutMillis);
116 
117           final AtomicBoolean accepted = new AtomicBoolean(false);
118           final CompletableFuture<RuntimeException> runtimeExceptionCF = new CompletableFuture<>();
119 
120           SocketAddress serverAddress = serverSock.getLocalSocketAddress();
121           serverAddressCF.complete(serverAddress);
122 
123           final Socket threadSocket = newSocket();
124           Thread t = new Thread(() -> {
125             int i = 0;
126             while (keepRunning.get()) {
127               i++;
128               try {
129                 Thread.sleep(connectDelayMillis);
130               } catch (InterruptedException e) {
131                 return;
132               }
133 
134               try {
135                 connectSocket(threadSocket, serverAddress);
136                 runtimeExceptionCF.complete(null);
137               } catch (SocketTimeoutException e) {
138                 if (!keepRunning.get()) {
139                   return;
140                 }
141 
142                 System.out.println("SocketTimeout, trying connect again (" + i + ")");
143                 continue;
144               } catch (TestAbortedWithImportantMessageException e) {
145                 runtimeExceptionCF.complete(e);
146               } catch (IOException e) {
147                 // ignore "connection reset by peer", etc. after connection was accepted
148                 if (!accepted.get()) {
149                   e.printStackTrace();
150                 }
151               }
152 
153               break; // NOPMD.AvoidBranchingStatementAsLastInLoop
154             }
155           });
156           t.setDaemon(true);
157           t.start();
158 
159           long time = System.currentTimeMillis();
160           try (Socket socket = serverSock.accept();) {
161             assertNotNull(socket);
162             accepted.set(true);
163           } catch (SocketTimeoutException e) {
164             String msg = checkKnownBugAcceptTimeout(e);
165             if (msg == null) {
166               throw e;
167             } else {
168               throw new TestAbortedWithImportantMessageException(
169                   MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg), e);
170             }
171           }
172           time = System.currentTimeMillis() - time;
173 
174           RuntimeException re = runtimeExceptionCF.get();
175           if (re != null) {
176             throw re;
177           }
178 
179           assertTrue(time >= connectDelayMillis && (time < timeoutMillis || (time
180               - connectDelayMillis) <= TIMING_INACCURACY_MILLIS),
181               "Timeout not properly honored. Accept succeeded after " + time + "ms vs. expected "
182                   + timeoutMillis + "ms");
183         }
184       });
185     } catch (AssertionFailedError e) {
186       String msg = checkKnownBugAcceptTimeout(serverAddressCF.getNow(null));
187       if (msg == null) {
188         throw e;
189       } else {
190         throw new TestAbortedWithImportantMessageException(
191             MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg), e);
192       }
193     } finally {
194       keepRunning.set(false);
195     }
196   }

230   @Test
231   public void testPendingAcceptCloseServerSocketImmediately() throws Exception {
232     testPendingAcceptCloseServerSocket(false);
233   }
234 
235   @Test
236   public void testPendingAcceptCloseServerSocketDelayed() throws Exception {
237     testPendingAcceptCloseServerSocket(true);
238   }
239 
240   private void testPendingAcceptCloseServerSocket(boolean delayed) throws Exception {
241     SocketAddress addr = newTempAddress();
242     ServerSocket ss = newServerSocketBindOn(addr);
243 
244     Runnable doClose = () -> {
245       try {
246         ss.close();
247       } catch (IOException e) {
248         TestStackTraceUtil.printStackTrace(e);
249       }
250     };
251 
252     if (delayed) {
253       TestAsyncUtil.runAsyncDelayed(1, TimeUnit.SECONDS, doClose);
254     } else {
255       doClose.run();
256     }
257 
258     assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
259       // Should be SocketClosedException or InvalidArgumentSocketException, but no guarantee
260       assertThrows(SocketException.class, () -> {
261         try (Socket unused = ss.accept()) {
262           fail("Should not be reached");
263         } catch (SocketException e) {
264           throw e;
265         }
266       });
267     });
268   }
