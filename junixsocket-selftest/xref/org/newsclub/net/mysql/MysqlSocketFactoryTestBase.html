30  import org.junit.jupiter.api.Test;
31  import org.newsclub.net.unix.AFUNIXSocketAddress;

33  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException;
34  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException.MessageType;
35  import com.mysql.cj.jdbc.exceptions.CommunicationsException;
36  
37  abstract class MysqlSocketFactoryTestBase {
38  
39    @Test
40    public void testDriverManagerConnectionToMissingServer() throws Exception {
41      AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
42      try (Connection unused = DriverManager.getConnection("jdbc:mysql://localhost/db?socketFactory="
43          + socketFactory() + "&junixsocket.file=" + addr.getPath())) {
44        fail(
45            "Should have thrown an exception since we're trying to connect to a non-existing database server");
46      } catch (CommunicationsException e) {
47        // expected
48      }
49    }
50  
51    @Test
52    public void testDriverManagerConnectionToMysqlSock() throws Exception {
53      MysqlCredentials creds = MysqlCredentials.getCredentials();
54  
55      AFUNIXSocketAddress addr = AFUNIXSocketAddress.of(new File("/tmp/mysql.sock"));
56      boolean connected = false;
57      try (Connection conn = DriverManager.getConnection("jdbc:mysql://localhost/mysql?socketFactory="
58          + socketFactory() + "&junixsocket.file=" + addr.getPath(), creds.getUser(), creds
59              .getPassword())) {
60        connected = true;
61  
62        try (PreparedStatement pstmt = conn.prepareStatement("SHOW TABLES");
63            ResultSet rs = pstmt.executeQuery()) {
64          if (rs.next()) {
65            assertNotNull(rs.getString(1));
66          }
67        }
68      } catch (SQLException e) {
69        if (connected) {
70          throw e;
71        } else {
72          throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_WITH_ISSUES,
73              "Could not connect to MySQL database for testing. Please correct the following definitions: "
74                  + creds.toString(), e);
75        }
76      }
77    }
78  
79    protected abstract String socketFactory();
80  }
