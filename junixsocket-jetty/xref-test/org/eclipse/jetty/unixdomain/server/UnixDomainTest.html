101   private void start(Handler handler) throws Exception {
102     server = new Server();
103     // UnixDomainServerConnector connector = new UnixDomainServerConnector(server, factories);
104     AFSocketServerConnector connector = new AFSocketServerConnector(server, factories);
105     String dir = System.getProperty("jetty.unixdomain.dir", System.getProperty("java.io.tmpdir"));
106     assertNotNull(dir);
107     File dirFile = new File(dir);
108     dirFile.mkdirs();
109     unixDomainPath = Files.createTempFile(dirFile.toPath(), "unix_", ".sock");
110     assertTrue(unixDomainPath.toAbsolutePath().toString().length() < 108,
111         "Unix-Domain path too long");
112     Files.delete(unixDomainPath);
113     // connector.setUnixDomainPath(unixDomainPath);
114     connector.setListenSocketAddress(AFUNIXSocketAddress.of(unixDomainPath));
115     server.addConnector(connector);
116     server.setHandler(handler);
117     server.start();
118   }

125   @Test
126   public void testHTTPOverUnixDomain() throws Exception {
127     String uri = "http://localhost:1234/path";
128     start(new AbstractHandler() {
129       @Override
130       @SuppressWarnings("deprecation")
131       public void handle(String target, Request jettyRequest, HttpServletRequest request,
132           HttpServletResponse response) {
133         jettyRequest.setHandled(true);
134 
135         // Verify the URI is preserved.
136         assertEquals(uri, request.getRequestURL().toString());
137 
138         EndPoint endPoint = jettyRequest.getHttpChannel().getEndPoint();
139 
140         // Verify the SocketAddresses.
141         SocketAddress local = endPoint.getLocalSocketAddress();
142         assertThat(local, Matchers.instanceOf(AFSocketAddress.class));
143         SocketAddress remote = endPoint.getRemoteSocketAddress();
144         if (remote != null) {
145           // remote should be null if not connected
146           assertThat(remote, Matchers.instanceOf(AFSocketAddress.class));
147         }
148 
149         // Verify that other address methods don't throw.
150         SocketAddress inetLocal = assertDoesNotThrow(endPoint::getLocalAddress);
151         // assertNull(local); // junixsocket's addresses extend InetSocketAddress
152         assertEquals(local, inetLocal);
153         SocketAddress inetRemote = assertDoesNotThrow(endPoint::getRemoteAddress);
154         // assertNull(remote); // junixsocket's addresses extend InetSocketAddress
155         assertEquals(remote, inetRemote);
156 
157         assertDoesNotThrow(endPoint::toString);
158       }
159     });
160 
161     // ClientConnector clientConnector = ClientConnector.forUnixDomain(unixDomainPath);
162     ClientConnector clientConnector = AFSocketClientConnector.withSocketAddress(AFUNIXSocketAddress
163         .of(unixDomainPath));
164     HttpClient httpClient = new HttpClient(new HttpClientTransportDynamic(clientConnector));
165     httpClient.start();
166     try {
167       ContentResponse response = httpClient.newRequest(uri).timeout(5, TimeUnit.SECONDS).send();
168 
169       assertEquals(HttpStatus.OK_200, response.getStatus());
170     } finally {
171       httpClient.stop();
172     }
173   }

208   @Test
209   public void testHTTPOverUnixDomainWithProxyProtocol() throws Exception {
210     String srcAddr = "/proxySrcAddr";
211     String dstAddr = "/proxyDstAddr";
212     factories = new ConnectionFactory[] {new ProxyConnectionFactory(), new HttpConnectionFactory()};
213     start(new AbstractHandler() {
214       @Override
215       public void handle(String target, Request jettyRequest, HttpServletRequest request,
216           HttpServletResponse response) {
217         jettyRequest.setHandled(true);
218         EndPoint endPoint = jettyRequest.getHttpChannel().getEndPoint();
219         assertThat(endPoint, Matchers.instanceOf(ProxyConnectionFactory.ProxyEndPoint.class));
220         // assertThat(endPoint.getLocalSocketAddress(),
221         // Matchers.instanceOf(unixDomainSocketAddressClass));
222         // assertThat(endPoint.getRemoteSocketAddress(),
223         // Matchers.instanceOf(unixDomainSocketAddressClass));
224         if ("/v1".equals(target)) {
225           // As PROXYv1 does not support UNIX, the wrapped EndPoint data is used.
226           Path localPath = toUnixDomainPath(endPoint.getLocalSocketAddress());
227           assertThat(localPath, Matchers.equalTo(unixDomainPath));
228         } else if ("/v2".equals(target)) {
229           SocketAddress localSocketAddress = endPoint.getLocalSocketAddress();
230           if (localSocketAddress != null) {
231             assertThat(toUnixDomainPath(localSocketAddress).toString(), Matchers.equalTo(separators(
232                 dstAddr)));
233           }
234           SocketAddress remoteSocketAddress = endPoint.getRemoteSocketAddress();
235           if (remoteSocketAddress != null) {
236             assertThat(toUnixDomainPath(remoteSocketAddress).toString(), Matchers.equalTo(
237                 separators(srcAddr)));
238           }
239         } else {
240           Assertions.fail("Invalid PROXY protocol version " + target);
241         }
242       }
243     });
244 
245     // Java 11+ portable way to implement SocketChannelWithAddress.Factory.
246     // ClientConnector clientConnector = ClientConnector.forUnixDomain(unixDomainPath);
247     ClientConnector clientConnector = AFSocketClientConnector.withSocketAddress(AFUNIXSocketAddress
248         .of(unixDomainPath));
249 
250     HttpClient httpClient = new HttpClient(new HttpClientTransportDynamic(clientConnector));
251     httpClient.start();
252     try {
253       // Try PROXYv1 with the PROXY information retrieved from the EndPoint.
254       // PROXYv1 does not support the UNIX family.
255       ContentResponse response1 = httpClient.newRequest("localhost", 0).path("/v1").tag(
256           new V1.Tag()).timeout(5, TimeUnit.SECONDS).send();
257 
258       assertEquals(HttpStatus.OK_200, response1.getStatus());
259 
260       // Try PROXYv2 with explicit PROXY information.
261       V2.Tag tag = new V2.Tag(V2.Tag.Command.PROXY, V2.Tag.Family.UNIX, V2.Tag.Protocol.STREAM,
262           srcAddr, 0, dstAddr, 0, null);
263       ContentResponse response2 = httpClient.newRequest("localhost", 0).path("/v2").tag(tag)
264           .timeout(5, TimeUnit.SECONDS).send();
265 
266       assertEquals(HttpStatus.OK_200, response2.getStatus());
267     } finally {
268       httpClient.stop();
269     }
270   }

274     server = new Server();
275     // UnixDomainServerConnector connector = new UnixDomainServerConnector(server, factories);
276     AFSocketServerConnector connector = new AFSocketServerConnector(server, factories);
277 
278     // connector.setUnixDomainPath(new File("/does/not/exist").toPath());
279     try {
280       connector.setListenSocketAddress(AFUNIXSocketAddress.of(new File("/does/not/exist")
281           .toPath()));
282     } catch (SocketException e) {
283       throw new IllegalStateException(e);
284     }
285 
286     server.addConnector(connector);
287     assertThrows(IOException.class, () -> server.start());
288   }

290   private static Path toUnixDomainPath(SocketAddress address) {
291     Objects.requireNonNull(address, "address");
292 
293     if (address instanceof AFUNIXSocketAddress) {
294       return new File(((AFUNIXSocketAddress) address).getPath()).toPath();
295     } else if (unixDomainSocketAddressClass != null) {
296       try {
297         return (Path) unixDomainSocketAddressClass.getMethod("getPath").invoke(address);
298       } catch (IllegalAccessException | InvocationTargetException | NoSuchMethodException
299           | SecurityException e) {
300         Assertions.fail(e);
301         throw new AssertionError(e);
302       }
303     } else {
304       throw new IllegalStateException("Unsupported socket address class " + address.getClass()
305           + ": " + address);
306     }
307   }

309   public static String separators(String path) {
310     StringBuilder ret = new StringBuilder();
311     for (char c : path.toCharArray()) {
312       if ((c == '/') || (c == '\\')) {
313         ret.append(File.separatorChar);
314       } else {
315         ret.append(c);
316       }
317     }
318     return ret.toString();
319   }
