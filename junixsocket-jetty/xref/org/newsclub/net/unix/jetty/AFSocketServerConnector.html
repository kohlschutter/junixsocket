130   /**
131    * Creates a new {@link AFSocketServerConnector}.
132    *
133    * @param server The server this connector will be added to. Must not be null.
134    * @param executor An executor for this connector or null to use the servers executor
135    * @param scheduler A scheduler for this connector or null to either a {@link Scheduler} set as a
136    *          server bean or if none set, then a new {@link ScheduledExecutorScheduler} instance.
137    * @param pool A buffer pool for this connector or null to either a {@link ByteBufferPool} set as
138    *          a server bean or none set, the new {code ArrayByteBufferPool} instance.
139    * @param acceptors the number of acceptor threads to use, or -1 for a default value. If 0, then
140    *          no acceptor threads will be launched and some other mechanism will need to be used to
141    *          accept new connections.
142    * @param selectors The number of selectors to use, or -1 for a default derived
143    * @param factories The Connection Factories to use.
144    */
145   @SuppressFBWarnings("EI_EXPOSE_REP2")
146   public AFSocketServerConnector(Server server, Executor executor, Scheduler scheduler,
147       ByteBufferPool pool, int acceptors, int selectors, ConnectionFactory... factories) {
148     super(server, executor, scheduler, pool, acceptors, factories.length > 0 ? factories
149         : new ConnectionFactory[] {new HttpConnectionFactory()});
150     this.server = server;
151     this.selectorManager = newSelectorManager(getExecutor(), getScheduler(), selectors);
152     addBean(selectorManager, true);
153 
154     this.selectorManagerListenerClass = findSelectorManagerListenerClass();
155   }
156 
157   @SuppressWarnings("unchecked")
158   private static Class<? extends EventListener> findSelectorManagerListenerClass() {
159     try {
160       return (Class<? extends EventListener>) Class.forName(
161           "org.eclipse.jetty.io.SelectorManager$SelectorManagerListener");
162     } catch (ClassNotFoundException e) {
163       return null;
164     }
165   }
166 
167   private SelectorManager newSelectorManager(Executor executor, Scheduler scheduler,
168       int selectors) {
169     return new AFSocketSelectorManager(executor, scheduler, selectors);
170   }

173    * Returns the Unix-Domain path this connector listens to.
174    *
175    * Added for compatibility with jetty's {@code UnixDomainServerConnector}.
176    *
177    * @return The Unix-Domain path this connector listens to.
178    * @deprecated Use {@link #getListenSocketAddress()} instead.
179    * @see #getListenSocketAddress()
180    */
181   @ManagedAttribute("The Unix-Domain path this connector listens to")
182   public Path getUnixDomainPath() {
183     if (listenSocketAddress instanceof AFUNIXSocketAddress) {
184       AFUNIXSocketAddress addr = (AFUNIXSocketAddress) listenSocketAddress;
185       if (addr.hasFilename()) {
186         try {
187           return addr.getFile().toPath();
188         } catch (FileNotFoundException e) {
189           return null;
190         }
191       }
192     }
193     return null;
194   }

310   @Override
311   protected void doStart() throws Exception {
312     if (selectorManagerListenerClass != null) {
313       getBeans(selectorManagerListenerClass).forEach(selectorManager::addEventListener);
314     }
315     serverChannel = open();
316     addBean(serverChannel);
317     super.doStart();
318   }
319 
320   @Override
321   protected void doStop() throws Exception {
322     super.doStop();
323     removeBean(serverChannel);
324     close();
325     if (selectorManagerListenerClass != null) {
326       getBeans(selectorManagerListenerClass).forEach(selectorManager::removeEventListener);
327     }
328   }

330   @Override
331   @SuppressWarnings("PMD.CognitiveComplexity")
332   protected void accept(int acceptorID) throws IOException {
333     ServerSocketChannel sc = this.serverChannel;
334     if (sc != null) {
335       try {
336         SocketChannel channel = sc.accept();
337         accepted(channel);
338       } catch (SocketException e) {
339         boolean takenOver = !sc.isOpen() || sc.getLocalAddress() == null;
340         if (!takenOver && sc instanceof AFServerSocketChannel<?>) {
341           takenOver = !((AFServerSocketChannel<?>) sc).isLocalSocketAddressValid();
342         }
343 
344         if (takenOver && isMayStopServer()) {
345           LOG.warn("Another server has taken over our address");
346           CompletableFuture.runAsync(this::checkServerStop);
347         }
348         throw (ClosedByInterruptException) new ClosedByInterruptException().initCause(e);
349       }
350     }
351   }
352 
353   private void checkServerStop() {
354     Connector[] connectors = server.getConnectors();
355 
356     if (connectors != null && !isMayStopServerForce()) {
357       for (Connector conn : connectors) {
358         if (conn != AFSocketServerConnector.this && conn.isRunning()) { // NOPMD.CompareObjectsWithEquals
359           return; // don't stop
360         }
361       }
362     }
363 
364     LOG.warn("Server has no other connectors; shutting down: " + server); // NOPMD
365 
366     try {
367       server.stop();
368     } catch (Exception e1) {
369       LOG.warn("Exception upon stopping " + server, e1); // NOPMD
370     }
371   }
372 
373   private void accepted(SocketChannel channel) throws IOException {
374     channel.configureBlocking(false);
375     configure(channel);
376     selectorManager.accept(channel);
377   }
378 
379   /**
380    * Configures an incoming {@link SocketChannel}, setting socket options such as receive and send
381    * buffer sizes.
382    *
383    * @param channel The socket channel to configure.
384    * @throws IOException on error.
385    */
386   protected void configure(SocketChannel channel) throws IOException {
387     channel.setOption(StandardSocketOptions.SO_REUSEADDR, true);
388     int rcvBufSize = getAcceptedReceiveBufferSize();
389     if (rcvBufSize > 0) {
390       channel.setOption(StandardSocketOptions.SO_RCVBUF, rcvBufSize);
391     }
392     int sndBufSize = getAcceptedSendBufferSize();
393     if (sndBufSize > 0) {
394       channel.setOption(StandardSocketOptions.SO_SNDBUF, sndBufSize);
395     }
396   }

404   private ServerSocketChannel open() throws IOException {
405     ServerSocketChannel sc = openServerSocketChannel();
406     if (getAcceptors() == 0) {
407       sc.configureBlocking(false);
408       acceptor.set(selectorManager.acceptor(sc));
409     }
410     return sc;
411   }
412 
413   private void close() throws IOException {
414     ServerSocketChannel sc = this.serverChannel;
415     this.serverChannel = null;
416     IO.close(sc);
417   }
418 
419   private ServerSocketChannel openServerSocketChannel() throws IOException {
420     ServerSocketChannel sc = null;
421     if (isInheritChannel()) {
422       Channel channel = System.inheritedChannel();
423       if (channel instanceof ServerSocketChannel) {
424         sc = (ServerSocketChannel) channel;
425       } else {
426         LOG.warn( // NOPMD.GuardLogStatement
427             "Unable to use System.inheritedChannel() {}. Trying a new ServerSocketChannel at {}",
428             channel, getListenSocketAddress());
429       }
430     }
431     if (sc == null) {
432       sc = bindServerSocketChannel();
433     }
434     return sc;
435   }
436 
437   private ServerSocketChannel bindServerSocketChannel() throws IOException {
438     AFSocketAddress socketAddress = listenSocketAddress;
439     AFServerSocketChannel<?> sc = socketAddress.getAddressFamily().newServerSocketChannel();
440 
441     try {
442       sc.bind(socketAddress, getAcceptQueueSize());
443       return sc;
444     } catch (IOException x) {
445       String message = String.format(Locale.ENGLISH, "Could not bind %s to %s",
446           AFSocketServerConnector.class.getSimpleName(), listenSocketAddress);
447       throw new IOException(message, x);
448     }
449   }

451   @Override
452   public void setAccepting(boolean accepting) {
453     super.setAccepting(accepting);
454     if (getAcceptors() == 0) {
455       return;
456     }
457     if (accepting) {
458       if (acceptor.get() == null) {
459         Closeable cl = selectorManager.acceptor(serverChannel);
460         if (!this.acceptor.compareAndSet(null, cl)) {
461           IO.close(cl);
462         }
463       }
464     } else {
465       Closeable cl = this.acceptor.get();
466       if (cl != null && this.acceptor.compareAndSet(cl, null)) {
467         IO.close(cl);
468       }
469     }
470   }

478   private final class AFSocketSelectorManager extends SelectorManager {
479     public AFSocketSelectorManager(Executor executor, Scheduler scheduler, int selectors) {
480       super(executor, scheduler, selectors);
481     }
482 
483     @Override
484     protected Selector newSelector() throws IOException {
485       SelectorProvider provider = listenSocketAddress.getAddressFamily().getSelectorProvider();
486       return provider.openSelector();
487     }
488 
489     @Override
490     protected void accepted(SelectableChannel channel) throws IOException {
491       AFSocketServerConnector.this.accepted((SocketChannel) channel);
492     }
493 
494     @Override
495     protected EndPoint newEndPoint(SelectableChannel channel, ManagedSelector selector,
496         SelectionKey selectionKey) {
497       SocketChannelEndPoint endPoint = new SocketChannelEndPoint((SocketChannel) channel, selector,
498           selectionKey, getScheduler());
499       endPoint.setIdleTimeout(getIdleTimeout());
500       return endPoint;
501     }
502 
503     @Override
504     public Connection newConnection(SelectableChannel channel, EndPoint endpoint,
505         Object attachment) {
506       return getDefaultConnectionFactory().newConnection(AFSocketServerConnector.this, endpoint);
507     }
508 
509     @Override
510     protected void endPointOpened(EndPoint endpoint) {
511       super.endPointOpened(endpoint);
512       onEndPointOpened(endpoint);
513     }
514 
515     @Override
516     protected void endPointClosed(EndPoint endpoint) {
517       onEndPointClosed(endpoint);
518       super.endPointClosed(endpoint);
519     }
520   }
