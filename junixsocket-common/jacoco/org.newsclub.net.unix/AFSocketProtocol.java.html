package org.newsclub.net.unix;

/**
 * Describes the "protocol" part of a socket.
 *
 * @author Christian Kohlschütter
 */
public enum AFSocketProtocol {
  /**
   * The "default" protocol.
   */
  DEFAULT(0);

  private final int id;

  AFSocketProtocol(int id) {
    this.id = id;
  }

  int getId() {
    return id;
  }
}
