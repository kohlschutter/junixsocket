18  package org.newsclub.net.unix;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.net.InetAddress;
23  import java.net.Socket;
24  import java.net.URLDecoder;
25  import java.net.UnknownHostException;
26  
27  import javax.net.SocketFactory;
28  
29  /**
30   * The base for a SocketFactory that connects to UNIX sockets.
31   * 
32   * Typically, the "hostname" is used as a reference to a socketFile on the file system. The actual
33   * mapping is left to the implementor.
34   * 
35   * Three default implementations are provided.
36   * 
37   * @see AFUNIXSocketFactory.FactoryArg
38   * @see AFUNIXSocketFactory.SystemProperty
39   * @see AFUNIXSocketFactory.URIScheme
40   */
41  public abstract class AFUNIXSocketFactory extends SocketFactory {
42    /**
43     * Translates a "host" string (and port) to an {@link AFUNIXSocketAddress}.
44     * 
45     * @param host The hostname
46     * @param port The port, or 0.
47     * @return The {@link AFUNIXSocketAddress}
48     * @throws IOException If there was a problem converting the hostname
49     * @throws NullPointerException If host was {@code null}.
50     */
51    protected abstract AFUNIXSocketAddress addressFromHost(String host, int port) throws IOException;
52  
53    /**
54     * Checks whether the given hostname is supported by this socket factory. If not, calls to
55     * createSocket will cause an UnknownHostException.
56     * 
57     * @param host The host to check.
58     * @return {@code true} if supported.
59     */
60    protected boolean isHostnameSupported(String host) {
61      return host != null;
62    }
63  
64    /**
65     * Checks whether the given {@link InetAddress} is supported by this socket factory. If not, calls
66     * to createSocket will cause an UnknownHostException.
67     * 
68     * By default, this only checks the hostname part of the address via
69     * {@link #isHostnameSupported(String)}.
70     * 
71     * @param address The address to check.
72     * @return {@code true} if supported.
73     */
74    protected boolean isInetAddressSupported(InetAddress address) {
75      return address != null && isHostnameSupported(address.getHostName());
76    }
77  
78    @Override
79    public Socket createSocket() throws IOException {
80      return AFUNIXSocket.newInstance(this);
81    }
82  
83    @Override
84    public Socket createSocket(String host, int port) throws IOException, UnknownHostException {
85      if (!isHostnameSupported(host)) {
86        throw new UnknownHostException();
87      }
88      if (port < 0) {
89        throw new IllegalArgumentException("Illegal port");
90      }
91  
92      AFUNIXSocketAddress socketAddress = addressFromHost(host, port);
93      String socketFile = socketAddress.getSocketFile();
94      if (socketFile == null || socketFile.isEmpty()) {
95        throw new UnknownHostException("Illegal socketFile");
96      }
97      File file = new File(socketFile);
98      if (!file.exists()) {
99        throw new UnknownHostException("Socketfile does not exist");
100     }
101     return AFUNIXSocket.connectTo(socketAddress);
102   }
103 
104   @Override
105   public Socket createSocket(String host, int port, InetAddress localHost, int localPort)
106       throws IOException, UnknownHostException {
107     if (!isHostnameSupported(host)) {
108       throw new UnknownHostException();
109     }
110     if (localPort < 0) {
111       throw new IllegalArgumentException("Illegal local port");
112     }
113     // NOTE: we simply ignore localHost and localPort
114     return createSocket(host, port);
115   }
116 
117   @Override
118   public Socket createSocket(InetAddress address, int port) throws IOException {
119     if (!isInetAddressSupported(address)) {
120       throw new UnknownHostException();
121     }
122     String hostname = address.getHostName();
123     if (!isHostnameSupported(hostname)) {
124       throw new UnknownHostException();
125     }
126     return createSocket(hostname, port);
127   }
128 
129   @Override
130   public Socket createSocket(InetAddress address, int port, InetAddress localAddress, int localPort)
131       throws IOException {
132     if (!isInetAddressSupported(address)) {
133       throw new UnknownHostException();
134     }
135     if (localAddress == null) {
136       throw new NullPointerException();
137     }
138     if (localPort < 0) {
139       throw new IllegalArgumentException("Illegal local port");
140     }
141     // NOTE: we simply ignore localAddress and localPort
142     return createSocket(address, port);
143   }

145   /**
146    * A socket factory that handles a custom hostname ("localhost", by default, and configured by the
147    * system property &quot;org.newsclub.net.unix.socket.hostname&quot;), forwarding all other
148    * requests to the fallback {@link SocketFactory}.
149    */
150   private abstract static class DefaultSocketHostnameSocketFactory extends AFUNIXSocketFactory {
151     private static final String PROP_SOCKET_HOSTNAME = "org.newsclub.net.unix.socket.hostname";
152 
153     @Override
154     protected final boolean isHostnameSupported(String host) {
155       return getDefaultSocketHostname().equals(host);
156     }
157 
158     private static String getDefaultSocketHostname() {
159       return System.getProperty(PROP_SOCKET_HOSTNAME, "localhost");
160     }
161   }
162 
163   /**
164    * A socket factory that handles a custom hostname ("junixsocket.localhost", by default, and
165    * configured by the system property &quot;org.newsclub.net.unix.socket.hostname&quot;),
166    * forwarding all other requests to the fallback {@link SocketFactory}.
167    * 
168    * The socket path is configured through an argument passed by to the constructor.
169    * 
170    * This is particularly useful for JDBC drivers that take a "socketFactory" and a
171    * "socketFactoryArg". The latter will be passed as a constructor argument.
172    */
173   public static final class FactoryArg extends DefaultSocketHostnameSocketFactory {
174     private final File socketFile;
175 
176     public FactoryArg(String socketPath) {
177       if (socketPath == null) {
178         throw new NullPointerException("Constructor argument was null");
179       }
180       this.socketFile = new File(socketPath);
181     }
182 
183     public FactoryArg(File file) {
184       if (file == null) {
185         throw new NullPointerException("Constructor argument was null");
186       }
187       this.socketFile = file;
188     }
189 
190     @Override
191     protected AFUNIXSocketAddress addressFromHost(String host, int port) throws IOException {
192       return new AFUNIXSocketAddress(socketFile, port);
193     }
194   }
195 
196   /**
197    * A socket factory that handles a custom hostname ("junixsocket.localhost", by default, and
198    * configured by the system property &quot;org.newsclub.net.unix.socket.hostname&quot;),
199    * forwarding all other requests to the fallback {@link SocketFactory}.
200    * 
201    * The socket path is configured through a system property,
202    * &quot;org.newsclub.net.unix.socket.default&quot;.
203    * 
204    * NOTE: While it is technically possible, it is highly discouraged to programmatically change the
205    * value of the property as it can lead to concurrency issues and undefined behavior.
206    */
207   public static final class SystemProperty extends DefaultSocketHostnameSocketFactory {
208     private static final String PROP_SOCKET_DEFAULT = "org.newsclub.net.unix.socket.default";
209 
210     @Override
211     protected AFUNIXSocketAddress addressFromHost(String host, int port) throws IOException {
212       String path = System.getProperty(PROP_SOCKET_DEFAULT);
213       if (path == null || path.isEmpty()) {
214         throw new IllegalStateException("Property not configured: " + PROP_SOCKET_DEFAULT);
215       }
216       File socketFile = new File(path);
217 
218       return new AFUNIXSocketAddress(socketFile, port);
219     }
220   }

222   /**
223    * A socket factory that handles special host names formatted as file:// URIs.
224    * 
225    * The file:// URI may also be specified in URL-encoded format, i.e., file:%3A%2F%2F etc.
226    * 
227    * You may also surround the URL with square brackets ("[" and "]"), whereas the closing bracket
228    * may be omitted.
229    * 
230    * NOTE: In some circumstances it is recommended to use "<code>[file:%3A%2F%2F</code>(...)", i.e.
231    * encoded and without the closing bracket. Since this is an invalid hostname, it will not trigger
232    * a DNS lookup, but can still be used within a JDBC Connection URL.
233    */
234   public static final class URIScheme extends AFUNIXSocketFactory {
235     private static final String FILE_SCHEME_PREFIX = "file://";
236     private static final String FILE_SCHEME_PREFIX_ENCODED = "file%";
237     private static final String FILE_SCHEME_LOCALHOST = "localhost";
238 
239     private static String stripBrackets(String host) {
240       if (host.startsWith("[")) {
241         if (host.endsWith("]")) {
242           host = host.substring(1, host.length() - 1);
243         } else {
244           host = host.substring(1);
245         }
246       }
247       return host;
248     }
249 
250     @Override
251     protected boolean isHostnameSupported(String host) {
252       host = stripBrackets(host);
253       return host.startsWith(FILE_SCHEME_PREFIX) || host.startsWith(FILE_SCHEME_PREFIX_ENCODED);
254     }
255 
256     @Override
257     protected AFUNIXSocketAddress addressFromHost(String host, int port) throws IOException {
258       host = stripBrackets(host);
259       if (host.startsWith(FILE_SCHEME_PREFIX_ENCODED)) {
260         try {
261           host = URLDecoder.decode(host, "UTF-8");
262         } catch (Exception e) {
263           throw (UnknownHostException) new UnknownHostException().initCause(e);
264         }
265       }
266       if (!host.startsWith(FILE_SCHEME_PREFIX)) {
267         throw new UnknownHostException();
268       }
269 
270       String path = host.substring(FILE_SCHEME_PREFIX.length());
271       if (path.isEmpty()) {
272         throw new UnknownHostException();
273       }
274       if (path.startsWith(FILE_SCHEME_LOCALHOST)) {
275         path = path.substring(FILE_SCHEME_LOCALHOST.length());
276       }
277       if (!path.startsWith("/")) {
278         throw new UnknownHostException();
279       }
280 
281       File socketFile = new File(path);
282       return new AFUNIXSocketAddress(socketFile, port);
283     }
284   }
