18  package org.newsclub.net.unix;
19  
20  import java.io.FileDescriptor;
21  import java.io.IOException;
22  import java.net.ProtocolFamily;
23  import java.net.SocketException;
24  import java.nio.channels.DatagramChannel;
25  
26  /**
27   * A {@link DatagramChannel} implementation that works with AF_UNIX Unix domain sockets.
28   *
29   * @author Christian Kohlschütter
30   */
31  public final class AFUNIXDatagramChannel extends AFDatagramChannel<AFUNIXSocketAddress> implements
32      AFUNIXSocketExtensions {
33    AFUNIXDatagramChannel(AFUNIXDatagramSocket socket) {
34      super(AFUNIXSelectorProvider.getInstance(), socket);
35    }
36  
37    /**
38     * Opens a datagram channel.
39     *
40     * @return The new channel
41     * @throws IOException if an I/O error occurs
42     */
43    public static AFUNIXDatagramChannel open() throws IOException {
44      return AFUNIXSelectorProvider.provider().openDatagramChannel();
45    }
46  
47    /**
48     * Opens a datagram channel.
49     *
50     * @param family The protocol family
51     * @return A new datagram channel
52     *
53     * @throws UnsupportedOperationException if the specified protocol family is not supported
54     * @throws IOException if an I/O error occurs
55     */
56    public static AFUNIXDatagramChannel open(ProtocolFamily family) throws IOException {
57      return AFUNIXSelectorProvider.provider().openDatagramChannel(family);
58    }
59  
60    // CPD-OFF
61  
62    @Override
63    public FileDescriptor[] getReceivedFileDescriptors() throws IOException {
64      return ((AFUNIXSocketExtensions) getAFSocket()).getReceivedFileDescriptors();
65    }
66  
67    @Override
68    public void clearReceivedFileDescriptors() {
69      ((AFUNIXSocketExtensions) getAFSocket()).clearReceivedFileDescriptors();
70    }
71  
72    @Override
73    public void setOutboundFileDescriptors(FileDescriptor... fdescs) throws IOException {
74      if (fdescs != null && fdescs.length > 0 && !isConnected()) {
75        throw new SocketException("Not connected");
76      }
77      ((AFUNIXSocketExtensions) getAFSocket()).setOutboundFileDescriptors(fdescs);
78    }
79  
80    @Override
81    public boolean hasOutboundFileDescriptors() {
82      return ((AFUNIXSocketExtensions) getAFSocket()).hasOutboundFileDescriptors();
83    }
84  
85    @Override
86    public AFUNIXSocketCredentials getPeerCredentials() throws IOException {
87      return ((AFUNIXSocketExtensions) getAFSocket()).getPeerCredentials();
88    }
89  }
