18  package org.newsclub.net.unix;
19  
20  import java.net.SocketOption;
21  import java.net.SocketOptions;
22  import java.net.StandardSocketOptions;
23  import java.util.Collections;
24  import java.util.HashMap;
25  import java.util.HashSet;
26  import java.util.Map;
27  import java.util.Set;
28  
29  /**
30   * Maps new SocketOption classes to the old integer-based scheme.
31   *
32   * @author Christian Kohlschütter
33   */
34  final class SocketOptionsMapper {
35    private static final Map<SocketOption<?>, SocketOptionRef> SOCKET_OPTIONS = new HashMap<>();
36    static final Set<SocketOption<?>> SUPPORTED_SOCKET_OPTIONS;
37  
38    static {
39      // Interesting fact of the day:
40      //
41      // The old int-optID API expects SocketException to be thrown for unknown values,
42      // and the new Java 9+ SocketOption API expects UnsupportedOperationException to be
43      // thrown.
44      //
45      // However, at least up to Java 11, the new API calls the old API under the hood
46      // (see SocketImpl), so in reality, SocketException is thrown then.
47      //
48      // In Java 15 (?), they refactored the class, and the automatic mapping no longer occurs
49      // for custom socket implementations, which means we have to roll our own mapping.
50      //
51      // As you see below, unlike Java 11's SocketImpl, our getOption/setOption implementation
52      // is not a series of chained if-statements. Instead, we have a central place where
53      // the mapping is defined using generics, maps and some Java boilerplate. I don't think
54      // it's necessarily faster but since we now have a single, central place where these mappings
55      // are defined, it feels cleaner and better to not repeat yourself.
56  
57      registerSocketOption(StandardSocketOptions.SO_KEEPALIVE, SocketOptions.SO_KEEPALIVE, false);
58      registerSocketOption(StandardSocketOptions.SO_SNDBUF, SocketOptions.SO_SNDBUF, true);
59      registerSocketOption(StandardSocketOptions.SO_RCVBUF, SocketOptions.SO_RCVBUF, true);
60      registerSocketOption(StandardSocketOptions.SO_REUSEADDR, SocketOptions.SO_REUSEADDR, true);
61      registerSocketOption(StandardSocketOptions.SO_LINGER, SocketOptions.SO_LINGER, true);
62      registerSocketOption(StandardSocketOptions.IP_TOS, SocketOptions.IP_TOS, false);
63      registerSocketOption(StandardSocketOptions.TCP_NODELAY, SocketOptions.TCP_NODELAY, false);
64  
65      Set<SocketOption<?>> supportedOptions = new HashSet<>();
66  
67      for (Map.Entry<SocketOption<?>, SocketOptionRef> en : SOCKET_OPTIONS.entrySet()) {
68        if (en.getValue().supported) {
69          supportedOptions.add(en.getKey());
70        }
71      }
72      SUPPORTED_SOCKET_OPTIONS = Collections.unmodifiableSet(supportedOptions);
73    }
74  
75    private static <T> void registerSocketOption(SocketOption<T> option, int socketOptionsId,
76        boolean supported) {
77      SOCKET_OPTIONS.put(option, new SocketOptionRef(socketOptionsId, supported));
78    }
79  
80    static Integer resolve(SocketOption<?> option) {
81      SocketOptionRef ref = SOCKET_OPTIONS.get(option);
82      if (ref == null) {
83        return null;
84      } else {
85        return ref.optionId;
86      }
87    }
88  
89    private static final class SocketOptionRef {
90      private final int optionId;
91      private final boolean supported;
92  
93      SocketOptionRef(int optionId, boolean supported) {
94        this.optionId = optionId;
95        this.supported = supported;
96      }
97    }
98  }
