41    private AFUNIXSocket(final AFUNIXSocketImpl impl, AFUNIXSocketFactory factory)
42        throws IOException {
43      super(impl);
44      this.socketFactory = factory;
45      if (factory == null) {
46        setIsCreated();
47      }
48    }

50    private void setIsCreated() throws IOException {
51      try {
52        NativeUnixSocket.setCreated(this);
53      } catch (LinkageError e) {
54        throw new IOException("Couldn't load native library", e);
55      }
56    }

96    /**
97     * Creates a new {@link AFUNIXSocket} and connects it to the given {@link AFUNIXSocketAddress}.

109   /**
110    * Binds this {@link AFUNIXSocket} to the given bindpoint. Only bindpoints of the type
111    * {@link AFUNIXSocketAddress} are supported.
112    */
113   @Override
114   public void bind(SocketAddress bindpoint) throws IOException {
115     super.bind(bindpoint);
116     this.addr = (AFUNIXSocketAddress) bindpoint;
117   }
118 
119   @Override
120   public void connect(SocketAddress endpoint) throws IOException {
121     connect(endpoint, 0);
122   }
123 
124   @Override
125   public void connect(SocketAddress endpoint, int timeout) throws IOException {
126     if (!(endpoint instanceof AFUNIXSocketAddress)) {
127       if (socketFactory != null) {
128         if (endpoint instanceof InetSocketAddress) {
129           InetSocketAddress isa = (InetSocketAddress) endpoint;
130 
131           String hostname = isa.getHostString();
132           if (socketFactory.isHostnameSupported(hostname)) {
133             endpoint = socketFactory.addressFromHost(hostname, isa.getPort());
134           }
135         }
136       }
137       if (!(endpoint instanceof AFUNIXSocketAddress)) {
138         throw new IllegalArgumentException("Can only connect to endpoints of type "
139             + AFUNIXSocketAddress.class.getName() + ", got: " + endpoint);
140       }
141     }
142     impl.connect(endpoint, timeout);
143     this.addr = (AFUNIXSocketAddress) endpoint;
144     NativeUnixSocket.setBound(this);
145     NativeUnixSocket.setConnected(this);
146   }
147 
148   @Override
149   public String toString() {
150     if (isConnected()) {
151       return "AFUNIXSocket[fd=" + impl.getFD() + ";path=" + addr.getSocketFile() + "]";
152     }
153     return "AFUNIXSocket[unconnected]";
154   }

180   /**
181    * Retrieves the "peer credentials" for this connection.
182    *
183    * These credentials may be useful to authenticate the other end of the socket (client or server).
184    *
185    * @return The peer's credentials.
186    * @throws IOException If there was an error returning these credentials.
187    */
188   public AFUNIXSocketCredentials getPeerCredentials() throws IOException {
189     if (isClosed() || !isConnected()) {

192     return impl.getPeerCredentials();
193   }

197     return super.isClosed() || (isConnected() && !impl.getFD().valid());

241   /**
242    * Sets a list of {@link FileDescriptor}s that should be sent as an ancillary message along with
243    * the next write.
244    * 
245    * NOTE: There can only be one set of file descriptors active until the write completes.
246    * 
247    * @param fdescs The file descriptors, or {@code null} if none.
248    * @throws IOException if the operation fails.
249    */
250   public void setOutboundFileDescriptors(FileDescriptor... fdescs) throws IOException {
251     if (fdescs == null || fdescs.length == 0) {
252       impl.setOutboundFileDescriptors(null);
253     } else {
254       int[] fds = new int[fdescs.length];
255       for (int i = 0, n = fdescs.length; i < n; i++) {
256         FileDescriptor fdesc = fdescs[i];
257         fds[i] = NativeUnixSocket.getFD(fdesc);
258       }
259       impl.setOutboundFileDescriptors(fds);
260     }
