37  /**
38   * A workaround to create an {@link InetAddress} for an {@link AFSocketAddress}.
39   *
40   * {@link DatagramPacket} internally requires InetAddress compatibility. Even if it pretends to
41   * accept {@link SocketAddress}es, it refuses anything other than {@link InetSocketAddress}
42   * <em>and</em> then even stores host and port separately.
43   *
44   * This implementation deserializes a specially crafted {@link InetAddress} with a hostname that
45   * encodes the raw bytes of an {@link AFSocketAddress}. We do this because the deserialization code
46   * path does not attempt DNS resolution (which would fail one way or another).
47   *
48   * The hostnames we use end with ".junixsocket", to distinguish them from regular hostnames.
49   *
50   * @author Christian Kohlschütter
51   */
52  class AFInetAddress {
53    private static final byte[] SERIALIZED_INET_ADDRESS_START = {
54        (byte) 0xac, (byte) 0xed, // STREAM_MAGIC
55        0x00, 0x05, // STREAM_VERSION
56        0x73, // TC_OBJECT
57        0x72, // TC_CLASSDESC,
58        0x00, 0x14, // length
59        'j', 'a', 'v', 'a', '.', 'n', 'e', 't', '.', //
60        'I', 'n', 'e', 't', 'A', 'd', 'd', 'r', 'e', 's', 's', // "java.net.InetAddress"
61  
62        0x2d, (byte) 0x9b, 0x57, (byte) 0xaf, (byte) 0x9f, (byte) 0xe3, (byte) 0xeb, (byte) 0xdb, //
63        // serialVersionUID for java.net.InetAddress
64  
65        0x03, // classDescFlags SC_WRITE_METHOD | SC_SERIALIZABLE
66        0x00, 0x03, // fieldCount (3)
67        'I', // int field
68        0x00, 0x07, // length (7)
69        'a', 'd', 'd', 'r', 'e', 's', 's', // "address"
70        'I', // int field
71        0x00, 0x06, // length (6)
72        'f', 'a', 'm', 'i', 'l', 'y', // "family"
73        'L', // Object field
74        0x00, 0x08, // length (8)
75        'h', 'o', 's', 't', 'N', 'a', 'm', 'e', // "hostName"
76        0x74, // (className1) TC_STRING
77        0x00, 0x12, // length (18)
78        'L', 'j', 'a', 'v', 'a', '/', 'l', 'a', 'n', 'g', '/', //
79        'O', 'b', 'j', 'e', 'c', 't', ';', // "Ljava/lang/Object;"
80        0x78, // TC_ENDBLOCKDATA,
81        0x70, // (superClassDesc) TC_NULL
82        0x7f, 0x00, 0x00, (byte) 0xaf, // "address" value: (int) 127.0.0.175 (0x7f0000af)
83        0x00, 0x00, 0x00, 0x01, // "family" value: (int) 1 (IPv4)
84        0x74, // "hostName" value is a TC_STRING
85        0x00, // high-byte of string length (always 0 in our case)
86        // low-byte of string length and string itself will be appended later
87        // followed by 0x78 (TC_ENDBLOCKDATA)
88    };
89  
90    private static final char PREFIX = '[';
91    private static final String MARKER_HEX_ENCODING = "%%";
92    static final String INETADDR_SUFFIX = ".junixsocket";
93  
94    /**
95     * Encodes a junixsocket socketAddress into a string that is (somewhat) guaranteed to not be
96     * resolved by java.net code.
97     *
98     * Implementation detail: The "[" prefix (with the corresponding "]" suffix missing from the
99     * input) should cause an early {@link UnknownHostException} be thrown, which is caught within
100    * {@link InetSocketAddress#InetSocketAddress(String, int)}, causing the hostname be marked as
101    * "unresolved" (without an address set).
102    *
103    * @param socketAddress The socket address.
104    * @return A string, to be used when calling
105    *         {@link InetSocketAddress#InetSocketAddress(String, int)}, etc.
106    */
107   static final String createUnresolvedHostname(byte[] socketAddress, AFAddressFamily<?> af) {
108     StringBuilder sb = new StringBuilder(1 + socketAddress.length + INETADDR_SUFFIX.length() + 8);
109     sb.append(PREFIX);
110     try {
111       sb.append(URLEncoder.encode(new String(socketAddress, StandardCharsets.ISO_8859_1),
112           StandardCharsets.ISO_8859_1.toString()));
113     } catch (UnsupportedEncodingException e) {
114       throw new IllegalStateException(e);
115     }
116     sb.append('.');
117     sb.append(af.getJuxString());
118     sb.append(INETADDR_SUFFIX);
119 
120     String str = sb.toString();
121 
122     if (str.length() < 64 || str.getBytes(StandardCharsets.UTF_8).length <= 255) {
123       return str;
124     }
125 
126     sb.setLength(0);
127     sb.append(PREFIX);
128     sb.append(MARKER_HEX_ENCODING);
129     for (int i = 0, n = socketAddress.length; i < n; i++) {
130       sb.append(String.format(Locale.ENGLISH, "%02x", socketAddress[i]));
131     }
132 
133     sb.append('.');
134     sb.append(af.getJuxString());
135     sb.append(INETADDR_SUFFIX);
136     return sb.toString();
137   }

139   /**
140    * Creates an InetAddress that is considered "resolved" internally (using a static loopback
141    * address), without actually having to resolve the address via DNS, thus still carrying the
142    * "hostname" field containing a hostname as returned by
143    * {@link #createUnresolvedHostname(byte[],AFAddressFamily)}.
144    *
145    * @param socketAddress The socket address.
146    * @return The {@link InetAddress}.
147    */
148   static final InetAddress wrapAddress(byte[] socketAddress, AFAddressFamily<?> af) {
149     Objects.requireNonNull(af);
150     if (socketAddress == null || socketAddress.length == 0) {
151       return null;
152     }
153 
154     byte[] bytes = createUnresolvedHostname(socketAddress, af).getBytes(StandardCharsets.UTF_8);
155     if (bytes.length > 255) {
156       throw new IllegalStateException("junixsocket address is too long to wrap as InetAddress");
157     }
158     byte[] serializedData = new byte[SERIALIZED_INET_ADDRESS_START.length + 1 + bytes.length + 1];
159     System.arraycopy(SERIALIZED_INET_ADDRESS_START, 0, serializedData, 0,
160         SERIALIZED_INET_ADDRESS_START.length);
161     serializedData[SERIALIZED_INET_ADDRESS_START.length] = (byte) bytes.length;
162     System.arraycopy(bytes, 0, serializedData, SERIALIZED_INET_ADDRESS_START.length + 1,
163         bytes.length);
164     serializedData[serializedData.length - 1] = 0x78; // TC_ENDBLOCKDATA
165 
166     try (ObjectInputStream ois = new ObjectInputStream(new ByteArrayInputStream(serializedData))) {
167       return (InetAddress) ois.readObject();
168     } catch (ClassNotFoundException | IOException e) {
169       throw new IllegalStateException(e);
170     }
171   }

173   static final byte[] unwrapAddress(InetAddress addr, AFAddressFamily<?> af)
174       throws SocketException {
175     Objects.requireNonNull(addr);
176 
177     if (!isSupportedAddress(addr, af)) {
178       throw new SocketException("Unsupported address");
179     }
180 
181     String hostname = addr.getHostName();
182     try {
183       return unwrapAddress(hostname, af);
184     } catch (IllegalArgumentException e) {
185       throw (SocketException) new SocketException("Unsupported address").initCause(e);
186     }
187   }
188 
189   static final byte[] unwrapAddress(String hostname, AFAddressFamily<?> af) throws SocketException {
190     Objects.requireNonNull(hostname);
191     if (!hostname.endsWith(INETADDR_SUFFIX)) {
192       throw new SocketException("Unsupported address");
193     }
194 
195     final int end = hostname.length() - INETADDR_SUFFIX.length();
196     char c;
197     int domDot = -1;
198     for (int i = end - 1; i >= 0; i--) {
199       c = hostname.charAt(i);
200       if (c == '.') {
201         domDot = i;
202         break;
203       }
204     }
205 
206     String juxString = hostname.substring(domDot + 1, end);
207     if (AFAddressFamily.getAddressFamily(juxString) != af) { // NOPMD
208       throw new SocketException("Incompatible address");
209     }
210 
211     String encodedHostname = hostname.substring(1, domDot);
212     if (encodedHostname.startsWith(MARKER_HEX_ENCODING)) {
213       // Hex-only encoding
214       int len = encodedHostname.length();
215       if ((len & 1) == 1) {
216         throw new IllegalStateException("Length of hex-encoded wrapping must be even");
217       }
218       byte[] unwrapped = new byte[(len - 2) / 2];
219       for (int i = 2, n = encodedHostname.length(), o = 0; i < n; i += 2, o++) {
220         int v = Integer.parseInt(encodedHostname.substring(i, i + 2), 16);
221         unwrapped[o] = (byte) (v & 0xFF);
222       }
223       return unwrapped;
224     } else {
225       // URL-encoding
226       try {
227         return URLDecoder.decode(encodedHostname, StandardCharsets.ISO_8859_1.toString()).getBytes(
228             StandardCharsets.ISO_8859_1);
229       } catch (UnsupportedEncodingException e) {
230         throw new IllegalStateException(e);
231       }
232     }
233   }
234 
235   static boolean isSupportedAddress(InetAddress addr, AFAddressFamily<?> af) {
236     if (addr instanceof Inet4Address && addr.isLoopbackAddress()) {
237       String hostname = addr.getHostName();
238       return hostname.endsWith(af.getJuxInetAddressSuffix());
239     }
240     return false;
241   }
