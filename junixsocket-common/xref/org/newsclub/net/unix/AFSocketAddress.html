102   /**
103    * Creates a new socket address.
104    *
105    * @param port The port.
106    * @param socketAddress The socket address in junixsocket-specific byte-array representation.
107    * @param nativeAddress The socket address in system-native representation.
108    * @param af The address family.
109    * @throws SocketException on error.
110    */
111   @SuppressWarnings({"cast", "this-escape"})
112   protected AFSocketAddress(int port, final byte[] socketAddress, ByteBuffer nativeAddress,
113       AFAddressFamily<?> af) throws SocketException {
114     /*
115      * Initializing the superclass with an unresolved hostname helps us pass the #equals and
116      * #hashCode checks, which unfortunately are declared final in InetSocketAddress.
117      *
118      * Using a resolved address (with the address bit initialized) would be ideal, but resolved
119      * addresses can only be IPv4 or IPv6 (at least as of Java 16 and earlier).
120      */
121     super(AFInetAddress.createUnresolvedHostname(socketAddress, af), port >= 0 && port <= 0xffff
122         ? port : 0);
123     if (socketAddress.length == 0) {
124       throw new SocketException("Illegal address length: " + socketAddress.length);
125     }
126 
127     this.nativeAddress = nativeAddress == null ? null : (ByteBuffer) (Object) nativeAddress
128         .duplicate().rewind();
129     if (port < -1) {
130       throw new IllegalArgumentException("port out of range");
131     } else if (port > 0xffff) {
132       if (!NativeUnixSocket.isLoaded()) {
133         throw (SocketException) new SocketException(
134             "Cannot set SocketAddress port - junixsocket JNI library is not available").initCause(
135                 NativeUnixSocket.unsupportedException());
136       }
137       NativeUnixSocket.setPort1(this, port);
138     }
139 
140     this.bytes = socketAddress.clone();
141     this.addressFamily = af;
142   }

186    * Wraps the socket name/peer name of a file descriptor as an {@link InetAddress}.
187    *
188    * @param fdesc The file descriptor.

193   protected static final InetAddress getInetAddress(FileDescriptor fdesc, boolean peerName,
194       AFAddressFamily<?> af) {
195     if (!fdesc.valid()) {
196       return null;
197     }
198     byte[] addr = NativeUnixSocket.sockname(af.getDomain(), fdesc, peerName);
199     if (addr == null) {
200       return null;
201     }
202     return AFInetAddress.wrapAddress(addr, af);

205   /**
206    * Gets the socket name/peer name of a file descriptor as an {@link AFSocketAddress}.
207    *
208    * @param <A> The corresponding address type.
209    * @param fdesc The file descriptor.
210    * @param peerName If {@code true}, the remote peer name (instead of the local name) is retrieved.
211    * @param port The port.
212    * @param af The address family.
213    * @return The {@link InetAddress}.
214    */
215   protected static final <A extends AFSocketAddress> @Nullable A getSocketAddress(
216       FileDescriptor fdesc, boolean peerName, int port, AFAddressFamily<A> af) {
217     if (!fdesc.valid()) {
218       return null;
219     }
220     byte[] addr = NativeUnixSocket.sockname(af.getDomain(), fdesc, peerName);
221     if (addr == null) {
222       return null;
223     }
224     try {
225       // FIXME we could infer the "port" from the path if the socket factory supports that
226       return AFSocketAddress.unwrap(AFInetAddress.wrapAddress(addr, af), port, af);
227     } catch (SocketException e) {
228       throw new IllegalStateException(e);
229     }
230   }

232   static final AFSocketAddress preprocessSocketAddress(
233       Class<? extends AFSocketAddress> supportedAddressClass, SocketAddress endpoint,
234       AFSocketAddressFromHostname<?> afh) throws SocketException {
235     Objects.requireNonNull(endpoint);
236     if (endpoint instanceof SentinelSocketAddress) {
237       return (SentinelSocketAddress) endpoint;
238     }
239 
240     if (!(endpoint instanceof AFSocketAddress)) {
241       if (afh != null) {
242         if (endpoint instanceof InetSocketAddress) {
243           InetSocketAddress isa = (InetSocketAddress) endpoint;
244 
245           String hostname = isa.getHostString();
246           if (afh.isHostnameSupported(hostname)) {
247             try {
248               endpoint = afh.addressFromHost(hostname, isa.getPort());
249             } catch (SocketException e) {
250               throw e;
251             }
252           }
253         }
254       }
255     }
256 
257     Objects.requireNonNull(endpoint);
258 
259     if (!supportedAddressClass.isAssignableFrom(endpoint.getClass())) {
260       throw new IllegalArgumentException("Can only connect to endpoints of type "
261           + supportedAddressClass.getName() + ", got: " + endpoint.getClass() + ": " + endpoint);
262     }
263 
264     return (AFSocketAddress) endpoint;
265   }

313   /**
314    * Resolves a junixsocket-specific byte-array representation of an {@link AFSocketAddress} to an
315    * actual {@link AFSocketAddress} instance, possibly reusing a cached instance.
316    *
317    * @param <A> The concrete {@link AFSocketAddress} that is supported by this type.
318    * @param socketAddress The socket address in junixsocket-specific byte-array representation.
319    * @param port The port.
320    * @param af The address family.
321    * @return The instance.
322    * @throws SocketException on error.
323    */
324   @SuppressWarnings({"unchecked", "null"})
325   protected static final <A extends AFSocketAddress> A resolveAddress(final byte[] socketAddress,
326       int port, AFAddressFamily<A> af) throws SocketException {
327     if (socketAddress.length == 0) {
328       throw new SocketException("Address cannot be empty");
329     }
330 
331     if (port == -1) {
332       port = 0;
333     }
334 
335     ByteBuffer direct = SOCKETADDRESS_BUFFER_TL.get();
336     int limit = NativeUnixSocket.isLoaded() ? NativeUnixSocket.bytesToSockAddr(af.getDomain(),
337         direct, socketAddress) : -1;
338     if (limit == -1) {
339       // not supported, but we can still create an address
340       return af.getAddressConstructor().newAFSocketAddress(port, socketAddress, null);
341     } else if (limit > SOCKADDR_MAX_LEN) {
342       throw new IllegalStateException("Unexpected address length");
343     }
344     direct.rewind();
345     direct.limit(limit);
346 
347     A instance;
348     synchronized (AFSocketAddress.class) {
349       Map<ByteBuffer, AFSocketAddress> map;
350       Map<Integer, Map<ByteBuffer, AFSocketAddress>> mapPorts = ADDRESS_CACHE.get(af);
351       if (mapPorts == null) {
352         instance = null;
353         mapPorts = new HashMap<>();
354         map = new HashMap<>();
355         mapPorts.put(port, map);
356         ADDRESS_CACHE.put(af, mapPorts);
357       } else {
358         map = mapPorts.get(port);
359         if (map == null) {
360           instance = null;
361           map = new HashMap<>();
362           mapPorts.put(port, map);
363         } else {
364           instance = (A) map.get(direct);
365         }
366       }
367 
368       if (instance == null) {
369         ByteBuffer key = newSockAddrKeyBuffer(limit);
370         key.put(direct);
371         key = key.asReadOnlyBuffer();
372 
373         instance = af.getAddressConstructor().newAFSocketAddress(port, socketAddress, key);
374 
375         map.put(key, instance);
376       }
377     }
378 
379     return instance;
380   }

382   @SuppressWarnings("null")
383   static final <A extends AFSocketAddress> A ofInternal(ByteBuffer socketAddressBuffer,
384       AFAddressFamily<A> af) throws SocketException {
385     synchronized (AFSocketAddress.class) {
386       Map<Integer, Map<ByteBuffer, AFSocketAddress>> mapPorts = ADDRESS_CACHE.get(af);
387       if (mapPorts != null) {
388         Map<ByteBuffer, AFSocketAddress> map = mapPorts.get(0); // FIXME get port, something like
389                                                                 // sockAddrToPort
390         if (map != null) {
391           @SuppressWarnings("unchecked")
392           A address = (A) map.get(socketAddressBuffer.rewind());
393           if (address != null) {
394             return address;
395           }
396         }
397       }
398 
399       if (!socketAddressBuffer.isDirect()) {
400         ByteBuffer buf = getNativeAddressDirectBuffer(Math.min(socketAddressBuffer.limit(),
401             SOCKADDR_MAX_LEN));
402         buf.put(socketAddressBuffer);
403         socketAddressBuffer = buf;
404       }
405       byte[] sockAddrToBytes = NativeUnixSocket.sockAddrToBytes(af.getDomain(),
406           socketAddressBuffer);
407       if (sockAddrToBytes == null) {
408         return null;
409       } else {
410         return AFSocketAddress.resolveAddress(sockAddrToBytes, 0, af);
411       }
412     }
413   }

486   static final int unwrapAddressDirectBufferInternal(ByteBuffer socketAddressBuffer,
487       SocketAddress address) throws SocketException {
488     if (!NativeUnixSocket.isLoaded()) {
489       throw new SocketException("Unsupported operation; junixsocket native library is not loaded");
490     }
491     Objects.requireNonNull(address);
492 
493     if (!(address instanceof AFSocketAddress)) {
494       Supplier<? extends AFSocketAddress> supp = AFUNIXSocketAddress.supportedAddressSupplier(
495           address);
496       address = supp == null ? null : supp.get();
497       if (address == null) {
498         throw new SocketException("Unsupported address");
499       }
500     }
501 
502     AFSocketAddress socketAddress = (AFSocketAddress) address;
503 
504     byte[] addr = socketAddress.getBytes();
505     int domain = socketAddress.getAddressFamily().getDomain();
506 
507     int len = NativeUnixSocket.bytesToSockAddr(domain, socketAddressBuffer, addr);
508     if (len == -1) {
509       throw new SocketException("Unsupported domain");
510     }
511     return len;
512   }

656   /**
657    * Returns a address string that can be used with {@code socat}'s {@code SOCKET-CONNECT},
658    * {@code SOCKET-LISTEN}, {@code SOCKET-DATAGRAM}, etc., address types, or {@code null} if the
659    * address type is not natively supported by this platform.
660    *
661    * This call is mostly suited for debugging purposes. The resulting string is specific to the
662    * platform the code is executed on, and thus may be different among platforms.
663    *
664    * @param socketType The socket type, or {@code null} to omit from string.
665    * @param socketProtocol The socket protocol, or {@code null} to omit from string.
666    * @return The string (such as 1:0:x2f746d702f796f).
667    * @throws IOException on error (a {@link SocketException} is thrown if the native address cannot
668    *           be accessed).
669    */
670   public @Nullable @SuppressWarnings("PMD.NPathComplexity") String toSocatAddressString(
671       AFSocketType socketType, AFSocketProtocol socketProtocol) throws IOException {
672 
673     if (SOCKADDR_NATIVE_FAMILY_OFFSET == -1 || SOCKADDR_NATIVE_DATA_OFFSET == -1) {
674       return null;
675     }
676     if (nativeAddress == null) {
677       throw (SocketException) new SocketException("Cannot access native address").initCause(
678           NativeUnixSocket.unsupportedException());
679     }
680     if (socketProtocol != null && socketProtocol.getId() != 0) {
681       throw new IOException("Protocol not (yet) supported"); // FIXME support additional protocols
682     }
683 
684     int family = (nativeAddress.get(SOCKADDR_NATIVE_FAMILY_OFFSET) & 0xFF);
685     int type = socketType == null ? -1 : NativeUnixSocket.sockTypeToNative(socketType.getId());
686     StringBuilder sb = new StringBuilder();
687     sb.append(family);
688     if (type != -1) {
689       sb.append(':');
690       sb.append(type);
691     }
692     if (socketProtocol != null) {
693       sb.append(':');
694       sb.append(socketProtocol.getId()); // FIXME needs native conversion
695     }
696     sb.append(":x");
697     int n = nativeAddress.limit();
698     while (n > 1 && nativeAddress.get(n - 1) == 0) {
699       n--;
700     }
701     for (int pos = SOCKADDR_NATIVE_DATA_OFFSET; pos < n; pos++) {
702       byte b = nativeAddress.get(pos);
703       sb.append(String.format(Locale.ENGLISH, "%02x", b));
704     }
705     return sb.toString();
706   }

709    * Checks if the given address could cover another address.
710    *
711    * By default, this is only true if both addresses are regarded equal using
712    * {@link #equals(Object)}.
713    *
714    * However, implementations may support "wildcard" addresses, and this method would compare a
715    * wildcard address against some non-wildcard address, for example.

717    * @param other The other address that could be covered by this address.
718    * @return {@code true} if the other address could be covered.
719    */

721     return this.equals(other);
