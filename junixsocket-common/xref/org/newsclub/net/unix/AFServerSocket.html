35  /**
36   * The server part of a junixsocket socket.
37   * 
38   * @param <A> The concrete {@link AFSocketAddress} that is supported by this type.
39   * @author Christian Kohlschütter

42      FileDescriptorAccess {

120   /**
121    * Creates a new AFServerSocket instance, using the given subclass constructor.
122    * 
123    * @param <A> The concrete {@link AFSocketAddress} that is supported by this type.
124    * @param instanceSupplier The subclass constructor.
125    * @param fdObj The file descriptor.
126    * @param localPort The local port.
127    * @param remotePort The remote port.
128    * @return The new instance.
129    * @throws IOException on error.
130    */
131   protected static <A extends AFSocketAddress> AFServerSocket<A> newInstance(
132       Constructor<A> instanceSupplier, FileDescriptor fdObj, int localPort, int remotePort)
133       throws IOException {
134     if (fdObj == null) {
135       return instanceSupplier.newInstance(null);
136     }
137 
138     int status = NativeUnixSocket.socketStatus(fdObj);
139     if (!fdObj.valid() || status == NativeUnixSocket.SOCKETSTATUS_INVALID) {
140       throw new SocketException("Not a valid socket");
141     }
142     AFServerSocket<A> socket = instanceSupplier.newInstance(fdObj);
143     socket.getAFImpl().updatePorts(localPort, remotePort);
144 
145     switch (status) {
146       case NativeUnixSocket.SOCKETSTATUS_CONNECTED:
147         throw new SocketException("Not a ServerSocket");
148       case NativeUnixSocket.SOCKETSTATUS_BOUND:
149         socket.bind(AFSocketAddress.INTERNAL_DUMMY_BIND);
150 
151         socket.setBoundEndpoint(AFSocketAddress.getSocketAddress(fdObj, false, localPort, socket
152             .addressFamily()));
153         break;
154       case NativeUnixSocket.SOCKETSTATUS_UNKNOWN:
155         break;
156       default:
157         throw new IllegalStateException("Invalid socketStatus response: " + status);
158     }
159 
160     socket.getAFImpl().setSocketAddress(socket.getLocalSocketAddress());
161     return socket;
162   }

215   /**
216    * Forces the address to be used for any subsequent call to {@link #bind(SocketAddress)} to be the
217    * given one, regardless of what'll be passed to {@link #bind(SocketAddress, int)}, but doesn't
218    * bind yet.
219    * 
220    * @param endpoint The forced endpoint address.
221    * @return This {@link AFServerSocket}.
222    */
223   public final AFServerSocket<A> forceBindAddress(SocketAddress endpoint) {
224     return bindHook((SocketAddress orig) -> {
225       return orig == null ? null : endpoint;
226     });
227   }

229   @SuppressWarnings("unchecked")
230   @Override
231   public final void bind(SocketAddress endpoint, int backlog) throws IOException {
232     if (isClosed()) {
233       throw new SocketException("Socket is closed");
234     }
235 
236     boolean bindErrorOk;
237     if (bindFilter != null) {
238       endpoint = bindFilter.apply(endpoint);
239       bindErrorOk = endpoint != null && isBound();
240     } else {
241       bindErrorOk = false;
242     }
243 
244     if (!(endpoint instanceof AFSocketAddress)) {
245       throw new IllegalArgumentException("Can only bind to endpoints of type "
246           + AFSocketAddress.class.getName() + ": " + endpoint);
247     }
248 
249     A endpointCast;
250     try {
251       endpointCast = (A) endpoint;
252     } catch (ClassCastException e) {
253       throw new IllegalArgumentException("Can only bind to specific endpoints", e);
254     }
255 
256     try {
257       getAFImpl().bind(endpoint, getReuseAddress() ? NativeUnixSocket.BIND_OPT_REUSE : 0);
258     } catch (SocketException e) {
259       if (bindErrorOk) {
260         // force-binding an address could mean double-binding the same address, that's OK.
261         return;
262       } else {
263         throw e;
264       }
265     }
266     setBoundEndpoint(getAFImpl().getLocalSocketAddress());
267     if (boundEndpoint == null) {
268       setBoundEndpoint(endpointCast);
269     }
270 
271     if (endpoint == AFSocketAddress.INTERNAL_DUMMY_BIND) { // NOPMD
272       return;
273     }
274 
275     implementation.listen(backlog);
276   }

278   @Override

285     return super.isClosed() || (isBound() && !implementation.getFD().valid());

288   @Override
289   public AFSocket<A> accept() throws IOException {
290     return accept1(true);
291   }

293   AFSocket<A> accept1(boolean throwOnFail) throws IOException {
294     AFSocket<A> as = newSocketInstance();
295 
296     boolean success = implementation.accept0(as.getAFImpl(false));
297     if (isClosed()) {
298       // We may have connected to the socket to unblock it
299       throw new SocketException("Socket is closed");
300     }
301 
302     if (!success) {
303       if (throwOnFail) {
304         if (getChannel().isBlocking()) {
305           // unexpected
306           return null;
307         } else {
308           // non-blocking socket, nothing to accept
309           throw new IllegalBlockingModeException();
310         }
311       } else {
312         return null;
313       }
314     }
315 
316     as.getAFImpl(true); // trigger create
317     as.connect(AFSocketAddress.INTERNAL_DUMMY_CONNECT);
318     as.getAFImpl().updatePorts(getAFImpl().getLocalPort1(), getAFImpl().getRemotePort());
319 
320     return as;
321   }

336   @Override
337   public synchronized void close() throws IOException {
338     if (isClosed()) {
339       return;
340     }
341 
342     boolean localSocketAddressValid = isLocalSocketAddressValid();
343 
344     AFSocketAddress endpoint = boundEndpoint;
345 
346     IOException superException = null;
347     try {
348       super.close();
349     } catch (IOException e) {
350       superException = e;
351     }
352     if (implementation != null) {
353       try {
354         implementation.close();
355       } catch (IOException e) {
356         if (superException == null) {
357           superException = e;
358         } else {
359           superException.addSuppressed(e);
360         }
361       }
362     }
363 
364     IOException ex = null;
365     try {
366       closeables.close(superException);
367     } finally {
368       if (endpoint != null && endpoint.hasFilename() && localSocketAddressValid
369           && isDeleteOnClose()) {
370         File f = endpoint.getFile();
371         if (!f.delete() && f.exists()) {
372           ex = new IOException("Could not delete socket file after close: " + f);
373         }
374       }
375     }
376     if (ex != null) {
377       throw ex;
378     }
379   }

408   @Override

412       setBoundEndpoint(getAFImpl().getLocalSocketAddress());

414     return boundEndpoint;

417   /**
418    * Checks if the local socket address returned by {@link #getLocalSocketAddress()} is still valid.
419    * 
420    * The address is no longer valid if the server socket has been closed, {@code null}, or another
421    * server socket has been bound on that address.
422    * 
423    * @return {@code true} iff still valid.
424    */
425   public boolean isLocalSocketAddressValid() {
426     if (isClosed()) {
427       return false;
428     }
429     @Nullable
430     A addr = getLocalSocketAddress();
431     if (addr == null) {
432       return false;
433     }
434     return addr.equals(getAFImpl().getLocalSocketAddress());
435   }
436 
437   final void setBoundEndpoint(@Nullable A addr) {
438     this.boundEndpoint = addr;
439     int port;
440     if (addr == null) {
441       port = -1;
442     } else {
443       port = addr.getPort();
444     }
445     getAFImpl().updatePorts(port, -1);
446   }
447 
448   @Override
449   public final int getLocalPort() {
450     if (boundEndpoint == null) {
451       setBoundEndpoint(getAFImpl().getLocalSocketAddress());
452     }
453     if (boundEndpoint == null) {
454       return -1;
455     } else {
456       return getAFImpl().getLocalPort1();
457     }
458   }

485   final AFSocketImpl<A> getAFImpl() {
486     if (created.compareAndSet(false, true)) {
487       try {
488         getSoTimeout(); // trigger create via java.net.Socket
489       } catch (IOException e) {
490         // ignore
491       }
492     }
493     return implementation;
494   }
