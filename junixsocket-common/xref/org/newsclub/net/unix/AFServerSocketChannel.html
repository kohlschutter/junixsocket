18  package org.newsclub.net.unix;
19  
20  import java.io.FileDescriptor;
21  import java.io.IOException;
22  import java.net.ServerSocket;
23  import java.net.SocketAddress;
24  import java.net.SocketOption;
25  import java.nio.channels.ServerSocketChannel;
26  import java.nio.channels.spi.SelectorProvider;
27  import java.util.Objects;
28  import java.util.Set;
29  
30  import org.eclipse.jdt.annotation.NonNull;
31  import org.eclipse.jdt.annotation.Nullable;
32  
33  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
34  
35  /**
36   * A selectable channel for stream-oriented listening sockets.
37   *
38   * @param <A> The concrete {@link AFSocketAddress} that is supported by this type.
39   * @author Christian Kohlschütter
40   */
41  public abstract class AFServerSocketChannel<A extends AFSocketAddress> extends ServerSocketChannel
42      implements FileDescriptorAccess, AFSomeSocketChannel {
43    private final @NonNull AFServerSocket<A> afSocket;
44  
45    /**
46     * Creates a new {@link AFServerSocketChannel} instance.
47     *
48     * @param socket The corresponding {@link ServerSocket}.
49     * @param sp The corresponding {@link SelectorProvider}.
50     */
51    @SuppressWarnings("all") // null
52    protected AFServerSocketChannel(AFServerSocket<A> socket, AFSelectorProvider<A> sp) {
53      super(sp);
54      this.afSocket = Objects.requireNonNull(socket);
55    }
56  
57    // CPD-OFF
58  
59    @SuppressWarnings("unchecked")
60    @Override
61    public <T> T getOption(SocketOption<T> name) throws IOException {
62      if (name instanceof AFSocketOption<?>) {
63        return getAFCore().getOption((AFSocketOption<T>) name);
64      }
65      Integer optionId = SocketOptionsMapper.resolve(name);
66      if (optionId == null) {
67        throw new UnsupportedOperationException("unsupported option");
68      } else {
69        return (T) afSocket.getAFImpl().getOption(optionId);
70      }
71    }
72  
73    @Override
74    public <T> AFServerSocketChannel<A> setOption(SocketOption<T> name, T value) throws IOException {
75      if (name instanceof AFSocketOption<?>) {
76        getAFCore().setOption((AFSocketOption<T>) name, value);
77        return this;
78      }
79      Integer optionId = SocketOptionsMapper.resolve(name);
80      if (optionId == null) {
81        throw new UnsupportedOperationException("unsupported option");
82      } else {
83        afSocket.getAFImpl().setOption(optionId, value);
84      }
85      return this;
86    }
87  
88    @Override
89    public final Set<SocketOption<?>> supportedOptions() {
90      return SocketOptionsMapper.SUPPORTED_SOCKET_OPTIONS;
91    }
92  
93    // CPD-ON
94  
95    @Override
96    public final AFServerSocketChannel<A> bind(SocketAddress local, int backlog) throws IOException {
97      afSocket.bind(local, backlog);
98      return this;
99    }
100 
101   @SuppressFBWarnings("EI_EXPOSE_REP")
102   @Override
103   public final AFServerSocket<A> socket() {
104     return afSocket;
105   }
106 
107   @Override
108   public AFSocketChannel<A> accept() throws IOException {
109     AFSocket<A> socket = afSocket.accept1(false);
110     return socket == null ? null : socket.getChannel();
111   }
112 
113   @Override
114   public final @Nullable A getLocalAddress() {
115     return getLocalSocketAddress();
116   }
117 
118   @Override
119   public final @Nullable A getLocalSocketAddress() {
120     return afSocket.getLocalSocketAddress();
121   }
122 
123   /**
124    * Checks if the local socket address returned by {@link #getLocalAddress()} is still valid.
125    *
126    * The address is no longer valid if the server socket has been closed, {@code null}, or another
127    * server socket has been bound on that address.
128    *
129    * @return {@code true} iff still valid.
130    */
131   public final boolean isLocalSocketAddressValid() {
132     return afSocket.isLocalSocketAddressValid();
133   }
134 
135   @Override
136   protected final void implCloseSelectableChannel() throws IOException {
137     afSocket.close();
138   }
139 
140   @Override
141   protected final void implConfigureBlocking(boolean block) throws IOException {
142     getAFCore().implConfigureBlocking(block);
143   }
144 
145   final AFSocketCore getAFCore() {
146     return afSocket.getAFImpl().getCore();
147   }
148 
149   @Override
150   public final FileDescriptor getFileDescriptor() throws IOException {
151     return afSocket.getFileDescriptor();
152   }
153 
154   /**
155    * Checks if this {@link AFServerSocketChannel}'s file should be removed upon {@link #close()}.
156    *
157    * Deletion is not guaranteed, especially when not supported (e.g., addresses in the abstract
158    * namespace).
159    *
160    * @return {@code true} if an attempt is made to delete the socket file upon {@link #close()}.
161    */
162   public final boolean isDeleteOnClose() {
163     return socket().isDeleteOnClose();
164   }
165 
166   /**
167    * Enables/disables deleting this {@link AFServerSocketChannel}'s file (or other resource type)
168    * upon {@link #close()}.
169    *
170    * Deletion is not guaranteed, especially when not supported (e.g., addresses in the abstract
171    * namespace).
172    *
173    * @param b Enabled if {@code true}.
174    */
175   public final void setDeleteOnClose(boolean b) {
176     socket().setDeleteOnClose(b);
177   }
178 
179   @Override
180   public void setShutdownOnClose(boolean enabled) {
181     socket().setShutdownOnClose(enabled);
182   }
183 }
