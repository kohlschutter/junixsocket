18  package org.newsclub.net.unix;
19  
20  import java.lang.ProcessBuilder.Redirect;
21  
22  /**
23   * Describes junixsocket capabilities the current environment (system platform, native library,
24   * etc.) may or may not support.
25   * 
26   * You can check whether your environment supports a given capability by calling
27   * {@link AFSocket#supports(AFSocketCapability)}.
28   */
29  public enum AFSocketCapability {
30    // see org_newsclub_net_unix_NativeUnixSocket.c in junixsocket-native
31  
32    /** Socket supports retrieving peer credentials. */
33    CAPABILITY_PEER_CREDENTIALS(0),
34  
35    /** Socket supports sending and receiving ancillary messages. */
36    CAPABILITY_ANCILLARY_MESSAGES(1),
37  
38    /** Socket supports passing file descriptors via ancillary messages. */
39    CAPABILITY_FILE_DESCRIPTORS(2),
40  
41    /** Socket addressing supports the abstract namespace (Linux). */
42    CAPABILITY_ABSTRACT_NAMESPACE(3),
43  
44    /** Support for AF_UNIX datagrams (not on Windows yet). */
45    CAPABILITY_UNIX_DATAGRAMS(4),
46  
47    /**
48     * A pair of interconnected sockets can be created natively as AF_UNIX sockets.
49     * 
50     * This currently not possible on Windows, but instead emulated via anonymous AF_INET ports when
51     * you use {@link AFSocketPair}. Other systems may provide partial implementations of pipe-based
52     * (i.e., non-socket) pairs.
53     * 
54     * This capability is specific to AF_UNIX sockets. Other sockets, such as AF_VSOCK, may not
55     * implement socketpair natively even if this capability is set, but would work-around that
56     * limitation in a similar fashion but maybe without resorting to AF_INET.
57     */
58    CAPABILITY_NATIVE_SOCKETPAIR(5),
59  
60    /**
61     * A file descriptor can be converted to {@link Redirect}.
62     * 
63     * This feature currently uses Java SDK internals that may change/disappear.
64     */
65    CAPABILITY_FD_AS_REDIRECT(6),
66  
67    /**
68     * Support for AF_TIPC.
69     * 
70     * Availability of this feature is checked upon launch and therefore loading the "tipc" kernel
71     * module at a later point may not be properly reflected.
72     */
73    CAPABILITY_TIPC(7),
74  
75    /**
76     * Support for AF_UNIX.
77     * 
78     * Availability of this feature is checked upon launch and therefore, on systems adding support at
79     * a later point, may not be properly reflected when checking at a later point.
80     * 
81     * NOTE: While this capability is typically supported on most systems that can actually load a
82     * junixsocket JNI library, it is unavailable for older Windows versions (such as 8.1, 10 before
83     * AFUNIX.SYS was included, etc.) and on systems where support for UNIX domain sockets is actively
84     * disabled. Therefore, it is still recommended to check for this capability.
85     */
86    CAPABILITY_UNIX_DOMAIN(8),
87  
88    /**
89     * Support for AF_VSOCK.
90     *
91     * Availability of this feature is checked upon launch and therefore enabling vsock at a later
92     * point may not be properly reflected.
93     * 
94     * @see #CAPABILITY_VSOCK_DGRAM
95     */
96    CAPABILITY_VSOCK(9),
97  
98    /**
99     * Support for AF_VSOCK datagrams (not all platforms/kernel versions or configurations support
100    * this).
101    *
102    * Availability of this feature is checked upon launch and therefore enabling vsock at a later
103    * point may not be properly reflected.
104    */
105   CAPABILITY_VSOCK_DGRAM(10),
106 
107   ; // end of list
108 
109   private final int bitmask;
110 
111   AFSocketCapability(int bit) {
112     this.bitmask = 1 << bit;
113   }
114 
115   int getBitmask() {
116     return bitmask;
117   }
118 }
