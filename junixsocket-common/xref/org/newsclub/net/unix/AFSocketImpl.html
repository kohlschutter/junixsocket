88      private void incPendingAccepts() throws SocketException {
89        if (pendingAccepts.incrementAndGet() >= Integer.MAX_VALUE) {
90          throw new SocketException("Too many pending accepts");
91        }
92      }

94      private void decPendingAccepts() throws SocketException {
95        pendingAccepts.decrementAndGet();
96      }

102     /**
103      * Unblock other threads that are currently waiting on accept, simply by connecting to the
104      * socket.
105      */
106     @Override
107     protected void unblockAccepts() {
108       if (socketAddress == null || socketAddress.getBytes() == null || inode.get() < 0) {
109         return;
110       }
111 
112       while (pendingAccepts.get() > 0) {
113         try {
114           FileDescriptor tmpFd = new FileDescriptor();
115 
116           try {
117             createSocket(tmpFd, AFSocketType.SOCK_STREAM);
118             ByteBuffer ab = socketAddress.getNativeAddressDirectBuffer();
119             NativeUnixSocket.connect(ab, ab.limit(), tmpFd, inode.get());
120           } catch (IOException e) {
121             // there's nothing more we can do to unlock these accepts
122             // (e.g., SocketException: No such file or directory)
123             return;
124           }
125           try {
126             NativeUnixSocket.shutdown(tmpFd, SHUT_RD_WR);
127           } catch (Exception e) {
128             // ignore
129           }
130           try {
131             NativeUnixSocket.close(tmpFd);
132           } catch (Exception e) {
133             // ignore
134           }
135         } catch (Exception e) {
136           // ignore
137         }
138 
139         // sleep a little to give the cleaners some CPU time to actually clean up
140         try {
141           Thread.sleep(5);
142         } catch (InterruptedException e) {
143           // ignore
144         }
145       }
146     }
147   }

232   final boolean accept0(SocketImpl socket) throws IOException {
233     FileDescriptor fdesc = core.validFdOrException();
234     if (isClosed()) {
235       throw new SocketException("Socket is closed");
236     } else if (!isBound()) {
237       throw new SocketException("Socket is not bound");
238     }
239 
240     AFSocketAddress socketAddress = core.socketAddress;
241     if (socketAddress == null) {
242       core.socketAddress = socketAddress = getLocalSocketAddress();
243     }
244     if (socketAddress == null) {
245       throw new SocketException("Socket is not bound");
246     }
247 
248     @SuppressWarnings("unchecked")
249     final AFSocketImpl<A> si = (AFSocketImpl<A>) socket;
250     try {
251       core.incPendingAccepts();
252       ByteBuffer ab = socketAddress.getNativeAddressDirectBuffer();
253       if (!NativeUnixSocket.accept(ab, ab.limit(), fdesc, si.fd, core.inode.get(), socketTimeout
254           .get())) {
255         return false;
256       }
257 
258       if (!isBound() || isClosed()) {
259         try {
260           NativeUnixSocket.shutdown(si.fd, SHUT_RD_WR);
261         } catch (Exception e) {
262           // ignore
263         }
264         try {
265           NativeUnixSocket.close(si.fd);
266         } catch (Exception e) {
267           // ignore
268         }
269         throw new SocketException("Socket is closed");
270       }
271     } finally {
272       core.decPendingAccepts();
273     }
274     si.setSocketAddress(socketAddress);
275     si.connected.set(true);
276 
277     return true;
278   }

300   final void bind(SocketAddress addr, int options) throws IOException {
301     if (addr == null) {
302       throw new IllegalArgumentException("Cannot bind to null address");
303     }
304     if (!(addr instanceof AFSocketAddress)) {
305       throw new SocketException("Cannot bind to this type of address: " + addr.getClass());
306     }
307 
308     bound.set(true);
309 
310     if (addr == AFSocketAddress.INTERNAL_DUMMY_BIND) { // NOPMD
311       core.inode.set(0);
312       return;
313     }
314 
315     AFSocketAddress socketAddress = (AFSocketAddress) addr;
316 
317     this.setSocketAddress(socketAddress);
318     ByteBuffer ab = socketAddress.getNativeAddressDirectBuffer();
319     core.inode.set(NativeUnixSocket.bind(ab, ab.limit(), fd, options));
320     core.validFdOrException();
321   }

358   final boolean connect0(SocketAddress addr, int connectTimeout) throws IOException {
359     if (addr == AFSocketAddress.INTERNAL_DUMMY_CONNECT) { // NOPMD
360       this.connected.set(true);
361       return true;
362     } else if (addr == AFSocketAddress.INTERNAL_DUMMY_CONNECT) { // NOPMD)
363       return false;
364     }
365 
366     if (!(addr instanceof AFSocketAddress)) {
367       throw new SocketException("Cannot connect to this type of address: " + addr.getClass());
368     }
369 
370     AFSocketAddress socketAddress = (AFSocketAddress) addr;
371     ByteBuffer ab = socketAddress.getNativeAddressDirectBuffer();
372     boolean success = NativeUnixSocket.connect(ab, ab.limit(), fd, -1);
373     if (success) {
374       setSocketAddress(socketAddress);
375       this.connected.set(true);
376     }
377     core.validFdOrException();
378     return success;
379   }

382   protected final void create(boolean stream) throws IOException {
383     if (isClosed()) {
384       throw new SocketException("Already closed");
385     }
386     if (fd.valid()) {
387       if (createType != null) {
388         if (createType.booleanValue() != stream) {
389           throw new IllegalStateException("Already created with different mode");
390         }
391       } else {
392         createType = stream;
393       }
394       return;
395     }
396     createType = stream;
397     createSocket(fd, stream ? AFSocketType.SOCK_STREAM : AFSocketType.SOCK_DGRAM);
398   }

437   private final class AFInputStreamImpl extends AFInputStream {
438     private volatile boolean streamClosed = false;
439     private final AtomicBoolean eofReached = new AtomicBoolean(false);
440 
441     private final int opt = (core.isBlocking() ? 0 : NativeUnixSocket.OPT_NON_BLOCKING);
442 
443     @Override
444     public int read(byte[] buf, int off, int len) throws IOException {
445       if (streamClosed) {
446         throw new IOException("This InputStream has already been closed.");
447       }
448       if (eofReached.get()) {
449         return -1;
450       }
451 
452       FileDescriptor fdesc = core.validFdOrException();
453       if (len == 0) {
454         return 0;
455       } else if (off < 0 || len < 0 || (len > buf.length - off)) {
456         throw new IndexOutOfBoundsException();
457       }
458 
459       try {
460         return NativeUnixSocket.read(fdesc, buf, off, len, opt, ancillaryDataSupport, socketTimeout
461             .get());
462       } catch (EOFException e) {
463         eofReached.set(true);
464         throw e;
465       }
466     }
467 
468     @Override
469     public int read() throws IOException {
470       FileDescriptor fdesc = core.validFdOrException();
471 
472       if (eofReached.get()) {
473         return -1;
474       }
475 
476       int byteRead = NativeUnixSocket.read(fdesc, null, 0, 1, opt, ancillaryDataSupport,
477           socketTimeout.get());
478       if (byteRead < 0) {
479         eofReached.set(true);
480         return -1;
481       } else {
482         return byteRead;
483       }
484     }
485 
486     @Override
487     public synchronized void close() throws IOException {
488       streamClosed = true;
489       FileDescriptor fdesc = core.validFd();
490       if (fdesc != null) {
491         NativeUnixSocket.shutdown(fdesc, SHUT_RD);
492       }
493 
494       closedInputStream = true;
495       checkClose();
496     }
497 
498     @Override
499     public int available() throws IOException {
500       if (streamClosed) {
501         throw new IOException("This InputStream has already been closed.");
502       }
503 
504       return AFSocketImpl.this.available();
505     }

509       return getFD();
510     }

513   private static boolean checkWriteInterruptedException(int bytesTransferred)
514       throws InterruptedIOException {
515     if (Thread.interrupted()) {
516       InterruptedIOException ex = new InterruptedIOException("Thread interrupted during write");
517       ex.bytesTransferred = bytesTransferred;
518       Thread.currentThread().interrupt();
519       throw ex;
520     }
521     return true;
522   }

524   private final class AFOutputStreamImpl extends AFOutputStream {
525     private volatile boolean streamClosed = false;
526 
527     private final int opt = (core.isBlocking() ? 0 : NativeUnixSocket.OPT_NON_BLOCKING);
528 
529     @Override
530     public void write(int oneByte) throws IOException {
531       FileDescriptor fdesc = core.validFdOrException();
532 
533       int written;
534       do {
535         written = NativeUnixSocket.write(fdesc, null, oneByte, 1, opt, ancillaryDataSupport);
536         if (written != 0) {
537           break;
538         }
539       } while (checkWriteInterruptedException(0));
540     }
541 
542     @Override
543     public void write(byte[] buf, int off, int len) throws IOException {
544       if (streamClosed) {
545         throw new SocketException("This OutputStream has already been closed.");
546       }
547       if (len < 0 || off < 0 || len > buf.length - off) {
548         throw new IndexOutOfBoundsException();
549       }
550       FileDescriptor fdesc = core.validFdOrException();
551       if (len == 0) {
552         return;
553       }
554 
555       int writtenTotal = 0;
556 
557       do {
558         final int written = NativeUnixSocket.write(fdesc, buf, off, len, opt, ancillaryDataSupport);
559         if (written < 0) {
560           throw new IOException("Unspecific error while writing");
561         }
562 
563         len -= written;
564         off += written;
565         writtenTotal += written;
566       } while (len > 0 && checkWriteInterruptedException(writtenTotal));
567     }
568 
569     @Override
570     public synchronized void close() throws IOException {
571       if (streamClosed) {
572         return;
573       }
574       streamClosed = true;
575       FileDescriptor fdesc = core.validFd();
576       if (fdesc != null) {
577         NativeUnixSocket.shutdown(fdesc, SHUT_WR);
578       }
579       closedOutputStream = true;
580       checkClose();
581     }

584     public FileDescriptor getFileDescriptor() throws IOException {
585       return getFD();
586     }

624   private Object getOption0(int optID) throws SocketException {
625     if (isClosed()) {
626       throw new SocketException("Socket is closed");
627     }
628     if (optID == SocketOptions.SO_REUSEADDR) {
629       return reuseAddr;
630     }
631 
632     FileDescriptor fdesc = core.validFdOrException();
633     return getOptionDefault(fdesc, optID, socketTimeout, addressFamily);
634   }

636   static final Object getOptionDefault(FileDescriptor fdesc, int optID, AtomicInteger acceptTimeout,
637       AFAddressFamily<?> af) throws SocketException {
638     try {
639       switch (optID) {
640         case SocketOptions.SO_KEEPALIVE:
641           try {
642             return (NativeUnixSocket.getSocketOptionInt(fdesc, optID) != 0);
643           } catch (SocketException e) {
644             // ignore
645             return false;
646           }
647         case SocketOptions.TCP_NODELAY:
648           return (NativeUnixSocket.getSocketOptionInt(fdesc, optID) != 0);
649         case SocketOptions.SO_TIMEOUT:
650           int v = Math.max(NativeUnixSocket.getSocketOptionInt(fdesc, 0x1005), NativeUnixSocket
651               .getSocketOptionInt(fdesc, 0x1006));
652           if (v == -1) {
653             // special value, meaning: do not override infinite timeout from native code
654             return 0;
655           }
656           return Math.max((acceptTimeout == null ? 0 : acceptTimeout.get()), v);
657         case SocketOptions.SO_LINGER:
658         case SocketOptions.SO_RCVBUF:
659         case SocketOptions.SO_SNDBUF:
660           return NativeUnixSocket.getSocketOptionInt(fdesc, optID);
661         case SocketOptions.IP_TOS:
662           return 0;
663         case SocketOptions.SO_BINDADDR:
664           return AFSocketAddress.getInetAddress(fdesc, false, af);
665         case SocketOptions.SO_REUSEADDR:
666           return false;
667         default:
668           throw new SocketException("Unsupported option: " + optID);
669       }
670     } catch (final SocketException e) {
671       throw e;
672     } catch (final Exception e) {
673       throw (SocketException) new SocketException("Could not get option").initCause(e);
674     }
675   }

683     if (isClosed()) {

736   static final void setOptionDefault(FileDescriptor fdesc, int optID, Object value,
737       AtomicInteger acceptTimeout) throws SocketException {
738     try {
739       switch (optID) {
740         case SocketOptions.SO_LINGER:
741 
742           if (value instanceof Boolean) {
743             final boolean b = (Boolean) value;
744             if (b) {
745               throw new SocketException("Only accepting Boolean.FALSE here");
746             }
747             NativeUnixSocket.setSocketOptionInt(fdesc, optID, -1);
748             return;
749           }
750           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
751           return;
752         case SocketOptions.SO_TIMEOUT: {
753           int timeout = expectInteger(value);
754           NativeUnixSocket.setSocketOptionInt(fdesc, 0x1005, timeout);
755           NativeUnixSocket.setSocketOptionInt(fdesc, 0x1006, timeout);
756           if (acceptTimeout != null) {
757             acceptTimeout.set(timeout);
758           }
759           return;
760         }
761         case SocketOptions.SO_RCVBUF:
762         case SocketOptions.SO_SNDBUF:
763           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
764           return;
765         case SocketOptions.SO_KEEPALIVE:
766           try {
767             NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectBoolean(value));
768           } catch (SocketException e) {
769             // ignore
770           }
771           return;
772         case SocketOptions.TCP_NODELAY:
773           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectBoolean(value));
774           return;
775         case SocketOptions.IP_TOS:
776           // ignore
777           return;
778         case SocketOptions.SO_REUSEADDR:
779           // ignore
780           return;
781         default:
782           throw new SocketException("Unsupported option: " + optID);
783       }
784     } catch (final SocketException e) {
785       throw e;
786     } catch (final Exception e) {
787       throw (SocketException) new SocketException("Error while setting option").initCause(e);
788     }
789   }

791   /**
792    * Shuts down both input and output at once. Equivalent to calling {@link #shutdownInput()} and
793    * {@link #shutdownOutput()}.
794    * 
795    * @throws IOException on error.
796    */
797   protected final void shutdown() throws IOException {
798     FileDescriptor fdesc = core.validFd();
799     if (fdesc != null) {
800       NativeUnixSocket.shutdown(fdesc, SHUT_RD_WR);
801       shutdownState = 0;
802     }
803   }

805   @Override
806   protected final void shutdownInput() throws IOException {
807     FileDescriptor fdesc = core.validFd();
808     if (fdesc != null) {
809       NativeUnixSocket.shutdown(fdesc, SHUT_RD);
810       shutdownState |= 1 << (SHUT_RD);
811       if (shutdownState == SHUTDOWN_RD_WR) {
812         NativeUnixSocket.shutdown(fdesc, SHUT_RD_WR);
813         shutdownState = 0;
814       }
815     }
816   }

820     FileDescriptor fdesc = core.validFd();

822       NativeUnixSocket.shutdown(fdesc, SHUT_WR);
823       shutdownState |= 1 << (SHUT_RD_WR);
824       if (shutdownState == SHUTDOWN_RD_WR) {
825         NativeUnixSocket.shutdown(fdesc, SHUT_RD_WR);
826         shutdownState = 0;
827       }
828     }

911   protected <T> void setOption(SocketOption<T> name, T value) throws IOException {
912     if (name instanceof AFSocketOption<?>) {
913       ((AFSocketImpl<?>) this).getCore().setOption((AFSocketOption<T>) name, value);
914       return;
915     }
916     Integer optionId = SocketOptionsMapper.resolve(name);
917     if (optionId == null) {
918       super.setOption(name, value);
919     } else {
920       setOption(optionId, value);
921     }
922   }

925   @Override

928       return ((AFSocketImpl<?>) this).getCore().getOption((AFSocketOption<T>) name);
929     }
930     Integer optionId = SocketOptionsMapper.resolve(name);
931     if (optionId == null) {
932       return super.getOption(name);
933     } else {
934       return (T) getOption(optionId);
935     }
