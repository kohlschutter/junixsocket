36  /**
37   * Describes an address family supported by junixsocket.
38   * 
39   * @param <A> The corresponding {@link AFSocketAddress} subclass.
40   * @author Christian Kohlschütter
41   */
42  public final class AFAddressFamily<A extends AFSocketAddress> {
43    private static final Map<String, AFAddressFamily<?>> AF_MAP = Collections.synchronizedMap(
44        new HashMap<>());
45    private static final Map<String, AFAddressFamily<?>> URI_SCHEMES = Collections.synchronizedMap(
46        new HashMap<>());
47    private static final AtomicBoolean DEFERRED_INIT_DONE = new AtomicBoolean(false);
48  
49    private final int domain;
50    private AFSocketAddressConstructor<A> addressConstructor;
51    private @Nullable Class<A> addressClass;
52    private final String juxString;
53    private final String juxInetAddressSuffix;
54    private final String addressClassname;
55  
56    private String selectorProviderClassname;
57  
58    private AFSocket.Constructor<A> socketConstructor;
59    private AFServerSocket.Constructor<A> serverSocketConstructor;
60    private AFSocketAddressConfig<A> addressConfig;
61  
62    static {
63      NativeUnixSocket.isLoaded(); // trigger init
64    }
65  
66    private AFAddressFamily(String juxString, int domain, String addressClassname) {
67      this.juxString = juxString;
68      this.domain = domain; // FIXME validate
69      this.addressClassname = addressClassname;
70      this.juxInetAddressSuffix = "." + juxString + AFInetAddress.INETADDR_SUFFIX;
71    }
72  
73    @SuppressWarnings("unchecked")
74    static synchronized <A extends AFSocketAddress> @NonNull AFAddressFamily<A> registerAddressFamily(
75        String juxString, int domain, String addressClassname) {
76      AFAddressFamily<?> af = AF_MAP.get(juxString);
77      if (af != null) {
78        if (af.getDomain() != domain) {
79          throw new IllegalStateException("Wrong domain for address family " + juxString + ": " + af
80              .getDomain() + " vs. " + domain);
81        }
82        return (AFAddressFamily<A>) af;
83      }
84  
85      af = new AFAddressFamily<>(juxString, domain, addressClassname);
86      AF_MAP.put(juxString, af);
87  
88      return (AFAddressFamily<A>) af;
89    }
90  
91    static synchronized void triggerInit() {
92      for (AFAddressFamily<?> af : new HashSet<>(AF_MAP.values())) {
93        if (af.addressClassname != null) {
94          try {
95            Class<?> clz = Class.forName(af.addressClassname);
96            clz.getMethod("addressFamily").invoke(null);
97          } catch (Exception e) {
98            // ignore
99          }
100       }
101     }
102   }
103 
104   static synchronized AFAddressFamily<?> getAddressFamily(String juxString) {
105     return AF_MAP.get(juxString);
106   }
107 
108   static AFAddressFamily<?> getAddressFamily(URI uri) {
109     checkDeferredInit();
110     Objects.requireNonNull(uri, "uri");
111     String scheme = uri.getScheme();
112     return URI_SCHEMES.get(scheme);
113   }
114 
115   static void checkDeferredInit() {
116     if (DEFERRED_INIT_DONE.compareAndSet(false, true)) {
117       NativeUnixSocket.isLoaded();
118       AFAddressFamily.triggerInit();
119     }
120   }

137   private void checkProvider() {
138     if (socketConstructor == null && selectorProviderClassname != null) {
139       try {
140         Class.forName(selectorProviderClassname);
141       } catch (ClassNotFoundException e) {
142         // ignore
143       }
144     }
145   }

147   AFSocket.Constructor<A> getSocketConstructor() {
148     checkProvider();
149     if (socketConstructor == null) {
150       throw new UnsupportedAddressTypeException();
151     }
152     return socketConstructor;
153   }

174   /**
175    * Registers an address family.
176    * 
177    * @param <A> The supported address type.
178    * @param juxString The sockaddr_* identifier as registered in native code.
179    * @param addressClass The supported address subclass.
180    * @param config The address-specific config object.
181    * @return The corresponding {@link AFAddressFamily} instance.
182    */
183   @SuppressWarnings({"unchecked", "rawtypes"})
184   public static synchronized <A extends AFSocketAddress> AFAddressFamily<A> registerAddressFamily(
185       String juxString, //
186       Class<A> addressClass, AFSocketAddressConfig<A> config) {
187     AFAddressFamily<?> af = getAddressFamily(juxString);
188     if (af == null) {
189       throw new IllegalStateException("Address family not supported by native code: " + juxString);
190     }
191     if (af.addressClassname != null && !addressClass.getName().equals(af.addressClassname)) {
192       throw new IllegalStateException("Unexpected classname for address family " + juxString + ": "
193           + addressClass.getName());
194     }
195     if (af.addressConstructor != null || af.addressClass != null) {
196       throw new IllegalStateException("Already registered: " + juxString);
197     }
198     af.addressConfig = (AFSocketAddressConfig) config;
199     af.addressConstructor = (AFSocketAddressConstructor) config.addressConstructor();
200     af.addressClass = (Class) addressClass;
201     af.selectorProviderClassname = config.selectorProviderClassname();
202 
203     for (String scheme : config.uriSchemes()) {
204       if (scheme.isEmpty()) {
205         throw new IllegalStateException("Invalid URI scheme; cannot register " + scheme + " for "
206             + juxString);
207 
208       }
209       if (URI_SCHEMES.containsKey(scheme)) {
210         throw new IllegalStateException("URI scheme already registered; cannot register " + scheme
211             + " for " + juxString);
212       }
213       URI_SCHEMES.put(scheme, af);
214     }
215 
216     return (AFAddressFamily<A>) af;
217   }

219   /**
220    * Registers an implementation.
221    * 
222    * @param <A> The supported address type.
223    * @param juxString The sockaddr_* identifier as registered in native code.
224    * @param addressFamily The supported address family as registered via
225    *          {@link #registerAddressFamily(String, Class, AFSocketAddressConfig)}.
226    * @param config The address family-specific configuration object.
227    * @return The corresponding {@link AFAddressFamily} instance.
228    */
229   @SuppressWarnings({"unchecked", "rawtypes", "PMD.ExcessiveParameterList"})
230   public static synchronized <A extends AFSocketAddress> AFAddressFamily<A> registerAddressFamilyImpl(
231       String juxString, //
232       AFAddressFamily<A> addressFamily, //
233       AFAddressFamilyConfig<A> config) {
234     Objects.requireNonNull(addressFamily);
235     Objects.requireNonNull(config);
236 
237     AFAddressFamily<?> af = getAddressFamily(juxString);
238     if (af == null) {
239       throw new IllegalStateException("Unknown address family: " + juxString);
240     }
241     if (addressFamily != af) { // NOPMD.CompareObjectsWithEquals
242       throw new IllegalStateException("Address family inconsistency: " + juxString);
243     }
244     if (af.socketConstructor != null) {
245       throw new IllegalStateException("Already registered: " + juxString);
246     }
247     af.socketConstructor = (AFSocket.Constructor) config.socketConstructor();
248     af.serverSocketConstructor = (AFServerSocket.Constructor) config.serverSocketConstructor();
249 
250     FileDescriptorCast.registerCastingProviders(config);
251 
252     return (AFAddressFamily<A>) af;
253   }

255   @SuppressWarnings("unchecked")
256   AFSocketImplExtensions<A> initImplExtensions(AncillaryDataSupport ancillaryDataSupport) {
257     switch (getDomain()) {
258       case NativeUnixSocket.DOMAIN_TIPC:
259         return (AFSocketImplExtensions<A>) new AFTIPCSocketImplExtensions(ancillaryDataSupport);
260       default:
261         throw new UnsupportedOperationException();
262     }
263   }

265   /**
266    * Creates a new, unconnected, unbound socket compatible with this socket address.
267    * 
268    * @return The socket instance.
269    * @throws IOException on error.
270    */
271   public AFSocket<?> newSocket() throws IOException {
272     try {
273       return getSocketConstructor().newInstance(null, null);
274     } catch (UnsupportedOperationException e) {
275       throw (SocketException) new SocketException().initCause(e);
276     }
277   }

279   /**
280    * Creates a new, unconnected, unbound server socket compatible with this socket address.
281    * 
282    * @return The server socket instance.
283    * @throws IOException on error.
284    */
285   public AFServerSocket<?> newServerSocket() throws IOException {
286     try {
287       return getServerSocketConstructor().newInstance(null);
288     } catch (UnsupportedOperationException e) {
289       throw (SocketException) new SocketException().initCause(e);
290     }
291   }

293   AFSocketAddress parseURI(URI u, int overridePort) throws SocketException {
294     if (addressConfig == null) {
295       throw new SocketException("Cannot instantiate addresses of type " + addressClass);
296     }
297     return addressConfig.parseURI(u, overridePort);
298   }

300   /**
301    * Returns the set of supported URI schemes that can be parsed to some {@link AFSocketAddress}.
302    * 
303    * The set is dependent on which {@link AFSocketAddress} implementations are registered with
304    * junixsocket.
305    * 
306    * @return The set of supported URI schemes.
307    */
308   public static synchronized Set<String> uriSchemes() {
309     checkDeferredInit();
310     return Collections.unmodifiableSet(URI_SCHEMES.keySet());
311   }
