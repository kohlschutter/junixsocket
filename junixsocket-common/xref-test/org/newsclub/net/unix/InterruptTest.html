40  @TestMethodOrder(MethodOrderer.MethodName.class)
41  public abstract class InterruptTest<A extends SocketAddress> extends SocketTestBase<A> {
42    protected InterruptTest(AddressSpecifics<A> asp) {
43      super(asp);
44    }
45  
46    @Test
47    public void testInterruptSocketVirtualThread() throws Exception {
48      if (!ThreadUtil.isVirtualThreadSupported()) {
49        throw new TestAbortedNotAnIssueException("Virtual Threads are not supported by this JVM");
50      }
51  
52      try (ServerSocket serverSocket = newServerSocketBindOn(newTempAddress())) {
53        CompletableFuture<Thread> myThread = new CompletableFuture<>();
54        CompletableFuture<Boolean> result = new CompletableFuture<>();
55  
56        ExecutorService executor = ThreadUtil.newVirtualThreadPerTaskExecutor();
57        executor.submit(() -> {
58          myThread.complete(Thread.currentThread());
59          boolean closedByInterrupt = false;
60          try {
61            serverSocket.accept();
62          } catch (SocketClosedByInterruptException e) {
63            if (Thread.interrupted()) {
64              closedByInterrupt = true;
65            } else {
66              e.printStackTrace();
67            }
68          } catch (SocketException e) {
69            if ("Closed by interrupt".equals(e.getMessage()) && Thread.interrupted() && serverSocket
70                .isClosed()) {
71              closedByInterrupt = true;
72            } else {
73              e.printStackTrace();
74            }
75          } catch (IOException e) {
76            e.printStackTrace();
77          } finally {
78            result.complete(closedByInterrupt);
79          }
80        });
81  
82        Thread t = myThread.get();
83        t.interrupt();
84        assertTrue(result.get(),
85            "Thread should have thrown a \"Closed by interrupt\" SocketException,"
86                + " the server socket should be closed, and the interrupt state should be set");
87  
88        executor.shutdownNow();
89        assertTrue(executor.awaitTermination(1, TimeUnit.SECONDS));
90      }
91    }
92  
93    @Test
94    public void testInterruptSocketChannelVirtualThread() throws Exception {
95      if (!ThreadUtil.isVirtualThreadSupported()) {
96        throw new TestAbortedNotAnIssueException("Virtual Threads are not supported by this JVM");
97      }
98  
99      assertTimeoutPreemptively(Duration.ofSeconds(10), () -> {
100       try (ServerSocketChannel serverSocketChannel = newServerSocketChannelBindOn(
101           newTempAddress())) {
102         serverSocketChannel.configureBlocking(true);
103 
104         CompletableFuture<Thread> myThread = new CompletableFuture<>();
105         CompletableFuture<Boolean> result = new CompletableFuture<>();
106 
107         ExecutorService executor = ThreadUtil.newVirtualThreadPerTaskExecutor();
108         executor.submit(() -> {
109           myThread.complete(Thread.currentThread());
110           boolean closedByInterrupt = false;
111           try {
112             serverSocketChannel.accept();
113           } catch (ClosedByInterruptException e) {
114             if (Thread.currentThread().isInterrupted()) {
115               closedByInterrupt = true;
116             } else {
117               e.printStackTrace();
118             }
119           } catch (IOException e) {
120             e.printStackTrace();
121           } finally {
122             result.complete(closedByInterrupt);
123           }
124         });
125 
126         Thread t = myThread.get();
127         t.interrupt();
128         // assertTrue(result.get(),
129         // "Thread should have thrown a \"Closed by interrupt\" SocketException, and the interrupt
130         // state should be set");
131 
132         executor.shutdownNow();
133         assertTrue(executor.awaitTermination(1, TimeUnit.SECONDS));
134       }
135     });
136   }
