65  /**
66   * This test measures throughput for sending and receiving messages over AF_UNIX, comparing
67   * implementations of junixsocket and JEP 380 (Java 16).
68   *
69   * The test is enabled by default (only runs for a very short time), and is also included within the
70   * self-test.
71   *
72   * The tests can be configured as follows (all system properties):
73   * <ul>
74   * <li><code>org.newsclub.net.unix.throughput-test.enabled</code> (0/1, default: 1)</li>
75   * <li><code>org.newsclub.net.unix.throughput-test.payload-size</code> (bytes, e.g., 8192)</li>
76   * <li><code>org.newsclub.net.unix.throughput-test.payload-size.datagram</code> (bytes, e.g., 2048;
77   * defaults to value specified with "payload-size" above)</li>
78   * <li><code>org.newsclub.net.unix.throughput-test.seconds</code> (default: 0)</li>
79   * </ul>
80   *
81   * @author Christian Kohlschütter
82   */

103   private static byte[] createTestData(int size) {
104     byte[] buf = new byte[size];
105     for (int i = 0; i < buf.length; i++) {
106       buf[i] = (byte) (i % 256);
107     }
108     return buf;
109   }

111   private static void reportResults(String testType, String s) {
112     if (NUM_SECONDS == 0) {
113       // Tests are too short to be meaningful (other than for code coverage) -- do not report
114       return;
115     }
116     System.out.println("ThroughputTest (" + testType + "): " + s);
117   }

119   @Test
120   public void testSocket() throws Exception {
121     assumeTrue(ENABLED > 0, "Throughput tests are disabled");
122     assumeTrue(PAYLOAD_SIZE > 0, "Payload must be positive");
123 
124     assertTimeoutPreemptively(Duration.ofSeconds(NUM_SECONDS + GRACE_TIME_NUM_SECONDS), () -> {
125       try (ServerThread serverThread = new ServerThread() {
126         @Override
127         protected void handleConnection(final Socket sock) throws IOException {
128           byte[] buf = new byte[PAYLOAD_SIZE];
129           int read;
130 
131           try (InputStream inputStream = sock.getInputStream();
132               OutputStream outputStream = sock.getOutputStream()) {
133             while ((read = inputStream.read(buf)) >= 0) {
134               outputStream.write(buf, 0, read);
135             }
136           }
137         }
138       }) {
139 
140         AtomicBoolean keepRunning = new AtomicBoolean(true);
141 
142         TestAsyncUtil.runAsyncDelayed(NUM_MILLISECONDS, TimeUnit.MILLISECONDS, () -> {
143           keepRunning.set(false);
144         });
145 
146         try (Socket sock = connectTo(serverThread.getServerAddress())) {
147           byte[] buf = createTestData(PAYLOAD_SIZE);
148 
149           try (InputStream inputStream = sock.getInputStream();
150               OutputStream outputStream = sock.getOutputStream()) {
151 
152             long readTotal = 0;
153             long time = System.currentTimeMillis();
154             while (keepRunning.get()) {
155               outputStream.write(buf);
156 
157               int remaining = buf.length;
158               int offset = 0;
159 
160               int read; // limited by net.local.stream.recvspace / sendspace etc.
161 
162               try {
163                 while (remaining > 0 && (read = inputStream.read(buf, offset, remaining)) >= 0) {
164                   if (read > 0) {
165                     remaining -= read;
166                     offset += read;
167                     readTotal += read;
168                   }
169                 }
170               } catch (SocketTimeoutException e) {
171                 if (keepRunning.get()) {
172                   throw e;
173                 }
174               }
175               assertEquals(0, remaining);
176             }
177             time = System.currentTimeMillis() - time;
178 
179             reportResults(stbTestType() + " byte[]", ((1000f * readTotal / time) / 1000f / 1000f)
180                 + " MB/s for payload size " + PAYLOAD_SIZE);
181           }
182         }
183       }
184     });
185   }

226   protected void runTestSocketChannel(String implId, SocketAddress sba, ServerSocketChannel ssc,
227       SupplierWithException<SocketChannel, IOException> sscSupp, boolean direct) throws Exception {
228     final AtomicBoolean keepRunning = new AtomicBoolean(true);
229 
230     try (ServerThread unused = new ServerThread() {
231 
232       @Override
233       protected ServerSocket startServer() throws IOException {
234         bindServerSocket(ssc, sba);
235         return null;
236       }
237 
238       @Override
239       public void shutdown() throws IOException {
240         super.shutdown();
241         ssc.close();
242       }
243 
244       @Override
245       protected void onServerSocketClose() {
246         keepRunning.set(false);
247         super.onServerSocketClose();
248       }
249 
250       @Override
251       protected void acceptAndHandleConnection() throws IOException {
252         ByteBuffer bb = direct ? ByteBuffer.allocateDirect(PAYLOAD_SIZE) : ByteBuffer.allocate(
253             PAYLOAD_SIZE);
254         try (SocketChannel sc = ssc.accept()) {
255           try {
256             bb.clear();
257             while (sc.read(bb) >= 0) {
258               bb.flip();
259               sc.write(bb);
260               bb.clear();
261             }
262           } catch (SocketException | ClosedChannelException | SocketTimeoutException e) {
263             if (keepRunning.get()) {
264               throw e;
265             } else {
266               // broken pipe (or connection reset by peer) is expected here
267             }
268           }
269         }
270       }
271 
272       @Override
273       protected void handleConnection(Socket sock) throws IOException {
274         throw new IllegalStateException();
275       }
276     }) {
277 
278       TestAsyncUtil.runAsyncDelayed(NUM_MILLISECONDS, TimeUnit.MILLISECONDS, () -> {
279         keepRunning.set(false);
280       });
281 
282       try (SocketChannel sc = sscSupp.get()) {
283         ByteBuffer bb = direct ? ByteBuffer.allocateDirect(PAYLOAD_SIZE) : ByteBuffer.allocate(
284             PAYLOAD_SIZE);
285 
286         long readTotal = 0;
287         long time = System.currentTimeMillis();
288         while (keepRunning.get()) {
289           bb.clear();
290           bb.put(createTestData(PAYLOAD_SIZE));
291           bb.flip();
292           long remaining = sc.write(bb);
293           bb.clear();
294 
295           long read; // limited by net.local.stream.recvspace / sendspace etc.
296           while (remaining > 0 && (read = sc.read(bb)) >= 0) {
297             remaining -= read;
298             readTotal += read;
299             bb.clear();
300           }
301           assertEquals(0, remaining);
302         }
303 
304         time = System.currentTimeMillis() - time;
305         reportResults(implId + " direct=" + direct, ((1000f * readTotal / time) / 1000f / 1000f)
306             + " MB/s for payload size " + PAYLOAD_SIZE);
307       }
308     }
309   }

313   @SuppressWarnings("PMD.CognitiveComplexity")
314   public void testDatagramPacket() throws Exception {
315     try {
316       assertTimeoutPreemptively(Duration.ofSeconds(NUM_SECONDS + GRACE_TIME_NUM_SECONDS), () -> {
317         SocketAddress dsAddr = newTempAddressForDatagram();
318         SocketAddress dcAddr = newTempAddressForDatagram();
319 
320         try (DatagramSocket ds = newDatagramSocket(); DatagramSocket dc = newDatagramSocket()) {
321           if (!ds.isBound()) {
322             ds.bind(dsAddr);
323           }
324           if (!dc.isBound()) {
325             dc.bind(dcAddr);
326           }
327 
328           dsAddr = ds.getLocalSocketAddress();
329           dcAddr = dc.getLocalSocketAddress();
330 
331           assertNotEquals(dsAddr, dcAddr);
332 
333           dc.connect(dsAddr);
334 
335           AtomicBoolean keepRunning = new AtomicBoolean(true);
336           TestAsyncUtil.runAsyncDelayed(NUM_MILLISECONDS, TimeUnit.MILLISECONDS, () -> {
337             keepRunning.set(false);
338           });
339 
340           AtomicLong readTotal = new AtomicLong();
341           long sentTotal = 0;
342 
343           new Thread() {
344             final DatagramPacket dp = new DatagramPacket(new byte[PAYLOAD_SIZE_DATAGRAM],
345                 PAYLOAD_SIZE_DATAGRAM);
346 
347             @Override
348             public void run() {
349               try {
350                 while (!Thread.interrupted() && !ds.isClosed()) {
351                   try {
352                     ds.receive(dp);
353                   } catch (SocketTimeoutException e) {
354                     continue;
355                   }
356                   int read = dp.getLength();
357                   if (read != PAYLOAD_SIZE_DATAGRAM && read != 0) {
358                     throw new IOException("Unexpected response length: " + read);
359                   }
360                   readTotal.addAndGet(dp.getLength());
361                 }
362               } catch (SocketException e) {
363                 if (keepRunning.get()) {
364                   TestStackTraceUtil.printStackTrace(e);
365                 }
366               } catch (IOException e) { // NOPMD.ExceptionAsFlowControl
367                 TestStackTraceUtil.printStackTrace(e);
368               }
369             }
370           }.start();
371 
372           long time = System.currentTimeMillis();
373 
374           DatagramPacket dp = new DatagramPacket(new byte[PAYLOAD_SIZE_DATAGRAM],
375               PAYLOAD_SIZE_DATAGRAM);
376           byte[] data = dp.getData();
377           for (int i = 0; i < data.length; i++) {
378             data[i] = (byte) i;
379           }
380 
381           while (keepRunning.get()) {
382             try {
383               dc.send(dp);
384             } catch (PortUnreachableException e) {
385               e.addSuppressed(new Exception(dp.getSocketAddress().toString()));
386               throw e;
387             }
388             sentTotal += PAYLOAD_SIZE_DATAGRAM;
389           }
390           time = System.currentTimeMillis() - time;
391           keepRunning.set(false);
392           ds.close(); // terminate server
393 
394           long readTotal0 = readTotal.get();
395 
396           reportResults(stbTestType() + " DatagramPacket", ((1000f * readTotal0 / time) / 1000f
397               / 1000f) + " MB/s for datagram payload size " + PAYLOAD_SIZE_DATAGRAM + "; " + String
398                   .format(Locale.ENGLISH, "%.1f%% packet loss", 100 * (1 - (readTotal0
399                       / (float) sentTotal))));
400         }
401       });
402     } catch (JUnitException e) {
403       // Ignore timeout failure (this is a throughput test only)
404       TestStackTraceUtil.printStackTrace(e);
405     }
406   }

474     SocketAddress dsAddr = newTempAddressForDatagram();
475     SocketAddress dcAddr = newTempAddressForDatagram();
476 
477     try (DatagramChannel ds = newDatagramChannel(); //
478         DatagramChannel dc = newDatagramChannel();) {

481         dsAddr = ds.getLocalAddress();
482       }
483       if (!dc.socket().isBound()) {
484         dc.bind(dcAddr);
485         dcAddr = dc.getLocalAddress();
486       }
487       dc.connect(dsAddr);
488       ds.connect(dcAddr);

492         sp = null;

494         sp = selectorProvider();

503   protected void testSocketDatagramChannel(String id, DatagramChannel ds, DatagramChannel dc, // NOPMD
504       SelectorProvider sp, boolean direct, boolean blocking) throws IOException {
505     // FIXME investigate why we need to add a few more bytes (82) than the payload
506     // the receiver blocks otherwise (not exactly the struct socket_addr_un).
507     // smells like some TCP/IP overhead ... (?)
508     // ds.setOption(StandardSocketOptions.SO_RCVBUF, (PAYLOAD_SIZE + 82));
509 
510     AtomicBoolean keepRunning = new AtomicBoolean(true);
511     TestAsyncUtil.runAsyncDelayed(NUM_MILLISECONDS, TimeUnit.MILLISECONDS, () -> {
512       keepRunning.set(false);
513 
514       try {
515         ds.close();
516       } catch (IOException e) {
517         TestStackTraceUtil.printStackTrace(e);
518       }
519       try {
520         dc.close();
521       } catch (IOException e) {
522         TestStackTraceUtil.printStackTrace(e);
523       }
524     });
525 
526     AtomicLong readTotal = new AtomicLong();
527     long sentTotal = 0;
528 
529     long time;
530 
531     CompletableFuture<Long> bytesRead = new CompletableFuture<>();
532     try (AbstractSelector readSelector = sp == null ? null : sp.openSelector()) {
533       new Thread() {
534         @Override
535         public void run() {
536           final ByteBuffer receiveBuffer = direct ? ByteBuffer.allocateDirect(PAYLOAD_SIZE_DATAGRAM)
537               : ByteBuffer.allocate(PAYLOAD_SIZE_DATAGRAM);
538           try {
539             SelectionKey key;
540             if (readSelector != null) {
541               key = ds.register(readSelector, SelectionKey.OP_READ);
542             } else {
543               key = null;
544             }
545             while (!Thread.interrupted() && keepRunning.get() && !bytesRead.isCancelled()) {
546               int read;
547               if (readSelector != null) {
548                 int numReady = readSelector.select(1000);
549                 if (numReady == 0) {
550                   continue;
551                 }
552                 assertEquals(1, numReady);
553                 Objects.requireNonNull(key);
554 
555                 // If we'd check for invalid keys, we could prevent the ClosedChannelException
556                 // if (!key.isValid()) {
557                 // break;
558                 // }
559               }
560               read = ds.read(receiveBuffer);
561               receiveBuffer.rewind();
562               if (read != PAYLOAD_SIZE_DATAGRAM && read != 0 && read != -1) {
563                 throw new IOException("Unexpected response length: " + read);
564               }
565               readTotal.addAndGet(read);
566             }
567             bytesRead.complete(readTotal.get());
568           } catch (ClosedChannelException | SocketException e) {
569             if (keepRunning.get()) {
570               keepRunning.set(false);
571               bytesRead.completeExceptionally(e);
572             } else {
573               bytesRead.complete(readTotal.get());
574             }
575           } catch (Exception e) { // NOPMD.ExceptionAsFlowControl
576             keepRunning.set(false);
577             bytesRead.completeExceptionally(e);
578           }
579         }
580       }.start();
581 
582       time = System.currentTimeMillis();
583 
584       final ByteBuffer sendBuffer = direct ? ByteBuffer.allocateDirect(PAYLOAD_SIZE_DATAGRAM)
585           : ByteBuffer.allocate(PAYLOAD_SIZE_DATAGRAM);
586 
587       try (AbstractSelector writeSelector = sp == null ? null : sp.openSelector()) { // NOPMD
588         if (sp != null) {
589           dc.register(writeSelector, SelectionKey.OP_WRITE);
590         }
591         while (keepRunning.get()) {
592           if (writeSelector != null) {
593             int numReady = writeSelector.select(1000);
594             if (numReady == 0) {
595               continue;
596             }
597             assertEquals(1, numReady);
598           }
599           int written;
600           try {
601             written = dc.write(sendBuffer);
602           } catch (SocketException | ClosedChannelException | SocketTimeoutException e) {
603             if (keepRunning.get()) {
604               throw e;
605             } else {
606               written = 0;
607             }
608           }
609 
610           if (written != PAYLOAD_SIZE_DATAGRAM && written != 0) {
611             throw new IOException("Unexpected written length: " + written);
612           }
613 
614           sentTotal += PAYLOAD_SIZE_DATAGRAM;
615           sendBuffer.rewind();
616         }
617       } finally {
618         time = System.currentTimeMillis() - time;
619         keepRunning.set(false);
620         ds.close(); // terminate server
621       }
622     }
623 
624     try {
625       bytesRead.get(NUM_MILLISECONDS, TimeUnit.MILLISECONDS);
626     } catch (TimeoutException e) {
627       if (NUM_SECONDS != 0) {
628         TestStackTraceUtil.printStackTrace(e);
629       }
630     } catch (InterruptedException | ExecutionException e) {
631       TestStackTraceUtil.printStackTrace(e);
632     }
633 
634     long readTotal0 = readTotal.get();
635 
636     reportResults(id + " direct=" + direct + ";blocking=" + blocking, ((1000f * readTotal0 / time)
637         / 1000f / 1000f) + " MB/s for datagram payload size " + PAYLOAD_SIZE_DATAGRAM + "; "
638         + String.format(Locale.ENGLISH, "%.1f%% packet loss", 100 * (1 - (readTotal0
639             / (float) sentTotal))));
640 
641   }
