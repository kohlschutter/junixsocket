43    @Test
44    public void testParseFail() throws Exception {
45      assertThrows(SocketException.class, () -> AFTIPCSocketAddress.of(URI.create("tipc://invalid")));
46      assertThrows(SocketException.class, () -> AFTIPCSocketAddress.of(URI.create(
47          "tipc://socket.23/")));
48    }

50    private AFTIPCSocketAddress validateAddress(String uri) throws IOException {
51      URI u = URI.create(uri);
52      AFTIPCSocketAddress address = AFTIPCSocketAddress.of(u);
53      URI u2 = address.toURI(u.getScheme(), null);
54      assertEquals(address, AFTIPCSocketAddress.of(u2));
55      return address;
56    }
57  
58    @Test
59    public void testSocketURI() throws Exception {
60      assertEquals(AFTIPCSocketAddress.ofSocket(1, 2), validateAddress("tipc://socket.1.2"));
61      assertEquals(AFTIPCSocketAddress.ofSocket(1, 2), validateAddress("tipc://socket.1.2.0"));
62      assertEquals(AFTIPCSocketAddress.ofSocket(1, 2), validateAddress("tipc://0-socket.1.2.0"));
63      assertEquals(AFTIPCSocketAddress.ofSocket(1, 2), validateAddress("tipc://0-3.1.2.0"));
64      assertEquals(AFTIPCSocketAddress.ofSocket(4, 1, 2), validateAddress("tipc://0-socket.1.2.0:4"));
65  
66      assertEquals(AFTIPCSocketAddress.ofSocket(-1, -1), validateAddress(
67          "tipc://socket.4294967295.4294967295"));
68      assertEquals(AFTIPCSocketAddress.ofSocket(-1, -1), validateAddress(
69          "tipc://socket.0xffffffff.0xffffffff"));
70    }
71  
72    @Test
73    public void testServiceURI() throws Exception {
74      assertEquals(AFTIPCSocketAddress.ofService(1, 2), validateAddress("tipc://1.2"));
75  
76      assertEquals(AFTIPCSocketAddress.ofService(1, 2), validateAddress("tipc://service.1.2"));
77      assertEquals(AFTIPCSocketAddress.ofService(1, 2), validateAddress("tipc://service.1.2.0"));
78      assertEquals(AFTIPCSocketAddress.ofService(Scope.SCOPE_NOT_SPECIFIED, 1, 2), validateAddress(
79          "tipc://0-service.1.2.0"));
80      assertEquals(AFTIPCSocketAddress.ofService(1, 2), validateAddress("tipc://2-2.1.2.0"));
81      assertEquals(AFTIPCSocketAddress.ofService(Scope.SCOPE_CLUSTER, 1, 2, 3), validateAddress(
82          "tipc://2-2.1.2.3"));
83      assertEquals(AFTIPCSocketAddress.ofService(4, Scope.SCOPE_CLUSTER, 1, 2, 3), validateAddress(
84          "tipc://2-2.1.2.3:4"));
85  
86      assertEquals(AFTIPCSocketAddress.ofService(Scope.SCOPE_CLUSTER, -1, -1, -1), validateAddress(
87          "tipc://service.4294967295.4294967295.4294967295"));
88      assertEquals(AFTIPCSocketAddress.ofService(12345, Scope.SCOPE_CLUSTER, -1, -1, -1),
89          validateAddress("tipc://service.4294967295.4294967295.4294967295:12345"));
90      assertEquals(AFTIPCSocketAddress.ofService(-1, -1), validateAddress(
91          "tipc://service.0xffffffff.0xffffffff"));
92    }

94    @Test
95    public void testServiceRangeURI() throws Exception {
96      assertEquals(AFTIPCSocketAddress.ofServiceRange(1, 2, 3), validateAddress(
97          "tipc://service-range.1.2.3"));
98      assertEquals(AFTIPCSocketAddress.ofServiceRange(1, 2, 2), validateAddress(
99          "tipc://service-range.1.2"));
100     assertEquals(AFTIPCSocketAddress.ofServiceRange(1, 2, 3), validateAddress(
101         "tipc://service-range.1.2.3"));
102     assertEquals(AFTIPCSocketAddress.ofServiceRange(Scope.SCOPE_NOT_SPECIFIED, 1, 2, 3),
103         validateAddress("tipc://0-service-range.1.2.3"));
104     assertEquals(AFTIPCSocketAddress.ofServiceRange(1, 2, 3), validateAddress("tipc://2-1.1.2.3"));
105     assertEquals(AFTIPCSocketAddress.ofServiceRange(4, Scope.SCOPE_CLUSTER, 1, 2, 3),
106         validateAddress("tipc://service-range.1.2.3:4"));
107     assertEquals(AFTIPCSocketAddress.ofServiceRange(4, Scope.SCOPE_CLUSTER, 1, 2, 3),
108         validateAddress("tipc://cluster-service-range.1.2.3:4"));
109     assertEquals(AFTIPCSocketAddress.ofServiceRange(4, Scope.SCOPE_NODE, 1, 2, 3), validateAddress(
110         "tipc://node-service-range.1.2.3:4"));
111 
112     assertEquals(AFTIPCSocketAddress.ofServiceRange(12345, Scope.SCOPE_CLUSTER, -1, -1, -1),
113         validateAddress("tipc://service-range.4294967295.4294967295.4294967295:12345"));
114     assertEquals(AFTIPCSocketAddress.ofServiceRange(-1, -1, -1), validateAddress(
115         "tipc://service-range.0xffffffff.0xffffffff"));
116     assertEquals(AFTIPCSocketAddress.ofServiceRange(-1, -1, -1), validateAddress(
117         "tipc://service-range.0xffffffff.0xffffffff.0xffffffff"));
118   }
