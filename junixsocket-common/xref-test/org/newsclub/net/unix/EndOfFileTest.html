53    @BeforeEach
54    public void setup() throws IOException {
55      server = AFUNIXServerSocket.newInstance();
56      server.bind(new AFUNIXSocketAddress(socketFile));

58      executor = Executors.newFixedThreadPool(2);
59    }

61    @AfterEach
62    public void teardown() {
63      try {
64        if (server != null) {
65          server.close();
66        }
67      } catch (IOException ignore) {
68        // ignore
69      }
70  
71      if (executor != null) {
72        executor.shutdown();
73      }
74    }

76    @SuppressWarnings("resource")
77    Socket[] connectToServer() throws Exception {
78      AFUNIXSocket clientSocket = AFUNIXSocket.newInstance();
79  
80      Future<Socket> serverAcceptFuture = executor.submit(new Callable<Socket>() {
81        @Override
82        public Socket call() throws Exception {
83          return server.accept();
84        }
85      });
86  
87      Thread.sleep(100);
88  
89      clientSocket.connect(new AFUNIXSocketAddress(socketFile));
90  
91      Socket serverSocket = serverAcceptFuture.get(100, TimeUnit.MILLISECONDS);
92  
93      return new Socket[] {serverSocket, clientSocket};
94    }

96    @Test
97    public void bidirectionalSanity() throws Exception {
98      assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
99        Socket[] sockets = connectToServer();
100       try (Socket serverSocket = sockets[0]; //
101           Socket clientSocket = sockets[1]) {
102         String input = null;
103         String output = null;
104 
105         OutputStreamWriter clientOutWriter = new OutputStreamWriter(clientSocket.getOutputStream(),
106             "UTF-8");
107         BufferedReader serverInReader = new BufferedReader(new InputStreamReader(serverSocket
108             .getInputStream(), "UTF-8"));
109         input = "TestStringOne";
110         clientOutWriter.write(input + "\n");
111         clientOutWriter.flush();
112         output = serverInReader.readLine();
113 
114         assertEquals(input, output, "Server output should match client input.");
115 
116         OutputStreamWriter serverOutWriter = new OutputStreamWriter(serverSocket.getOutputStream(),
117             "UTF-8");
118         BufferedReader clientInReader = new BufferedReader(new InputStreamReader(clientSocket
119             .getInputStream(), "UTF-8"));
120         input = "TestStringTwo";
121         serverOutWriter.write(input + "\n");
122         serverOutWriter.flush();
123         output = clientInReader.readLine();
124 
125         assertEquals(input, output, "Client output should match server input.");
126       }
127     });
128   }

216   @Test
217   @SuppressWarnings("resource")
218   public void serverWriteToSocketClosedByClient() throws Exception {
219     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
220       Socket[] sockets = connectToServer();
221       Socket serverSocket = sockets[0];
222       Socket clientSocket = sockets[1];
223 
224       clientSocket.close();
225       // wait for close to propagate
226       Thread.sleep(100);
227 
228       IOException exception = null;
229       try {
230         /*
231          * http://www.unixguide.net/network/socketfaq/2.1.shtml http://www.faqs.org/rfcs/rfc793.html
232          * 
233          * The TCP RFC allows the open side to continue sending data. In most (all?)
234          * implementations, the closed side will respond with a RST. For this reason, it takes two
235          * writes to cause an IOException with TCP sockets. (or more, if there is latency)
236          * 
237          * However, it is expected that the write give an IOException as soon as possible - which
238          * means it is OK for our socket implementation to give an IOException on the first write.
239          */
240         serverSocket.getOutputStream().write(new byte[] {'1'});
241         Thread.sleep(100);
242         serverSocket.getOutputStream().write(new byte[] {'2'});
243       } catch (IOException e) {
244         exception = e;
245       }
246 
247       assertNotNull(exception,
248           "Server should see an IOException when writing to a socket that the client closed.");
249 
250       try {
251         serverSocket.close();
252       } catch (IOException ignore) {
253         // ignore
254       }
255     });
256   }
