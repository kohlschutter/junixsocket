49  @SuppressFBWarnings({

51  public abstract class EndOfFileTest<A extends SocketAddress> extends SocketTestBase<A> {
52    protected ServerSocket server;
53    protected ExecutorService executor;
54  
55    protected EndOfFileTest(AddressSpecifics<A> asp) {
56      super(asp);
57    }
58  
59    @BeforeEach
60    public void setUp() throws IOException {
61      server = newServerSocket();
62      server.bind(getServerBindAddress());
63  
64      executor = Executors.newFixedThreadPool(2);
65    }
66  
67    @AfterEach
68    public void tearDown() throws IOException {
69      try {
70        if (server != null) {
71          server.close();
72        }
73      } catch (IOException ignore) {
74        // ignore
75      }
76  
77      if (executor != null) {
78        executor.shutdown();
79      }
80    }
81  
82    protected Socket[] connect() throws Exception {
83      Socket clientSocket = newSocket();
84  
85      Future<Socket> serverAcceptFuture = executor.submit(new Callable<Socket>() {
86        @Override
87        public Socket call() throws Exception {
88          return server.accept();
89        }
90      });
91  
92      Thread.sleep(100);
93  
94      clientSocket.connect(server.getLocalSocketAddress());
95  
96      Socket serverSocket = serverAcceptFuture.get(5, TimeUnit.SECONDS);
97  
98      return new Socket[] {serverSocket, clientSocket};
99    }

101   @Test
102   public void bidirectionalSanity() throws Exception {
103     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
104       Socket[] sockets = connect();
105       try (Socket serverSocket = sockets[0]; //
106           Socket clientSocket = sockets[1]) {
107         String input;
108         String output;
109 
110         OutputStreamWriter clientOutWriter = new OutputStreamWriter(clientSocket.getOutputStream(),
111             "UTF-8");
112         BufferedReader serverInReader = new BufferedReader(new InputStreamReader(serverSocket
113             .getInputStream(), "UTF-8"));
114         input = "TestStringOne";
115         clientOutWriter.write(input + "\n");
116         clientOutWriter.flush();
117         output = serverInReader.readLine();
118 
119         assertEquals(input, output, "Server output should match client input.");
120 
121         OutputStreamWriter serverOutWriter = new OutputStreamWriter(serverSocket.getOutputStream(),
122             "UTF-8");
123         BufferedReader clientInReader = new BufferedReader(new InputStreamReader(clientSocket
124             .getInputStream(), "UTF-8"));
125         input = "TestStringTwo";
126         serverOutWriter.write(input + "\n");
127         serverOutWriter.flush();
128         output = clientInReader.readLine();
129 
130         assertEquals(input, output, "Client output should match server input.");
131       }
132     });
133   }

221   @Test
222   @SuppressWarnings("resource")
223   public void serverWriteToSocketClosedByClient() throws Exception {
224     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
225       Socket[] sockets = connect();
226       Socket serverSocket = sockets[0];
227       Socket clientSocket = sockets[1];
228 
229       clientSocket.close();
230       // wait for close to propagate
231       Thread.sleep(100);
232 
233       IOException exception = null;
234       try {
235         /*
236          * http://www.unixguide.net/network/socketfaq/2.1.shtml http://www.faqs.org/rfcs/rfc793.html
237          * 
238          * The TCP RFC allows the open side to continue sending data. In most (all?)
239          * implementations, the closed side will respond with a RST. For this reason, it takes two
240          * writes to cause an IOException with TCP sockets. (or more, if there is latency)
241          * 
242          * However, it is expected that the write give an IOException as soon as possible - which
243          * means it is OK for our socket implementation to give an IOException on the first write.
244          */
245         serverSocket.getOutputStream().write(new byte[] {'1'});
246         Thread.sleep(100);
247         serverSocket.getOutputStream().write(new byte[] {'2'});
248       } catch (IOException e) {
249         exception = e;
250       }
251 
252       assertNotNull(exception,
253           "Server should see an IOException when writing to a socket that the client closed.");
254 
255       try {
256         serverSocket.close();
257       } catch (IOException ignore) {
258         // ignore
259       }
260     });
261   }
