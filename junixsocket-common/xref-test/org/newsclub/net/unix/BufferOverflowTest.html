42  /**
43   * See https://code.google.com/p/junixsocket/issues/detail?id=20
44   */
45  public class BufferOverflowTest {
46    private File socketFile;
47    private ServerSocket server;
48    private ExecutorService executor;
49  
50    @BeforeEach
51    public void setup() throws IOException {
52      String explicitFile = System.getProperty("org.newsclub.net.unix.testsocket");
53      if (explicitFile != null) {
54        socketFile = new File(explicitFile);
55      } else {
56        socketFile = new File(new File(System.getProperty("java.io.tmpdir")),
57            "junixsocket-test.sock");
58      }
59  
60      server = AFUNIXServerSocket.newInstance();
61      server.bind(new AFUNIXSocketAddress(socketFile));
62  
63      executor = Executors.newFixedThreadPool(2);
64    }
65  
66    @AfterEach
67    public void teardown() {
68      try {
69        if (server != null) {
70          server.close();
71        }
72      } catch (IOException e) {
73        // ignore
74      }
75  
76      if (executor != null) {
77        executor.shutdown();
78      }
79    }
80  
81    @SuppressWarnings("resource")
82    Socket[] connectToServer() throws Exception {
83      AFUNIXSocket clientSocket = AFUNIXSocket.newInstance();
84  
85      Future<Socket> serverAcceptFuture = executor.submit(new Callable<Socket>() {
86        @Override
87        public Socket call() throws Exception {
88          return server.accept();
89        }
90      });
91  
92      Thread.sleep(100);
93  
94      clientSocket.connect(new AFUNIXSocketAddress(socketFile));
95  
96      Socket serverSocket = serverAcceptFuture.get(100, TimeUnit.MILLISECONDS);
97  
98      return new Socket[] {serverSocket, clientSocket};
99    }
100 
101   @Test
102   public void readOverflow() throws Exception {
103     assertTimeout(Duration.ofSeconds(2), () -> {
104       Socket[] sockets = connectToServer();
105       try (Socket serverSocket = sockets[0]; //
106           Socket clientSocket = sockets[1];) {
107 
108         byte[] input = new byte[16];
109         byte[] output = new byte[15];
110 
111         try (OutputStream clientOutStream = clientSocket.getOutputStream(); //
112             InputStream serverInStream = serverSocket.getInputStream()) {
113           clientOutStream.write(input, 0, 16);
114           int numRead = serverInStream.read(output, 0, 16);
115           assertTrue(numRead <= 15, "Buffer Overflow not detected");
116         }
117       }
118     });
119   }
120 
121   @Test
122   public void writeOverflow() throws Exception {
123     assertTimeout(Duration.ofSeconds(2), () -> {
124       Socket[] sockets = connectToServer();
125       try (Socket serverSocket = sockets[0]; //
126           Socket clientSocket = sockets[1];) {
127 
128         byte[] input = new byte[15];
129         byte[] output = new byte[16];
130 
131         try (OutputStream clientOutStream = clientSocket.getOutputStream(); //
132             InputStream serverInStream = serverSocket.getInputStream()) {
133           try {
134             clientOutStream.write(input, 0, 16);
135             int numRead = serverInStream.read(output, 0, 15);
136             assertEquals(15, numRead);
137             serverInStream.read();
138             fail("The call to read should have thrown an IndexOutOfBoundsException");
139           } catch (IndexOutOfBoundsException ex) {
140             // expected
141           }
142         }
143       }
144     });
145   }
