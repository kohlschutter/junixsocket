18  package org.newsclub.net.unix.tipc;
19  
20  import java.io.Closeable;
21  import java.io.IOException;
22  import java.nio.ByteBuffer;
23  import java.nio.channels.SelectionKey;
24  import java.nio.channels.Selector;
25  import java.util.concurrent.atomic.AtomicBoolean;
26  
27  import org.newsclub.net.unix.AFSocketType;
28  import org.newsclub.net.unix.AFTIPCSocketAddress;
29  
30  /**
31   * Provides access to the TIPC topology service.
32   *
33   * @author Christian Kohlschütter
34   */
35  public class AFTIPCTopologyWatcher implements Closeable {
36    private final int defaultTimeout;
37    private final AFTIPCDatagramChannel channel;
38    private final Selector selector;
39    private boolean doLoop;
40    private final AtomicBoolean running = new AtomicBoolean(false);
41  
42    /**
43     * Creates an {@link AFTIPCTopologyWatcher} whose subscription requests do not time out by
44     * default.
45     *
46     * @throws IOException on error.
47     */
48    public AFTIPCTopologyWatcher() throws IOException {
49      this(AFTIPCTopologySubscription.TIPC_WAIT_FOREVER);
50    }
51  
52    /**
53     * Creates an {@link AFTIPCTopologyWatcher} whose subscription requests use the given default
54     * timeout.
55     *
56     * @param defaultTimeoutSeconds The timeout in seconds (or
57     *          {@link AFTIPCTopologySubscription#TIPC_WAIT_FOREVER};
58     * @throws IOException on error.
59     */
60    public AFTIPCTopologyWatcher(int defaultTimeoutSeconds) throws IOException {
61      this.defaultTimeout = defaultTimeoutSeconds;
62      this.channel = AFTIPCDatagramSocket.newInstance(AFSocketType.SOCK_SEQPACKET).getChannel();
63      this.selector = channel.provider().openSelector();
64      channel.connect(AFTIPCSocketAddress.ofTopologyService());
65      channel.configureBlocking(false);
66    }
67  
68    /**
69     * Watches for all port changes.
70     *
71     * @return The subscription object.
72     * @throws IOException on error.
73     * @see #cancelSubscription(AFTIPCTopologySubscription)
74     */
75    public final AFTIPCTopologySubscription addPortSubscription() throws IOException {
76      return addPortSubscription(0, ~0);
77    }
78  
79    /**
80     * Watches for port changes of the given port ("port" meaning TIPC port, not TCP).
81     *
82     * @param port The port.
83     * @return The subscription object.
84     * @throws IOException on error.
85     * @see #cancelSubscription(AFTIPCTopologySubscription)
86     */
87    public final AFTIPCTopologySubscription addPortSubscription(int port) throws IOException {
88      return addPortSubscription(port, port);
89    }
90  
91    /**
92     * Watches for port changes within the given range ("port" meaning TIPC port, not TCP).
93     *
94     * @param lower The lower value of the port range.
95     * @param upper The upper value of the port range.
96     * @return The subscription object.
97     * @throws IOException on error.
98     * @see #cancelSubscription(AFTIPCTopologySubscription)
99     */
100   public final AFTIPCTopologySubscription addPortSubscription(int lower, int upper)
101       throws IOException {
102     return sendMessage(new AFTIPCTopologySubscription(AFTIPCTopologySubscription.TIPC_NODE_STATE,
103         lower, upper, AFTIPCTopologySubscription.Flags.TIPC_SUB_PORTS, defaultTimeout,
104         AFTIPCTopologySubscription.USR_EMPTY));
105   }
106 
107   /**
108    * Watches for all link state changes.
109    *
110    * @return The subscription object.
111    * @throws IOException on error.
112    * @see #cancelSubscription(AFTIPCTopologySubscription)
113    */
114   public final AFTIPCTopologySubscription addLinkStateSubscription() throws IOException {
115     return sendMessage(new AFTIPCTopologySubscription(AFTIPCTopologySubscription.TIPC_LINK_STATE, 0,
116         ~0, AFTIPCTopologySubscription.Flags.NONE, defaultTimeout,
117         AFTIPCTopologySubscription.USR_EMPTY));
118   }
119 
120   /**
121    * Watches for service changes of the given service type, matching any instance.
122    *
123    * @param type The service type.
124    * @return The subscription object.
125    * @throws IOException on error.
126    * @see #cancelSubscription(AFTIPCTopologySubscription)
127    */
128   public final AFTIPCTopologySubscription addServiceSubscription(int type) throws IOException {
129     return addServiceSubscription(type, 0, ~0);
130   }
131 
132   /**
133    * Watches for service changes of the given service type, matching only the specified instance.
134    *
135    * @param type The service type.
136    * @param instance The instance to match.
137    * @return The subscription object.
138    * @throws IOException on error.
139    * @see #cancelSubscription(AFTIPCTopologySubscription)
140    */
141   public final AFTIPCTopologySubscription addServiceSubscription(int type, int instance)
142       throws IOException {
143     return addServiceSubscription(type, instance, instance);
144   }
145 
146   /**
147    * Watches for service changes of the given service type and instance range.
148    *
149    * @param type The service type.
150    * @param lower The lower value of the instance range.
151    * @param upper The upper value of the instance range.
152    * @return The subscription object.
153    * @throws IOException on error.
154    * @see #cancelSubscription(AFTIPCTopologySubscription)
155    */
156   public final AFTIPCTopologySubscription addServiceSubscription(int type, int lower, int upper)
157       throws IOException {
158     return sendMessage(new AFTIPCTopologySubscription(type, lower, upper,
159         AFTIPCTopologySubscription.Flags.TIPC_SUB_SERVICE, defaultTimeout,
160         AFTIPCTopologySubscription.USR_EMPTY));
161   }
162 
163   /**
164    * Cancels a previously added service subscription.
165    *
166    * @param sub The subscription to cancel.
167    * @throws IOException on error.
168    */
169   public final void cancelSubscription(AFTIPCTopologySubscription sub) throws IOException {
170     sendMessage(sub.toCancellation());
171   }
172 
173   /**
174    * Sends a manually crafted subscription message to the TIPC topology server. You usually don't
175    * need to do this directly; use the
176    * {@code #addPortSubscription(int, int)}/{@link #cancelSubscription(AFTIPCTopologySubscription)}
177    * methods instead.
178    *
179    * @param sub The subscription message.
180    * @return The very message.
181    * @throws IOException on error.
182    */
183   public final AFTIPCTopologySubscription sendMessage(AFTIPCTopologySubscription sub)
184       throws IOException {
185     channel.write(sub.toBuffer());
186     return sub;
187   }
188 
189   /**
190    * Runs a receive loop until {@link #stopLoop()} or {@link #close()} is called.
191    *
192    * This method returns after the run loop terminates.
193    *
194    * @throws IOException on error.
195    */
196   @SuppressWarnings("null")
197   public final void runLoop() throws IOException {
198     if (!running.compareAndSet(false, true)) {
199       throw new IllegalStateException("Already running");
200     }
201 
202     ByteBuffer buf = ByteBuffer.allocate(64);
203     SelectionKey key = channel.register(selector, SelectionKey.OP_READ);
204 
205     try {
206       doLoop = true;
207       while (!Thread.interrupted() && doLoop) {
208         int n = selector.select();
209         if (!key.isValid() || !doLoop) {
210           break;
211         }
212         if (n > 0) {
213           channel.receive(buf);
214           @SuppressWarnings("cast")
215           AFTIPCTopologyEvent event = AFTIPCTopologyEvent.readFromBuffer((ByteBuffer) buf.flip());
216           onEvent(event);
217           buf.clear();
218         }
219       }
220     } finally {
221       key.cancel();
222       running.set(false);
223     }
224   }
225 
226   /**
227    * Called for every event encountered by the run loop.
228    *
229    * @param event The event.
230    * @throws IOException on error. Any exception will terminate the run loop.
231    * @see #runLoop()
232    */
233   protected void onEvent(AFTIPCTopologyEvent event) throws IOException {
234   }
235 
236   /**
237    * Called upon {@link #close()}.
238    *
239    * @throws IOException on error.
240    */
241   protected void onClose() throws IOException {
242   }
243 
244   /**
245    * Checks if the watcher run loop is running.
246    *
247    * @return {@code true} if running.
248    * @see #runLoop()
249    */
250   public boolean isRunning() {
251     return running.get();
252   }
253 
254   /**
255    * Stops the run loop.
256    */
257   public final void stopLoop() {
258     doLoop = false;
259     selector.wakeup();
260   }
261 
262   /**
263    * Closes this instance.
264    */
265   @Override
266   public final void close() throws IOException {
267     channel.close();
268     stopLoop();
269     onClose();
270   }
271 }
