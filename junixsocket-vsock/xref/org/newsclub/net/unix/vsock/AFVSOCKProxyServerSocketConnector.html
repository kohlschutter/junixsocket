18  package org.newsclub.net.unix.vsock;
19  
20  import java.io.File;
21  import java.io.IOException;
22  
23  import org.newsclub.net.unix.AFServerSocket;
24  import org.newsclub.net.unix.AFServerSocketConnector;
25  import org.newsclub.net.unix.AFSocketAddress;
26  import org.newsclub.net.unix.AFUNIXSocketAddress;
27  import org.newsclub.net.unix.AFVSOCKSocketAddress;
28  import org.newsclub.net.unix.AddressUnavailableSocketException;
29  
30  /**
31   * Provides access to AF_VSOCK connections that aren't directly accessible but exposed via a
32   * proxying/multiplexing Unix domain socket.
33   *
34   * @author Christian Kohlschütter
35   * @see #openFirecrackerStyleConnector(File, int)
36   * @see #openDirectConnector()
37   */
38  public final class AFVSOCKProxyServerSocketConnector implements
39      AFServerSocketConnector<AFVSOCKSocketAddress, AFSocketAddress> {
40    private static final AFServerSocketConnector<AFVSOCKSocketAddress, AFSocketAddress> DIRECT_CONNECTOR =
41        new AFServerSocketConnector<AFVSOCKSocketAddress, AFSocketAddress>() {
42  
43          @Override
44          public AFServerSocket<? extends AFSocketAddress> bind(AFVSOCKSocketAddress addr)
45              throws IOException {
46            return addr.newForceBoundServerSocket();
47          }
48        };
49  
50    private final String listenAddressPrefix;
51    private final int allowedCID;
52  
53    private AFVSOCKProxyServerSocketConnector(String listenAddressPrefix, int allowedCID) {
54      this.listenAddressPrefix = listenAddressPrefix;
55      this.allowedCID = allowedCID;
56    }
57  
58    /**
59     * Returns an instance that is configured to support
60     * [Firecracker-style](https://github.com/firecracker-microvm/firecracker/blob/main/docs/vsock.md)
61     * Unix domain sockets.
62     *
63     * @param listenAddressPrefix The prefix of any listening socket. The actual socket will have
64     *          <code>_<em>vsockPort</em></code> appended to it (with {@code vsockPort} being replaced
65     *          by the corresponding port number).
66     * @param allowedCID The permitted CID, or {@link AFVSOCKSocketAddress#VMADDR_CID_ANY} for "any".
67     * @return The instance.
68     */
69    public static AFServerSocketConnector<AFVSOCKSocketAddress, AFSocketAddress> openFirecrackerStyleConnector(
70        File listenAddressPrefix, int allowedCID) {
71      return new AFVSOCKProxyServerSocketConnector(listenAddressPrefix.getAbsolutePath(), allowedCID);
72    }
73  
74    /**
75     * Returns an instance that is configured to connect directly to the given address.
76     *
77     * @return The direct instance.
78     */
79    public static AFServerSocketConnector<AFVSOCKSocketAddress, AFSocketAddress> openDirectConnector() {
80      return DIRECT_CONNECTOR;
81    }
82  
83    @Override
84    public AFServerSocket<?> bind(AFVSOCKSocketAddress addr) throws IOException {
85      int cid = addr.getVSOCKCID();
86      if (cid != allowedCID && cid != AFVSOCKSocketAddress.VMADDR_CID_ANY
87          && allowedCID != AFVSOCKSocketAddress.VMADDR_CID_ANY) {
88        throw new AddressUnavailableSocketException("Factory does not cover CID " + cid);
89      }
90  
91      return AFUNIXSocketAddress.of(new File(listenAddressPrefix + "_" + addr.getVSOCKPort()))
92          .newForceBoundServerSocket();
93    }
94  }
