18  package org.newsclub.net.unix;
19  
20  import static org.junit.jupiter.api.Assertions.assertFalse;
21  import static org.junit.jupiter.api.Assertions.assertTrue;
22  import static org.junit.jupiter.api.Assertions.fail;
23  
24  import java.io.IOException;
25  import java.net.ServerSocket;
26  import java.net.Socket;
27  import java.net.SocketException;
28  
29  import org.junit.jupiter.api.Test;
30  
31  /**
32   * Tests breaking out of accept.
33   * 
34   * @see <a href="http://code.google.com/p/junixsocket/issues/detail?id=6">Issue 6</a>
35   */
36  public class CancelAcceptTest extends SocketTestBase {
37    private boolean serverSocketClosed = false;
38  
39    public CancelAcceptTest() throws IOException {
40      super();
41    }
42  
43    @Test
44    public void issue6test1() throws Exception {
45      serverSocketClosed = false;
46  
47      final ServerThread st = new ServerThread() {
48  
49        @Override
50        protected void handleConnection(final Socket sock) throws IOException {
51        }
52  
53        @Override
54        protected void onServerSocketClose() {
55          serverSocketClosed = true;
56        }
57      };
58  
59      try (AFUNIXSocket sock = connectToServer()) {
60        // open and close
61      }
62      try (AFUNIXSocket sock = connectToServer()) {
63        // open and close
64      }
65  
66      @SuppressWarnings("resource")
67      final ServerSocket servSock = st.getServerSocket();
68  
69      assertFalse(serverSocketClosed && !servSock.isClosed(),
70          "ServerSocket should not be closed now");
71      servSock.close();
72      try {
73        try (AFUNIXSocket sock = connectToServer()) {
74          // open and close
75        }
76        fail("Did not throw SocketException");
77      } catch (SocketException e) {
78        // as expected
79      }
80      assertTrue(serverSocketClosed || servSock.isClosed(), "ServerSocket should be closed now");
81  
82      try {
83        try (AFUNIXSocket sock = connectToServer()) {
84          fail("ServerSocket should have been closed already");
85        }
86        fail("Did not throw SocketException");
87      } catch (SocketException e) {
88        // as expected
89      }
90    }
91  }
