63    private static void assertChangeToNonBlocking(ServerSocketChannel as) throws IOException {
64      assertTrue(as.isBlocking());
65      as.configureBlocking(false);
66      assertFalse(as.isBlocking());
67    }

69    private static void assertSelect(int expected, Selector sel, boolean block) throws IOException {
70      int now = sel.selectNow();
71      if (now != 0) {
72        assertEquals(expected, now);
73      }
74  
75      if (block) {
76        assertEquals(expected, now = Math.max(now, sel.select()));
77      }
78      assertEquals(expected, Math.max(now, sel.selectNow()));
79      assertEquals(expected, sel.selectedKeys().size());
80    }

82    @Test
83    public void testNonBlockingAccept() throws IOException, InterruptedException {
84      assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
85        Selector sscSel = selectorProvider().openSelector();
86        assertTrue(sscSel.selectedKeys().isEmpty());
87  
88        try (ServerSocketChannel ssc = selectorProvider().openServerSocketChannel()) {
89          bindServerSocket(ssc, newTempAddress());
90          assertChangeToNonBlocking(ssc);
91  
92          SelectionKey key = ssc.register(sscSel, SelectionKey.OP_ACCEPT);
93  
94          assertEquals(Collections.singleton(key), sscSel.keys());
95          assertTrue(sscSel.selectedKeys().isEmpty());
96  
97          assertNull(ssc.accept());
98          assertEquals(0, sscSel.selectNow());
99  
100         try (SocketChannel sc = selectorProvider().openSocketChannel()) {
101           CompletableFuture<Boolean> future = new CompletableFuture<>();
102           new Thread(() -> {
103             try {
104               // connect from a different thread
105               // TIPC doesn't like being called from the same thread
106               future.complete(connectSocket(sc, ssc.getLocalAddress()));
107             } catch (RuntimeException | IOException e) {
108               future.completeExceptionally(e);
109             }
110           }).start();
111 
112           assertSelect(1, sscSel, true);
113           assertEquals(Collections.singleton(key), sscSel.selectedKeys());
114 
115           SocketChannel sscsc = ssc.accept();
116           assertNotNull(sscsc);
117 
118           assertTrue(future.get());
119 
120           assertNull(ssc.accept());
121         }
122       }
123 
124       assertSelect(0, sscSel, false);
125     });
126   }

128   @Test
129   public void testCancelSelect() throws Exception {
130     Selector selector = selectorProvider().openSelector();
131     CompletableFuture<Integer> cf = new CompletableFuture<>();
132 
133     new Thread() {
134       @Override
135       public void run() {
136         int num;
137         try {
138           num = selector.select();
139         } catch (IOException e) {
140           cf.completeExceptionally(e);
141           return;
142         }
143         cf.complete(num);
144       }
145     }.start();
146 
147     selector.wakeup();
148     try {
149       assertEquals(0, cf.get(5, TimeUnit.SECONDS));
150     } catch (ExecutionException e) {
151       if ((e.getCause() instanceof SocketException) && e.getCause().getMessage().contains(
152           "closed")) {
153         // OK
154       } else {
155         throw e;
156       }
157     }
158   }

160   private Future<Integer> newHelloClient(SocketAddress serverAddr, Semaphore sema) {
161     return Executors.newFixedThreadPool(1).submit(() -> {
162       try (Socket sock = connectTo(serverAddr); //
163           OutputStream out = sock.getOutputStream()) {
164         out.write("Hello".getBytes(StandardCharsets.UTF_8));
165         out.flush();
166 
167         if (sema != null) {
168           sema.tryAcquire(1, 10000, TimeUnit.MILLISECONDS);
169         }
170       }
171       return 0;
172     });
173   }

233   @SuppressWarnings({
234       "PMD.CognitiveComplexity", "PMD.NcssCount", "PMD.CyclomaticComplexity",
235       "PMD.ExcessiveMethodLength", "PMD.NPathComplexity"})
236   private void testConnectionClose(boolean clientCloseImmediately, boolean checkInvalid)
237       throws Exception {
238     ByteBuffer buffer = ByteBuffer.allocate(512);
239 
240     SelectorProvider provider = selectorProvider();
241     try (ServerSocketChannel server = provider.openServerSocketChannel()) {
242       bindServerSocket(server, newTempAddress());
243       server.configureBlocking(false);
244 
245       Selector selector = provider.openSelector();
246       server.register(selector, SelectionKey.OP_ACCEPT);
247 
248       final Semaphore mayCloseSema = new Semaphore(0);
249       final Future<Integer> threadFuture = newHelloClient(server.getLocalAddress(),
250           clientCloseImmediately ? null : mayCloseSema);
251 
252       int numAcceptable = 0;
253       int numReadable = 0;
254       int numClosedChannelException = 0;
255 
256       long timeout = 5000;
257       selectLoop : while (timeout > 0) {
258         long time = System.currentTimeMillis();
259         int n;
260         while ((n = selector.select(timeout)) != 0) {
261           assertTrue(n > 0);
262           for (SelectionKey key : selector.selectedKeys()) {
263             if (numAcceptable > 3 || numReadable > 2) {
264               break selectLoop;
265             }
266             if (checkInvalid && !key.isValid()) {
267               key.cancel();
268               timeout = 10;
269               continue;
270             }
271             if (key.isAcceptable()) {
272               SocketChannel channel = server.accept();
273               if (channel == null) {
274                 continue;
275               }
276               numAcceptable++;
277               channel.configureBlocking(false);
278               channel.register(selector, SelectionKey.OP_READ);
279               if (!clientCloseImmediately) {
280                 assertNotNull(channel.getLocalAddress());
281               } else {
282                 channel.getLocalAddress(); // just trigger
283               }
284               channel.getRemoteAddress(); // just trigger
285             }
286             if (key.isReadable()) {
287               numReadable++;
288               SocketChannel channel = (SocketChannel) key.channel();
289               try {
290                 buffer.clear();
291                 int numRead = channel.read(buffer);
292                 switch (numReadable) {
293                   case 1:
294                     assertEquals("Hello", new String(buffer.array(), 0, numRead,
295                         StandardCharsets.UTF_8));
296                     break;
297                   case 2:
298                     fail("Should have thrown ClosedChannelException");
299                     break;
300                   default:
301                     fail("Should not have been reached");
302                     break;
303                 }
304                 break selectLoop;
305               } catch (ClosedChannelException e) {
306                 // channel.close(); // not strictly necessary -- will be closed via Cleaner
307                 numClosedChannelException++;
308                 if (!checkInvalid) {
309                   key.cancel();
310                   if (numReadable >= 2) {
311                     break selectLoop;
312                   }
313                 }
314               }
315             }
316           }
317         }
318         timeout -= (System.currentTimeMillis() - time);
319       }
320 
321       mayCloseSema.release();
322 
323       if (numAcceptable == 0 && threadFuture.isDone()) {
324         try {
325           threadFuture.get();
326         } catch (ExecutionException e) {
327           Throwable t = e.getCause();
328           if (t instanceof RuntimeException) {
329             throw (RuntimeException) t; // NOPMD.PreserveStackTrace
330           }
331         }
332       }
333       assertEquals(1, numAcceptable);
334 
335       if (!checkInvalid) {
336         // invalid
337       } else {
338         if (numClosedChannelException == 0) { // either the client closed and data didn't make it
339                                               // through, or the second select correctly said
340                                               // nothing was readable
341           if (clientCloseImmediately) {
342             if (numReadable != 0) {
343               assertEquals(1, numReadable);
344             }
345           }
346         } else {
347           assertEquals(2, numReadable);
348           assertEquals(1, numClosedChannelException);
349         }
350       }
351 
352       threadFuture.get(1, TimeUnit.SECONDS);
353     }
354   }
