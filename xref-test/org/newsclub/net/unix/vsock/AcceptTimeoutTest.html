18  package org.newsclub.net.unix.vsock;
19  
20  import java.io.IOException;
21  import java.net.SocketAddress;
22  import java.net.SocketTimeoutException;
23  
24  import org.junit.jupiter.api.Test;
25  import org.newsclub.net.unix.AFSocketCapability;
26  import org.newsclub.net.unix.AFSocketCapabilityRequirement;
27  import org.newsclub.net.unix.AFVSOCKSocketAddress;
28  import org.opentest4j.TestAbortedException;
29  
30  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
31  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException;
32  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException.MessageType;
33  
34  @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_VSOCK)
35  @SuppressFBWarnings("NM_SAME_SIMPLE_NAME_AS_SUPERCLASS")
36  public final class AcceptTimeoutTest extends
37      org.newsclub.net.unix.AcceptTimeoutTest<AFVSOCKSocketAddress> {
38  
39    public AcceptTimeoutTest() {
40      super(AFVSOCKAddressSpecifics.INSTANCE);
41    }
42  
43    @Override
44    @Test
45    public void testTimeoutAfterDelay() throws Exception {
46      try {
47        super.testTimeoutAfterDelay();
48      } catch (SocketTimeoutException e) {
49        e.printStackTrace();
50        try {
51          newInterconnectedSockets();
52        } catch (TestAbortedException e2) {
53          TestAbortedWithImportantMessageException e3 = new TestAbortedWithImportantMessageException(
54              MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
55              AFVSOCKAddressSpecifics.KERNEL_NOT_CONFIGURED, summaryImportantMessage(
56                  AFVSOCKAddressSpecifics.KERNEL_NOT_CONFIGURED), e);
57          e3.addSuppressed(e2);
58          throw e3; // NOPMD.PreserveStackTrace
59        }
60      }
61    }
62  
63    /**
64     * Subclasses may override this to tell that there is a known issue with "Accept timeout after
65     * delay" when a SocketTimeoutException was thrown.
66     *
67     * @param e The exception
68     * @return An explanation iff this should not cause a test failure but trigger "With issues".
69     */
70    @Override
71    protected String checkKnownBugAcceptTimeout(SocketTimeoutException e) {
72      String message = e.getMessage();
73      if (message != null && message.contains("navailable")) {
74        return "Server accept timed out. VSOCK may not be available";
75      } else {
76        return null;
77      }
78    }
79  
80    @Override
81    protected String checkKnownBugAcceptTimeout(SocketAddress serverAddress) {
82      boolean vsockNotAvailable = false;
83  
84      Integer cid = null;
85      try {
86        if ((cid = AFVSOCKSocket.getLocalCID()) == AFVSOCKSocketAddress.VMADDR_CID_ANY) {
87          vsockNotAvailable = true;
88        }
89      } catch (IOException e) {
90        e.printStackTrace();
91        vsockNotAvailable = true;
92      }
93  
94      if (vsockNotAvailable) {
95        return "Server accept timed out. " + (cid != null ? "Local CID=" + cid
96            : "Local CID could not be retrieved") + ". VSOCK may not be available";
97      } else if ((serverAddress instanceof AFVSOCKSocketAddress)) {
98        AFVSOCKSocketAddress sa = (AFVSOCKSocketAddress) serverAddress;
99        if (sa.getVSOCKCID() == AFVSOCKSocketAddress.VMADDR_CID_LOCAL) {
100         // seen on Amazon EC2
101         return "Server accept timed out. Requested VMADDR_CID_LOCAL with local CID=" + cid
102             + ". VSOCK may not be available";
103       }
104     }
105     return null;
106   }
107 }
