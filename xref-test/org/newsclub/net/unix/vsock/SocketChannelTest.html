18  package org.newsclub.net.unix.vsock;
19  
20  import java.io.IOException;
21  import java.net.SocketAddress;
22  import java.net.SocketException;
23  import java.net.SocketTimeoutException;
24  import java.nio.channels.ServerSocketChannel;
25  import java.nio.channels.SocketChannel;
26  
27  import org.newsclub.net.unix.AFSocketCapability;
28  import org.newsclub.net.unix.AFSocketCapabilityRequirement;
29  import org.newsclub.net.unix.AFVSOCKSocketAddress;
30  import org.newsclub.net.unix.InvalidSocketException;
31  
32  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
33  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException;
34  import com.kohlschutter.testutil.TestAbortedWithImportantMessageException.MessageType;
35  
36  @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_VSOCK)
37  @SuppressFBWarnings("NM_SAME_SIMPLE_NAME_AS_SUPERCLASS")
38  public final class SocketChannelTest extends
39      org.newsclub.net.unix.SocketChannelTest<AFVSOCKSocketAddress> {
40  
41    public SocketChannelTest() {
42      super(AFVSOCKAddressSpecifics.INSTANCE);
43    }
44  
45    /**
46     * Subclasses may override this to tell that there is a known issue with "accept".
47     *
48     * @param e The exception
49     * @return An explanation iff this should not cause a test failure but trigger "With issues".
50     */
51    @Override
52    protected String checkKnownBugAcceptFailure(SocketException e) {
53      if (e instanceof InvalidSocketException) {
54        return "Server accept failed. VSOCK may not be available";
55      } else {
56        return null;
57      }
58    }
59  
60    /**
61     * Subclasses may override this to tell that there is a known issue with "accept".
62     *
63     * @param e The exception
64     * @return An explanation iff this should not cause a test failure but trigger "With issues".
65     */
66    @Override
67    protected String checkKnownBugAcceptFailure(SocketTimeoutException e) {
68      return "Server accept failed. VSOCK may not be available";
69    }
70  
71    @Override
72    protected String checkKnownBugFirstAcceptCallNotTerminated() {
73      // seen in virtualized Linux environments with CID=3
74      return AFVSOCKAddressSpecifics.KERNEL_NOT_CONFIGURED;
75    }
76  
77    @Override
78    protected void handleBind(ServerSocketChannel ssc, SocketAddress sa) throws IOException {
79      try {
80        super.handleBind(ssc, sa);
81      } catch (InvalidSocketException e) {
82        String msg = "Could not bind AF_VSOCK server socket to CID=" + ((AFVSOCKSocketAddress) sa)
83            .getVSOCKCID() + "; check kernel capabilities.";
84        throw (TestAbortedWithImportantMessageException) new TestAbortedWithImportantMessageException(
85            MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg)).initCause(
86                e);
87      }
88    }
89  
90    @Override
91    protected boolean handleConnect(SocketChannel sc, SocketAddress sa) throws IOException {
92      try {
93        return super.handleConnect(sc, sa);
94      } catch (InvalidSocketException e) {
95        String msg = "Could not connect AF_VSOCK socket to CID=" + ((AFVSOCKSocketAddress) sa)
96            .getVSOCKCID() + "; check kernel capabilities.";
97        throw (TestAbortedWithImportantMessageException) new TestAbortedWithImportantMessageException(
98            MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg)).initCause(
99                e);
100     }
101   }
102 }
