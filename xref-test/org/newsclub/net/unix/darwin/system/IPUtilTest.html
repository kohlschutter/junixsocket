18  package org.newsclub.net.unix.darwin.system;
19  
20  import static org.junit.jupiter.api.Assertions.assertEquals;
21  
22  import java.nio.ByteBuffer;
23  
24  import org.junit.jupiter.api.Test;
25  
26  public class IPUtilTest {
27  
28    @Test
29    public void testIPHeaderChecksum() {
30      // example from https://www.thegeekstuff.com/2012/05/ip-header-checksum/
31      ByteBuffer bb = ByteBuffer.wrap(new byte[] {
32          0x45, 0x00, //
33          0x00, 0x3c, //
34          0x1c, 0x46, //
35          0x40, 0x00, //
36          0x40, 0x06, //
37          0x00, 0x00, //
38          (byte) 0xac, 0x10, //
39          0x0a, 0x63, //
40          (byte) 0xac, 0x10, //
41          0x0a, 0x0c});
42  
43      assertEquals(0xB1E6, IPUtil.checksumIPv4header(bb, 0, bb.remaining()));
44    }
45  }
