51  /**
52   * Some base functionality for socket tests.
53   *
54   * This class provides access to the {@link AddressSpecifics} methods for the socket implementation
55   * under test. It is essential to use these wrapper methods in tests instead of directly calling the
56   * {@link AFSocket} etc. methods: Some socket implementations (and sometimes only in certain
57   * kernel/environment configurations) may expose unexpected behavior that is otherwise hard to
58   * catch.
59   *
60   * This is especially relevant when connecting/binding sockets (see
61   * {@link #connectSocket(Socket, SocketAddress)}, #bindServerSocket(ServerSocket, SocketAddress)},
62   * etc.)
63   *
64   * @author Christian Kohlschuetter
65   */

122   public static File newTempFile(String name) {
123     File f;
124     try {
125       f = (name == null) ? File.createTempFile("jutest", ".sock") : new File(name);
126       f.deleteOnExit(); // always delete on exit to clean-up sockets created under that name
127     } catch (IOException e) {
128       throw new IllegalStateException("Can't create temporary file", e);
129     }
130     if (!f.delete()) {
131       throw new IllegalStateException("Could not delete temporary file that we just created: " + f);
132     }
133     return f;
134   }

136   protected ServerSocket startServer() throws IOException {
137     caller = new Exception("Test server stacktrace");
138     final ServerSocket server = newServerSocket();
139     SocketAddress bindAddr = getServerBindAddress();
140     try {
141       asp.bindServerSocket(server, getServerBindAddress());
142     } catch (BindException e) {
143       if (asp instanceof JavaAddressSpecifics && ((InetSocketAddress) bindAddr).getPort() == 0) {
144         asp.bindServerSocket(server, null);
145       } else {
146         throw e;
147       }
148     }
149     return server;
150   }

168   protected abstract class ServerThread extends Thread implements AutoCloseable {
169     private final ServerSocket serverSocket;
170     private volatile Exception exception = null;
171     private volatile Error error = null;
172     private final AtomicBoolean loop = new AtomicBoolean(true);
173     private final Semaphore sema = new Semaphore(1);
174     private final Semaphore readySema = new Semaphore(0);
175 
176     @SuppressFBWarnings("SC_START_IN_CTOR")
177     protected ServerThread() throws IOException, InterruptedException {
178       super();
179       serverSocket = startServer(); // NOPMD
180       setDaemon(true);
181 
182       start();
183       readySema.acquire();
184     }

189     }

206     /**
207      * Stops the server.
208      *
209      * @throws IOException on error.
210      */
211     public void shutdown() throws IOException {
212       stopAcceptingConnections();
213       if (serverSocket != null) {
214         onServerSocketClose();
215         serverSocket.close();
216       }
217     }

219     /**
220      * Callback used to handle a connection call.
221      *
222      * After returning from this call, the socket is closed.
223      *
224      * Use {@link #stopAcceptingConnections()} to stop accepting new calls.
225      *
226      * @param sock The socket to handle.
227      * @throws IOException upon error.
228      */
229     protected abstract void handleConnection(Socket sock) throws IOException;
230 
231     /**
232      * Called from within {@link #handleConnection(Socket)} to tell the server to no longer accept
233      * new calls and to terminate the server thread.
234      *
235      * Note that this will lead to existing client connections to be closed.
236      *
237      * If you want to deny new connections but finish your work on the client side (in another
238      * thread), then please use semaphores etc. to ensure reaching a safe state before calling this
239      * method.
240      */
241     protected void stopAcceptingConnections() {
242       loop.set(false);
243     }
244 
245     protected void onServerSocketClose() {
246       stopAcceptingConnections();
247     }

268     /**
269      * Called upon receiving an exception that may be handled specifically.
270      *
271      * @param e The exception
272      * @return {@link ExceptionHandlingDecision#RAISE} if we should handle the exception somehow,
273      *         {@link ExceptionHandlingDecision#IGNORE} if we should pretend the exception never
274      *         occurred.
275      */
276     protected ExceptionHandlingDecision handleException(Exception e) {
277       return ExceptionHandlingDecision.RAISE;
278     }
279 
280     protected void acceptAndHandleConnection() throws IOException {
281       boolean acceptSuccess = false;
282       sema.release();
283       try (Socket sock = serverSocket.accept()) {
284         try {
285           sema.acquire();
286         } catch (InterruptedException e) {
287           throw (InterruptedIOException) new InterruptedIOException(e.getMessage()).initCause(e);
288         }
289         acceptSuccess = true;
290 
291         if (sock instanceof AFSocket<?>) {
292           AFSocket<?> afs = (AFSocket<?>) sock;
293           if (shouldDoConnectionCheckUponAccept() && afs.checkConnectionClosed()) {
294             acceptSuccess = false;
295           }
296         }
297 
298         if (acceptSuccess) {
299           handleConnection(sock);
300         }
301       } catch (IOException e) { // NOPMD.ExceptionAsFlowControl
302         if (!acceptSuccess) {
303           // ignore: connection closed before accept could complete
304           if (serverSocket.isClosed()) {
305             stopAcceptingConnections();
306           }
307         } else {
308           throw e;
309         }
310       } finally {
311         if (acceptSuccess) {
312           sema.release();
313         }
314       }
315     }
316 
317     @Override
318     @SuppressWarnings("PMD.AvoidInstanceofChecksInCatchClause")
319     public final void run() {
320       try {
321         loop.set(true);
322         readySema.release();
323         onServerReady();
324         while (loop.get()) {
325           acceptAndHandleConnection();
326         }
327       } catch (Exception e) {
328         if (!loop.get()) {
329           // ignore
330         } else if (handleException(e) != ExceptionHandlingDecision.IGNORE) {
331           if (e instanceof TimeoutException
332               && caller instanceof TestAbortedWithImportantMessageException) {
333             caller.addSuppressed(e);
334             exception = caller;
335           } else {
336             e.addSuppressed(caller);
337             exception = e;
338           }
339         }
340       } catch (Error e) {
341         error = e;
342       }
343     }
344 
345     /**
346      * Called right before starting the accept loop.
347      */
348     protected void onServerReady() {
349     }
350 
351     /**
352      * Checks if there were any exceptions thrown during the lifetime of this ServerThread.
353      *
354      * NOTE: This call blocks until the Thread actually terminates.
355      *
356      * @throws Exception upon error.
357      */
358     public void checkException() throws Exception {
359       boolean serverStillRunning = !sema.tryAcquire(30, TimeUnit.SECONDS);
360       if (error != null) {
361         throw error;
362       }
363       if (exception != null) {
364         throw exception;
365       }
366       if (serverStillRunning) {
367         throw new IllegalStateException("SocketTestBase server still running after 30 seconds");
368       }
369     }
370   }

493   /**
494    * Returns the Linux kernel's major and minor version as an integer array (i.e., {@code 5.10.2 ->
495    * int[]{5,10}}), or {@code null} if the running system isn't Linux or the version could not be
496    * determined.
497    *
498    * @return The running Linux kernels' major and minor version as an integer array, or
499    *         {@code null}.
500    */
501   protected int[] getLinuxMajorMinorVersion() {
502     if (!"Linux".equals(System.getProperty("os.name"))) {
503       return null; // NOPMD.PMD.ReturnEmptyCollectionRatherThanNull
504     }
505     Matcher m = PAT_LINUX_VERSION.matcher(System.getProperty("os.version", ""));
506     if (m.find()) {
507       return new int[] {Integer.parseInt(m.group(1)), Integer.parseInt(m.group(2))};
508     } else {
509       return null; // NOPMD.PMD.ReturnEmptyCollectionRatherThanNull
510     }
511   }
