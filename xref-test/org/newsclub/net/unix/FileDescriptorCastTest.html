132   @Test
133   public void testRandomAccessFile() throws Exception {
134     File tempFile = AFUNIXSocketAddress.newTempPath(false);
135     try (RandomAccessFile raf = new RandomAccessFile(tempFile, "rw");
136         FileChannel fc = FileDescriptorCast.using(raf.getFD()).as(FileChannel.class)) {
137       ByteBuffer bb = ByteBuffer.allocate(4); // big endian
138       bb.putInt(0x12345678);
139       bb.flip();
140       fc.write(bb);
141 
142       assertEquals(4, raf.length());
143       assertEquals(4, raf.getFilePointer());
144 
145       raf.seek(2);
146       assertEquals(2, fc.position());
147       bb.clear();
148       assertEquals(2, fc.read(bb));
149       assertEquals(0x5678, bb.getShort());
150     } finally {
151       IOUtil.delete(tempFile);
152     }
153   }

155   @Test
156   public void testPipe() throws Exception {
157     try (AFPipe pipe = AFPipe.open();
158         AFPipe.SourceChannel source = pipe.source();
159         AFPipe.SinkChannel sink = pipe.sink();) {
160 
161       ByteBuffer buf = ByteBuffer.allocate(32);
162       buf.order(ByteOrder.LITTLE_ENDIAN);
163       buf.putInt(0xeeff);
164       buf.flip();
165 
166       assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
167         sink.write(buf);
168 
169         OutputStream badOut = FileDescriptorCast.using(source.getFileDescriptor()).as(
170             OutputStream.class);
171         try {
172           badOut.write(0x22);
173         } catch (IOException e) {
174           // expected but not guaranteed to be thrown
175         }
176 
177         FileDescriptorCast fdc = FileDescriptorCast.using(source.getFileDescriptor());
178         InputStream in = fdc.as(InputStream.class);
179 
180         int available = in.available();
181         if (available != 0) {
182           assertEquals(4, available);
183         }
184         assertEquals(0xFF, in.read());
185         assertEquals(0xEE, in.read());
186         assertEquals(0, in.read());
187         assertEquals(0, in.read());
188 
189         OutputStream out = FileDescriptorCast.using(sink.getFileDescriptor()).as(
190             OutputStream.class);
191         out.write(0x22);
192         out.write(0x33);
193 
194         buf.clear();
195         int numRead = source.read(buf);
196         assertEquals(2, numRead);
197         buf.flip();
198         assertEquals(0x3322, buf.getShort());
199       });
200     }

203   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DOMAIN)
204   @Test
205   public void testCastGeneric() throws Exception {
206     try (AFUNIXServerSocketChannel ussc = AFUNIXServerSocketChannel.open()) {
207       ussc.bind(AFUNIXSocketAddress.ofNewTempFile());
208 
209       AFGenericServerSocketChannel gssc = FileDescriptorCast.using(ussc.getFileDescriptor()).as(
210           AFGenericServerSocketChannel.class);
211 
212       CompletableFuture<@Nullable ConnectionResult> cf = CompletableFuture.supplyAsync(() -> {
213         try {
214           AFGenericSocketChannel sc = gssc.accept();
215           ByteBuffer bb = ByteBuffer.allocate(64);
216           int r = sc.read(bb);
217           if (r != 1) {
218             throw new IllegalStateException("Unexpected result: " + r + " bytes read");
219           }
220 
221           return new ConnectionResult(bb.get(0), sc.getLocalSocketAddress(), sc
222               .getRemoteSocketAddress());
223         } catch (IOException e) {
224           throw new IllegalStateException(e);
225         }
226       });
227 
228       try (AFUNIXSocket sock = AFUNIXSocket.connectTo(ussc.getLocalAddress())) {
229         AFGenericSocket gs = FileDescriptorCast.using(sock.getFileDescriptor()).as(
230             AFGenericSocket.class);
231 
232         try {
233           gs.getOutputStream().write(123);
234         } catch (BrokenPipeSocketException e) {
235           try {
236             cf.get();
237           } catch (Exception e2) {
238             e.addSuppressed(e2);
239           }
240           throw e;
241         }
242 
243         AFGenericSocketAddress lsa = gs.getLocalSocketAddress();
244         if (lsa != null) {
245           assertEquals(AFGenericSocketAddress.class, lsa.getClass());
246         }
247         AFGenericSocketAddress rsa = gs.getRemoteSocketAddress();
248         if (rsa != null) {
249           assertEquals(AFGenericSocketAddress.class, rsa.getClass());
250         }
251 
252         ConnectionResult cr = Objects.requireNonNull(cf.get());
253         assertEquals(123, cr.firstByte);
254         if (lsa != null) {
255           compareGenericAddresses(lsa, cr.remoteSocketAddress);
256         }
257         if (rsa != null) {
258           compareGenericAddresses(rsa, cr.localSocketAddress);
259         }
260       }
261     }
262   }
263 
264   private void compareGenericAddresses(AFGenericSocketAddress rsa, SocketAddress lsa) {
265     if (rsa.toBytes().length < 2 && !rsa.equals(lsa)) {
266       // observed on AIX
267       // this is more or less equivalent to rsa == null
268     } else {
269       if (!rsa.equals(lsa) && lsa instanceof AFGenericSocketAddress) {
270         // On Windows, we may get the same address back but padding bytes
271         // may be omitted. Therefore, just compare the common prefix.
272 
273         byte[] bytesRemote = rsa.getBytes();
274         byte[] bytesLocal = ((AFGenericSocketAddress) lsa).getBytes();
275         int minLength = Math.min(bytesRemote.length, bytesLocal.length);
276 
277         boolean ok = true;
278         for (int i = 0; i < minLength; i++) {
279           if (bytesRemote[i] != bytesLocal[i]) {
280             ok = false;
281             break;
282           }
283         }
284         if (!ok) {
285           assertEquals(rsa, lsa);
286         }
287       } else {
288         assertEquals(rsa, lsa);
289       }
290     }
291   }

294   @Test
295   public void testCastGenericDuplicating() throws Exception {
296     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
297     Path p = addr.getFile().toPath();
298     try (AFUNIXServerSocketChannel ussc = AFUNIXServerSocketChannel.open()) {
299       ussc.bind(addr);
300 
301       FileDescriptorCast fdc = FileDescriptorCast.duplicating(ussc.getFileDescriptor());
302       if (fdc == null) {
303         throw new TestAbortedNotAnIssueException("FileDescriptCast.duplicating not supported");
304       }
305 
306       // also won't delete the file because we told to it not delete above, otherwise
307       // we would be able to bind but not connect (this is an AF_UNIX-specific issue)
308 
309       AFGenericServerSocketChannel gssc = fdc.as(AFGenericServerSocketChannel.class);
310 
311       CompletableFuture<@Nullable ConnectionResult> cf = CompletableFuture.supplyAsync(() -> {
312         try {
313           AFGenericSocketChannel sc = gssc.accept();
314           ByteBuffer bb = ByteBuffer.allocate(64);
315           int r = sc.read(bb);
316           if (r != 1) {
317             throw new IllegalStateException("Unexpected result: " + r + " bytes read");
318           }
319 
320           return new ConnectionResult(bb.get(0), sc.getLocalSocketAddress(), sc
321               .getRemoteSocketAddress());
322         } catch (IOException e) {
323           throw new IllegalStateException(e);
324         }
325       });
326 
327       try (AFUNIXSocket sock = AFUNIXSocket.connectTo(ussc.getLocalAddress())) {
328         AFGenericSocket gs = FileDescriptorCast.using(sock.getFileDescriptor()).as(
329             AFGenericSocket.class);
330 
331         try {
332           gs.getOutputStream().write(123);
333         } catch (BrokenPipeSocketException e) {
334           try {
335             cf.get();
336           } catch (Exception e2) {
337             e.addSuppressed(e2);
338           }
339           throw e;
340         }
341 
342         AFGenericSocketAddress lsa = gs.getLocalSocketAddress();
343         if (lsa != null) {
344           assertEquals(AFGenericSocketAddress.class, lsa.getClass());
345         }
346         AFGenericSocketAddress rsa = gs.getRemoteSocketAddress();
347         if (rsa != null) {
348           assertEquals(AFGenericSocketAddress.class, rsa.getClass());
349         }
350 
351         ConnectionResult cr = Objects.requireNonNull(cf.get());
352         assertEquals(123, cr.firstByte);
353         if (lsa != null) {
354           compareGenericAddresses(lsa, cr.remoteSocketAddress);
355         }
356         if (rsa != null) {
357           compareGenericAddresses(rsa, cr.localSocketAddress);
358         }
359       }
360     } finally {
361       Files.deleteIfExists(p);
362     }
363   }
