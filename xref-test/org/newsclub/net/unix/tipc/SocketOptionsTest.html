59    @Test
60    public void testTIPCConnTimeout() throws Exception {
61      Socket socket = newSocket();
62      @SuppressWarnings("null")
63      int timeout = socket.getOption(AFTIPCSocketOptions.TIPC_CONN_TIMEOUT);
64  
65      int newTimeout = getRandom().nextInt(8000) + 1;
66      if (newTimeout == timeout) {
67        newTimeout++;
68      }
69      socket.setOption(AFTIPCSocketOptions.TIPC_CONN_TIMEOUT, newTimeout);
70      assertEquals(newTimeout, socket.getOption(AFTIPCSocketOptions.TIPC_CONN_TIMEOUT));
71      socket.setOption(AFTIPCSocketOptions.TIPC_CONN_TIMEOUT, timeout);
72      assertEquals(timeout, socket.getOption(AFTIPCSocketOptions.TIPC_CONN_TIMEOUT));
73    }

138         return;

140 
141       socket.setOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN, AFTIPCGroupRequest.with(4711, 0,
142           Scope.SCOPE_NODE, GroupRequestFlags.NONE));
143       assertEquals(4711, socket.getOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN).getType());
144 
145       try {
146         socket.write(ByteBuffer.allocate(64));
147       } catch (NoRouteToHostException e) {
148         // expected as long as there's no other member in the group
149       }
150 
151       socket.setOption(AFTIPCSocketOptions.TIPC_GROUP_LEAVE, AFTIPCSocketOptions.VOID);
152     }

155   @Test
156   public void testGroupLoopback() throws Exception {
157     try (AFTIPCDatagramChannel socket = (AFTIPCDatagramChannel) newDatagramChannel()) {
158       try {
159         assertEquals(AFTIPCGroupRequest.NONE, socket.getOption(
160             AFTIPCSocketOptions.TIPC_GROUP_JOIN));
161       } catch (SocketException e) {
162         // on older kernels that don't support group requests
163         assumeTrue(false, "Not supported by kernel");
164         return;
165       }
166 
167       socket.setOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN, AFTIPCGroupRequest.with(4712, 0,
168           Scope.SCOPE_NODE, GroupRequestFlags.GROUP_LOOPBACK));
169       assertEquals(4712, socket.getOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN).getType());
170 
171       ByteBuffer sendBuffer = ByteBuffer.allocate(8192);
172       try {
173         socket.write(sendBuffer); // should not throw thanks to
174                                   // GroupRequestFlags.GROUP_LOOPBACK
175       } catch (NoRouteToHostException e) {
176         throw new TestAbortedWithImportantMessageException(
177             MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
178             "Kernel may be too old for full TIPC support", e);
179       }
180 
181       ByteBuffer recvBuffer = ByteBuffer.allocate(16384);
182       int len = socket.read(recvBuffer);
183       if (len > 0) {
184         // We've received the undeliverable message as loopback
185         assertEquals(sendBuffer.position(), len);
186         assertEquals(recvBuffer.position(), len);
187       }
188       sendBuffer.clear();
189       recvBuffer.clear();
190 
191       socket.send(sendBuffer, null); // should not throw thanks to
192                                      // GroupRequestFlags.GROUP_LOOPBACK
193 
194       AFTIPCSocketAddress sa = socket.receive(recvBuffer);
195       assertEquals(socket.getLocalAddress(), sa); // loopback message from self
196 
197       if (len > 0) {
198         // We've received the undeliverable message as loopback
199         assertEquals(sendBuffer.position(), len);
200         assertEquals(recvBuffer.position(), len);
201       }
202 
203       socket.setOption(AFTIPCSocketOptions.TIPC_GROUP_LEAVE, AFTIPCSocketOptions.VOID);
204     }
205   }

209     try (DatagramChannel socket1 = newDatagramChannel();
210         DatagramChannel socket2 = newDatagramChannel();) {
211 
212       // enforce reliable communication
213       socket1.setOption(AFTIPCSocketOptions.TIPC_DEST_DROPPABLE, false);
214       socket2.setOption(AFTIPCSocketOptions.TIPC_DEST_DROPPABLE, false);
215 
216       ByteBuffer sendBuffer = ByteBuffer.allocate(8192);
217       ByteBuffer recvBuffer = ByteBuffer.allocate(16384);
218       socket1.send(sendBuffer, socket2.getLocalAddress());
219 
220       assertEquals(socket1.getLocalAddress(), socket2.receive(recvBuffer));
221     }

224   @Test
225   public void testGroupCommunication() throws Exception {
226     try (DatagramChannel socket1 = newDatagramChannel();
227         DatagramChannel socket2 = newDatagramChannel();) {
228 
229       // enforce reliable communication
230       socket1.setOption(AFTIPCSocketOptions.TIPC_DEST_DROPPABLE, false);
231       socket2.setOption(AFTIPCSocketOptions.TIPC_DEST_DROPPABLE, false);
232 
233       socket1.setOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN, AFTIPCGroupRequest.with(4713, 0,
234           Scope.SCOPE_NODE, GroupRequestFlags.NONE));
235       socket2.setOption(AFTIPCSocketOptions.TIPC_GROUP_JOIN, AFTIPCGroupRequest.with(4713, 0,
236           Scope.SCOPE_NODE, GroupRequestFlags.NONE));
237 
238       ByteBuffer sendBuffer = ByteBuffer.allocate(8192);
239       ByteBuffer recvBuffer = ByteBuffer.allocate(16384);
240       int sent = socket1.write(sendBuffer);
241 
242       CompletableFuture<SocketAddress> receiveAddrFuture = new CompletableFuture<>();
243       try {
244         assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
245           receiveAddrFuture.complete(socket2.receive(recvBuffer));
246         });
247       } catch (AssertionFailedError e) {
248         throw new TestAbortedWithImportantMessageException(
249             MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
250             "Kernel may be too old for full TIPC support", e);
251       }
252       assertEquals(socket1.getLocalAddress(), receiveAddrFuture.get());
253       assertEquals(sent, sendBuffer.position());
254       assertEquals(sent, recvBuffer.position());
255     }
