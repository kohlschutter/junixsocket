18  package org.newsclub.net.unix;

20  import java.io.IOException;
21  import java.net.InetAddress;
22  import java.net.InetSocketAddress;
23  import java.net.ServerSocket;
24  import java.net.Socket;
25  import java.util.concurrent.Callable;
26  import java.util.concurrent.Executors;
27  import java.util.concurrent.Future;
28  import java.util.concurrent.TimeUnit;
29  
30  import org.junit.jupiter.api.AfterEach;
31  import org.junit.jupiter.api.BeforeEach;
32  
33  /**
34   * This test ensures that the test cases written for the junixsocket also pass for the standard Java
35   * Internet sockets. If they do not, the tests may be incorrect. As much as possible, junixsocket's
36   * AFUnixSocket should emulate the regular Java Internet sockets so they can be used with all
37   * existing tools.
38   * 
39   * See http://code.google.com/p/junixsocket/issues/detail?id=9
40   * 
41   * @author Derrick Rice (April, 2010)
42   */
43  public class EndOfFileJavaTest extends EndOfFileTest {
44    int port;
45  
46    @Override
47    @BeforeEach
48    public void setup() throws IOException {
49      final int desiredPort;
50      String explicitPort = System.getProperty("org.newsclub.net.unix.testport");
51      if (explicitPort != null) {
52        desiredPort = Integer.valueOf(explicitPort);
53      } else {
54        desiredPort = 0;
55      }
56  
57      server = new ServerSocket();
58      server.bind(new InetSocketAddress(InetAddress.getLoopbackAddress(), desiredPort), 1);
59      port = server.getLocalPort();
60      executor = Executors.newFixedThreadPool(2);
61    }
62  
63    @Override
64    @AfterEach
65    public void teardown() {
66      super.teardown();
67    }
68  
69    @SuppressWarnings("resource")
70    @Override
71    Socket[] connectToServer() throws Exception {
72      Socket clientSocket = new Socket();
73  
74      Future<Socket> serverAcceptFuture = executor.submit(new Callable<Socket>() {
75        @Override
76        public Socket call() throws Exception {
77          return server.accept();
78        }
79      });
80  
81      clientSocket.connect(new InetSocketAddress(InetAddress.getLoopbackAddress(), port));
82  
83      Socket serverSocket = serverAcceptFuture.get(100, TimeUnit.MILLISECONDS);
84  
85      return new Socket[] {serverSocket, clientSocket};
86    }
87  }
