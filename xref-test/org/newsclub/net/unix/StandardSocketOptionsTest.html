79    @Test
80    public void testUnconnectedServerSocketOptions() throws Exception {
81      try (ServerSocket sock = newServerSocketBindOn(newTempAddress(), true); //
82          TestState<ServerSocket> state = new TestStateServerSocket(sock);) {
83  
84        // supported
85        state.testSocketOption(StandardSocketOptions.SO_RCVBUF, null, 1024, null);
86        state.testSocketOption(StandardSocketOptions.SO_REUSEADDR, null, true, true);
87  
88        // unsupported; throws an exception
89        state.testSocketOption(StandardSocketOptions.IP_MULTICAST_IF, null, NetworkInterface
90            .getByIndex(0), null);
91        state.testSocketOption(StandardSocketOptions.IP_MULTICAST_LOOP, null, true, null);
92        state.testSocketOption(StandardSocketOptions.IP_MULTICAST_TTL, null, 123, null);
93        state.testSocketOption(StandardSocketOptions.SO_BROADCAST, null, true, null);
94        state.testSocketOption(StandardSocketOptions.SO_REUSEPORT, null, true, null);
95  
96        // Java 8 mostly: cannot use on server sockets that are not connected to a socket
97        state.testSocketOption(StandardSocketOptions.TCP_NODELAY, null, true, null);
98        state.testSocketOption(StandardSocketOptions.SO_SNDBUF, null, 4096, null);
99        state.testSocketOption(StandardSocketOptions.SO_LINGER, null, 123, null);
100       state.testSocketOption(StandardSocketOptions.SO_KEEPALIVE, null, null, null);
101       // Make sure we acknowledge them as "covered" nevertheless
102       state.coveredSupportedOptions.addAll(Set.of(StandardSocketOptions.TCP_NODELAY,
103           StandardSocketOptions.SO_SNDBUF, StandardSocketOptions.SO_LINGER,
104           StandardSocketOptions.SO_KEEPALIVE));
105 
106       // always 0
107       state.testSocketOption(StandardSocketOptions.IP_TOS, 0, 3, 0);
108     }
109   }

113     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
114       try (ServerThread serverThread = new ServerThread() {
115 
116         @Override
117         protected void handleConnection(Socket sock) throws IOException {
118           try (TestState<Socket> state = new TestStateSocket(sock)) {

121             state.testSocketOption(StandardSocketOptions.SO_SNDBUF, null, 8192, null);
122             state.testSocketOption(StandardSocketOptions.SO_LINGER, null, 123, null);
123             state.testSocketOption(StandardSocketOptions.SO_KEEPALIVE, null, true, null);
124             try (OutputStream out = sock.getOutputStream()) {
125               out.write(0xAF);
126             }
127           }
128         }
129       }; //
130           Socket sock = connectTo(serverThread.getServerAddress());
131           InputStream in = sock.getInputStream();//
132       ) {
133         assertEquals(0xAF, in.read());
134       }
135     });

138   abstract class TestState<S extends Closeable> implements Closeable {
139     protected final S sock;
140     private final Set<SocketOption<?>> coveredSupportedOptions = new HashSet<>();
141 
142     protected TestState(S sock) {
143       this.sock = sock;
144     }
145 
146     @SuppressWarnings("null")
147     @Override
148     public void close() throws AssertionError, IOException {
149       Set<SocketOption<?>> sockSupportedOptions = supportedOptions();
150 
151       // Ignore these -- they're a standard option we have to silently "support" (by doing nothing)
152       coveredSupportedOptions.removeAll(IGNORABLE_OPTIONS);
153 
154       for (SocketOption<?> opt : sockSupportedOptions) {
155         if (!coveredSupportedOptions.contains(opt)) {
156           assumeTrue(false, "Test did not cover supported option: " + opt);
157         }
158       }
159       for (SocketOption<?> opt : coveredSupportedOptions) {
160         if (!sockSupportedOptions.contains(opt)) {
161           fail(sock.getClass() + "#supportedOptions does not cover supported option: " + opt);
162         }
163       }
164     }
165 
166     private <T> void testSocketOption(SocketOption<T> option, T oldValExpected, T setVal,
167         T newValExpected) {
168       boolean unsupportedIsNotAFailure = (newValExpected == null);
169       try {
170 
171         T prevVal = checkUnsupported(option, unsupportedIsNotAFailure, () -> {
172           return getOption(option);
173         });
174         if (oldValExpected != null) {
175           assertEquals(oldValExpected, prevVal, () -> "Unexpected old value for " + option);
176         }
177         try {
178           assertEquals(sock, checkUnsupported(option, unsupportedIsNotAFailure, () -> {
179             return setOption(option, setVal);
180           }));
181 
182           T newVal = checkUnsupported(option, unsupportedIsNotAFailure, () -> {
183             return getOption(option);
184           });
185           if (newValExpected != null) {
186             assertEquals(newValExpected, newVal, () -> "Unexpected new value for " + option);
187           }
188         } finally {
189           // make sure to set old val again after the test
190           checkUnsupported(option, unsupportedIsNotAFailure, () -> {
191             return setOption(option, prevVal);
192           });
193         }
194       } catch (UnsupportedOperationException e) {
195         // already handled
196         return;
197       }
198 
199       coveredSupportedOptions.add(option);
200     }
201 
202     protected abstract <T> T getOption(SocketOption<T> option) throws IOException;
203 
204     protected abstract <T> Closeable setOption(SocketOption<T> option, T value) throws IOException; // NOPMD
205 
206     protected abstract Set<SocketOption<?>> supportedOptions() throws IOException;
207 
208     protected <R> R checkUnsupported(SocketOption<?> option, boolean unsupportedIsNotAFailure,
209         Callable<R> r) {
210       try {
211         return r.call();
212       } catch (UnsupportedOperationException e) {
213         // ignore
214         if (!unsupportedIsNotAFailure) {
215           softAssertions.fail("Unsupported socket option: " + option, e);
216         }
217         throw e;
218       } catch (Exception e) {
219         if (unsupportedIsNotAFailure && e.getMessage().toLowerCase(Locale.ENGLISH).contains(
220             "support")) {
221           // unsupported, not supported, supports only, etc.
222         } else {
223           softAssertions.fail("Unsupported socket option: " + option, e);
224         }
225         throw new UnsupportedOperationException(e);
226       }
227     }
228   }
