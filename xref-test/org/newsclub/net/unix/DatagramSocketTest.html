43  @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
44  @SuppressFBWarnings({
45      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
46  public abstract class DatagramSocketTest<A extends SocketAddress> extends SocketTestBase<A> {
47    protected DatagramSocketTest(AddressSpecifics<A> asp) {
48      super(asp);
49    }
50  
51    private void assertUnconnectedDatagramSocket(DatagramSocket ds) {
52      assertFalse(ds.isClosed());
53      assertFalse(ds.isConnected());
54      assertEquals(0, ds.getLocalPort());
55      assertUnconnectedUnbound(ds);
56    }
57  
58    protected void assertUnconnectedUnbound(DatagramSocket ds) {
59      assertFalse(ds.isBound());
60      assertTrue(ds.getLocalAddress().isAnyLocalAddress());
61      assertNull(ds.getLocalSocketAddress());
62    }
63  
64    private static void assertClosedDatagramSocket(DatagramSocket ds) {
65      assertTrue(ds.isClosed());
66      assertNull(ds.getLocalAddress());
67      assertNull(ds.getLocalSocketAddress());
68      assertEquals(-1, ds.getLocalPort());
69    }
70  
71    private void assertBoundDatagramSocket(DatagramSocket ds, SocketAddress boundAddr)
72        throws SocketException {
73      assertTrue(ds.isBound());
74      assertFalse(ds.isClosed());
75      assertEquals(0, ds.getLocalPort());
76      assertFalse(ds.getLocalAddress().isAnyLocalAddress());
77      assertBoundAddrIdenticalToLocalAddress(ds, boundAddr);
78    }
79  
80    protected void assertBoundAddrIdenticalToLocalAddress(DatagramSocket ds, SocketAddress boundAddr)
81        throws SocketException {
82      assertEquals(boundAddr, unwrap(ds.getLocalAddress(), 0));
83      assertEquals(boundAddr, ds.getLocalSocketAddress());
84    }
85  
86    private void assertConnectedDatagramSocket(DatagramSocket ds, SocketAddress boundAddr,
87        SocketAddress remoteAddr) throws SocketException {
88      assertTrue(ds.isConnected());
89      assertFalse(ds.isClosed());
90      assertRemoteAddress(ds, remoteAddr);
91      if (boundAddr != null) {
92        assertBoundDatagramSocket(ds, boundAddr);
93      } else {
94        assertEquals(0, ds.getLocalPort());
95      }
96    }
97  
98    protected void assertRemoteAddress(DatagramSocket ds, SocketAddress remoteAddr)
99        throws SocketException {
100     assertEquals(remoteAddr, ds.getRemoteSocketAddress());
101   }
102 
103   private static void assertDatagramPacketAddress(DatagramPacket dp, SocketAddress addr0) {
104     AFSocketAddress addr = (AFSocketAddress) addr0;
105     assertEquals(addr.wrapAddress(), dp.getAddress());
106     assertEquals(new InetSocketAddress(addr.wrapAddress(), addr.getPort()), dp.getSocketAddress());
107     assertEquals(0, dp.getPort());
108   }

110   @Test
111   public void testBindConnect() throws SocketException, IOException, InterruptedException {
112     AFSocketAddress ds1Addr = (AFSocketAddress) newTempAddress();
113     AFSocketAddress ds2Addr = (AFSocketAddress) newTempAddress();
114     assertNotEquals(ds1Addr, ds2Addr);
115 
116     try (DatagramSocket ds1 = newDatagramSocket(); DatagramSocket ds2 = newDatagramSocket()) {
117       assertUnconnectedDatagramSocket(ds1);
118 
119       ds1.bind(ds1Addr);
120       assertNull(ds1.getRemoteSocketAddress());
121 
122       assertBoundDatagramSocket(ds1, ds1Addr);
123 
124       if (!ds2.isBound()) {
125         ds2.bind(ds2Addr);
126       }
127       ds1.connect(ds2Addr);
128 
129       assertConnectedDatagramSocket(ds1, ds1Addr, ds2Addr);
130 
131       byte[] data = "Hello world!".getBytes(StandardCharsets.UTF_8);
132       byte[] buf = new byte[512];
133       System.arraycopy(data, 0, buf, 0, data.length);
134       DatagramPacket dp1 = new DatagramPacket(buf, data.length);
135       ds1.send(dp1);
136 
137       DatagramPacket dp2 = new DatagramPacket(new byte[1024], 3, 1021);
138       dp2.setPort(123); // should be cleared
139       ds2.receive(dp2);
140 
141       byte[] received = new byte[dp2.getLength()];
142       System.arraycopy(dp2.getData(), dp2.getOffset(), received, 0, dp2.getLength());
143       assertArrayEquals(data, received);
144 
145       assertDatagramPacketAddress(dp2, ds1Addr);
146 
147       // dp2.setSocketAddress(ds1Addr); // doesn't work :-(
148       dp2.setAddress(ds1Addr.wrapAddress());
149       ds2.send(dp2);
150 
151       dp1.setLength(100); // maximum package length
152       ds1.receive(dp1);
153       assertEquals(12, dp1.getLength());
154       assertEquals(ds2Addr.wrapAddress(), dp1.getAddress());
155 
156       ds1.close();
157       assertClosedDatagramSocket(ds1);
158 
159       ds2.close();
160       assertClosedDatagramSocket(ds2);
161     }
162   }

187   @Test
188   public void testReadTimeout() throws IOException {
189     SocketAddress dsAddr = newTempAddress();
190 
191     assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
192       try (DatagramSocket ds = newDatagramSocket()) {
193         ds.setSoTimeout(50);
194         ds.bind(dsAddr);
195         assertThrows(SocketTimeoutException.class, () -> {
196           ds.receive(AFDatagramUtil.datagramWithCapacity(64));
197         });
198       }
199     });
200   }

202   @Test
203   public void testPeekTimeout() throws IOException {
204     SocketAddress dsAddr = newTempAddress();
205 
206     assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
