48  @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
49  @SuppressFBWarnings({
50      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})

52    protected DatagramSocketTest(AddressSpecifics<A> asp) {
53      super(asp);
54    }

56    private void assertUnconnectedDatagramSocket(DatagramSocket ds) {
57      assertFalse(ds.isClosed());
58      assertFalse(ds.isConnected());
59      assertEquals(0, ds.getLocalPort());
60      assertUnconnectedUnbound(ds);
61    }
62  
63    protected void assertUnconnectedUnbound(DatagramSocket ds) {
64      assertFalse(ds.isBound());
65      assertTrue(ds.getLocalAddress().isAnyLocalAddress());
66      assertNull(ds.getLocalSocketAddress());
67    }
68  
69    private static void assertClosedDatagramSocket(DatagramSocket ds) {
70      assertTrue(ds.isClosed());
71      assertNull(ds.getLocalAddress());
72      assertNull(ds.getLocalSocketAddress());
73      assertEquals(-1, ds.getLocalPort());
74    }
75  
76    private void assertBoundDatagramSocket(DatagramSocket ds, SocketAddress boundAddr)
77        throws SocketException {
78      assertTrue(ds.isBound());
79      assertFalse(ds.isClosed());
80      assertEquals(0, ds.getLocalPort());
81      assertFalse(ds.getLocalAddress().isAnyLocalAddress());
82      assertBoundAddrIdenticalToLocalAddress(ds, boundAddr);
83    }
84  
85    protected void assertBoundAddrIdenticalToLocalAddress(DatagramSocket ds, SocketAddress boundAddr)
86        throws SocketException {
87      assertEquals(boundAddr, unwrap(ds.getLocalAddress(), 0));
88      assertEquals(boundAddr, ds.getLocalSocketAddress());
89    }
90  
91    private void assertConnectedDatagramSocket(DatagramSocket ds, SocketAddress boundAddr,
92        SocketAddress remoteAddr) throws SocketException {
93      assertTrue(ds.isConnected());
94      assertFalse(ds.isClosed());
95      assertRemoteAddress(ds, remoteAddr);
96      if (boundAddr != null) {
97        assertBoundDatagramSocket(ds, boundAddr);
98      } else {
99        assertEquals(0, ds.getLocalPort());
100     }
101   }
102 
103   protected void assertRemoteAddress(DatagramSocket ds, SocketAddress remoteAddr)
104       throws SocketException {
105     assertEquals(remoteAddr, ds.getRemoteSocketAddress());
106   }
107 
108   private static void assertDatagramPacketAddress(DatagramPacket dp, SocketAddress addr0) {
109     AFSocketAddress addr = (AFSocketAddress) addr0;
110     assertEquals(addr.wrapAddress(), dp.getAddress());
111     assertEquals(new InetSocketAddress(addr.wrapAddress(), addr.getPort()), dp.getSocketAddress());
112     assertEquals(0, dp.getPort());
113   }

115   @Test
116   public void testBindConnect() throws Exception {
117     AFSocketAddress ds1Addr = (AFSocketAddress) newTempAddressForDatagram();
118     AFSocketAddress ds2Addr = (AFSocketAddress) newTempAddressForDatagram();
119 
120     try (DatagramSocket ds1 = newDatagramSocket(); DatagramSocket ds2 = newDatagramSocket()) {
121       assertUnconnectedDatagramSocket(ds1);
122 
123       ds1.bind(ds1Addr);
124       ds1Addr = (AFSocketAddress) ds1.getLocalSocketAddress();
125       assertNull(ds1.getRemoteSocketAddress());
126 
127       assertBoundDatagramSocket(ds1, ds1Addr);
128 
129       if (!ds2.isBound()) {
130         ds2.bind(ds2Addr);
131         ds2Addr = (AFSocketAddress) ds2.getLocalSocketAddress();
132       }
133 
134       assertNotEquals(ds1Addr, ds2Addr);
135 
136       ds1.connect(ds2Addr);
137 
138       try {
139         assertConnectedDatagramSocket(ds1, ds1Addr, ds2Addr);
140       } catch (AssertionFailedError e) {
141         if (TestUtil.isHaikuOS()) {
142           throw TestUtil.haikuBug18534(e);
143         } else {
144           throw e;
145         }
146       }
147 
148       byte[] data = "Hello world!".getBytes(StandardCharsets.UTF_8);
149       byte[] buf = new byte[512];
150       System.arraycopy(data, 0, buf, 0, data.length);
151       DatagramPacket dp1 = new DatagramPacket(buf, data.length);
152       ds1.send(dp1);
153 
154       DatagramPacket dp2 = new DatagramPacket(new byte[1024], 3, 1021);
155       dp2.setPort(123); // should be cleared
156       ds2.receive(dp2);
157 
158       byte[] received = new byte[dp2.getLength()];
159       System.arraycopy(dp2.getData(), dp2.getOffset(), received, 0, dp2.getLength());
160       assertArrayEquals(data, received);
161 
162       assertDatagramPacketAddress(dp2, ds1Addr);
163 
164       // dp2.setSocketAddress(ds1Addr); // doesn't work :-(
165       dp2.setAddress(ds1Addr.wrapAddress());
166       ds2.send(dp2);
167 
168       dp1.setLength(100); // maximum package length
169       ds1.receive(dp1);
170       assertEquals(12, dp1.getLength());
171       assertEquals(ds2Addr.wrapAddress(), dp1.getAddress());
172 
173       ByteBuffer bb1 = ByteBuffer.allocate(512);
174       ByteBuffer bb2 = ByteBuffer.allocateDirect(512);
175 
176       bb1.putLong(0xF00BAA);
177       bb1.flip();
178 
179       ds1.getChannel().send(bb1, null); // can specify null here but no address, since it's already
180                                         // connected
181 
182       SocketAddress receivedFrom = ds2.getChannel().receive(bb2);
183       assertEquals(ds1Addr, receivedFrom);
184       bb2.flip();
185       assertEquals(0xF00BAA, bb2.getLong());
186 
187       ds1.close();
188       assertClosedDatagramSocket(ds1);
189 
190       ds2.close();
191       assertClosedDatagramSocket(ds2);
192     }
193   }

195   @Test
196   public void testChannelSendTo() throws Exception {
197     AFSocketAddress ds1Addr = (AFSocketAddress) newTempAddressForDatagram();
198     AFSocketAddress ds2Addr = (AFSocketAddress) newTempAddressForDatagram();
199     try (DatagramChannel dc1 = newDatagramChannel(); //
200         DatagramChannel dc2 = newDatagramChannel()) {
201 
202       // dc1 is neither connected nor bound
203       dc1.bind(ds1Addr);
204       dc2.bind(ds2Addr);
205 
206       ByteBuffer bb1 = ByteBuffer.allocate(512);
207       ByteBuffer bb2 = ByteBuffer.allocateDirect(512);
208 
209       bb1.putLong(0xF00BAA);
210       bb1.flip();
211 
212       dc1.send(bb1, ds2Addr);
213 
214       SocketAddress receivedFrom = dc2.receive(bb2);
215       bb2.flip();
216       assertEquals(0xF00BAA, bb2.getLong());
217 
218       assertExpectedSocketAddressFromDatagramChannelReceive(ds1Addr, receivedFrom);
219     }
220   }
221 
222   @SuppressWarnings("PMD.PreserveStackTrace")
223   protected void assertExpectedSocketAddressFromDatagramChannelReceive(SocketAddress expected,
224       SocketAddress received) {
225     try {
226       assertEquals(expected, received);
227     } catch (AssertionFailedError e) {
228       throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
229           "DatagramChannel.receive did not return expected the sender address; "
230               + "this may be a limitation of your system environment.");
231     }
232   }

236     SocketAddress dsAddr = newTempAddressForDatagram();
237 
238     assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
239       try (DatagramSocket ds = newDatagramSocket()) {
