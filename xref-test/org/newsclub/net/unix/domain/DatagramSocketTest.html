18  package org.newsclub.net.unix.domain;
19  
20  import static org.junit.jupiter.api.Assertions.assertEquals;
21  
22  import java.nio.ByteBuffer;
23  import java.nio.charset.StandardCharsets;
24  
25  import org.junit.jupiter.api.Test;
26  import org.newsclub.net.unix.AFSocketCapability;
27  import org.newsclub.net.unix.AFSocketCapabilityRequirement;
28  import org.newsclub.net.unix.AFSocketType;
29  import org.newsclub.net.unix.AFUNIXDatagramChannel;
30  import org.newsclub.net.unix.AFUNIXSocketAddress;
31  import org.newsclub.net.unix.AFUNIXSocketPair;
32  import org.newsclub.net.unix.OperationNotSupportedSocketException;
33  
34  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
35  import com.kohlschutter.testutil.TestAbortedNotAnIssueException;
36  
37  @AFSocketCapabilityRequirement({
38      AFSocketCapability.CAPABILITY_UNIX_DOMAIN, AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS})
39  @SuppressFBWarnings("NM_SAME_SIMPLE_NAME_AS_SUPERCLASS")
40  public final class DatagramSocketTest extends
41      org.newsclub.net.unix.DatagramSocketTest<AFUNIXSocketAddress> {
42  
43    public DatagramSocketTest() {
44      super(AFUNIXAddressSpecifics.INSTANCE);
45    }
46  
47    @Test
48    public void testSeqPacket() throws Exception {
49      AFUNIXSocketPair<AFUNIXDatagramChannel> pair;
50      try {
51        pair = AFUNIXSocketPair.openDatagram(AFSocketType.SOCK_SEQPACKET);
52      } catch (OperationNotSupportedSocketException e) {
53        throw new TestAbortedNotAnIssueException("SEQPACKET not supported", e);
54      }
55  
56      String msg = "Hello World";
57  
58      ByteBuffer buf = ByteBuffer.wrap(msg.getBytes(StandardCharsets.UTF_8));
59      ByteBuffer dst = ByteBuffer.allocate(64);
60      pair.getSocket1().send(buf, null);
61      int r = pair.getSocket2().read(dst);
62      assertEquals(buf.limit(), r);
63      dst.flip();
64      assertEquals(buf.limit(), dst.limit());
65  
66      assertEquals(msg, StandardCharsets.UTF_8.decode(dst).toString());
67    }
68  }
