35  @SuppressFBWarnings({
36      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
37  public abstract class SocketPairTest<A extends SocketAddress> extends SocketTestBase<A> {
38    // CPD-OFF
39  
40    protected SocketPairTest(AddressSpecifics<A> asp) {
41      super(asp);
42    }
43  
44    @Test
45    public void testSocketPair() throws Exception {
46      CloseablePair<? extends SocketChannel> pair = newSocketPair();
47  
48      SocketChannel sc1 = pair.getFirst();
49      SocketChannel sc2 = pair.getSecond();
50  
51      Objects.requireNonNull(sc1);
52      Objects.requireNonNull(sc2);
53  
54      assertTrue(sc1.isConnected());
55      assertTrue(sc2.isConnected());
56  
57      assertNotEquals(pair.getFirst(), pair.getSecond());
58      assertNotEquals(pair.getFirst().socket(), pair.getSecond().socket());
59  
60      if (sc1 instanceof AFUNIXSocketChannel) {
61        assertEquals(((AFUNIXSocketChannel) sc1).getPeerCredentials(), ((AFUNIXSocketChannel) sc2)
62            .getPeerCredentials());
63      }
64  
65      ByteBuffer bb = ByteBuffer.allocate(4096);
66      bb.putInt(0x04030201);
67      bb.flip();
68      sc1.write(bb);
69  
70      ByteBuffer bb2 = ByteBuffer.allocate(4096);
71      sc2.read(bb2);
72      bb2.flip();
73      assertEquals(0x04030201, bb2.getInt());
74  
75      if (getServerBindAddress() instanceof AFUNIXSocketAddress) {
76        assertNull(pair.getFirst().getLocalAddress());
77        assertNull(pair.getSecond().getLocalAddress());
78        assertNull(pair.getFirst().getRemoteAddress());
79        assertNull(pair.getSecond().getRemoteAddress());
80      } else {
81        assertEquals(pair.getFirst().getLocalAddress(), pair.getSecond().getRemoteAddress());
82        assertEquals(pair.getSecond().getLocalAddress(), pair.getFirst().getRemoteAddress());
83      }
84    }
85
