35  /**
36   * A base implementation for a simple, multi-threaded socket server.
37   * 
38   * This class supports both unix and "regular" sockets.
39   * 
40   * @author Christian Kohlschütter
41   */
42  public abstract class AFUNIXSocketServer {
43    private final SocketAddress listenAddress;
44  
45    private int maxConcurrentConnections = Runtime.getRuntime().availableProcessors();
46    private int serverTimeout = 0; // by default, the server doesn't timeout.
47    private int socketTimeout = (int) TimeUnit.SECONDS.toMillis(60);
48    private int serverBusyTimeout = (int) TimeUnit.SECONDS.toMillis(1);
49  
50    private Thread listenThread = null;
51    private ServerSocket serverSocket;
52    private boolean stopRequested = false;
53  
54    private final Object connectionsMonitor = new Object();
55    private ForkJoinPool connectionPool;
56  
57    public AFUNIXSocketServer(SocketAddress listenAddress) {
58      if (listenAddress == null) {
59        throw new NullPointerException("listenAddress");
60      }
61      this.listenAddress = listenAddress;
62    }

73    }

80      synchronized (this) {
81        if (serverSocket != null) {
82          throw new IllegalStateException("Already configured");

85      }
86    }

104   /**
105    * Checks if the server is running.
106    * 
107    * @return {@code true} if the server is alive.
108    */
109   public boolean isRunning() {
110     synchronized (this) {
111       return (listenThread != null && listenThread.isAlive());
112     }
113   }
114 
115   /**
116    * Starts the server.
117    */
118   public void start() {
119     synchronized (this) {
120       if (isRunning()) {
121         return;
122       }
123       if (connectionPool == null) {
124         connectionPool = new ForkJoinPool(maxConcurrentConnections,
125             ForkJoinPool.defaultForkJoinWorkerThreadFactory, null, true);
126       }
127 
128       Thread t = new Thread(AFUNIXSocketServer.this.toString() + " listening thread") {
129         @Override
130         public void run() {
131           try {
132             listen();
133           } catch (Exception e) {
134             onListenException(e);
135           }
136         }
137       };
138       t.start();
139 
140       listenThread = t;
141     }
142   }
143 
144   protected ServerSocket newServerSocket() throws IOException {
145     if (listenAddress instanceof AFUNIXSocketAddress) {
146       return AFUNIXServerSocket.newInstance();
147     } else {
148       return new ServerSocket();
149     }
150   }
151 
152   @SuppressWarnings("resource")
153   private void listen() throws IOException {
154     ServerSocket server;
155     synchronized (this) {
156       if (serverSocket != null) {
157         throw new IllegalStateException("The server is already listening");
158       }
159       serverSocket = newServerSocket();
160       server = serverSocket;
161     }
162     onServerStarting();
163 
164     try {
165       server.bind(listenAddress);
166       onServerBound(listenAddress);
167       server.setSoTimeout(serverTimeout);
168 
169       long busyStartTime = 0;
170       acceptLoop : while (!stopRequested && !Thread.interrupted()) {
171         try {
172           while (!stopRequested && connectionPool
173               .getActiveThreadCount() >= maxConcurrentConnections) {
174             if (busyStartTime == 0) {
175               busyStartTime = System.currentTimeMillis();
176             }
177             onServerBusy(busyStartTime);
178 
179             synchronized (connectionsMonitor) {
180               try {
181                 connectionsMonitor.wait(serverBusyTimeout);
182               } catch (InterruptedException e) {
183                 throw new InterruptedIOException("Interrupted while waiting on server resources");
184               }
185             }
186           }
187           busyStartTime = 0;
188 
189           if (stopRequested || server == null) {
190             break;
191           }
192 
193           onServerReady(connectionPool.getActiveThreadCount());
194 
195           Socket socket;
196           try {
197             socket = server.accept();
198           } catch (SocketException e) {
199             if (server.isClosed()) {
200               // already closed, ignore
201               break acceptLoop;
202             } else {
203               onSocketExceptionDuringAccept(e);
204               throw e;
205             }
206           }
207           try {
208             socket.setSoTimeout(socketTimeout);
209           } catch (SocketException e) {
210             // Connection closed before we could do anything
211             onSocketExceptionAfterAccept(socket, e);
212             socket.close();
213 
214             continue acceptLoop;
215           }
216 
217           onSubmitted(socket, submit(socket, server, connectionPool));
218         } catch (SocketTimeoutException e) {
219           if (!connectionPool.isQuiescent()) {
220             continue acceptLoop;
221           } else {
222             onServerShuttingDown();
223             connectionPool.shutdown();
224             break acceptLoop;
225           }
226         }
227       }
228     } finally {
229       stop();
230       onServerStopped(server);
231     }
232   }
233 
234   /**
235    * Stops the server.
236    * 
237    * @throws IOException If there was an error.
238    */
239   public void stop() throws IOException {
240     synchronized (this) {
241       stopRequested = true;
242 
243       ServerSocket theServerSocket = serverSocket;
244       serverSocket = null;
245       if (theServerSocket == null) {
246         return;
247       }
248       theServerSocket.close();
249     }
250   }
251 
252   private Future<?> submit(Socket socket, ServerSocket server, ExecutorService executor) {
253     return executor.submit(new Runnable() {
254       @Override
255       public void run() {
256         onBeforeServingSocket(socket);
257 
258         try {
259           doServeSocket(socket);
260         } catch (Exception e) {
261           onServingException(socket, e);
262         } finally {
263           // Notify the server's accept thread that we handled the connection
264           synchronized (connectionsMonitor) {
265             connectionsMonitor.notifyAll();
266           }
267           try {
268             socket.close();
269           } catch (IOException e) {
270             // ignore
271           }
272           onAfterServingSocket(socket);
273         }
274       }
275     });
276   }
