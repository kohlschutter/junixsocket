18  package org.newsclub.net.unix.rmi;
19  
20  import java.io.Closeable;
21  import java.io.IOException;
22  import java.lang.ref.WeakReference;
23  import java.rmi.NoSuchObjectException;
24  import java.rmi.RemoteException;
25  import java.rmi.ServerException;
26  import java.util.ArrayList;
27  import java.util.BitSet;
28  import java.util.Iterator;
29  import java.util.List;
30  import java.util.Objects;
31  import java.util.Random;
32  import java.util.concurrent.ExecutorService;
33  import java.util.concurrent.Executors;
34  import java.util.stream.IntStream;
35  
36  /**
37   * A very simple implementation of a {@link AFUNIXRMIService}.
38   * 
39   * @author Christian Kohlschütter
40   */
41  final class AFUNIXRMIServiceImpl implements AFUNIXRMIService {
42    private final BitSet ports = new BitSet(1000);
43    private final WeakReference<AFUNIXNaming> naming;
44    private final List<WeakReference<Closeable>> closeAtShutdown = new ArrayList<>();
45  
46    public AFUNIXRMIServiceImpl(AFUNIXNaming naming) {
47      this.naming = new WeakReference<>(naming);
48    }
49  
50    private/* synchronized */int randomPort() {
51      final Random random = new Random();
52      int maxRandom = ports.size();
53  
54      int port;
55      for (int i = 0; i < 3; i++) {
56        port = ports.nextClearBit(random.nextInt(maxRandom));
57        if (port < maxRandom) {
58          return port;
59        } else {
60          maxRandom = port;
61          if (maxRandom == 0) {
62            break;
63          }
64        }
65      }
66      return ports.nextClearBit(0);
67    }
68  
69    @Override
70    public synchronized int newPort() throws IOException {
71      int port = randomPort();
72      ports.set(port);
73      port += AFUNIXRMIPorts.ANONYMOUS_PORT_BASE;
74      return port;
75    }
76  
77    @Override
78    public synchronized void returnPort(int port) throws IOException {
79      ports.clear(port - AFUNIXRMIPorts.ANONYMOUS_PORT_BASE);
80    }
81  
82    @Override
83    public IntStream openPorts() throws RemoteException {
84      return ports.stream().map((int v) -> {
85        return v + AFUNIXRMIPorts.ANONYMOUS_PORT_BASE;
86      });
87    }
88  
89    @Override
90    public void shutdown() throws RemoteException {
91      AFUNIXNaming namingInstance = naming.get();
92      if (namingInstance != null) {
93        if (!namingInstance.isRemoteShutdownAllowed()) {
94          throw new ServerException("Remote shutdown is disabled");
95        }
96        namingInstance.shutdownRegistry();
97      }
98    }
99  
100   @Override
101   public boolean isShutdownAllowed() throws RemoteException {
102     AFUNIXNaming namingInstance = naming.get();
103     if (namingInstance != null) {
104       return namingInstance.isRemoteShutdownAllowed();
105     } else {
106       return true;
107     }
108   }
109 
110   @Override
111   public void registerForShutdown(Closeable closeable) throws RemoteException {
112     synchronized (closeAtShutdown) {
113       unregisterForShutdown(closeable);
114       closeAtShutdown.add(new WeakReference<Closeable>(closeable));
115     }
116   }
117 
118   @SuppressWarnings("resource")
119   @Override
120   public void unregisterForShutdown(Closeable closeable) throws RemoteException {
121     synchronized (closeAtShutdown) {
122       Objects.requireNonNull(closeable);
123       for (Iterator<WeakReference<Closeable>> it = closeAtShutdown.iterator(); it.hasNext();) {
124         if (it.next().get() == closeable) {
125           it.remove();
126           return;
127         }
128       }
129     }
130   }
131 
132   void shutdownRegisteredCloseables() {
133     List<WeakReference<Closeable>> list;
134     synchronized (closeAtShutdown) {
135       list = new ArrayList<>(closeAtShutdown);
136       closeAtShutdown.clear();
137     }
138 
139     ExecutorService executor = Executors.newCachedThreadPool();
140     for (WeakReference<Closeable> ref : list) {
141       executor.submit(new Runnable() {
142         @Override
143         public void run() {
144           @SuppressWarnings("resource")
145           Closeable cl = ref.get();
146           if (cl == null) {
147             return;
148           }
149           try {
150             cl.close();
151           } catch (NoSuchObjectException e) {
152             // ignore
153           } catch (IOException e) {
154             e.printStackTrace();
155           }
156         }
157       });
158     }
159     executor.shutdown();
160   }
161 }
