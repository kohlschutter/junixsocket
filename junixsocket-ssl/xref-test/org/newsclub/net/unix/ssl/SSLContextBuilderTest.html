408   private void runServerAndClient(AFUNIXSocketAddress addr, SSLSocketFactory serverSocketFactory,
409       SSLSocketFactory clientSocketFactory, CompletableFuture<Exception> serverException,
410       CompletableFuture<Exception> clientException, boolean expectClientTrouble) throws Exception {
411 
412     TestingAFSocketServer<AFUNIXSocketAddress> server =
413         new TestingAFSocketServer<AFUNIXSocketAddress>(addr) {
414           @Override
415           protected void doServeSocket(AFSocket<? extends AFUNIXSocketAddress> plainSocket)
416               throws IOException {
417             Exception caught = null;
418             try (SSLSocket sslSocket = (SSLSocket) serverSocketFactory.createSocket(plainSocket,
419                 "localhost.junixsocket", plainSocket.getPort(), false)) {
420 
421               try (InputStream in = sslSocket.getInputStream();
422                   OutputStream out = sslSocket.getOutputStream();) {
423 
424                 int v = in.read();
425                 assertEquals('!', v);
426 
427                 out.write("Hello World".getBytes(StandardCharsets.UTF_8));
428                 out.flush();
429               }
430             } catch (Exception e) {
431               caught = e;
432             } finally {
433               serverException.complete(caught);
434             }
435           }
436         };
437 
438     try {
439       server.startAndWaitToBecomeReady();
440 
441       Exception caught = null;
442       try (AFUNIXSocket plainSocket = AFUNIXSocket.connectTo(addr);
443           SSLSocket sslSocket = (SSLSocket) clientSocketFactory.createSocket(plainSocket,
444               "localhost.junixsocket", plainSocket.getPort(), false)) {
445 
446         try (InputStream in = sslSocket.getInputStream();
447             OutputStream out = sslSocket.getOutputStream()) {
448           out.write('!');
449           out.flush();
450 
451           byte[] by = new byte[11];
452           int offset = 0;
453           int r;
454           while (offset < by.length && (r = in.read(by, offset, by.length - offset)) >= 0) {
455             offset += r;
456           }
457           assertEquals("Hello World", new String(by, 0, offset, StandardCharsets.UTF_8));
458         } catch (SocketException e) {
459           if (expectClientTrouble) {
460             // ignore
461           } else {
462             throw e;
463           }
464         }
465       } catch (Exception e) {
466         caught = e;
467       } finally {
468         clientException.complete(caught);
469       }
470     } finally {
471       server.stop();
472 
473       TestUtil.throwMoreInterestingThrowableThanSocketException(() -> serverException.getNow(null),
474           () -> clientException.getNow(null));
475 
476       server.checkThrowable();
477     }
478   }
