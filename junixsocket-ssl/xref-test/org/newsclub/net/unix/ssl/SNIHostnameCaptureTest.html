141   private void testSNISuccess(TestSSLConfiguration configuration, boolean expectDefault,
142       boolean serverNullDefaultHostname, boolean clientEmptyDefaultHostname) throws Exception {
143     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
144 
145     try {
146       SSLSocketFactory serverSocketFactory = configuration.configure(SSLContextBuilder.forServer()) //
147           .withKeyStore(TestResourceUtil.getRequiredResource(SSLContextBuilderTest.class,
148               "juxserver.p12"), () -> "serverpass".toCharArray()) //
149           .buildAndDestroyBuilder().getSocketFactory();
150 
151       assertTimeoutPreemptively(Duration.ofSeconds(5), () -> {
152         runServerAndClient(configuration, addr, serverSocketFactory, expectDefault,
153             serverNullDefaultHostname, clientEmptyDefaultHostname);
154       });
155     } catch (KnownJavaBugIOException e) {
156       throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
157           e.getMessage(), e);
158     } catch (Exception | Error e) {
159       throw configuration.handleException(e);
160     } finally {
161       Files.deleteIfExists(addr.getFile().toPath());
162     }
163   }

165   private static boolean isBouncyCastleJSSE(SSLSocketFactory socketFactory) {
166     SSLContext context;
167     if (socketFactory instanceof BuilderSSLSocketFactory) {
168       context = ((BuilderSSLSocketFactory) socketFactory).getContext();
169     } else {
170       context = null;
171     }
172     Provider p;
173     if (context != null && (p = context.getProvider()) != null && "BCJSSE".equals(p.getName())) {
174       return true;
175     } else {
176       return false;
177     }
178   }

180   @SuppressWarnings({"PMD.CognitiveComplexity", "PMD.NcssCount", "PMD.NPathComplexity"})
181   private void runServerAndClient(TestSSLConfiguration configuration, AFUNIXSocketAddress addr,
182       SSLSocketFactory serverSocketFactory, boolean expectDefault,
183       boolean serverNullDefaultHostname, boolean clientEmptyDefaultHostname) throws Exception {
184 
185     CompletableFuture<Throwable> failureServer = new CompletableFuture<>();
186     CompletableFuture<Throwable> failureClient = new CompletableFuture<>();
187 
188     // We cannot reuse SSLContext due to a potential Java bug, see
189     // https://github.com/kohlschuetter/snisslbug
190     SSLSocketFactory clientSocketFactory = initClientSocketFactory(configuration);
191 
192     AtomicBoolean gotHostnameInTime = new AtomicBoolean(false);
193 
194     TestingAFSocketServer<AFUNIXSocketAddress> server =
195         new TestingAFSocketServer<AFUNIXSocketAddress>(addr) {
196 
197           @Override
198           protected void doServeSocket(AFSocket<? extends AFUNIXSocketAddress> plainSocket)
199               throws IOException {
200 
201             boolean sniBroken = false;
202 
203             try (SSLSocket sslSocket = (SSLSocket) serverSocketFactory.createSocket(plainSocket,
204                 "localhost.junixsocket", plainSocket.getPort(), false)) {
205 
206               SNIHostnameCapture hnc = serverNullDefaultHostname ? //
207                   SNIHostnameCapture.configure(sslSocket, SNIHostnameCapture.ACCEPT_ANY_HOSTNAME) : //
208                   SNIHostnameCapture.configure(sslSocket, SNIHostnameCapture.ACCEPT_ANY_HOSTNAME,
209                       () -> "defaulthost");
210 
211               assertFalse(hnc.isComplete());
212               assertThrows(IllegalStateException.class, () -> hnc.getHostname());
213 
214               // NOTE: starting a handshake doesn't mean it's completed before reading the 1st byte!
215               sslSocket.startHandshake();
216 
217               hnc.isComplete(1, TimeUnit.SECONDS);
218               gotHostnameInTime.set(hnc.isComplete());
219 
220               try {
221                 if (expectDefault) {
222                   if (serverNullDefaultHostname && clientEmptyDefaultHostname) {
223                     assertNull(hnc.getHostname());
224                   } else if (clientEmptyDefaultHostname) {
225                     assertEquals("defaulthost", hnc.getHostname());
226                   } else {
227                     assertEquals("localhost.junixsocket", hnc.getHostname());
228                   }
229                 } else {
230                   if ("defaulthost".equals(hnc.getHostname()) && AFSocket.isRunningOnAndroid()) {
231                     // This is a known edge case.
232                     failureServer.complete(new TestAbortedWithImportantMessageException(
233                         MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
234                         "Android seems to not properly send the SNI hostname request"));
235                   } else {
236                     assertEquals("subdomain.example.com", hnc.getHostname());
237                   }
238                 }
239               } catch (AssertionFailedError e) {
240                 if (configuration.isSniBroken()) {
241                   sniBroken = true;
242                 } else {
243                   throw e;
244                 }
245               }
246 
247               try (InputStream in = sslSocket.getInputStream();
248                   OutputStream out = sslSocket.getOutputStream();) {
249                 int v = in.read();
250                 assertEquals('?', v);
251                 if (!hnc.isComplete()) {
252                   fail("Handshake is not marked complete, but data can be read");
253                 }
254 
255                 CompletableFuture<UnsupportedOperationException> cf =
256                     new CompletableFuture<UnsupportedOperationException>();
257                 try {
258                   assertEquals(hnc.getHostnameFromSSLSession(sslSocket, cf::complete), hnc
259                       .getHostname());
260                 } catch (AssertionFailedError e) {
261                   if (cf.isDone() && isBouncyCastleJSSE(serverSocketFactory)) {
262                     // BouncyCastle doesn't support this
263                   } else {
264                     throw e;
265                   }
266                 }
267 
268                 assertEquals(expectDefault ? 1 : 0, in.read());
269                 assertEquals(serverNullDefaultHostname ? 1 : 0, in.read());
270                 assertEquals(clientEmptyDefaultHostname ? 1 : 0, in.read());
271 
272                 out.write("Hello World".getBytes(StandardCharsets.UTF_8));
273                 out.flush();
274               }
275               if (!gotHostnameInTime.get()) {
276                 fail("Handshake was not marked complete in time, but eventually was");
277               }
278               if (sniBroken) {
279                 throw new TestAbortedWithImportantMessageException(
280                     MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
281                     "SNI support is knowingly incomplete for " + configuration);
282               }
283             } catch (Exception | Error e) {
284               failureServer.complete(e);
285             }
286             if (!gotHostnameInTime.get()) {
287               failureServer.complete(new AssertionFailedError(
288                   "Handshake was not marked complete in time"));
289             }
290 
291           }
292         };
293 
294     try {
295       server.startAndWaitToBecomeReady();
296 
297       try (AFUNIXSocket plainSocket = AFUNIXSocket.connectTo(addr);
298           SSLSocket sslSocket = (SSLSocket) clientSocketFactory.createSocket(plainSocket,
299               clientEmptyDefaultHostname ? "" : "localhost.junixsocket", plainSocket.getPort(),
300               false)) {
301 
302         if (!expectDefault) {
303           SSLParametersUtil.setSNIServerName(sslSocket, new SNIHostName("subdomain.example.com"));
304         }
305 
306         try (InputStream in = sslSocket.getInputStream();
307             OutputStream out = sslSocket.getOutputStream()) {
308           out.write('?');
309 
310           out.write(expectDefault ? 1 : 0);
311           out.write(serverNullDefaultHostname ? 1 : 0);
312           out.write(clientEmptyDefaultHostname ? 1 : 0);
313 
314           out.flush();
315 
316           byte[] by = new byte[11];
317           int offset = 0;
318           int r;
319           while (offset < by.length && (r = in.read(by, offset, by.length - offset)) >= 0) {
320             offset += r;
321           }
322           assertEquals("Hello World", new String(by, 0, offset, StandardCharsets.UTF_8));
323         } catch (Exception e) {
324           failureClient.complete(e);
325         }
326       }
327     } finally {
328       server.stop();
329 
330       TestUtil.throwMoreInterestingThrowableThanSocketException(() -> failureServer.getNow(null),
331           () -> failureClient.getNow(null));
332 
333       server.checkThrowable();
334     }
335   }

348   private void testWonkySSLSocket(int matchType) throws Exception {
349     SSLSocket sslSocket = new SSLSocket() {
350       private HandshakeCompletedListener hcl;
351       private SSLParameters params = new SSLParameters();
352 
353       @Override
354       @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED_NO_SIDE_EFFECT")
355       public void startHandshake() throws IOException {
356         for (SNIMatcher sm : getSSLParameters().getSNIMatchers()) {
357           sm.matches(new SNIHostName("hostname"));
358           sm.matches(new SNIServerName(23, "WAT".getBytes(StandardCharsets.US_ASCII)) {
359 
360           });
361           sm.matches(new SNIHostName("hostname"));
362           sm.matches(new SNIHostName("anotherHostname"));
363         }
