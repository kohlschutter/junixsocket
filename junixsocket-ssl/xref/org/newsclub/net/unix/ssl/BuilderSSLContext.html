21  import java.security.NoSuchAlgorithmException;
22  import java.security.SecureRandom;
23  import java.util.function.Function;
24  
25  import javax.net.SocketFactory;
26  import javax.net.ssl.KeyManager;
27  import javax.net.ssl.SSLContext;
28  import javax.net.ssl.SSLContextSpi;
29  import javax.net.ssl.SSLEngine;
30  import javax.net.ssl.SSLParameters;
31  import javax.net.ssl.SSLServerSocketFactory;

34  import javax.net.ssl.TrustManager;

36  /**
37   * An {@link SSLContext} wrapper that applies settings specified with {@link SSLContextBuilder}.
38   *
39   * @author Christian Kohlschütter
40   */
41  final class BuilderSSLContext extends SSLContext {
42    BuilderSSLContext(boolean clientMode, SSLContext context,
43        Function<SSLParameters, SSLParameters> parametersFunction, SocketFactory socketFactory) {
44      super(new ConfiguredSSLContextSpi(clientMode, context, parametersFunction, socketFactory),
45          context.getProvider(), context.getProtocol());
46    }
47  
48    /**
49     * Calls {@link SSLContext#init(KeyManager[], TrustManager[], SecureRandom)} with the given
50     * parameters. If there was a known problem with initializing the default {@link SecureRandom}
51     * (when {@code null} was specified), try {@link SecureRandom#getInstanceStrong()} instance.
52     *
53     * @param context The context to initialize.
54     * @param km The key managers, or {@code null}.
55     * @param tm The trust managers, or {@code null}.
56     * @param sr The secure random, or {@code null}.
57     * @throws KeyManagementException on error.
58     */
59    static void initContext(SSLContext context, KeyManager[] km, TrustManager[] tm, SecureRandom sr)
60        throws KeyManagementException {
61      try {
62        context.init(km, tm, sr);
63      } catch (IllegalStateException e) {
64        // In certain configurations, BouncyCastle may fail to detect a SecureRandom named "DEFAULT"
65        if (sr == null && e.getMessage().contains("SecureRandom not available")) {
66          try {
67            // Use any strong SecureRandom instance as fallback
68            sr = SecureRandom.getInstanceStrong();
69          } catch (NoSuchAlgorithmException e1) {
70            e.addSuppressed(e1);
71            throw e;
72          }
73          context.init(km, tm, sr);
74        } else {
75          throw e;
76        }
77      }
78    }
79  
80    private static final class ConfiguredSSLContextSpi extends SSLContextSpi {
81      private final SSLSocketFactory socketFactory;
82      private final SSLServerSocketFactory serverSocketFactory;
83  
84      private final SSLContext context;
85      private final SSLParameters params;
86      private final boolean clientMode;
87  
88      private ConfiguredSSLContextSpi(boolean clientMode, SSLContext context,
89          Function<SSLParameters, SSLParameters> parametersFunction, SocketFactory socketFactory) {
90        super();
91        this.clientMode = clientMode;
92        this.context = context;
93  
94        SSLParameters p = context.getDefaultSSLParameters();
95        if (parametersFunction != null) {
96          p = parametersFunction.apply(p);
97        }
98  
99        this.params = p;
100       this.socketFactory = new BuilderSSLSocketFactory(clientMode, context, context
101           .getSocketFactory(), p, socketFactory);
102       this.serverSocketFactory = new BuilderSSLServerSocketFactory(context.getServerSocketFactory(),
103           p);
104     }
105 
106     @Override
107     protected void engineInit(KeyManager[] km, TrustManager[] tm, SecureRandom sr)
108         throws KeyManagementException {
109       initContext(context, km, tm, sr);
110     }
111 
112     @Override
113     protected SSLSocketFactory engineGetSocketFactory() {
114       return socketFactory;
115     }
116 
117     @Override
118     protected SSLServerSocketFactory engineGetServerSocketFactory() {
119       return serverSocketFactory;
120     }
121 
122     @Override
123     protected SSLEngine engineCreateSSLEngine() {
124       return init(context.createSSLEngine());
125     }
126 
127     @Override
128     protected SSLEngine engineCreateSSLEngine(String host, int port) {
129       return init(context.createSSLEngine(host, port));
130     }
131 
132     @Override
133     protected SSLSessionContext engineGetServerSessionContext() {
134       return context.getServerSessionContext();
135     }
136 
137     @Override
138     protected SSLSessionContext engineGetClientSessionContext() {
139       return context.getClientSessionContext();
140     }
141 
142     private SSLEngine init(SSLEngine engine) {
143       engine.setEnabledProtocols(params.getProtocols());
144       engine.setEnabledCipherSuites(params.getCipherSuites());
145       engine.setUseClientMode(clientMode);
146 
147       return engine;
148     }
149   }
