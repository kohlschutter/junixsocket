70    private List<LibraryCandidate> tryProviderClass(String providerClassname, String artifactName)
71        throws IOException, ClassNotFoundException {
72      Class<?> providerClass = Class.forName(providerClassname);
73  
74      String version = getArtifactVersion(providerClass, artifactName);
75      String libraryNameAndVersion = LIBRARY_NAME + "-" + version;
76  
77      return findLibraryCandidates(artifactName, libraryNameAndVersion, providerClass);
78    }

80    public static String getJunixsocketVersion() throws IOException {
81      // NOTE: This can't easily be tested from within the junixsocket-common Maven build
82  
83      String v = BuildProperties.getBuildProperties().get("git.build.version");
84      if (v != null && !v.startsWith("$")) {
85        return v;
86      }
87  
88      return getArtifactVersion(AFSocket.class, "junixsocket-common");
89    }

91    private static String getArtifactVersion(Class<?> providerClass, String... artifactNames)
92        throws IOException {
93      for (String artifactName : artifactNames) {
94        Properties p = new Properties();
95        String resource = "/META-INF/maven/com.kohlschutter.junixsocket/" + artifactName
96            + "/pom.properties";
97        try (InputStream in = providerClass.getResourceAsStream(resource)) {
98          if (in == null) {
99            throw new FileNotFoundException("Could not find resource " + resource + " relative to "
100               + providerClass);
101         }
102         p.load(in);
103         String version = p.getProperty("version");
104 
105         Objects.requireNonNull(version, "Could not read version from pom.properties");
106         return version;
107       }
108     }
109     throw new IllegalStateException("No artifact names specified");
110   }

112   private abstract static class LibraryCandidate implements Closeable {
113     protected final String libraryNameAndVersion;
114 
115     protected LibraryCandidate(String libraryNameAndVersion) {
116       this.libraryNameAndVersion = libraryNameAndVersion;
117     }
118 
119     @SuppressFBWarnings("THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION")
120     abstract String load() throws Exception;
121 
122     @Override
123     public abstract void close();
124 
125     @Override
126     public String toString() {
127       return super.toString() + "[" + libraryNameAndVersion + "]";
128     }
129   }

131   private static final class StandardLibraryCandidate extends LibraryCandidate {
132     StandardLibraryCandidate(String version) {
133       super(version == null ? LIBRARY_NAME : LIBRARY_NAME + "-" + version);
134     }
135 
136     @Override
137     @SuppressFBWarnings("THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION")
138     String load() throws Exception, LinkageError {
139       if (libraryNameAndVersion != null) {
140         System.loadLibrary(libraryNameAndVersion);
141         return libraryNameAndVersion;
142       }
143       return null;
144     }
145 
146     @Override
147     public void close() {
148     }
149 
150     @Override
151     public String toString() {
152       return super.toString() + "(standard library path)";
153     }
154   }

156   private static final class ClasspathLibraryCandidate extends LibraryCandidate {
157     private final String artifactName;
158     private final URL library;
159     private final String path;
160 
161     ClasspathLibraryCandidate(String artifactName, String libraryNameAndVersion, String path,
162         URL library) {
163       super(libraryNameAndVersion);
164       this.artifactName = artifactName;
165       this.path = path;
166       this.library = library;
167     }
168 
169     @Override
170     synchronized String load() throws IOException, LinkageError {
171       if (libraryNameAndVersion == null) {
172         return null;
173       }
174 
175       File libDir = TEMP_DIR;
176 
177       for (int attempt = 0; attempt < 3; attempt++) {
178         File libFile;
179         try {
180           libFile = File.createTempFile("libtmp", System.mapLibraryName(libraryNameAndVersion),
181               libDir);
182           try (InputStream libraryIn = library.openStream();
183               OutputStream out = new FileOutputStream(libFile)) { // NOPMD UseTryWithResources
184             byte[] buf = new byte[4096];
185             int read;
186             while ((read = libraryIn.read(buf)) >= 0) {
187               out.write(buf, 0, read);
188             }
189           }
190         } catch (IOException e) {
191           throw e;
192         }
193 
194         try {
195           System.load(libFile.getAbsolutePath());
196         } catch (UnsatisfiedLinkError e) {
197           String message = e.getMessage().toLowerCase(Locale.getDefault());
198           if (!message.contains("perm")) {
199             throw e;
200           }
201 
202           // Operation not permitted; permission denied; EPERM...
203           // -> tmp directory may be mounted with "noexec", try loading from user.home, user.dir
204 
205           switch (attempt) {
206             case 0:
207               libDir = new File(System.getProperty("user.home", "."));
208               break;
209             case 1:
210               libDir = new File(System.getProperty("user.dir", "."));
211               break;
212             default:
213               throw e;
214           }
215 
216           continue;
217         } finally {
218           if (!libFile.delete()) {
219             libFile.deleteOnExit();
220           }
221         }
222 
223         // If we reach this, then we were able to load the library
224         break; // NOPMD.AvoidBranchingStatementAsLastInLoop
225       }
226       return artifactName + "/" + libraryNameAndVersion;
227     }
228 
229     @Override
230     public void close() {
231     }
232 
233     @Override
234     public String toString() {
235       return super.toString() + "(" + artifactName + ":" + path + ")";
236     }
237   }

243   @SuppressFBWarnings("THROWS_METHOD_THROWS_RUNTIMEEXCEPTION")
244   private static synchronized void setLoaded0(String library) {
245     if (LOADED.compareAndSet(false, true)) {
246       NativeUnixSocket.setLoaded(true);
247       AFSocket.loadedLibrary = library;
248       try {
249         NativeUnixSocket.init();
250       } catch (RuntimeException e) {
251         throw e;
252       } catch (Exception e) {
253         throw new IllegalStateException(e);
254       }
255     }
256   }

258   private Throwable loadLibraryOverride() {
259     String libraryOverride = System.getProperty(PROP_LIBRARY_OVERRIDE, "");
260     String libraryOverrideForce = System.getProperty(PROP_LIBRARY_OVERRIDE_FORCE, "false");
261 
262     if (libraryOverride.isEmpty() && libraryOverrideForce.startsWith("/")) {
263       libraryOverride = libraryOverrideForce;
264       libraryOverrideForce = "true";
265     }
266 
267     if (!libraryOverride.isEmpty()) {
268       try {
269         System.load(libraryOverride);
270         setLoaded(libraryOverride);
271         return null;
272       } catch (Exception | LinkageError e) {
273         if (Boolean.parseBoolean(libraryOverrideForce)) {
274           throw e;
275         }
276         return e;
277       }
278     } else {
279       return new Exception("No library specified with -D" + PROP_LIBRARY_OVERRIDE + "=");
280     }
281   }

283   private static Object loadLibrarySyncMonitor() {
284     Object monitor = NativeLibraryLoader.class.getClassLoader(); // NOPMD
285     if (monitor == null) {
286       // bootstrap classloader?
287       return NativeLibraryLoader.class;
288     } else {
289       return monitor;
290     }
291   }

293   @SuppressWarnings("null")
294   public synchronized void loadLibrary() {
295     synchronized (loadLibrarySyncMonitor()) { // NOPMD We want to lock this class' classloader.
296       if (LOADED.get()) {
297         // Already loaded
298         return;
299       }
300 
301       NativeUnixSocket.initPre();
302 
303       // set -Dorg.newsclub.net.unix.library.override.force=provided to assume that
304       // we already have loaded the library via System.load, etc.
305       if ("provided".equals(System.getProperty(PROP_LIBRARY_OVERRIDE_FORCE, ""))) {
306         setLoaded("provided");
307         return;
308       }
309 
310       boolean provided = false;
311       try {
312         NativeUnixSocket.noop();
313         provided = true;
314       } catch (UnsatisfiedLinkError | Exception e) {
315         // expected unless we manually loaded the library
316       }
317       if (provided) {
318         setLoaded("provided");
319         return;
320       }
321 
322       if (Boolean.parseBoolean(System.getProperty(PROP_LIBRARY_DISABLE, "false"))) {
323         throw initCantLoadLibraryError(Collections.singletonList(new UnsupportedOperationException(
324             "junixsocket disabled by System.property " + PROP_LIBRARY_DISABLE)));
325       }
326 
327       List<Throwable> suppressedThrowables = new ArrayList<>();
328       Throwable ex = loadLibraryOverride();
329       if (ex == null) {
330         return;
331       }
332       suppressedThrowables.add(ex);
333 
334       List<LibraryCandidate> candidates = initLibraryCandidates(suppressedThrowables);
335 
336       String loadedLibraryId = null;
337       for (LibraryCandidate candidate : candidates) {
338         try {
339           if ((loadedLibraryId = candidate.load()) != null) {
340             break;
341           }
342         } catch (Exception | LinkageError e) {
343           suppressedThrowables.add(e);
344         }
345       }
346 
347       for (LibraryCandidate candidate : candidates) {
348         candidate.close();
349       }
350 
351       if (loadedLibraryId == null) {
352         throw initCantLoadLibraryError(suppressedThrowables);
353       }
354 
355       setLoaded(loadedLibraryId);
356     }
357   }

359   private UnsatisfiedLinkError initCantLoadLibraryError(List<Throwable> suppressedThrowables) {
360     String message = "Could not load native library " + LIBRARY_NAME + " for architecture "
361         + ARCHITECTURE_AND_OS;
362 
363     String cp = System.getProperty("java.class.path", "");
364     if (cp.contains("junixsocket-native-custom/target-eclipse") || cp.contains(
365         "junixsocket-native-common/target-eclipse")) {
366       message += "\n\n*** ECLIPSE USERS ***\nIf you're running from within Eclipse, "
367           + "please close the projects \"junixsocket-native-common\" and \"junixsocket-native-custom\"\n";
368     }
369 
370     UnsatisfiedLinkError e = new UnsatisfiedLinkError(message);
371     if (suppressedThrowables != null) {
372       for (Throwable suppressed : suppressedThrowables) {
373         e.addSuppressed(suppressed);
374       }
375     }
376     throw e;
377   }

379   private List<LibraryCandidate> initLibraryCandidates(List<Throwable> suppressedThrowables) {
380     List<LibraryCandidate> candidates = new ArrayList<>();
381     try {
382       String version = getArtifactVersion(getClass(), "junixsocket-common", "junixsocket-core");
383       if (version != null) {
384         candidates.add(new StandardLibraryCandidate(version));
385       }
386     } catch (Exception e) {
387       suppressedThrowables.add(e);
388     }
389 
390     try {
391       candidates.addAll(tryProviderClass("org.newsclub.lib.junixsocket.custom.NarMetadata",
392           "junixsocket-native-custom"));
393     } catch (Exception e) {
394       suppressedThrowables.add(e);
395     }
396     try {
397       candidates.addAll(tryProviderClass("org.newsclub.lib.junixsocket.common.NarMetadata",
398           "junixsocket-native-common"));
399     } catch (Exception e) {
400       suppressedThrowables.add(e);
401     }
402 
403     candidates.add(new StandardLibraryCandidate(null));
404 
405     return candidates;
406   }

408   private static String lookupArchProperty(String key, String defaultVal) {
409     return System.getProperty(key, defaultVal).replaceAll("[ /\\\\'\";:\\$]", "");
410   }

412   private static List<String> architectureAndOS() {
413     String arch = lookupArchProperty("os.arch", "UnknownArch");
414     String osName = lookupArchProperty("os.name", "UnknownOS");
415 
416     List<String> list = new ArrayList<>();
417     if (IS_ANDROID) {
418       // Android identifies itself as os.name="Linux"
419       // let's probe for an Android-specific library first
420       list.add(arch + "-Android");
421     }
422     list.add(arch + "-" + osName);
423     if (osName.startsWith("Windows") && !"Windows10".equals(osName)) {
424       list.add(arch + "-" + "Windows10");
425     }
426 
427     return list;
428   }

430   private static boolean checkAndroid() {
431     String vmName = lookupArchProperty("java.vm.name", "UnknownVM");
432     String vmSpecVendor = lookupArchProperty("java.vm.specification.vendor",
433         "UnknownSpecificationVendor");
434 
435     return ("Dalvik".equals(vmName) || vmSpecVendor.contains("Android"));
436   }

446   private static URL validateResourceURL(URL url) {
447     if (url == null) {
448       return null;
449     }
450     try (InputStream in = url.openStream()) {
451       return url;
452     } catch (IOException e) {
453       return null;
454     }
455   }

457   private List<LibraryCandidate> findLibraryCandidates(String artifactName,
458       String libraryNameAndVersion, Class<?> providerClass) {
459     String mappedName = System.mapLibraryName(libraryNameAndVersion);
460 
461     String[] prefixes = mappedName.startsWith("lib") ? new String[] {""} : new String[] {"", "lib"};
462 
463     List<LibraryCandidate> list = new ArrayList<>();
464     for (String archOs : ARCHITECTURE_AND_OS) {
465       for (String compiler : new String[] {"clang", "gcc"
466           // "gpp", "g++", "linker", "clang", "gcc", "cc", "CC", "icpc", "icc", "xlC", "xlC_r",
467           // "msvc",
468           // "icl", "ecpc", "ecc"
469       }) {
470         for (String prefix : prefixes) {
471           String path = "/lib/" + archOs + "-" + compiler + "/jni/" + prefix + mappedName;
472 
473           URL url;
474 
475           url = validateResourceURL(providerClass.getResource(path));
476           if (url != null) {
477             list.add(new ClasspathLibraryCandidate(artifactName, libraryNameAndVersion, path, url));
478           }
479 
480           // NOTE: we have to try .nodeps version _after_ trying the properly linked one.
481           // While the former may throw an UnsatisfiedLinkError, this one may just terminate the VM
482           // with a "symbol lookup error"
483           String nodepsPath = nodepsPath(path);
484           if (nodepsPath != null) {
485             url = validateResourceURL(providerClass.getResource(nodepsPath));
486             if (url != null) {
487               list.add(new ClasspathLibraryCandidate(artifactName, libraryNameAndVersion,
488                   nodepsPath, url));
489             }
490           }
491         }
492       }
493     }
494     return list;
495   }

497   private String nodepsPath(String path) {
498     int lastDot = path.lastIndexOf('.');
499     if (lastDot == -1) {
500       return null;
501     } else {
502       return path.substring(0, lastDot) + ".nodeps" + path.substring(lastDot);
503     }
504   }
