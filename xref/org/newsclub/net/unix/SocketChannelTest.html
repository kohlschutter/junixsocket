57    @Test
58    public void testNonBlockingConnect() throws IOException {
59      SocketAddress sa = newTempAddress();
60  
61      ServerSocketChannel ssc = selectorProvider().openServerSocketChannel();
62      ssc.configureBlocking(false);
63      bindServerSocket(ssc, sa, 1);
64      sa = ssc.getLocalAddress();
65  
66      {
67        SocketChannel sc = selectorProvider().openSocketChannel();
68        sc.configureBlocking(false);
69        if (!handleConnect(sc, sa)) {
70          // connect pending
71          assertTrue(sc.isConnected() || sc.isConnectionPending());
72          long now = System.currentTimeMillis();
73          do {
74            if (sc.finishConnect()) {
75              break;
76            }
77            try {
78              Thread.sleep(100);
79            } catch (InterruptedException e) {
80              break;
81            }
82            if ((System.currentTimeMillis() - now) > 1000) {
83              fail("Non-blocking connect not connected after 1s");
84              break;
85            }
86          } while (!Thread.interrupted());
87          assertTrue(sc.finishConnect());
88        }
89        assertTrue(sc.isConnected());
90        assertFalse(sc.isConnectionPending());
91      }
92    }

104   @SuppressWarnings({
105       "PMD.ExcessiveMethodLength", "PMD.CognitiveComplexity", "PMD.NPathComplexity",
106       "PMD.CyclomaticComplexity"})
107   private void testDoubleBind(boolean reuseAddress) throws Exception {
108     SocketAddress sa0 = newTempAddress();
109 
110     final Future<SocketChannel> acceptCall;
111     final Future<SocketChannel> acceptCall2;
112     Future<Void> connectCall = null;
113 
114     AtomicBoolean socketDomainWillAcceptCallOnFirstBind = new AtomicBoolean(true);
115 
116     try (ServerSocketChannel ssc1 = selectorProvider().openServerSocketChannel()) {
117       bindServerSocket(ssc1, sa0, 1);
118       final SocketAddress sa = resolveAddressForSecondBind(sa0, ssc1);
119 
120       AtomicBoolean connectMustSucceed = new AtomicBoolean(false);
121       AtomicBoolean wasRebound = new AtomicBoolean(false);
122 
123       acceptCall = TestAsyncUtil.supplyAsync(() -> {
124         try {
125           SocketChannel sc;
126           try {
127             sc = ssc1.accept();
128           } catch (ClosedChannelException | SocketClosedException e) {
129             if (wasRebound.get()) {
130               // The system terminated our accept because another socket was rebound
131               // This may not occur on all systems, but we have to handle it.
132               return null;
133             } else {
134               throw e;
135             }
136           }
137           socketDomainWillAcceptCallOnFirstBind.set(false);
138           Objects.requireNonNull(sc);
139           if (reuseAddress && !connectMustSucceed.get()) {
140             // fail("Did not throw SocketException"); // no longer thrown in Sonoma 14.2.1?
141           }
142           return sc;
143         } catch (ClosedChannelException | SocketException e) { // NOPMD.ExceptionAsFlowControl
144           String msg = checkKnownBugAcceptFailure(e);
145           if (msg != null) {
146             throw new TestAbortedWithImportantMessageException(
147                 MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg), e);
148           }
149           if (reuseAddress) {
150             // expected (Software caused connection abort)
151           } else {
152             fail(e);
153           }
154         } catch (SocketTimeoutException e) {
155           String msg = checkKnownBugAcceptFailure(e);
156           if (msg != null) {
157             throw new TestAbortedWithImportantMessageException(
158                 MessageType.TEST_ABORTED_SHORT_WITH_ISSUES, msg, summaryImportantMessage(msg), e);
159           }
160           fail(e);
161         } catch (IOException e) { // NOPMD.ExceptionAsFlowControl
162           fail(e);
163         }
164         return null;
165       });
166 
167       try (ServerSocketChannel ssc2 = selectorProvider().openServerSocketChannel()) {
168         try {
169           ssc2.setOption(StandardSocketOptions.SO_REUSEADDR, reuseAddress);
170         } catch (UnsupportedOperationException e) {
171           // ignore
172         }
173 
174         try {
175           wasRebound.set(true);
176           bindServerSocket(ssc2, sa, 1);
177           if (!reuseAddress && !socketDomainPermitsDoubleBind()) {
178             fail("Did not throw expected SocketException (Address already in use)");
179           }
180         } catch (ClosedChannelException | SocketException e) {
181           if (!reuseAddress) {
182             // expected
183           } else {
184             // permissible, depending on socket domain
185             // but connecting to the first server must succeed
186             connectMustSucceed.set(true);
187           }
188         }
189 
190         if (reuseAddress) {
191           acceptCall2 = TestAsyncUtil.supplyAsync(() -> {
192             try {
193               SocketChannel sc = ssc2.accept();
194               socketDomainWillAcceptCallOnFirstBind.set(false);
195               Objects.requireNonNull(sc);
196               return sc;
197             } catch (InvalidArgumentSocketException e) {
198               if (!acceptCall.isDone()) {
199                 socketDomainWillAcceptCallOnFirstBind.set(true);
200                 // some socket domains will permit a bind but not another accept until the first one
201                 // is done (e.g., VSOCK).
202               } else {
203                 fail(e);
204               }
205             } catch (IOException e) {
206               fail(e);
207             }
208             return null;
209           });
210         } else {
211           acceptCall2 = null;
212         }
213 
214         // unblock accept of any successful bind
215         if (!acceptCall.isDone() && socketDomainWillAcceptCallOnFirstBind.get()) {
216           connectCall = TestAsyncUtil.supplyAsync(() -> {
217             try {
218               newSocket().connect(sa);
219             } catch (ClosedChannelException | SocketClosedException e) {
220               // ignore
221             } catch (SocketException e) {
222               if (connectMustSucceed.get()) {
223                 fail("Connect should have succeeded", e);
224               } else {
225                 // ignore
226               }
227             } catch (IOException e) {
228               fail(e);
229             }
230           });
231         }
232       }
233     }
234 
235     // Assert that eventually all accept jobs have terminated.
236     if (acceptCall2 != null) {
237       try {
238         acceptCall2.get(5, TimeUnit.SECONDS);
239       } catch (ExecutionException e) {
240         // ignore socket closed etc.
241       } catch (TimeoutException e) {
242         fail("Second accept call did not terminate");
243       }
244     }
245 
246     String triggerWithIssues = null;
247 
248     try {
249       acceptCall.get(5, TimeUnit.SECONDS);
250     } catch (ExecutionException e) {
251       // ignore socket closed etc.
252       if (e.getCause() instanceof TestAbortedWithImportantMessageException) {
253         throw (TestAbortedWithImportantMessageException) e.getCause(); // NOPMD.PreserveStackTrace
254       } else {
255         throw e;
256       }
257     } catch (TimeoutException e) {
258       triggerWithIssues = checkKnownBugFirstAcceptCallNotTerminated();
259       if (triggerWithIssues == null) {
260         fail("First accept call did not terminate");
261       }
262     }
263     if (triggerWithIssues != null) {
264       throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_SHORT_WITH_ISSUES,
265           triggerWithIssues, summaryImportantMessage(triggerWithIssues));
266     }
267 
268     if (connectCall != null) {
269       try {
270         connectCall.get(5, TimeUnit.SECONDS);
271       } catch (ExecutionException e) {
272         // ignore socket closed etc.
273       } catch (TimeoutException e) {
274         fail("Connect call did not terminate");
275       }
276     }
277   }

317   @Test
318   public void testByteBufferWithPositionOffset() throws Exception {
319     SocketAddress sa = newTempAddress();
320 
321     final int bb1Offset = 32;
322     final int bb2Offset = 1;
323 
324     byte[] data = new byte[96];
325     getRandom().nextBytes(data);
326 
327     try (ServerSocketChannel ssc = selectorProvider().openServerSocketChannel()) {
328       handleBind(ssc, sa);
329 
330       ByteBuffer bb1 = ByteBuffer.allocate(data.length + bb1Offset);
331       bb1.position(bb1Offset);
332 
333       bb1.put(data);
334       bb1.flip();
335 
336       bb1.position(bb1Offset);
337       assertEquals(bb1Offset, bb1.position());
338 
339       TestAsyncUtil.runAsync(() -> {
340         try (SocketChannel sc = ssc.accept()) {
341           int written = 0;
342           while (bb1.hasRemaining()) {
343             written += sc.write(bb1);
344           }
345           assertEquals(data.length, written);
346         } catch (IOException e) {
347           fail(e);
348         }
349       });
350 
351       SocketChannel sc = selectorProvider().openSocketChannel();
352 
353       ByteBuffer bb2 = ByteBuffer.allocate(data.length + bb2Offset);
354 
355       assertTrue(handleConnect(sc, ssc.getLocalAddress()));
356 
357       bb2.position(bb2Offset);
358 
359       int read = 0;
360       int r;
361       do {
362         r = sc.read(bb2);
363         if (r == -1) {
364           break;
365         }
366         read += r;
367       } while (bb2.hasRemaining());
368       assertEquals(data.length, read);
369 
370       assertEquals(bb1.capacity(), bb1.position());
371       assertEquals(bb1.capacity(), bb1.limit());
372       assertEquals(data.length + bb2Offset, bb2.position());
373       assertEquals(bb2.capacity(), bb2.limit());
374 
375       bb1.position(bb1Offset);
376 
377       for (int i = 0, n = data.length; i < n; i++) {
378         assertEquals(bb1.get(bb1Offset + i), bb2.get(bb2Offset + i), "at pos " + i);
379       }
380     }
381   }

458   public void testBindNull() throws Exception {
459     try (ServerSocketChannel sc = newServerSocketChannel()) {
460       ServerSocket s = socketIfPossible(sc);
461       assertTrue(s == null || !s.isBound());
462       try {
463         sc.bind(null);
464       } catch (UnsupportedOperationException e) {
465         if (mayTestBindNullThrowUnsupportedOperationException()) {
466           // OK
467           return;
468         } else {
469           throw e;
470         }
471       }
472       assertTrue(s == null || s.isBound());
473 
474       SocketAddress addr = sc.getLocalAddress();
475       if (!mayTestBindNullHaveNullLocalSocketAddress()) {
476         assertNotNull(addr);
477       }
478 
479       cleanupTestBindNull(sc, addr);
480     }
