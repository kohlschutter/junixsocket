46    @Test
47    public void testNonBlockingConnect() throws IOException {
48      SocketAddress sa = newTempAddress();
49  
50      ServerSocketChannel ssc = selectorProvider().openServerSocketChannel();
51      ssc.configureBlocking(false);
52      bindServerSocket(ssc, sa, 1);
53      sa = ssc.getLocalAddress();
54  
55      {
56        SocketChannel sc = selectorProvider().openSocketChannel();
57        sc.configureBlocking(false);
58        if (!sc.connect(sa)) {
59          // connect pending
60          assertTrue(sc.isConnected() || sc.isConnectionPending());
61          long now = System.currentTimeMillis();
62          do {
63            if (sc.finishConnect()) {
64              break;
65            }
66            try {
67              Thread.sleep(100);
68            } catch (InterruptedException e) {
69              break;
70            }
71            if ((System.currentTimeMillis() - now) > 1000) {
72              fail("Non-blocking connect not connected after 1s");
73              break;
74            }
75          } while (!Thread.interrupted());
76          assertTrue(sc.finishConnect());
77        }
78        assertTrue(sc.isConnected());
79        assertFalse(sc.isConnectionPending());
80      }
81    }

83    @Test
84    public void testDoubleBindAddressNotReusable() throws Exception {
85      testDoubleBind(false);
86    }

88    @Test
89    public void testDoubleBindAddressReusable() throws Exception {
90      testDoubleBind(true);
91    }

93    @SuppressWarnings({"PMD.ExcessiveMethodLength", "PMD.CognitiveComplexity", "PMD.NPathComplexity"})
94    private void testDoubleBind(boolean reuseAddress) throws Exception {
95      SocketAddress sa0 = newTempAddress();
96  
97      final CompletableFuture<SocketChannel> acceptCall;
98      final CompletableFuture<SocketChannel> acceptCall2;
99      CompletableFuture<Void> connectCall = null;
100 
101     AtomicBoolean socketDomainWillAcceptCallOnFirstBind = new AtomicBoolean(true);
102 
103     try (ServerSocketChannel ssc1 = selectorProvider().openServerSocketChannel()) {
104       bindServerSocket(ssc1, sa0, 1);
105       final SocketAddress sa = resolveAddressForSecondBind(sa0, ssc1);
106 
107       AtomicBoolean connectMustSucceed = new AtomicBoolean(false);
108 
109       acceptCall = CompletableFuture.supplyAsync(() -> {
110         try {
111           SocketChannel sc = ssc1.accept();
112           socketDomainWillAcceptCallOnFirstBind.set(false);
113           Objects.requireNonNull(sc);
114           if (reuseAddress && !connectMustSucceed.get()) {
115             fail("Did not throw SocketException");
116           }
117           return sc;
118         } catch (SocketException e) {
119           if (reuseAddress) {
120             // expected (Software caused connection abort)
121           } else {
122             fail(e);
123           }
124         } catch (IOException e) {
125           fail(e);
126         }
127         return null;
128       });
129 
130       try (ServerSocketChannel ssc2 = selectorProvider().openServerSocketChannel()) {
131         ssc2.socket().setReuseAddress(reuseAddress);
132 
133         try {
134           bindServerSocket(ssc2, sa, 1);
135           if (!reuseAddress && !socketDomainPermitsDoubleBind()) {
136             fail("Did not throw expected SocketException (Address already in use)");
137           }
138         } catch (SocketException e) {
139           if (!reuseAddress) {
140             // expected
141           } else {
142             // permissible, depending on socket domain
143             // but connecting to the first server must succeed
144             connectMustSucceed.set(true);
145           }
146         }
147 
148         if (reuseAddress) {
149           acceptCall2 = CompletableFuture.supplyAsync(() -> {
150             try {
151               SocketChannel sc = ssc2.accept();
152               socketDomainWillAcceptCallOnFirstBind.set(false);
153               Objects.requireNonNull(sc);
154               return sc;
155             } catch (InvalidArgumentSocketException e) {
156               if (!acceptCall.isDone()) {
157                 socketDomainWillAcceptCallOnFirstBind.set(true);
158                 // some socket domains will permit a bind but not another accept until the first one
159                 // is done (e.g., VSOCK).
160               } else {
161                 fail(e);
162               }
163             } catch (IOException e) {
164               fail(e);
165             }
166             return null;
167           });
168         } else {
169           acceptCall2 = null;
170         }
171 
172         // unblock accept of any successful bind
173         if (!acceptCall.isDone() && socketDomainWillAcceptCallOnFirstBind.get()) {
174           connectCall = CompletableFuture.runAsync(() -> {
175             try {
176               newSocket().connect(sa);
177             } catch (SocketException e) {
178               if (connectMustSucceed.get()) {
179                 fail("Connect should have succeeded", e);
180               } else {
181                 // ignore
182               }
183             } catch (IOException e) {
184               fail(e);
185             }
186           });
187         }
188       }
189     }
190 
191     // Assert that eventually all accept jobs have terminated.
192     if (acceptCall2 != null) {
193       try {
194         acceptCall2.get(5, TimeUnit.SECONDS);
195       } catch (ExecutionException e) {
196         // ignore socket closed etc.
197       } catch (TimeoutException e) {
198         fail("Second accept call did not terminate");
199       }
200     }
201 
202     String triggerWithIssues = null;
203 
204     try {
205       acceptCall.get(5, TimeUnit.SECONDS);
206     } catch (ExecutionException e) {
207       // ignore socket closed etc.
208     } catch (TimeoutException e) {
209       triggerWithIssues = checkKnownBugFirstAcceptCallNotTerminated();
210       if (triggerWithIssues == null) {
211         fail("First accept call did not terminate");
212       }
213     }
214     if (connectCall != null) {
215       try {
216         connectCall.get(5, TimeUnit.SECONDS);
217       } catch (ExecutionException e) {
218         // ignore socket closed etc.
219       } catch (TimeoutException e) {
220         fail("Connect call did not terminate");
221       }
222     }
223 
224     if (triggerWithIssues != null) {
225       throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_WITH_ISSUES,
226           triggerWithIssues);
227     }
228   }

230   /**
231    * Subclasses may override this to tell that there is a known issue with "First accept call did
232    * not terminate".
233    * 
234    * @return An explanation iff this should not cause a test failure but trigger "With issues".
235    */
236   protected String checkKnownBugFirstAcceptCallNotTerminated() {
237     return null;
238   }

240   /**
241    * Returns the temporary address usable to binding on for a second bind.
242    * 
243    * Depending on the socket domain, a wildcard address may be permittable or not for a second bind.
244    * 
245    * @param originalAddress The original temporary address (e.g., a wildcard address).
246    * @param ssc The socket that was bound to that address.
247    * @return The local bound address, or the {@code originalAddress}.
248    * @throws IOException on error.
249    * @see #testDoubleBindAddressReusable()
250    */
251   protected SocketAddress resolveAddressForSecondBind(SocketAddress originalAddress,
252       ServerSocketChannel ssc) throws IOException {
253     return ssc.getLocalAddress();
254   }

256   /**
257    * Override to declare that a certain socket domain permits double-binding an address,
258    * particularly when the address is comparable to a wildcard address.
259    * 
260    * @return {@code true} iff double-binding the same address is allowed.
261    * @see #testDoubleBindAddressReusable()
262    */
263   protected boolean socketDomainPermitsDoubleBind() {
264     return false;
265   }
