18  package org.newsclub.net.unix;
19  
20  import static org.junit.jupiter.api.Assertions.assertFalse;
21  import static org.junit.jupiter.api.Assertions.assertTrue;
22  import static org.junit.jupiter.api.Assertions.fail;
23  
24  import java.io.IOException;
25  import java.net.SocketAddress;
26  import java.nio.channels.ServerSocketChannel;
27  import java.nio.channels.SocketChannel;
28  
29  import org.junit.jupiter.api.Test;
30  
31  public abstract class SocketChannelTest<A extends SocketAddress> extends SocketTestBase<A> {
32    protected SocketChannelTest(AddressSpecifics<A> asp) {
33      super(asp);
34    }
35  
36    @Test
37    public void testNonBlockingConnect() throws IOException {
38      SocketAddress sa = newTempAddress();
39  
40      ServerSocketChannel ssc = selectorProvider().openServerSocketChannel();
41      ssc.configureBlocking(false);
42      ssc.bind(sa, 1);
43  
44      {
45        SocketChannel sc = selectorProvider().openSocketChannel();
46        sc.configureBlocking(false);
47        if (!sc.connect(sa)) {
48          // connect pending
49          assertTrue(sc.isConnected() || sc.isConnectionPending());
50          long now = System.currentTimeMillis();
51          do {
52            if (sc.finishConnect()) {
53              break;
54            }
55            try {
56              Thread.sleep(100);
57            } catch (InterruptedException e) {
58              break;
59            }
60            if ((System.currentTimeMillis() - now) > 1000) {
61              fail("Non-blocking connect not connected after 1s");
62              break;
63            }
64          } while (!Thread.interrupted());
65          assertTrue(sc.finishConnect());
66        }
67        assertTrue(sc.isConnected());
68        assertFalse(sc.isConnectionPending());
69      }
70    }
71  
72  }
