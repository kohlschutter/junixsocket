47    @Test
48    public void testFileScheme() throws Exception {
49      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
50          "file:/tmp/yo")));
51      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
52          "file:///tmp/yo")));
53      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
54          "file://localhost/tmp/yo")));
55    }

57    @Test
58    public void testUnixScheme() throws Exception {
59      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
60          "unix:/tmp/yo")));
61      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
62          "unix:///tmp/yo")));
63      assertEquals(AFUNIXSocketAddress.of(new File("/tmp/yo")), AFUNIXSocketAddress.of(URI.create(
64          "unix://localhost/tmp/yo")));

67    @Test
68    public void testHttpUnix() throws Exception {
69      assertEquals(AFUNIXSocketAddress.of(new File("/var/run/docker.sock")), AFUNIXSocketAddress.of(
70          URI.create("http+unix://%2Fvar%2Frun%2Fdocker.sock/info")));
71      assertEquals(AFUNIXSocketAddress.of(new File("/var/run/docker.sock"), 8123), AFUNIXSocketAddress
72          .of(URI.create("http+unix://%2Fvar%2Frun%2Fdocker.sock:8123/info")));
73      assertEquals(AFUNIXSocketAddress.of(new File("/var/run/docker.sock"), 8123), AFUNIXSocketAddress
74          .of(URI.create(
75              "http+unix://foo:bar%40@%2Fvar%2Frun%2Fdocker.sock:8123/info?q1=a1&q2=a2#frag%40")));
76      assertEquals(AFUNIXSocketAddress.of(new File("test"), 80), AFUNIXSocketAddress.of(URI.create(
77          "http+unix://test:80/info")));
78    }

80    private String fixWindowsEncodedPaths(String path) {
81      return path.replace("%5C", "%2F");
82    }
83  
84    private void assertsEqualUnixURIs(URI u1, URI u2) {
85      assertEquals(fixWindowsEncodedPaths(u1.toString()), fixWindowsEncodedPaths(u2.toString()));
86    }
87  
88    private void assertParseURI(URI u) throws Exception {
89      AFUNIXSocketAddress addr = (AFUNIXSocketAddress) AFSocketAddress.of(u);
90      URI u2 = addr.toURI(u.getScheme(), u);
91      assertsEqualUnixURIs(u, u2);
92    }
93  
94    @Test
95    public void testParseURIandBack() throws Exception {
96      assertParseURI(URI.create(
97          "http+unix://foo:bar%40@%2Fvar%2Frun%2Fdocker.sock:8123/info?q1=a1&q2=a2+%40#frag%40"));
98      assertParseURI(URI.create(
99          "http+unix://%2Fvar%2Frun%2Fdocker.sock:8123/info?q1=a1&q2=a2+%40#frag%40"));
100     assertParseURI(URI.create(
101         "http+unix://%2Fvar%2Frun%2Fdocker.sock/info?q1=a1&q2=a2+%40#frag%40"));
102     assertParseURI(URI.create("http+unix://%2Fvar%2Frun%2Fdocker.sock/info?q1="));
103     assertParseURI(URI.create("http+unix://%2Fvar%2Frun%2Fdocker.sock:1234"));
104   }

106   @Test
107   public void testURITemplate() throws Exception {
108     URI socketURI = URI.create("unix://%2Fvar%2Frun%2Fdocker.sock");
109     URI httpURI = URI.create("http://localhost/some/path?q=");
110     AFSocketAddress a = AFSocketAddress.of(socketURI);
111     assertEquals("https+unix://%2Fvar%2Frun%2Fdocker.sock/some/path?q=", fixWindowsEncodedPaths(a
112         .toURI("https+unix", httpURI).toString()));

125   public void testSocatString() throws Exception {
126     String socatString = AFUNIXSocketAddress.of(new File("/tmp/yo")).toSocatAddressString(null,
127         AFSocketProtocol.DEFAULT);

141   @Test
142   public void testSerialize() throws Exception {
143     try (ByteArrayOutputStream bos = new ByteArrayOutputStream();
144         ObjectOutputStream oos = new ObjectOutputStream(bos)) {
145 
146       AFUNIXSocketAddress addr = AFUNIXSocketAddress.of(URI.create("file:/tmp/yo"));
147       oos.writeObject(addr);
148       oos.flush();
149 
150       try (ObjectInputStream oin = new ObjectInputStream(new ByteArrayInputStream(bos
151           .toByteArray()))) {
152         AFUNIXSocketAddress addr2 = (AFUNIXSocketAddress) oin.readObject();
153         assertEquals(addr, addr2);
154         assertEquals(addr.getAddressFamily(), addr2.getAddressFamily());
155       }
156     }
