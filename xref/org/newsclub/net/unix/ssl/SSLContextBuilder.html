364   private KeyManager[] buildKeyManagers(KeyManagerFactory kmf) throws GeneralSecurityException,
365       IOException, UnrecoverableKeyException {
366     if (keyStoreUrl == null) {
367       // use default
368       return null; // NOPMD.ReturnEmptyCollectionRatherThanNull
369     }
370 
371     KeyStore ks = keyStoreSupplier.get();
372 
373     char[] password = keyStorePassword == null ? null : keyStorePassword.get();
374     try (InputStream in = keyStoreUrl.openStream()) {
375       ks.load(in, password);
376       kmf.init(ks, password);
377     } catch (IOException e) {
378       throw wrapIOExceptionIfJDKBug(e);
379     } finally {
380       clear(password);
381     }
382 
383     return kmf.getKeyManagers();
384   }

386   /**
387    * For a given {@link IOException} thrown from within {@link SSLContextBuilder}, check if it is
388    * due to a known JDK bug, and if so, wrap that exception in a {@link KnownJavaBugIOException}
389    * with a proper explanation.
390    *
391    * @param e The exception to check/wrap.
392    * @return The exception, or a {@link KnownJavaBugIOException}.
393    */
394   public static IOException wrapIOExceptionIfJDKBug(IOException e) {
395     String msg = e.getMessage();
396     if (msg == null) {
397       return e;
398     }
399     if (msg.contains("data isn't an object ID (tag = 48)")) {
400       return knownJDKBug(e, "JDK-8202837", "8u312", "11.0.3");
401     } else if (msg.contains("HmacPBESHA256 not available")) {
402       return knownJDKBug(e, "JDK-8267701", "8u301", "11.0.12",
403           "alternatively try running Java with -Dkeystore.pkcs12.legacy");
404     } else if (msg.contains("HmacPBESHA256")) {
405       if ("IBM J9 VM".equals(System.getProperty("java.vm.name"))) {
406         String vmVersion = System.getProperty("java.runtime.version", "");
407         if (vmVersion.startsWith("8.0.") && vmVersion.compareTo("8.0.8.") < 0) { // NOPMD
408           return new KnownJavaBugIOException(
409               "Bug JDK-8267701 detected -- please upgrade your Java release to at least IBM SDK 8.0.8.0; "
410                   + "details here: https://www.ibm.com/support/pages/troubleshooting-unable-open-pkcs12-keystores-due-unrecoverablekeyexception",
411               e);
412         }
413       }
414       return new KnownJavaBugIOException(
415           "Bug JDK-8267701 detected -- please upgrade your Java release; "
416               + "alternatively try running Java with -Dkeystore.pkcs12.legacy", e);
417     }
418     return e;
419   }

421   private static KnownJavaBugIOException knownJDKBug(Exception e, String bugId, String java8Version,
422       String java11Version) {
423     return knownJDKBug(e, bugId, java8Version, java11Version, null);
424   }

426   private static KnownJavaBugIOException knownJDKBug(Exception e, String bugId, String java8Version,
427       String java11Version, String hint) {
428     if (hint == null) {
429       hint = "";
430     } else {
431       hint = "; " + hint;
432     }
433     String specVersion = System.getProperty("java.specification.version", "");
434     if (specVersion.startsWith("1.")) {
435       return new KnownJavaBugIOException("Bug " + bugId + " detected -- please upgrade to Java "
436           + java8Version + " or newer" + hint, e);
437     } else if ("9".equals(specVersion) || "10".equals(specVersion) || "11".equals(specVersion)) {
438       return new KnownJavaBugIOException("Bug " + bugId + " detected -- please upgrade to Java "
439           + java11Version + " or newer" + hint, e);
440     } else {
441       return new KnownJavaBugIOException("Bug " + bugId
442           + " detected -- please upgrade your Java release" + hint, e);
443     }
444   }

447       GeneralSecurityException {
448 
449     if (trustManagerUrl == null) {
450       // use default
451       return null; // NOPMD.ReturnEmptyCollectionRatherThanNull
452     }

477   /**
478    * Try to initialize the SSLContext with a provider specified by classname.
479    *
480    * If {@code providerString} contains a comma-separated list, all providers in the list are
481    * initialized (if possible), but the first one used to initialize the {@link SSLContext}.
482    *
483    * @param e The original exception when trying to resolve the provider by ID.
484    * @param protocol The desired protocol.
485    * @param providerString The provider string with the classname(s).
486    * @return The context.
487    * @throws NoSuchAlgorithmException on error.
488    * @throws NoSuchProviderException on error.
489    */
490   @SuppressFBWarnings("DCN_NULLPOINTER_EXCEPTION")
491   private static SSLContext tryInitContextFallback(NoSuchProviderException e, String protocol,
492       String providerString) throws NoSuchAlgorithmException, NoSuchProviderException {
493     try {
494       List<Provider> providers = new ArrayList<>();
495       for (String pn : providerString.split(",")) {
496         pn = pn.trim(); // NOPMD.AvoidReassigningLoopVariables
497         if (pn.isEmpty()) {
498           continue;
499         }
500         Class<?> klazz = Class.forName(pn);
501         if (!Provider.class.isAssignableFrom(klazz)) {
502           continue;
503         }
504         Provider p = (Provider) klazz.getConstructor().newInstance();
505         providers.add(p);
506 
507         try {
508           Security.addProvider(p);
509         } catch (SecurityException | NullPointerException e1) {
510           e.addSuppressed(e1);
511         }
512       }
513       if (providers.isEmpty()) {
514         throw e;
515       } else {
516         return SSLContext.getInstance(protocol, Objects.requireNonNull(providers.get(0)).getName());
517       }
518     } catch (InstantiationException | IllegalAccessException | IllegalArgumentException
519         | InvocationTargetException | NoSuchMethodException | SecurityException
520         | ClassNotFoundException | NullPointerException e1) {
521       e.addSuppressed(e1);
522       throw e; // NOPMD.PreserveStackTrace
523     }
524   }

526   /**
527    * Builds an {@link SSLContext} using the current builder state.
528    * <p>
529    * <b>IMPORTANT:</b> Use {@link #buildAndDestroyBuilder()} to ensure sensitive information, such
530    * as passwords, are properly destroyed/removed from memory.
531    *
532    * @return The new {@link SSLContext} instance.
533    * @throws GeneralSecurityException on error.
534    * @throws IOException on error.
535    * @see #buildAndDestroyBuilder()
536    */
537   public SSLContext build() throws GeneralSecurityException, IOException {
538     SSLContext sslContext;
539     if (provider == null) {
540       sslContext = SSLContext.getInstance(protocol);
541     } else if (provider instanceof String) {
542       String providerString = (String) provider;
543       try {
544         sslContext = SSLContext.getInstance(protocol, providerString);
545       } catch (NoSuchProviderException e) {
546         sslContext = tryInitContextFallback(e, protocol, providerString);
547       }
548     } else {
549       sslContext = SSLContext.getInstance(protocol, (Provider) provider);
550     }
551 
552     SSLFunction<KeyManagerFactory, KeyManager[]> km = this.keyManager;
553     if (km == null) {
554       km = this::buildKeyManagers;
555     }
556 
557     SSLFunction<TrustManagerFactory, TrustManager[]> tm = this.trustManager;
558     if (tm == null) {
559       tm = this::buildTrustManagers;
560     }
561 
562     KeyManager[] kms = km.apply(buildKeyManagerFactory());
563     TrustManager[] tms = tm.apply(buildTrustManagerFactory());
564 
565     BuilderSSLContext.initContext(sslContext, kms, tms, secureRandom);
566 
567     return new BuilderSSLContext(clientMode, sslContext, parametersFunction, socketFactory);
568   }

570   /**
571    * Builds an {@link SSLContext} using the current builder state, and destroys the builder's state,
572    * to reduce the chance of information leakage.
573    *
574    * @return The new {@link SSLContext} instance.
575    * @throws GeneralSecurityException on error.
576    * @throws IOException on error.
577    * @throws DestroyFailedException on error.
578    * @see #destroy()
579    */
580   public SSLContext buildAndDestroyBuilder() throws GeneralSecurityException, IOException,
581       DestroyFailedException {
582     try {
583       SSLContext context = build();
584       destroy();
585       return context;
586     } catch (IOException e) {
587       throw wrapIOExceptionIfJDKBug(e);
588     }
589   }

591   /**
592    * Destroys the state of this builder and all key-/trust-related settings specified.
593    *
594    * @throws DestroyFailedException on error (thrown at the end, to increase level of destruction).
595    */
596   public void destroy() throws DestroyFailedException {
597     DestroyFailedException dfe = null;
598 
599     for (Object o : new Object[] {
600         keyStorePassword, trustManagerPassword, keyManager, trustManager}) {
601       if (o instanceof Destroyable) {
602         Destroyable d = (Destroyable) o;
603         if (!d.isDestroyed()) {
604           try {
605             d.destroy();
606           } catch (DestroyFailedException e) {
607             if (dfe == null) {
608               dfe = e;
609             } else {
610               dfe.addSuppressed(e);
611             }
612           }
613         }
614       }
615     }
616 
617     this.keyStoreUrl = null;
618     this.keyStorePassword = null;
619     this.keyManager = null;
620 
621     this.trustManagerUrl = null;
622     this.trustManagerPassword = null;
623     this.trustManager = null;
624 
625     this.parametersFunction = null;
626     this.secureRandom = null;
627 
628     if (dfe != null) {
629       throw dfe;
630     }
631   }

633   private static Provider resolveProviderIfPossible(String className) {
634     Class<?> klazz;
635     try {
636       klazz = Class.forName(className);
637     } catch (ClassNotFoundException e) {
638       return null; // NOPMD
639     }
640     try {
641       return (Provider) klazz.getConstructor().newInstance();
642     } catch (InstantiationException | IllegalAccessException | IllegalArgumentException
643         | InvocationTargetException | NoSuchMethodException | SecurityException
644         | ClassCastException e) {
645       throw new IllegalStateException("Cannot instantiate provider '" + className
646           + "', despite being on the classpath", e);
647     }
648   }
