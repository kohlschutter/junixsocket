277   @Override
278   public String toString() {
279     int port = getPort();
280 
281     byte[] bytes = getBytes();
282     if (bytes.length != (3 * 4)) {
283       return getClass().getName() + "[" + (port == 0 ? "" : "port=" + port) + ";UNKNOWN" + "]";
284     }
285 
286     ByteBuffer bb = ByteBuffer.wrap(bytes);
287     int reserved1 = bb.getInt();
288     int vsockPort = bb.getInt();
289     int cid = bb.getInt();
290 
291     String vsockPortString;
292     if (vsockPort >= -1) {
293       vsockPortString = Integer.toString(vsockPort);
294     } else {
295       vsockPortString = String.format(Locale.ENGLISH, "0x%08x", vsockPort);
296     }
297 
298     String typeString = (reserved1 == 0 ? "" : "reserved1=" + reserved1 + ";") + "vsockPort="
299         + vsockPortString + ";cid=" + cid;
300 
301     return getClass().getName() + "[" + (port == 0 ? "" : "port=" + port + ";") + typeString + "]";
302   }

337   @SuppressWarnings("cast")
338   private static byte[] toBytes(int port, int cid) {
339     ByteBuffer bb = ByteBuffer.allocate(3 * 4);
340     bb.putInt(0); // svm_reserved1
341     bb.putInt(port); // svm_port
342     bb.putInt(cid); // svm_cid
343     return (byte[]) bb.flip().array();
344   }

398   /**
399    * Returns an {@link AFVSOCKSocketAddress} for the given URI, if possible.
400    * 
401    * @param uri The URI.
402    * @param overridePort The port to forcibly use, or {@code -1} for "don't override".
403    * @return The address.
404    * @throws SocketException if the operation fails.
405    */
406   @SuppressWarnings({
407       "PMD.CognitiveComplexity", "PMD.CyclomaticComplexity", "PMD.ExcessiveMethodLength",
408       "PMD.NcssCount", "PMD.NPathComplexity", "PMD.ShortMethodName"})
409   public static AFVSOCKSocketAddress of(URI uri, int overridePort) throws SocketException {
410     switch (uri.getScheme()) {
411       case "vsock":
412       case "http+vsock":
413       case "https+vsock":
414         break;
415       default:
416         throw new SocketException("Unsupported URI scheme: " + uri.getScheme());
417     }
418 
419     String host = uri.getHost();
420     if (host == null) {
421       host = uri.getAuthority();
422       if (host != null) {
423         int at = host.indexOf('@');
424         if (at >= 0) {
425           host = host.substring(at + 1);
426         }
427       }
428     }
429     if (host == null) {
430       throw new SocketException("Cannot get hostname from URI: " + uri);
431     }
432 
433     try {
434       Matcher m = PAT_VSOCK_URI_HOST_AND_PORT.matcher(host);
435       if (!m.matches()) {
436         throw new SocketException("Invalid VSOCK URI: " + uri);
437       }
438 
439       String cidStr = m.group("cid");
440       String portStr = m.group("port");
441       String javaPortStr = m.group("javaPort");
442 
443       int cid;
444       switch (cidStr == null ? "" : cidStr) {
445         case "":
446         case "any":
447           cid = VMADDR_CID_ANY;
448           break;
449         case "hypervisor":
450           cid = VMADDR_CID_HYPERVISOR;
451           break;
452         case "local":
453           cid = VMADDR_CID_LOCAL;
454           break;
455         case "host":
456           cid = VMADDR_CID_HOST;
457           break;
458         default:
459           cid = parseInt(cidStr);
460           break;
461       }
462 
463       int port;
464       switch (portStr == null ? "" : portStr) {
465         case "any":
466         case "":
467           port = VMADDR_PORT_ANY;
468           break;
469         default:
470           port = parseInt(portStr);
471           break;
472       }
473 
474       int javaPort = overridePort != -1 ? overridePort : uri.getPort();
475       if (javaPortStr != null && !javaPortStr.isEmpty()) {
476         javaPort = parseInt(javaPortStr);
477       }
478 
479       return ofPortAndCID(javaPort, port, cid);
480     } catch (IllegalArgumentException e) {
481       throw (SocketException) new SocketException("Invalid VSOCK URI: " + uri).initCause(e);
482     }
483   }

485   @Override
486   @SuppressWarnings({"PMD.CognitiveComplexity", "PMD.CompareObjectsWithEquals"})
487   public URI toURI(String scheme, URI template) throws IOException {
488     switch (scheme) {
489       case "vsock":
490       case "http+vsock":
491       case "https+vsock":
492         break;
493       default:
494         return super.toURI(scheme, template);
495     }
496 
497     byte[] bytes = getBytes();
498     if (bytes.length != (3 * 4)) {
499       return super.toURI(scheme, template);
500     }
501 
502     StringBuilder sb = new StringBuilder();
503 
504     String portStr;
505     int port;
506     switch ((port = getVSOCKPort())) {
507       case VMADDR_PORT_ANY:
508         portStr = "any";
509         break;
510       default:
511         portStr = Integer.toUnsignedString(port);
512         break;
513     }
514 
515     sb.append(portStr);
516     sb.append('.');
517     String cidStr;
518     int cid;
519     switch ((cid = getVSOCKCID())) {
520       case VMADDR_CID_ANY:
521         cidStr = "any";
522         break;
523       case VMADDR_CID_HYPERVISOR:
524         cidStr = "hypervisor";
525         break;
526       case VMADDR_CID_LOCAL:
527         cidStr = "local";
528         break;
529       case VMADDR_CID_HOST:
530         cidStr = "host";
531         break;
532       default:
533         cidStr = Integer.toUnsignedString(cid);
534         break;
535     }
536 
537     sb.append(cidStr);
538 
539     return new HostAndPort(sb.toString(), getPort()).toURI(scheme, template);
540   }

542   private static int parseInt(String v) {
543     if (v.startsWith("0x")) {
544       return Integer.parseUnsignedInt(v.substring(2), 16);
545     } else if (v.startsWith("-")) {
546       return Integer.parseInt(v);
547     } else {
548       return Integer.parseUnsignedInt(v);
549     }
550   }

552   /**
553    * Checks if the given address could cover another address.
554    * 
555    * By default, this is only true if both addresses are regarded equal using
556    * {@link #equals(Object)}.
557    * 
558    * However, implementations may support "wildcard" addresses, and this method would compare a
559    * wildcard address against some non-wildcard address, for example.
560    * 
561    * @param covered The other address that could be covered by this address.
562    * @return {@code true} if the other address could be covered.
563    */
564   @Override
565   public boolean covers(AFSocketAddress covered) {
566     if (super.covers(covered)) {
567       return true;
568     } else if (covered instanceof AFVSOCKSocketAddress) {
569       AFVSOCKSocketAddress other = (AFVSOCKSocketAddress) covered;
570 
571       if (getVSOCKCID() == VMADDR_CID_ANY) {
572         if (getVSOCKPort() == VMADDR_PORT_ANY) {
573           return true;
574         } else {
575           return getVSOCKPort() == other.getVSOCKPort();
576         }
577       } else if (getVSOCKPort() == VMADDR_PORT_ANY) {
578         return getVSOCKCID() == other.getVSOCKCID();
579       }
580     }
581 
582     return equals(covered);
583   }
