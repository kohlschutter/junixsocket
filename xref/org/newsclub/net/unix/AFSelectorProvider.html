18  package org.newsclub.net.unix;
19  
20  import java.io.IOException;
21  import java.net.ProtocolFamily;
22  import java.net.SocketAddress;
23  import java.nio.channels.spi.AbstractSelector;
24  import java.nio.channels.spi.SelectorProvider;
25  
26  import org.eclipse.jdt.annotation.NonNull;
27  
28  /**
29   * Service-provider class for junixsocket selectors and selectable channels.
30   *
31   * @param <A> The concrete {@link AFSocketAddress} that is supported by this type.
32   */
33  public abstract class AFSelectorProvider<A extends AFSocketAddress> extends SelectorProvider {
34  
35    /**
36     * Constructs a new {@link AFSelectorProvider}.
37     */
38    protected AFSelectorProvider() {
39      super();
40    }
41  
42    /**
43     * Constructs a new pipe.
44     *
45     * @param selectable {@code true} if the pipe should be selectable.
46     * @return The pipe.
47     * @throws IOException on error.
48     */
49    private AFPipe newPipe(boolean selectable) throws IOException {
50      return new AFPipe(this, selectable);
51    }
52  
53    /**
54     * Constructs a new socket pair from two sockets.
55     *
56     * @param <Y> The type of the pair.
57     * @param s1 Some socket, the first one.
58     * @param s2 Some socket, the second one.
59     * @return The pair.
60     */
61    protected abstract <Y extends AFSomeSocket> AFSocketPair<Y> newSocketPair(Y s1, Y s2);
62  
63    /**
64     * Constructs a new socket.
65     *
66     * @return The socket instance.
67     * @throws IOException on error.
68     */
69    protected abstract AFSocket<A> newSocket() throws IOException;
70  
71    /**
72     * Returns the protocol family supported by this implementation.
73     *
74     * @return The protocol family.
75     */
76    protected abstract ProtocolFamily protocolFamily();
77  
78    /**
79     * Returns the address family supported by this implementation.
80     *
81     * @return The address family.
82     */
83    protected abstract AFAddressFamily<@NonNull A> addressFamily();
84  
85    /**
86     * Returns the domain ID for the supported protocol, as specified by {@link NativeUnixSocket}.
87     *
88     * @return The domain ID.
89     */
90    protected final int domainId() {
91      return addressFamily().getDomain();
92    }
93  
94    /**
95     * Opens a socket pair of interconnected channels.
96     *
97     * @return The new channel pair.
98     * @throws IOException on error.
99     */
100   @SuppressWarnings("resource")
101   public AFSocketPair<? extends AFSocketChannel<A>> openSocketChannelPair() throws IOException {
102     AFSocketChannel<A> s1 = openSocketChannel();
103     AFSocketChannel<A> s2 = openSocketChannel();
104 
105     NativeUnixSocket.socketPair(domainId(), NativeUnixSocket.SOCK_STREAM, s1.getAFCore().fd, s2
106         .getAFCore().fd);
107 
108     s1.socket().internalDummyConnect();
109     s2.socket().internalDummyConnect();
110 
111     return newSocketPair(s1, s2);
112   }
113 
114   /**
115    * Opens a socket pair of interconnected datagram channels.
116    *
117    * @return The new channel pair.
118    * @throws IOException on error.
119    */
120   @SuppressWarnings("resource")
121   public AFSocketPair<? extends AFDatagramChannel<A>> openDatagramChannelPair() throws IOException {
122     ProtocolFamily pf = protocolFamily();
123     AFDatagramChannel<A> s1 = openDatagramChannel(pf);
124     AFDatagramChannel<A> s2 = openDatagramChannel(pf);
125 
126     NativeUnixSocket.socketPair(domainId(), NativeUnixSocket.SOCK_STREAM, s1.getAFCore().fd, s2
127         .getAFCore().fd);
128 
129     s1.socket().internalDummyBind();
130     s2.socket().internalDummyBind();
131     s1.socket().internalDummyConnect();
132     s2.socket().internalDummyConnect();
133 
134     return newSocketPair(s1, s2);
135   }
136 
137   @Override
138   public abstract AFDatagramChannel<A> openDatagramChannel() throws IOException;
139 
140   @Override
141   public AFDatagramChannel<A> openDatagramChannel(ProtocolFamily family) throws IOException {
142     if (!protocolFamily().name().equals(family.name())) {
143       throw new UnsupportedOperationException("Unsupported protocol family");
144     }
145     return openDatagramChannel();
146   }
147 
148   @Override
149   public final AFPipe openPipe() throws IOException {
150     return newPipe(false);
151   }
152 
153   /**
154    * Opens a pipe with support for selectors.
155    *
156    * @return The new pipe
157    * @throws IOException on error.
158    */
159   final AFPipe openSelectablePipe() throws IOException {
160     return newPipe(true);
161   }
162 
163   @Override
164   public final AbstractSelector openSelector() throws IOException {
165     return new AFSelector(this);
166   }
167 
168   @Override
169   public abstract AFServerSocketChannel<A> openServerSocketChannel() throws IOException;
170 
171   /**
172    * Opens a server-socket channel bound on the given {@link SocketAddress}.
173    *
174    * @param sa The socket address to bind on.
175    * @return The new channel
176    * @throws IOException on error.
177    */
178   public abstract AFServerSocketChannel<A> openServerSocketChannel(SocketAddress sa)
179       throws IOException;
180 
181   @Override
182   public AFSocketChannel<A> openSocketChannel() throws IOException {
183     return newSocket().getChannel();
184   }
185 
186   /**
187    * Opens a socket channel connected to the given {@link SocketAddress}.
188    *
189    * @param sa The socket address to connect to.
190    * @return The new channel
191    * @throws IOException on error.
192    */
193   public abstract AFSocketChannel<A> openSocketChannel(SocketAddress sa) throws IOException;
194 }
