41  /**
42   * Provides object-oriented access to file descriptors via {@link InputStream}, {@link Socket},
43   * etc., depending on the file descriptor type.
44   * <p>
45   * Typical usage:
46   * </p>
47   * <pre><code>
48   * FileDescriptor fd;
49   * 
50   * // succeeds if fd refers to an AF_UNIX stream socket
51   * AFUNIXSocket socket = FileDescriptorCast.using(fd).as(AFUNIXSocket.class); 
52   * 
53   * // succeeds if fd refers to an AF_UNIX datagram socket
54   * AFUNIXDatagramChannel channel = FileDescriptorCast.using(fd).as(AFUNIXDatagramChannel.class); 
55   * 
56   * // always succeeds
57   * InputStream in = FileDescriptorCast.using(fd).as(InputStream.class); 
58   * OutputStream in = FileDescriptorCast.using(fd).as(OutputStream.class); 
59   * </code></pre>
60   * <p>
61   * IMPORTANT: On some platforms (e.g., Solaris, Illumos) you may need to re-apply a read timeout
62   * (e.g., using {@link Socket#setSoTimeout(int)}) after obtaining the socket.
63   * </p>
64   * <p>
65   * Note that you may also lose Java port information for {@link AFSocketAddress} implementations
66   * that do not encode this information directly (such as {@link AFUNIXSocketAddress} and
67   * {@link AFTIPCSocketAddress}).
68   * </p>
69   * 
70   * @author Christian Kohlschütter
71   */
72  public final class FileDescriptorCast implements FileDescriptorAccess {
73    private static final Map<Class<?>, CastingProviderMap> PRIMARY_TYPE_PROVIDERS_MAP = Collections
74        .synchronizedMap(new HashMap<>());
75  
76    private final FileDescriptor fdObj;
77  
78    private int localPort = 0;
79    private int remotePort = 0;
80  
81    private static final Function<FileDescriptor, FileInputStream> FD_IS_PROVIDER = System
82        .getProperty("osv.version") != null ? LenientFileInputStream::new : FileInputStream::new;
83  
84    private static final CastingProviderMap GLOBAL_PROVIDERS_FINAL = new CastingProviderMap() {
85  
86      @Override
87      protected void addProviders() {
88        // FileDescriptor and Object cannot be overridden
89        addProvider(FileDescriptor.class, new CastingProvider<FileDescriptor>() {
90          @Override
91          public FileDescriptor provideAs(FileDescriptorCast fdc,
92              Class<? super FileDescriptor> desiredType) throws IOException {
93            return fdc.getFileDescriptor();
94          }
95        });
96      }
97    };
98  
99    private static final CastingProviderMap GLOBAL_PROVIDERS = new CastingProviderMap() {
100     @Override
101     protected void addProviders() {
102       addProvider(WritableByteChannel.class, new CastingProvider<WritableByteChannel>() {
103         @SuppressWarnings("resource")
104         @Override
105         public WritableByteChannel provideAs(FileDescriptorCast fdc,
106             Class<? super WritableByteChannel> desiredType) throws IOException {
107           return new FileOutputStream(fdc.getFileDescriptor()).getChannel();
108         }
109       });
110       addProvider(ReadableByteChannel.class, new CastingProvider<ReadableByteChannel>() {
111         @SuppressWarnings("resource")
112         @Override
113         public ReadableByteChannel provideAs(FileDescriptorCast fdc,
114             Class<? super ReadableByteChannel> desiredType) throws IOException {
115           return FD_IS_PROVIDER.apply(fdc.getFileDescriptor()).getChannel();
116         }
117       });
118 
119       addProvider(FileChannel.class, new CastingProvider<FileChannel>() {
120         @Override
121         public FileChannel provideAs(FileDescriptorCast fdc, Class<? super FileChannel> desiredType)
122             throws IOException {
123           return RAFChannelProvider.getFileChannel(fdc.getFileDescriptor());
124         }
125       });
126 
127       addProvider(FileOutputStream.class, new CastingProvider<FileOutputStream>() {
128         @Override
129         public FileOutputStream provideAs(FileDescriptorCast fdc,
130             Class<? super FileOutputStream> desiredType) throws IOException {
131           return new FileOutputStream(fdc.getFileDescriptor());
132         }
133       });
134       addProvider(FileInputStream.class, new CastingProvider<FileInputStream>() {
135         @Override
136         public FileInputStream provideAs(FileDescriptorCast fdc,
137             Class<? super FileInputStream> desiredType) throws IOException {
138           return FD_IS_PROVIDER.apply(fdc.getFileDescriptor());
139         }
140       });
141 
142       if (AFSocket.supports(AFSocketCapability.CAPABILITY_FD_AS_REDIRECT)) {
143         addProvider(Redirect.class, new CastingProvider<Redirect>() {
144           @Override
145           public Redirect provideAs(FileDescriptorCast fdc, Class<? super Redirect> desiredType)
146               throws IOException {
147 
148             Redirect red = NativeUnixSocket.initRedirect(fdc.getFileDescriptor());
149             if (red == null) {
150               throw new ClassCastException("Cannot access file descriptor as " + desiredType);
151             }
152             return red;
153           }
154         });
155       }
156     }
157   };

159   private final CastingProviderMap cpm;
160 
161   private FileDescriptorCast(FileDescriptor fdObj, CastingProviderMap cpm) {
162     this.fdObj = Objects.requireNonNull(fdObj);
163     this.cpm = Objects.requireNonNull(cpm);
164   }

166   private static void registerCastingProviders(Class<?> primaryType, CastingProviderMap cpm) {
167     Objects.requireNonNull(primaryType);
168     CastingProviderMap prev;
169     if ((prev = PRIMARY_TYPE_PROVIDERS_MAP.put(primaryType, cpm)) != null) {
170       PRIMARY_TYPE_PROVIDERS_MAP.put(primaryType, prev);
171       throw new IllegalStateException("Already registered: " + primaryType);
172     }
173   }
174 
175   static <A extends AFSocketAddress> void registerCastingProviders(
176       AFAddressFamilyConfig<A> config) {
177     Class<? extends AFSocket<A>> socketClass = config.socketClass();
178     Class<? extends AFDatagramSocket<A>> datagramSocketClass = config.datagramSocketClass();
179 
180     registerCastingProviders(socketClass, new CastingProviderMap() {
181 
182       @SuppressWarnings("null")
183       @Override
184       protected void addProviders() {
185         addProviders(GLOBAL_PROVIDERS);
186 
187         final CastingProvider<AFSocket<A>> cpSocket = (fdc, desiredType) -> AFSocket.newInstance(
188             config.socketConstructor(), (AFSocketFactory<A>) null, fdc.getFileDescriptor(),
189             fdc.localPort, fdc.remotePort);
190         final CastingProvider<AFServerSocket<A>> cpServerSocket = (fdc,
191             desiredType) -> AFServerSocket.newInstance(config.serverSocketConstructor(), fdc
192                 .getFileDescriptor(), fdc.localPort, fdc.remotePort);
193 
194         addProvider(socketClass, cpSocket);
195         addProvider(config.serverSocketClass(), cpServerSocket);
196 
197         addProvider(config.socketChannelClass(), (fdc, desiredType) -> cpSocket.provideAs(fdc,
198             AFSocket.class).getChannel());
199         addProvider(config.serverSocketChannelClass(), (fdc, desiredType) -> cpServerSocket
200             .provideAs(fdc, AFServerSocket.class).getChannel());
201       }
202     });
203 
204     registerCastingProviders(datagramSocketClass, new CastingProviderMap() {
205 
206       @SuppressWarnings("null")
207       @Override
208       protected void addProviders() {
209         addProviders(GLOBAL_PROVIDERS);
210 
211         final CastingProvider<AFDatagramSocket<A>> cpDatagramSocket = (fdc,
212             desiredType) -> AFDatagramSocket.newInstance(config.datagramSocketConstructor(), fdc
213                 .getFileDescriptor(), fdc.localPort, fdc.remotePort);
214 
215         addProvider(datagramSocketClass, cpDatagramSocket);
216 
217         addProvider(config.datagramChannelClass(), (fdc, desiredType) -> cpDatagramSocket.provideAs(
218             fdc, AFDatagramSocket.class).getChannel());
219       }
220     });
221   }

223   private abstract static class CastingProviderMap {
224     private final Map<Class<?>, CastingProvider<?>> providers = new HashMap<>();
225     private final Set<Class<?>> classes = Collections.unmodifiableSet(providers.keySet());
226 
227     protected CastingProviderMap() {
228       addProviders();
229 
230       addProviders(GLOBAL_PROVIDERS_FINAL);
231     }
232 
233     protected abstract void addProviders();
234 
235     protected final <T> void addProvider(Class<T> type, CastingProvider<?> cp) {
236       Objects.requireNonNull(type);
237 
238       addProvider0(type, cp);
239     }
240 
241     private void addProvider0(Class<?> type, CastingProvider<?> cp) {
242       if (providers.put(type, cp) != cp) { // NOPMD
243         for (Class<?> cl : type.getInterfaces()) {
244           addProvider0(cl, cp);
245         }
246         Class<?> scl = type.getSuperclass();
247         if (scl != null) {
248           addProvider0(scl, cp);
249         }
250       }
251     }
252 
253     protected final void addProviders(CastingProviderMap other) {
254       if (other == null || other == this) { // NOPMD
255         return;
256       }
257       this.providers.putAll(other.providers);
258     }
259 
260     @SuppressWarnings("unchecked")
261     public <T> CastingProvider<? extends T> get(Class<T> desiredType) {
262       return (CastingProvider<? extends T>) providers.get(desiredType);
263     }
264   }
265 
266   @FunctionalInterface
267   private interface CastingProvider<T> {
268     T provideAs(FileDescriptorCast fdc, Class<? super T> desiredType) throws IOException;
269   }

271   /**
272    * Creates a {@link FileDescriptorCast} using the given file descriptor.
273    * 
274    * @param fdObj The file descriptor.
275    * @return The {@link FileDescriptorCast} instance.
276    * @throws IOException on error, especially if the given file descriptor is invalid or
277    *           unsupported.
278    */
279   public static FileDescriptorCast using(FileDescriptor fdObj) throws IOException {
280     if (!fdObj.valid()) {
281       throw new IOException("Not a valid file descriptor");
282     }
283     Class<?> primaryType = NativeUnixSocket.isLoaded() ? NativeUnixSocket.primaryType(fdObj) : null;
284     if (primaryType == null) {
285       primaryType = FileDescriptor.class;
286     }
287 
288     triggerInit();
289 
290     CastingProviderMap map = PRIMARY_TYPE_PROVIDERS_MAP.get(primaryType);
291     return new FileDescriptorCast(fdObj, map == null ? GLOBAL_PROVIDERS : map);
292   }
293 
294   private static void triggerInit() {
295     AFUNIXSocketAddress.addressFamily().getClass(); // trigger registration
296     AFTIPCSocketAddress.addressFamily().getClass(); // trigger registration
297   }

299   /**
300    * Registers the given port number as the "local port" for this file descriptor.
301    * 
302    * Important: This only changes the state of this instance. The actual file descriptor is not
303    * affected.
304    * 
305    * @param port The port to assign to (must be &gt;= 0).
306    * @return This instance.
307    */
308   public FileDescriptorCast withLocalPort(int port) {
309     if (port < 0) {
310       throw new IllegalArgumentException();
311     }
312     this.localPort = port;
313     return this;
314   }
315 
316   /**
317    * Registers the given port number as the "remote port" for this file descriptor.
318    * 
319    * Important: This only changes the state of this instance. The actual file descriptor is not
320    * affected.
321    * 
322    * @param port The port to assign to (must be &gt;= 0).
323    * @return This instance.
324    */
325   public FileDescriptorCast withRemotePort(int port) {
326     if (port < 0) {
327       throw new IllegalArgumentException();
328     }
329     this.remotePort = port;
330     return this;
331   }

333   /**
334    * Casts this instance to the desired type.
335    * 
336    * @param <K> The desired type.
337    * @param desiredType The class of the desired type.
338    * @return s An instance of the desired type.
339    * @throws IOException if there was a problem while casting.
340    * @throws ClassCastException if the cast cannot be legally made.
341    * @see #availableTypes()
342    * @see #isAvailable(Class)
343    */
344   @SuppressWarnings("PMD.ShortMethodName")
345   public @NonNull <K> K as(Class<K> desiredType) throws IOException {
346     Objects.requireNonNull(desiredType);
347 
348     CastingProvider<? extends K> provider = cpm.get(desiredType);
349     if (provider != null) {
350       K obj = desiredType.cast(provider.provideAs(this, desiredType));
351       Objects.requireNonNull(obj);
352       return obj;
353     } else {
354       throw new ClassCastException("Cannot access file descriptor as " + desiredType);
355     }
356   }
357 
358   /**
359    * Checks if the instance can be cast as the given desired type (using {@link #as(Class)}).
360    * 
361    * @param desiredType The class of the desired type.
362    * @return {@code true} if the cast can be made.
363    * @throws IOException on error.
364    * @see #as(Class)
365    */
366   public boolean isAvailable(Class<?> desiredType) throws IOException {
367     return cpm.providers.containsKey(desiredType);
368   }
369 
370   /**
371    * Returns a collection of available types this instance can be cast to (using
372    * {@link #as(Class)}).
373    * 
374    * @return The collection of available types.
375    */
376   public Set<Class<?>> availableTypes() {
377     return cpm.classes;
378   }

380   @Override
381   @SuppressFBWarnings("EI_EXPOSE_REP")
382   public FileDescriptor getFileDescriptor() {
383     return fdObj;
384   }

386   private static final class LenientFileInputStream extends FileInputStream {
387     private LenientFileInputStream(FileDescriptor fdObj) {
388       super(fdObj);
389     }
390 
391     @Override
392     public int available() throws IOException {
393       try {
394         return super.available();
395       } catch (IOException e) {
396         String msg = e.getMessage();
397         if ("Invalid seek".equals(msg)) {
398           // OSv may not like FileInputStream#availabe() on pipe fds.
399           return 0;
400         }
401         throw e;
402       }
403     }
404   }
