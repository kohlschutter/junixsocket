47  /**
48   * Provides object-oriented access to file descriptors via {@link InputStream}, {@link Socket},
49   * etc., depending on the file descriptor type.
50   * <p>
51   * Typical usage:
52   * </p>
53   * <pre><code>
54   * FileDescriptor fd;
55   *
56   * // succeeds if fd refers to an AF_UNIX stream socket
57   * AFUNIXSocket socket = FileDescriptorCast.using(fd).as(AFUNIXSocket.class);
58   *
59   * // succeeds if fd refers to an AF_UNIX datagram socket
60   * AFUNIXDatagramChannel channel = FileDescriptorCast.using(fd).as(AFUNIXDatagramChannel.class);
61   *
62   * // always succeeds
63   * InputStream in = FileDescriptorCast.using(fd).as(InputStream.class);
64   * OutputStream in = FileDescriptorCast.using(fd).as(OutputStream.class);
65   * </code></pre>
66   * <p>
67   * <b>Important notes</b>
68   * <ol>
69   * <li>On some platforms (e.g., Solaris, Illumos) you may need to re-apply a read timeout (e.g.,
70   * using {@link Socket#setSoTimeout(int)}) after obtaining the socket.</li>
71   * <li>You may lose Java port information for {@link AFSocketAddress} implementations that do not
72   * encode this information directly (such as {@link AFUNIXSocketAddress} and
73   * {@link AFTIPCSocketAddress}).</li>
74   * <li>The "blocking" state of a socket may be forcibly changed to "blocking" when performing the
75   * cast, especially when casting to {@link Socket}, {@link DatagramSocket} or {@link ServerSocket}
76   * and any of their subclasses where "blocking" is the expected state.</li>
77   * <li>When calling {@link #using(FileDescriptor)} for a {@link FileDescriptor} obtained from
78   * another socket or other resource in the same JVM (i.e., not from another process), especially for
79   * sockets provided by junixsocket itself, there is a chance that the garbage collector may clean up
80   * the original socket at an opportune moment, thereby closing the resource underlying the shared
81   * {@link FileDescriptor} prematurely.
82   * <p>
83   * This is considered an edge-case, and deliberately not handled automatically for performance and
84   * portability reasons: We would have to do additional reference counting on all FileDescriptor
85   * instances, either through patching {@code FileCleanable} or a shared data structure.
86   * <p>
87   * The issue can be prevented by keeping a reference to the original object, such as keeping it in
88   * an enclosing try-with-resources block or as a member variable, for example. Alternatively, using
89   * a "duplicate" file descriptor (via {@link #duplicating(FileDescriptor)}) circumvents this
90   * problem, at the cost of using additional system resources.</li>
91   * <li>As a consequence of the previous point: For {@link #using(FileDescriptor)}: when casting file
92   * descriptors that belong to a junixsocket-controlled sockets, the target socket is configured in a
93   * way such that garbage collection will not automatically close the target's underlying file
94   * descriptor (but still potentially any file descriptors received from other processes via
95   * ancillary messages).</li>
96   * <li>The same restrictions as for {@link #using(FileDescriptor)} apply to
97   * {@link #unsafeUsing(int)} as well.</li>
98   * </ol>
99   *
100  * @author Christian Kohlsch√ºtter
101  */
102 @SuppressWarnings("PMD.CouplingBetweenObjects")

223   private static int getFdIfPossible(FileDescriptor fd) {
224     if (!NativeUnixSocket.isLoaded()) {
225       return -1;
226     }
227     try {
228       if (!fd.valid()) {
229         return -1;
230       }
231       return NativeUnixSocket.getFD(fd);
232     } catch (IOException e) {
233       return -1;
234     }
235   }

238     Objects.requireNonNull(primaryType);
239     CastingProviderMap prev;

241       PRIMARY_TYPE_PROVIDERS_MAP.put(primaryType, prev);
242       throw new IllegalStateException("Already registered: " + primaryType);
243     }

344     protected abstract void addProviders();
345 
346     protected final <T> void addProvider(Class<T> type, CastingProvider<?> cp) {
347       Objects.requireNonNull(type);
348 
349       addProvider0(type, cp);
350     }
351 
352     private void addProvider0(Class<?> type, CastingProvider<?> cp) {
353       if (providers.put(type, cp) != cp) { // NOPMD
354         for (Class<?> cl : type.getInterfaces()) {
355           addProvider0(cl, cp);
356         }
357         Class<?> scl = type.getSuperclass();
358         if (scl != null) {
359           addProvider0(scl, cp);
360         }
361       }
362     }
363 
364     protected final void addProviders(CastingProviderMap other) {
365       if (other == null || other == this) { // NOPMD
366         return;
367       }
368       this.providers.putAll(other.providers);
369     }
370 
371     @SuppressWarnings("unchecked")
372     public <T> CastingProvider<? extends T> get(Class<T> desiredType) {
373       return (CastingProvider<? extends T>) providers.get(desiredType);
374     }
375   }

388   /**
389    * Creates a {@link FileDescriptorCast} using the given file descriptor.
390    * <p>
391    * Note that if any resource that also references this {@link FileDescriptor} is
392    * garbage-collected, the cleanup for that object may close the referenced {@link FileDescriptor},
393    * thereby resulting in premature connection losses, etc. See {@link #duplicating(FileDescriptor)}
394    * for a solution to this problem.
395    *
396    * @param fdObj The file descriptor.
397    * @return The {@link FileDescriptorCast} instance.
398    * @throws IOException on error, especially if the given file descriptor is invalid or
399    *           unsupported.
400    */
401   public static FileDescriptorCast using(FileDescriptor fdObj) throws IOException {
402     if (!fdObj.valid()) {
403       throw new IOException("Not a valid file descriptor");
404     }
405 
406     Class<?> primaryType = NativeUnixSocket.isLoaded() ? NativeUnixSocket.primaryType(fdObj) : null;
407     if (primaryType == null) {
408       primaryType = FileDescriptor.class;
409     }
410 
411     triggerInit();
412 
413     CastingProviderMap map = PRIMARY_TYPE_PROVIDERS_MAP.get(primaryType);
414     return new FileDescriptorCast(fdObj, map == null ? GLOBAL_PROVIDERS : map);
415   }
416 
417   /**
418    * Creates a {@link FileDescriptorCast} using a <em>duplicate</em> of the given file descriptor.
419    * <p>
420    * Duplicating a file descriptor is performed at the system-level, which means an additional file
421    * descriptor pointing to the same resource as the original is created by the operating system.
422    * <p>
423    * The advantage of using {@link #duplicating(FileDescriptor)} over {@link #using(FileDescriptor)}
424    * is that neither implicit garbage collection nor an explicit call to {@link Closeable#close()}
425    * on a resource owning the original {@link FileDescriptor} affects the availability of the
426    * resource from the target of the cast.
427    *
428    * @param fdObj The file descriptor to duplicate.
429    * @return The {@link FileDescriptorCast} instance.
430    * @throws IOException on error, especially if the given file descriptor is invalid or
431    *           unsupported, or if duplicating fails or is unsupported.
432    */
433   public static FileDescriptorCast duplicating(FileDescriptor fdObj) throws IOException {
434     if (!fdObj.valid()) {
435       throw new IOException("Not a valid file descriptor");
436     }
437 
438     FileDescriptor duplicate = NativeUnixSocket.duplicate(fdObj, new FileDescriptor());
439     if (duplicate == null) {
440       throw new IOException("Could not duplicate file descriptor");
441     }
442     return using(duplicate);
443   }

445   /**
446    * Creates a {@link FileDescriptorCast} using the given native file descriptor value.
447    * <p>
448    * This method is inherently unsafe as it may
449    * <ol>
450    * <li>make assumptions on the internal system representation of a file descriptor (which differs
451    * between Windows and Unix, for example).</li>
452    * <li>provide access to resources that are otherwise not accessible</li>
453    * </ol>
454    * <p>
455    * Note that attempts are made to reuse {@link FileDescriptor#in}, {@link FileDescriptor#out}, and
456    * {@link FileDescriptor#err}, respectively.
457    *
458    * @param fd The system-native file descriptor value.
459    * @return The {@link FileDescriptorCast} instance.
460    * @throws IOException on error, especially if the given file descriptor is invalid or
461    *           unsupported, or when "unsafe" operations are unavailable or manually disabled for the
462    *           current environment.
463    */
464   @Unsafe
465   public static FileDescriptorCast unsafeUsing(int fd) throws IOException {
466     AFSocket.ensureUnsafeSupported();
467 
468     FileDescriptor fdObj;
469     if (fd == -1) {
470       throw new IOException("Not a valid file descriptor");
471     } else if (fd == FD_IN) {
472       fdObj = FileDescriptor.in;
473     } else if (fd == FD_OUT) {
474       fdObj = FileDescriptor.out;
475     } else if (fd == FD_ERR) {
476       fdObj = FileDescriptor.err;
477     } else {
478       fdObj = null;
479     }
480 
481     if (fdObj != null) {
482       int check = getFdIfPossible(fdObj);
483       if (fd == check) {
484         return using(fdObj);
485       }
486     }
487 
488     fdObj = new FileDescriptor();
489     NativeUnixSocket.initFD(fdObj, fd);
490 
491     return using(fdObj);
492   }

495     for (AFAddressFamily<?> family : new AFAddressFamily<?>[] {
496         AFUNIXSocketAddress.addressFamily(), //

498         AFVSOCKSocketAddress.addressFamily(), //
499         AFSYSTEMSocketAddress.addressFamily(), //
500     }) {
501       Objects.requireNonNull(family.getClass()); // trigger init
502     }
503   }

539   /**
540    * Casts this instance to the desired type.
541    *
542    * @param <K> The desired type.
543    * @param desiredType The class of the desired type.
544    * @return s An instance of the desired type.
545    * @throws IOException if there was a problem while casting.
546    * @throws ClassCastException if the cast cannot be legally made.
547    * @see #availableTypes()
548    * @see #isAvailable(Class)
549    */
550   @SuppressWarnings("PMD.ShortMethodName")
551   public @NonNull <K> K as(Class<K> desiredType) throws IOException {
552     Objects.requireNonNull(desiredType);
553 
554     CastingProvider<? extends K> provider = cpm.get(desiredType);
555     if (provider != null) {
556       K obj = desiredType.cast(provider.provideAs(this, desiredType));
557       Objects.requireNonNull(obj);
558       return obj;
559     } else {
560       throw new ClassCastException("Cannot access file descriptor as " + desiredType);
561     }
562   }

592   private static final class LenientFileInputStream extends FileInputStream {
593     private LenientFileInputStream(FileDescriptor fdObj) {
594       super(fdObj);
595     }
596 
597     @Override
598     public int available() throws IOException {
599       try {
600         return super.available();
601       } catch (IOException e) {
602         String msg = e.getMessage();
603         if ("Invalid seek".equals(msg)) {
604           // OSv may not like FileInputStream#availabe() on pipe fds.
605           return 0;
606         }
607         throw e;
608       }
609     }
610   }

662   /**
663    * Reconfigures the Java-side of the socket/socket channel such that its state is compatible with
664    * the native socket's state. This is necessary to properly configure blocking/non-blocking state,
665    * as that is cached on the Java side.
666    * <p>
667    * If {@code isChannel} is false, then we want to cast to a {@link Socket}, {@link DatagramSocket}
668    * or {@link ServerSocket}, which means blocking I/O is desired. If the underlying native socket
669    * is configured non-blocking, we need to reset the state to "blocking" accordingly.
670    * <p>
671    * If {@code isChannel} is true, then we want to cast to a {@link SocketChannel},
672    * {@link DatagramChannel} or {@link ServerSocketChannel}, in which case the blocking state should
673    * be preserved, if possible. It is then up to the user to check blocking state via
674    * {@link AbstractSelectableChannel#isBlocking()} prior to using the socket.
675    * <p>
676    * Note that on Windows, it may be impossible to query the blocking state from an external socket,
677    * so the state is always forcibly set to "blocking".
678    *
679    * @param <S> The type.
680    * @param isChannel The desired cast type (socket=set to blocking, or channel=preserve state).
681    * @param socketChannel The channel.
682    * @throws IOException on error.
683    */
684   private static <@NonNull S extends AFSomeSocketChannel> void reconfigure(boolean isChannel,
685       S socketChannel) throws IOException {
686     if (isChannel) {
687       reconfigureKeepBlockingState(socketChannel);
688     } else {
689       reconfigureSetBlocking(socketChannel);
690     }
691   }
692 
693   private static <@NonNull S extends AFSomeSocketChannel> void reconfigureKeepBlockingState(
694       S socketChannel) throws IOException {
695     int result = NativeUnixSocket.checkBlocking(socketChannel.getFileDescriptor());
696 
697     boolean blocking;
698     switch (result) {
699       case 0:
700         blocking = false;
701         break;
702       case 1:
703         blocking = true;
704         break;
705       case 2:
706         // need to reconfigure/forcibly override any cached result -> set to blocking by default
707         socketChannel.configureBlocking(false);
708         socketChannel.configureBlocking(true);
709         return;
710       default:
711         throw new OperationNotSupportedSocketException("Invalid blocking state");
712     }
713 
714     socketChannel.configureBlocking(blocking);
715   }

718       S socketChannel) throws IOException {
719     int result = NativeUnixSocket.checkBlocking(socketChannel.getFileDescriptor());
720 
721     switch (result) {
722       case 0:
723         // see below
724         break;
725       case 1:
726         // already blocking, nothing to do
727         return;
728       case 2:
729         // need to reconfigure/forcibly override any cached result -> set to blocking by default
730         // see below
731         break;
732       default:
733         throw new OperationNotSupportedSocketException("Invalid blocking state");
734     }
735 
736     socketChannel.configureBlocking(false);
737     socketChannel.configureBlocking(true);
738   }
