133   private static SocketException handleSocketException(SocketException e, String msg)
134       throws SocketException, IOException {
135 
136     final String shortMsg;
137     if (e instanceof AddressUnavailableSocketException) {
138       shortMsg = ACCESS_DENIED;
139     } else if (e instanceof InvalidSocketException) {
140       shortMsg = KERNEL_TOO_OLD;
141     } else {
142       return e;
143     }
144 
145     if (AFVSOCKSocket.getLocalCID() == AFVSOCKSocketAddress.VMADDR_CID_HOST) {
146       throw new TestAbortedWithImportantMessageException(
147           MessageType.TEST_ABORTED_SHORT_INFORMATIONAL, msg == null || msg.isEmpty() ? shortMsg
148               : shortMsg + ": " + msg, e);
149     } else {
150       return e;
151     }
152   }
153 
154   private static SocketException handleSocketException(SocketException e, SocketAddress addr)
155       throws IOException {
156     if (!(addr instanceof AFVSOCKSocketAddress)) {
157       return e;
158     }
159     AFVSOCKSocketAddress sa = (AFVSOCKSocketAddress) addr;
160     switch (sa.getVSOCKCID()) {
161       case AFVSOCKSocketAddress.VMADDR_CID_HOST:
162       case AFVSOCKSocketAddress.VMADDR_CID_LOCAL:
163         return handleSocketException(e, "Cannot connect to addresses with CID=" + sa.getVSOCKCID());
164       default:
165         return e;
166     }
167   }
