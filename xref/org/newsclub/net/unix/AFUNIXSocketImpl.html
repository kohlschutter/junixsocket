103   protected final void finalize() {
104     try {
105       // prevent socket file descriptor leakage
106       close();
107     } catch (Throwable t) {
108       // nothing that can be done here
109     }
110 
111     // Also close all file descriptors that we've received over the wire
112     try {
113       synchronized (closeableFileDescriptors) {
114         for (FileDescriptor fd : closeableFileDescriptors.keySet()) {
115           NativeUnixSocket.close(fd);
116         }
117       }
118     } catch (Throwable t) {
119       // nothing that can be done here
120     }
121   }

199   private void unblockAccepts() {
200     while (pendingAccepts.get() > 0) {
201       try {
202         FileDescriptor tmpFd = new FileDescriptor();
203 
204         try {
205           NativeUnixSocket.connect(socketAddress.getBytes(), tmpFd, inode);
206         } catch (IOException e) {
207           // there's nothing we can do to unlock these accepts
208           return;
209         }
210         try {
211           NativeUnixSocket.shutdown(tmpFd, SHUT_RD_WR);
212         } catch (Exception e) {
213           // ignore
214         }
215         try {
216           NativeUnixSocket.close(tmpFd);
217         } catch (Exception e) {
218           // ignore
219         }
220       } catch (Exception e) {
221         // ignore
222       }
223     }
224   }

226   @Override
227   protected final synchronized void close() throws IOException {
228     boolean wasBound = bound;
229     bound = false;
230 
231     FileDescriptor fdesc = validFd();
232     if (fdesc != null) {
233       NativeUnixSocket.shutdown(fdesc, SHUT_RD_WR);
234 
235       closed = true;
236       if (wasBound && socketAddress != null && socketAddress.getBytes() != null && inode >= 0) {
237         unblockAccepts();
238       }
239 
240       NativeUnixSocket.close(fdesc);
241     }
242     closed = true;
243   }

309   private class AFUNIXInputStream extends InputStream {
310     private volatile boolean streamClosed = false;
311     private boolean eofReached = false;
312 
313     @Override
314     public int read(byte[] buf, int off, int len) throws IOException {
315       if (streamClosed) {
316         throw new IOException("This InputStream has already been closed.");
317       }
318       FileDescriptor fdesc = validFdOrException();
319       if (len == 0) {
320         return 0;
321       } else if (off < 0 || len < 0 || (len > buf.length - off)) {
322         throw new IndexOutOfBoundsException();
323       }
324 
325       return NativeUnixSocket.read(AFUNIXSocketImpl.this, fdesc, buf, off, len,
326           ancillaryReceiveBuffer);
327     }
328 
329     @Override
330     public int read() throws IOException {
331       if (eofReached) {
332         return -1;
333       }
334       final byte[] buf1 = new byte[1];
335       final int numRead = read(buf1, 0, 1);
336       if (numRead <= 0) {
337         eofReached = true;
338         return -1;
339       } else {
340         return buf1[0] & 0xFF;
341       }
342     }
343 
344     @Override
345     public synchronized void close() throws IOException {
346       streamClosed = true;
347       FileDescriptor fdesc = validFd();
348       if (fdesc != null) {
349         NativeUnixSocket.shutdown(fdesc, SHUT_RD);
350       }
351 
352       closedInputStream = true;
353       checkClose();
354     }
355 
356     @Override
357     public int available() throws IOException {
358       if (streamClosed) {
359         throw new IOException("This InputStream has already been closed.");
360       }
361       FileDescriptor fdesc = validFdOrException();
362 
363       return NativeUnixSocket.available(fdesc);
364     }
365   }

367   private class AFUNIXOutputStream extends OutputStream {
368     private volatile boolean streamClosed = false;
369 
370     @Override
371     public void write(int oneByte) throws IOException {
372       final byte[] buf1 = {(byte) oneByte};
373       write(buf1, 0, 1);
374     }
375 
376     @Override
377     public void write(byte[] buf, int off, int len) throws IOException {
378       if (streamClosed) {
379         throw new SocketException("This OutputStream has already been closed.");
380       }
381       if (len < 0 || off < 0 || len > buf.length - off) {
382         throw new IndexOutOfBoundsException();
383       }
384       FileDescriptor fdesc = validFdOrException();
385       int writtenTotal = 0;
386 
387       while (len > 0) {
388         if (Thread.interrupted()) {
389           InterruptedIOException ex = new InterruptedIOException("Thread interrupted during write");
390           ex.bytesTransferred = writtenTotal;
391           Thread.currentThread().interrupt();
392           throw ex;
393         }
394 
395         final int written = NativeUnixSocket.write(AFUNIXSocketImpl.this, fdesc, buf, off, len,
396             pendingFileDescriptors);
397         if (written < 0) {
398           throw new IOException("Unspecific error while writing");
399         }
400         len -= written;
401         off += written;
402         writtenTotal += written;
403       }
404     }
405 
406     @Override
407     public synchronized void close() throws IOException {
408       if (streamClosed) {
409         return;
410       }
411       streamClosed = true;
412       FileDescriptor fdesc = validFd();
413       if (fdesc != null) {
414         NativeUnixSocket.shutdown(fdesc, SHUT_WR);
415       }
416       closedOutputStream = true;
417       checkClose();
418     }
419   }

429   private synchronized FileDescriptor validFd() {
430     if (closed) {
431       return null;
432     }
433     FileDescriptor descriptor = this.fd;
434     if (descriptor != null) {
435       if (descriptor.valid()) {
436         return descriptor;
437       }
438     }
439     return null;

448   private static int expectInteger(Object value) throws SocketException {

468   @Override
469   public Object getOption(int optID) throws SocketException {
470     if (optID == SocketOptions.SO_REUSEADDR) {
471       return reuseAddr;
472     }
473 
474     FileDescriptor fdesc = validFdOrException();
475     try {
476       switch (optID) {
477         case SocketOptions.SO_KEEPALIVE:
478         case SocketOptions.TCP_NODELAY:
479           return NativeUnixSocket.getSocketOptionInt(fdesc, optID) != 0 ? true : false;
480         case SocketOptions.SO_TIMEOUT:
481           return Math.max(timeout, Math.max(NativeUnixSocket.getSocketOptionInt(fdesc, 0x1005),
482               NativeUnixSocket.getSocketOptionInt(fdesc, 0x1006)));
483         case SocketOptions.SO_LINGER:
484         case SocketOptions.SO_RCVBUF:
485         case SocketOptions.SO_SNDBUF:
486           return NativeUnixSocket.getSocketOptionInt(fdesc, optID);
487         default:
488           throw new SocketException("Unsupported option: " + optID);
489       }
490     } catch (final SocketException e) {
491       throw e;
492     } catch (final Exception e) {
493       throw (SocketException) new SocketException("Error while getting option").initCause(e);
494     }
495   }

497   @Override
498   public void setOption(int optID, Object value) throws SocketException {
499     if (optID == SocketOptions.SO_REUSEADDR) {
500       reuseAddr = expectBoolean(value) == 0 ? false : true;
501       return;
502     }
503 
504     FileDescriptor fdesc = validFdOrException();
505     try {
506       switch (optID) {
507         case SocketOptions.SO_LINGER:
508 
509           if (value instanceof Boolean) {
510             final boolean b = (Boolean) value;
511             if (b) {
512               throw new SocketException("Only accepting Boolean.FALSE here");
513             }
514             NativeUnixSocket.setSocketOptionInt(fdesc, optID, -1);
515             return;
516           }
517           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
518           return;
519         case SocketOptions.SO_TIMEOUT:
520           this.timeout = expectInteger(value);
521           NativeUnixSocket.setSocketOptionInt(fdesc, 0x1005, timeout);
522           NativeUnixSocket.setSocketOptionInt(fdesc, 0x1006, timeout);
523           return;
524         case SocketOptions.SO_RCVBUF:
525         case SocketOptions.SO_SNDBUF:
526           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
527           return;
528         case SocketOptions.SO_KEEPALIVE:
529         case SocketOptions.TCP_NODELAY:
530           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectBoolean(value));
531           return;
532         default:
533           throw new SocketException("Unsupported option: " + optID);
534       }
535     } catch (final SocketException e) {
536       throw e;
537     } catch (final Exception e) {
538       throw (SocketException) new SocketException("Error while setting option").initCause(e);
539     }
540   }

616   public final FileDescriptor[] getReceivedFileDescriptors() {
617     if (receivedFileDescriptors.isEmpty()) {
618       return null;
619     }
620     List<FileDescriptor[]> copy = new ArrayList<>(receivedFileDescriptors);
621     if (copy.isEmpty()) {
622       return null;
623     }
624     receivedFileDescriptors.removeAll(copy);
625     int count = 0;
626     for (FileDescriptor[] fds : copy) {
627       count += fds.length;
628     }
629     if (count == 0) {
630       return null;
631     }
632     FileDescriptor[] oneArray = new FileDescriptor[count];
633     int offset = 0;
634     for (FileDescriptor[] fds : copy) {
635       System.arraycopy(fds, 0, oneArray, offset, fds.length);
636       offset += fds.length;
637     }
638     return oneArray;
639   }

645   // called from native code
646   final void receiveFileDescriptors(int[] fds) throws IOException {
647     if (fds == null || fds.length == 0) {
648       return;
649     }
650     final int fdsLength = fds.length;
651     FileDescriptor[] descriptors = new FileDescriptor[fdsLength];
652     for (int i = 0; i < fdsLength; i++) {
653       final FileDescriptor fdesc = new FileDescriptor();
654       NativeUnixSocket.initFD(fdesc, fds[i]);
655       descriptors[i] = fdesc;
656 
657       closeableFileDescriptors.put(fdesc, fds[i]);
658 
659       @SuppressWarnings("resource")
660       final Closeable cleanup = new Closeable() {
661 
662         @Override
663         public void close() throws IOException {
664           closeableFileDescriptors.remove(fdesc);
665         }
666       };
667       NativeUnixSocket.attachCloseable(fdesc, cleanup);
668     }
669 
670     this.receivedFileDescriptors.add(descriptors);
671   }
