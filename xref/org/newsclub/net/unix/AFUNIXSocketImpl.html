190   private final class AFUNIXInputStream extends InputStream {
191     private volatile boolean streamClosed = false;
192 
193     @Override
194     public int read(byte[] buf, int off, int len) throws IOException {
195       if (streamClosed) {
196         throw new IOException("This InputStream has already been closed.");
197       }
198       FileDescriptor fdesc = validFdOrException();
199       if (len == 0) {
200         return 0;
201       } else if (off < 0 || len < 0 || (len > buf.length - off)) {
202         throw new IndexOutOfBoundsException();
203       }
204 
205       try {
206         return NativeUnixSocket.read(fdesc, buf, off, len);
207       } catch (final IOException e) {
208         throw (IOException) new IOException(e.getMessage() + " at " + AFUNIXSocketImpl.this
209             .toString()).initCause(e);
210       }
211     }
212 
213     @Override
214     public int read() throws IOException {
215       final byte[] buf1 = new byte[1];
216       final int numRead = read(buf1, 0, 1);
217       if (numRead <= 0) {
218         return -1;
219       } else {
220         return buf1[0] & 0xFF;
221       }
222     }
223 
224     @Override
225     public synchronized void close() throws IOException {
226       streamClosed = true;
227       FileDescriptor fdesc = validFd();
228       if (fdesc != null) {
229         NativeUnixSocket.shutdown(fdesc, SHUT_RD);
230       }
231 
232       closedInputStream = true;
233       checkClose();
234     }
235 
236     @Override
237     public int available() throws IOException {
238       if (streamClosed) {
239         throw new IOException("This InputStream has already been closed.");
240       }
241       FileDescriptor fdesc = validFdOrException();
242 
243       return NativeUnixSocket.available(fdesc);
244     }
245   }

247   private final class AFUNIXOutputStream extends OutputStream {
248     private volatile boolean streamClosed = false;
249 
250     @Override
251     public void write(int oneByte) throws IOException {
252       final byte[] buf1 = new byte[] {(byte) oneByte};
253       write(buf1, 0, 1);
254     }
255 
256     @Override
257     public void write(byte[] buf, int off, int len) throws IOException {
258       if (streamClosed) {
259         throw new SocketException("This OutputStream has already been closed.");
260       }
261       if (len < 0 || off < 0 || len > buf.length - off) {
262         throw new IndexOutOfBoundsException();
263       }
264       FileDescriptor fdesc = validFdOrException();
265       int writtenTotal = 0;
266       try {
267         while (len > 0) {
268           if (Thread.interrupted()) {
269             InterruptedIOException ex = new InterruptedIOException(
270                 "Thread interrupted during write");
271             ex.bytesTransferred = writtenTotal;
272             Thread.currentThread().interrupt();
273             throw ex;
274           }
275           final int written = NativeUnixSocket.write(fdesc, buf, off, len);
276           if (written < 0) {
277             throw new IOException("Unspecific error while writing");
278           }
279           len -= written;
280           off += written;
281           writtenTotal += written;
282         }
283       } catch (final IOException e) {
284         throw (IOException) new IOException(e.getMessage() + " at " + AFUNIXSocketImpl.this
285             .toString()).initCause(e);
286       }
287     }
288 
289     @Override
290     public synchronized void close() throws IOException {
291       if (streamClosed) {
292         return;
293       }
294       streamClosed = true;
295       FileDescriptor fdesc = validFd();
296       if (fdesc != null) {
297         NativeUnixSocket.shutdown(fdesc, SHUT_WR);
298       }
299       closedOutputStream = true;
300       checkClose();
301     }
302   }

304   private FileDescriptor validFdOrException() throws SocketException {
305     FileDescriptor fdesc = validFd();
306     if (fdesc == null) {
307       throw new SocketException("Not open");
308     }
309     return fdesc;
310   }
311 
312   private synchronized FileDescriptor validFd() {
313     if (closed) {
314       return null;
315     }
316     FileDescriptor descriptor = this.fd;
317     if (descriptor != null) {
318       if (descriptor.valid()) {
319         return descriptor;
320       }
321     }
322     return null;
323   }

331   private static int expectInteger(Object value) throws SocketException {
332     try {
333       return (Integer) value;
334     } catch (final ClassCastException e) {
335       throw (SocketException) new SocketException("Unsupported value: " + value).initCause(e);
336     } catch (final NullPointerException e) {
337       throw (SocketException) new SocketException("Value must not be null").initCause(e);
338     }
339   }

341   private static int expectBoolean(Object value) throws SocketException {
342     try {
343       return ((Boolean) value).booleanValue() ? 1 : 0;
344     } catch (final ClassCastException e) {
345       throw (SocketException) new SocketException("Unsupported value: " + value).initCause(e);
346     } catch (final NullPointerException e) {
347       throw (SocketException) new SocketException("Value must not be null").initCause(e);
348     }

351   @Override
352   public Object getOption(int optID) throws SocketException {
353     FileDescriptor fdesc = validFdOrException();
354     try {
355       switch (optID) {
356         case SocketOptions.SO_KEEPALIVE:
357         case SocketOptions.TCP_NODELAY:
358           return NativeUnixSocket.getSocketOptionInt(fdesc, optID) != 0 ? true : false;
359         case SocketOptions.SO_LINGER:
360         case SocketOptions.SO_TIMEOUT:
361         case SocketOptions.SO_RCVBUF:
362         case SocketOptions.SO_SNDBUF:
363           return NativeUnixSocket.getSocketOptionInt(fdesc, optID);
364         default:
365           throw new SocketException("Unsupported option: " + optID);
366       }
367     } catch (final SocketException e) {
368       throw e;
369     } catch (final Exception e) {
370       throw (SocketException) new SocketException("Error while getting option").initCause(e);
371     }
372   }

374   @Override
375   public void setOption(int optID, Object value) throws SocketException {
376     FileDescriptor fdesc = validFdOrException();
377     try {
378       switch (optID) {
379         case SocketOptions.SO_LINGER:
380 
381           if (value instanceof Boolean) {
382             final boolean b = (Boolean) value;
383             if (b) {
384               throw new SocketException("Only accepting Boolean.FALSE here");
385             }
386             NativeUnixSocket.setSocketOptionInt(fdesc, optID, -1);
387             return;
388           }
389           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
390           return;
391         case SocketOptions.SO_RCVBUF:
392         case SocketOptions.SO_SNDBUF:
393         case SocketOptions.SO_TIMEOUT:
394           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectInteger(value));
395           return;
396         case SocketOptions.SO_KEEPALIVE:
397         case SocketOptions.TCP_NODELAY:
398           NativeUnixSocket.setSocketOptionInt(fdesc, optID, expectBoolean(value));
399           return;
400         default:
401           throw new SocketException("Unsupported option: " + optID);
402       }
403     } catch (final SocketException e) {
404       throw e;
405     } catch (final Exception e) {
406       throw (SocketException) new SocketException("Error while setting option").initCause(e);
407     }
408   }

426   /**
427    * Changes the behavior to be somewhat lenient with respect to the specification.
428    * 
429    * In particular, we ignore calls to {@link Socket#getTcpNoDelay()} and
430    * {@link Socket#setTcpNoDelay(boolean)}.
431    */
432   static class Lenient extends AFUNIXSocketImpl {
433     Lenient() {
434       super();
435     }
436 
437     @Override
438     public void setOption(int optID, Object value) throws SocketException {
439       try {
440         super.setOption(optID, value);
441       } catch (SocketException e) {
442         switch (optID) {
443           case SocketOptions.TCP_NODELAY:
444             return;
445           default:
446             throw e;
447         }
448       }
449     }
450 
451     @Override
452     public Object getOption(int optID) throws SocketException {
453       try {
454         return super.getOption(optID);
455       } catch (SocketException e) {
456         switch (optID) {
457           case SocketOptions.TCP_NODELAY:
458           case SocketOptions.SO_KEEPALIVE:
459             return false;
460           default:
461             throw e;
462         }
463       }
464     }
