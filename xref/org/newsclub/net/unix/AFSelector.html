47    private final Set<AFSelectionKey> keysRegistered = ConcurrentHashMap.newKeySet();
48    private final Set<SelectionKey> keysRegisteredPublic = Collections.unmodifiableSet(

51    private final SelectionKeySet keysSelectedPublic = new SelectionKeySet();

56      super(provider);

62    @Override
63    protected SelectionKey register(AbstractSelectableChannel ch, int ops, Object att) {
64      AFSelectionKey key = new AFSelectionKey(this, ch, ops, att);
65      synchronized (this) {
66        pollFd = null;
67        keysRegistered.add(key);
68      }
69      return key;

87    @Override
88    public int select(long timeout) throws IOException {
89      if (timeout > Integer.MAX_VALUE) {
90        timeout = Integer.MAX_VALUE;
91      } else if (timeout < 0) {
92        throw new IllegalArgumentException("Timeout must not be negative");
93      }
94  
95      return select0((int) timeout);
96    }

98    @Override
99    public int select() throws IOException {
100     try {
101       return select0(-1);
102     } catch (SocketTimeoutException e) {
103       return 0;
104     }
105   }

107   @SuppressWarnings("PMD.CognitiveComplexity")
108   private int select0(int timeout) throws IOException {
109     PollFd pfd;
110     synchronized (this) {
111       if (!isOpen()) {
112         throw new ClosedSelectorException();
113       }
114       pfd = pollFd = initPollFd(pollFd);
115       numKeysSelected.set(0);
116     }
117     int num;
118     try {
119       begin();
120       num = NativeUnixSocket.poll(pfd, timeout);
121     } finally {
122       end();
123     }
124     synchronized (this) {
125       numKeysSelected.set(0);
126       pfd = pollFd;
127       if (pfd != null) {
128         AFSelectionKey[] keys = pfd.keys;
129         if (keys != null) {
130           for (AFSelectionKey key : keys) {
131             if (key != null && key.hasOpInvalid()) {
132               SelectableChannel ch = key.channel();
133               if (ch != null && ch.isOpen()) {
134                 ch.close();
135               }
136             }
137           }
138         }
139       }
140       if (num > 0) {
141         consumeAllBytesAfterPoll();
142         setOpsReady(pfd); // updates keysSelected and numKeysSelected
143       }
144       return numKeysSelected.get();
145     }
146   }
147 
148   private synchronized void consumeAllBytesAfterPoll() throws IOException {
149     if (pollFd == null) {
150       return;
151     }
152     if ((pollFd.rops[0] & SelectionKey.OP_READ) == 0) {
153       return;
154     }
155     int maxReceive;
156     int bytesReceived;
157 
158     int options = selectorPipe.getOptions();
159 
160     synchronized (pipeMsgReceiveBuffer) {
161       pipeMsgReceiveBuffer.clear();
162       maxReceive = pipeMsgReceiveBuffer.remaining();
163       bytesReceived = NativeUnixSocket.receive(pollFd.fds[0], pipeMsgReceiveBuffer, 0, maxReceive,
164           null, options, null, 1);
165     }
166 
167     if (bytesReceived == maxReceive && maxReceive > 0) {
168       // consume all pending bytes
169       int read;
170       do {
171         if ((read = NativeUnixSocket.poll(selectorPipePollFd, 0)) > 0) {
172           synchronized (pipeMsgReceiveBuffer) {
173             pipeMsgReceiveBuffer.clear();
174             read = NativeUnixSocket.receive(selectorPipePollFd.fds[0], pipeMsgReceiveBuffer, 0,
175                 maxReceive, null, options, null, 1);
176           }
177         }
178       } while (read == maxReceive && read > 0);
179     }
180   }

182   private synchronized void setOpsReady(PollFd pfd) {
183     int numSelected = 0;
184     if (pfd != null) {
185       for (int i = 1; i < pfd.rops.length; i++) {
186         int rops = pfd.rops[i];
187         AFSelectionKey key = pfd.keys[i];
188         key.setOpsReady(rops);
189         if (rops != 0 && !key.isRemovedFromSelected() && key.isValid()) {
190           numSelected++;
191         }
192       }
193     }
194     numKeysSelected.set(numSelected);
195   }
196 
197   @SuppressWarnings("resource")
198   private PollFd initPollFd(PollFd existingPollFd) throws IOException {
199     synchronized (this) {
200       for (Iterator<AFSelectionKey> it = keysRegistered.iterator(); it.hasNext();) {
201         AFSelectionKey key = it.next();
202         key.setOpsReady(0);
203         if (!key.getAFCore().fd.valid()) {
204           key.removeFromSelected(true);
205           key.cancelNoRemove();
206           it.remove();
207           existingPollFd = null;
208         } else {
209           key.removeFromSelected(false);
210         }
211       }
212 
213       if (existingPollFd != null && //
214           existingPollFd.keys != null && //
215           (existingPollFd.keys.length - 1) == keysRegistered.size()) {
216         boolean needsUpdate = false;
217         int i = 1;
218         for (AFSelectionKey key : keysRegistered) {
219           if (existingPollFd.keys[i] != key || !key.isValid()) { // NOPMD
220             needsUpdate = true;
221             break;
222           }
223           existingPollFd.ops[i] = key.interestOps();
224 
225           i++;
226         }
227 
228         if (!needsUpdate) {
229           return existingPollFd;
230         }
231       }
232 
233       int size = keysRegistered.size() + 1;
234       FileDescriptor[] fds = new FileDescriptor[size];
235       int[] ops = new int[size];
236 
237       AFSelectionKey[] keys = new AFSelectionKey[size];
238       fds[0] = selectorPipe.sourceFD();
239       ops[0] = SelectionKey.OP_READ;
240 
241       int i = 1;
242       for (AFSelectionKey key : keysRegistered) {
243         keys[i] = key;
244         fds[i] = key.getAFCore().fd;
245         ops[i] = key.interestOps();
246         i++;
247       }
248       return new PollFd(keys, fds, ops);
249     }
250   }

252   @Override
253   protected void implCloseSelector() throws IOException {
254     wakeup();
255     Set<SelectionKey> keys;
256     synchronized (this) {
257       keys = keys();
258       keysRegistered.clear();
259     }
260     for (SelectionKey key : keys) {
261       ((AFSelectionKey) key).cancelNoRemove();
262     }
263     selectorPipe.close();
264   }

266   @Override
267   public Selector wakeup() {
268     if (isOpen()) {
269       try {
270         synchronized (pipeMsgWakeUp) {
271           pipeMsgWakeUp.clear();
272           try {
273             selectorPipe.sink().write(pipeMsgWakeUp);
274           } catch (SocketException e) {
275             if (selectorPipe.sinkFD().valid()) {
276               throw e;
277             } else {
278               // ignore (Broken pipe, etc)
279             }
280           }
281         }
282       } catch (IOException e) { // NOPMD.ExceptionAsFlowControl
283         // FIXME throw as runtimeexception?
284         StackTraceUtil.printStackTrace(e);
285       }
286     }
287     return this;
288   }
289 
290   synchronized void remove(AFSelectionKey key) {
291     key.removeFromSelected(true);
292     deregister(key);
293     pollFd = null;
294   }
295 
296   private void deregister(AFSelectionKey key) {
297     // super.deregister unnecessarily casts SelectionKey to AbstractSelectionKey, and
298     // ((AbstractSelectableChannel)key.channel()).removeKey(key); is not visible.
299     // so we have to resort to some JNI trickery...
300     try {
301       NativeUnixSocket.deregisterSelectionKey((AbstractSelectableChannel) key.channel(), key);
302     } catch (ClassCastException e) {
303       // because our key isn't an AbstractSelectableKey, internal invalidation fails
304       // but at that point, the key is deregistered
305     }
306   }

308   static final class PollFd {
309     // accessed from native code
310     final FileDescriptor[] fds;
311     // accessed from native code
312     final int[] ops;
313     // accessed from native code
314     final int[] rops;
315 
316     final AFSelectionKey[] keys;
317 
318     PollFd(FileDescriptor pipeSourceFd) {
319       this(pipeSourceFd, SelectionKey.OP_READ);
320     }
321 
322     PollFd(FileDescriptor pipeSourceFd, int op) {
323       this.fds = new FileDescriptor[] {pipeSourceFd};
324       this.ops = new int[] {op};
325       this.rops = new int[1];
326       this.keys = null;
327     }
328 
329     @SuppressWarnings("PMD.ArrayIsStoredDirectly")
330     PollFd(AFSelectionKey[] keys, FileDescriptor[] fds, int[] ops) {
331       this.keys = keys;
332       if (fds.length != ops.length) {
333         throw new IllegalStateException();
334       }
335       this.fds = fds;
336       this.ops = ops;
337       this.rops = new int[ops.length];
338     }
339   }

341   private final class SelectionKeySet extends AbstractSet<SelectionKey> {
342     SelectionKeySet() {
343       super();
344     }
345 
346     @Override
347     public boolean add(SelectionKey e) {
348       throw new UnsupportedOperationException();
349     }
350 
351     @Override
352     public boolean addAll(Collection<? extends SelectionKey> c) {
353       throw new UnsupportedOperationException();
354     }
355 
356     @Override
357     public boolean remove(Object o) {
358       if (!(o instanceof AFSelectionKey)) {
359         return false;
360       }
361       AFSelectionKey key = (AFSelectionKey) o;
362       if (!keysRegistered.contains(key) || key.isRemovedFromSelected()) {
363         return false;
364       }
365       key.setOpsReady(0);
366       key.removeFromSelected(true);
367       return true;
368     }
369 
370     @Override
371     public Iterator<SelectionKey> iterator() {
372       if (isEmpty()) {
373         return Collections.emptyIterator();
374       }
375       Iterator<AFSelectionKey> it = keysRegistered.iterator();
376       return new Iterator<SelectionKey>() {
377         AFSelectionKey currentKey = null;
378         AFSelectionKey nextKey = next0();
379 
380         @Override
381         public boolean hasNext() {
382           return nextKey != null;
383         }
384 
385         private AFSelectionKey next0() {
386           while (it.hasNext()) {
387             AFSelectionKey key = it.next();
388             if (key.isSelected() && !key.isRemovedFromSelected()) {
389               return key;
390             }
391           }
392           return null;
393         }
394 
395         @Override
396         public SelectionKey next() {
397           if (nextKey == null) {
398             throw new NoSuchElementException();
399           }
400           currentKey = nextKey;
401           nextKey = next0();
402           return currentKey;
403         }
404 
405         @Override
406         public void remove() {
407           if (currentKey != null) {
408             SelectionKeySet.this.remove(currentKey);
409             currentKey = null;
410           }
411         }
412       };
413     }
414 
415     @Override
416     public int size() {
417       return numKeysSelected.get();
418     }
419 
420     @Override
421     public boolean isEmpty() {
422       return size() == 0;
423     }
424 
425     @Override
426     public boolean contains(Object o) {
427       if (!(o instanceof AFSelectionKey)) {
428         return false;
429       }
430       AFSelectionKey key = (AFSelectionKey) o;
431       return keysRegistered.contains(key) && !key.isCancelled();
432     }
433 
434     @Override
435     public boolean containsAll(Collection<?> c) {
436       for (Object o : c) {
437         if (!contains(o)) {
438           return false;
439         }
440       }
441       return true;
442     }
443 
444     @Override
445     public void clear() {
446       for (AFSelectionKey key : keysRegistered) {
447         if (key.isSelected()) {
448           key.setOpsReady(0);
449           key.removeFromSelected(true);
450         }
451       }
452     }
453   }
