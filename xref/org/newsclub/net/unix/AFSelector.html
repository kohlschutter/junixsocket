38  final class AFSelector extends AbstractSelector {
39    private final AFPipe selectorPipe;
40    private final PollFd selectorPipePollFd;
41  
42    private final ByteBuffer pipeMsgWakeUp = ByteBuffer.allocate(1);
43    private final ByteBuffer pipeMsgReceiveBuffer = ByteBuffer.allocateDirect(256);
44  
45    private final Map<AFSelectionKey, Integer> keysRegistered = new ConcurrentHashMap<>();
46    private final Set<AFSelectionKey> keysRegisteredKeySet = keysRegistered.keySet();
47    private final Set<SelectionKey> keysRegisteredPublic = Collections.unmodifiableSet(
48        keysRegisteredKeySet);
49  
50    private final AtomicInteger selectCount = new AtomicInteger(0);
51    private final MapValueSet<SelectionKey, Integer> selectedKeysSet =
52        new MapValueSet<SelectionKey, Integer>(keysRegistered, selectCount::get, 0);
53    private final Set<SelectionKey> selectedKeysPublic = new UngrowableSet<>(selectedKeysSet);
54  
55    private PollFd pollFd = null;

57    AFSelector(AFSelectorProvider<?> provider) throws IOException {
58      super(provider);
59  
60      this.selectorPipe = AFUNIXSelectorProvider.getInstance().openSelectablePipe();
61      this.selectorPipePollFd = new PollFd(selectorPipe.sourceFD());
62    }

64    @Override
65    protected SelectionKey register(AbstractSelectableChannel ch, int ops, Object att) {
66      AFSelectionKey key = new AFSelectionKey(this, ch, ops, att);
67      synchronized (this) {
68        pollFd = null;
69        selectedKeysSet.markRemoved(key);
70      }
71      return key;
72    }

76      return keysRegisteredPublic;

79    @Override
80    public Set<SelectionKey> selectedKeys() {
81      return selectedKeysPublic;

84    @Override
85    public int selectNow() throws IOException {
86      return select0(0);

89    @Override
90    public int select(long timeout) throws IOException {
91      if (timeout > Integer.MAX_VALUE) {
92        timeout = Integer.MAX_VALUE;
93      } else if (timeout < 0) {
94        throw new IllegalArgumentException("Timeout must not be negative");
95      }
96  
97      return select0((int) timeout);
98    }

100   @Override
101   public int select() throws IOException {
102     try {
103       return select0(-1);
104     } catch (SocketTimeoutException e) {
105       return 0;
106     }
107   }

109   @SuppressWarnings("PMD.CognitiveComplexity")
110   private int select0(int timeout) throws IOException {
111     PollFd pfd;
112 
113     int selectId = updateSelectCount();
114 
115     synchronized (this) {
116       if (!isOpen()) {
117         throw new ClosedSelectorException();
118       }
119 
120       pfd = pollFd = initPollFd(pollFd);
121     }
122     int num;
123     try {
124       begin();
125       num = NativeUnixSocket.poll(pfd, timeout);
126     } finally {
127       end();
128     }
129     synchronized (this) {
130       pfd = pollFd;
131       if (pfd != null) {
132         AFSelectionKey[] keys = pfd.keys;
133         if (keys != null) {
134           for (AFSelectionKey key : keys) {
135             if (key != null && key.hasOpInvalid()) {
136               SelectableChannel ch = key.channel();
137               if (ch != null && ch.isOpen()) {
138                 ch.close();
139               }
140             }
141           }
142         }
143       }
144       if (num > 0) {
145         consumeAllBytesAfterPoll();
146         setOpsReady(pfd, selectId); // updates keysSelected and numKeysSelected
147       }
148       return selectedKeysSet.size();
149     }
150   }
151 
152   private synchronized void consumeAllBytesAfterPoll() throws IOException {
153     if (pollFd == null) {
154       return;
155     }
156     if ((pollFd.rops[0] & SelectionKey.OP_READ) == 0) {
157       return;
158     }
159     int maxReceive;
160     int bytesReceived;
161 
162     int options = selectorPipe.getOptions();
163 
164     synchronized (pipeMsgReceiveBuffer) {
165       pipeMsgReceiveBuffer.clear();
166       maxReceive = pipeMsgReceiveBuffer.remaining();
167       bytesReceived = NativeUnixSocket.receive(pollFd.fds[0], pipeMsgReceiveBuffer, 0, maxReceive,
168           null, options, null, 1);
169     }
170 
171     if (bytesReceived == maxReceive && maxReceive > 0) {
172       // consume all pending bytes
173       int read;
174       do {
175         if ((read = NativeUnixSocket.poll(selectorPipePollFd, 0)) > 0) {
176           synchronized (pipeMsgReceiveBuffer) {
177             pipeMsgReceiveBuffer.clear();
178             read = NativeUnixSocket.receive(selectorPipePollFd.fds[0], pipeMsgReceiveBuffer, 0,
179                 maxReceive, null, options, null, 1);
180           }
181         }
182       } while (read == maxReceive && read > 0);
183     }
184   }

186   private int updateSelectCount() {
187     int selectId = selectCount.incrementAndGet();
188     if (selectId == 0) {
189       // overflow (unlikely)
190       selectedKeysSet.markAllRemoved();
191       selectId = selectCount.incrementAndGet();
192     }
193     return selectId;
194   }
195 
196   private void setOpsReady(PollFd pfd, int selectId) {
197     if (pfd != null) {
198       for (int i = 1; i < pfd.rops.length; i++) {
199         int rops = pfd.rops[i];
200         AFSelectionKey key = pfd.keys[i];
201         key.setOpsReady(rops);
202         if (rops != 0 && keysRegistered.containsKey(key)) {
203           keysRegistered.put(key, selectId);
204         }
205       }
206     }
207   }
208 
209   @SuppressWarnings({"resource", "PMD.CognitiveComplexity"})
210   private PollFd initPollFd(PollFd existingPollFd) throws IOException {
211     synchronized (this) {
212       for (Iterator<AFSelectionKey> it = keysRegisteredKeySet.iterator(); it.hasNext();) {
213         AFSelectionKey key = it.next();
214         if (!key.getAFCore().fd.valid() || !key.isValid()) {
215           key.cancelNoRemove();
216           it.remove();
217           existingPollFd = null;
218         } else {
219           key.setOpsReady(0);
220         }
221       }
222 
223       if (existingPollFd != null && //
224           existingPollFd.keys != null && //
225           (existingPollFd.keys.length - 1) == keysRegistered.size()) {
226         boolean needsUpdate = false;
227         int i = 1;
228         for (AFSelectionKey key : keysRegisteredKeySet) {
229           if (existingPollFd.keys[i] != key || !key.isValid()) { // NOPMD
230             needsUpdate = true;
231             break;
232           }
233           existingPollFd.ops[i] = key.interestOps();
234 
235           i++;
236         }
237 
238         if (!needsUpdate) {
239           return existingPollFd;
240         }
241       }
242 
243       int keysToPoll = keysRegistered.size();
244       for (AFSelectionKey key : keysRegisteredKeySet) {
245         if (!key.isValid()) {
246           keysToPoll--;
247         }
248       }
249 
250       int size = keysToPoll + 1;
251       FileDescriptor[] fds = new FileDescriptor[size];
252       int[] ops = new int[size];
253 
254       AFSelectionKey[] keys = new AFSelectionKey[size];
255       fds[0] = selectorPipe.sourceFD();
256       ops[0] = SelectionKey.OP_READ;
257 
258       int i = 1;
259       for (AFSelectionKey key : keysRegisteredKeySet) {
260         if (!key.isValid()) {
261           continue;
262         }
263         keys[i] = key;
264         fds[i] = key.getAFCore().fd;
265         ops[i] = key.interestOps();
266         i++;
267       }
268       return new PollFd(keys, fds, ops);
269     }
270   }

272   @Override
273   protected void implCloseSelector() throws IOException {
274     wakeup();
275     Set<SelectionKey> keys;
276     synchronized (this) {
277       keys = keys();
278       keysRegistered.clear();
279     }
280     for (SelectionKey key : keys) {
281       ((AFSelectionKey) key).cancelNoRemove();
282     }
283     selectorPipe.close();
284   }

286   @Override
287   public Selector wakeup() {
288     if (isOpen()) {
289       try {
290         synchronized (pipeMsgWakeUp) {
291           pipeMsgWakeUp.clear();
292           try {
293             selectorPipe.sink().write(pipeMsgWakeUp);
294           } catch (SocketException e) {
295             if (selectorPipe.sinkFD().valid()) {
296               throw e;
297             } else {
298               // ignore (Broken pipe, etc)
299             }
300           }
301         }
302       } catch (IOException e) { // NOPMD.ExceptionAsFlowControl
303         // FIXME throw as runtimeexception?
304         StackTraceUtil.printStackTrace(e);
305       }
306     }
307     return this;
308   }
309 
310   synchronized void remove(AFSelectionKey key) {
311     selectedKeysSet.remove(key);
312     deregister(key);
313     pollFd = null;
314   }
315 
316   private void deregister(AFSelectionKey key) {
317     // super.deregister unnecessarily casts SelectionKey to AbstractSelectionKey, and
318     // ((AbstractSelectableChannel)key.channel()).removeKey(key); is not visible.
319     // so we have to resort to some JNI trickery...
320     try {
321       NativeUnixSocket.deregisterSelectionKey((AbstractSelectableChannel) key.channel(), key);
322     } catch (ClassCastException e) {
323       // because our key isn't an AbstractSelectableKey, internal invalidation fails
324       // but at that point, the key is deregistered
325     }
326   }

328   static final class PollFd {
329     // accessed from native code
330     final FileDescriptor[] fds;
331     // accessed from native code
332     final int[] ops;
333     // accessed from native code
334     final int[] rops;
335 
336     final AFSelectionKey[] keys;
337 
338     PollFd(FileDescriptor pipeSourceFd) {
339       this(pipeSourceFd, SelectionKey.OP_READ);
340     }
341 
342     PollFd(FileDescriptor pipeSourceFd, int op) {
343       this.fds = new FileDescriptor[] {pipeSourceFd};
344       this.ops = new int[] {op};
345       this.rops = new int[1];
346       this.keys = null;
347     }
348 
349     @SuppressWarnings("PMD.ArrayIsStoredDirectly")
350     PollFd(AFSelectionKey[] keys, FileDescriptor[] fds, int[] ops) {
351       this.keys = keys;
352       if (fds.length != ops.length) {
353         throw new IllegalStateException();
354       }
355       this.fds = fds;
356       this.ops = ops;
357       this.rops = new int[ops.length];
358     }
359   }
