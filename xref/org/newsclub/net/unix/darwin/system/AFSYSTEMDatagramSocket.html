18  package org.newsclub.net.unix.darwin.system;
19  
20  import java.io.FileDescriptor;
21  import java.io.IOException;
22  import java.net.DatagramSocket;
23  
24  import org.newsclub.net.unix.AFDatagramSocket;
25  import org.newsclub.net.unix.AFSYSTEMSocketAddress;
26  import org.newsclub.net.unix.AFSYSTEMSocketImplExtensions;
27  import org.newsclub.net.unix.AFSocketType;
28  
29  /**
30   * A {@link DatagramSocket} implementation that works with {@code AF_SYSTEM} sockets.
31   *
32   * @author Christian Kohlschütter
33   */
34  public final class AFSYSTEMDatagramSocket extends AFDatagramSocket<AFSYSTEMSocketAddress> implements
35      AFSYSTEMSocketExtensions {
36  
37    AFSYSTEMDatagramSocket(FileDescriptor fd) throws IOException {
38      this(fd, AFSocketType.SOCK_DGRAM);
39    }
40  
41    AFSYSTEMDatagramSocket(FileDescriptor fd, AFSocketType socketType) throws IOException {
42      super(new AFSYSTEMDatagramSocketImpl(fd, socketType));
43    }
44  
45    @Override
46    protected AFSYSTEMDatagramChannel newChannel() {
47      return new AFSYSTEMDatagramChannel(this);
48    }
49  
50    /**
51     * Returns a new {@link AFSYSTEMDatagramSocket} instance, using the default
52     * {@link AFSocketType#SOCK_DGRAM} socket type.
53     *
54     * @return The new instance.
55     * @throws IOException on error.
56     */
57    public static AFSYSTEMDatagramSocket newInstance() throws IOException {
58      return (AFSYSTEMDatagramSocket) newInstance(AFSYSTEMDatagramSocket::new);
59    }
60  
61    /**
62     * Returns a new {@link AFSYSTEMDatagramSocket} instance for the given socket type.
63     *
64     * @param socketType The socket type.
65     * @return The new instance.
66     * @throws IOException on error.
67     */
68    public static AFSYSTEMDatagramSocket newInstance(AFSocketType socketType) throws IOException {
69      return (AFSYSTEMDatagramSocket) newInstance((fd) -> {
70        return new AFSYSTEMDatagramSocket(fd, socketType);
71      });
72    }
73  
74    @Override
75    public AFSYSTEMDatagramChannel getChannel() {
76      return (AFSYSTEMDatagramChannel) super.getChannel();
77    }
78  
79    @Override
80    protected AFSYSTEMSocketImplExtensions getImplExtensions() {
81      return (AFSYSTEMSocketImplExtensions) super.getImplExtensions();
82    }
83  
84    /**
85     * Retrieves the kernel control ID given a kernel control name.
86     *
87     * An {@link IOException} is thrown if the ID is invalid or could not be accessed due to access
88     * restrictions.
89     *
90     * @param name The control name
91     * @return The control Id.
92     * @throws IOException on error.
93     */
94    public int getNodeIdentity(String name) throws IOException {
95      return getImplExtensions().getKernelControlId(getFileDescriptor(), name);
96    }
97  
98    /**
99     * Retrieves the kernel control ID given a kernel control name.
100    *
101    * An {@link IOException} is thrown if the ID is invalid or could not be accessed due to access
102    * restrictions.
103    *
104    * @param name The control name
105    * @return The control Id.
106    * @throws IOException on error.
107    */
108   public int getNodeIdentity(WellKnownKernelControlNames name) throws IOException {
109     return getNodeIdentity(name.getControlName());
110   }
111 
112   @Override
113   protected AFDatagramSocket<AFSYSTEMSocketAddress> newDatagramSocketInstance() throws IOException {
114     return new AFSYSTEMDatagramSocket(null);
115   }
116 }
