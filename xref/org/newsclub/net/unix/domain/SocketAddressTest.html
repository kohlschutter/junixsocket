49    @Test
50    public void testPort() throws IOException {
51      assertEquals(0, AFUNIXSocketAddress.of(new File("/tmp/whatever")).getPort());
52      assertEquals(123, AFUNIXSocketAddress.of(new File("/tmp/whatever"), 123).getPort());
53      assertEquals(44444, AFUNIXSocketAddress.of(new File("/tmp/whatever"), 44444).getPort());
54  
55      try {
56        AFUNIXSocketAddress.of(new File("/tmp/whatever"), -2);
57        fail("Expected IllegalArgumentException for illegal port");
58      } catch (final IllegalArgumentException e) {
59        // expected
60      }
61      try {
62        AFUNIXSocketAddress.of(new File("/tmp/whatever"), 65536);
63      } catch (final IllegalArgumentException e) {
64        fail("AFUNIXSocketAddress supports ports larger than 65535");
65      }
66    }
67  
68    @Test
69    public void testPath() throws Exception {
70      // as of junixsocket 2.4.0, a different canonical path doesn't matter
71      assertEquals("/tmp/whatever", AFUNIXSocketAddress.of(new File("/tmp/whatever")).getPath()
72          .replace(File.separatorChar, '/'));
73      assertEquals("whatever", AFUNIXSocketAddress.of(new File("whatever")).getPath().replace(
74          File.separatorChar, '/'));
75  
76      byte[] bytes = AFUNIXSocketAddress.of(new File("/tmp/whatever")).getPathAsBytes();
77      for (int i = 0; i < bytes.length; i++) {
78        if (bytes[i] == (byte) File.separatorChar) {
79          bytes[i] = '/';
80        }
81      }
82      assertArrayEquals("/tmp/whatever".getBytes(Charset.defaultCharset()), bytes);
83    }
84  
85    @Test
86    public void testAbstractNamespace() throws Exception {
87      AFUNIXSocketAddress address = AFUNIXSocketAddress.inAbstractNamespace(
88          "test\n\u000b\u0000\u007f");
89      if ("z/OS".equals(System.getProperty("os.name"))) {
90        // FIXME: check bytes in EBCDIC
91      } else {
92        byte[] addressBytes = {0, 't', 'e', 's', 't', '\n', '\u000b', '\u0000', '\u007f'};
93        assertArrayEquals(addressBytes, address.getPathAsBytes());
94      }
95      assertEquals(0, address.getPort());
96      if ("z/OS".equals(System.getProperty("os.name"))) {
97        // FIXME: check bytes in EBCDIC
98      } else {
99        assertEquals("@test..@.", address.getPath());
100       assertEquals("[path=\\x00test\\x0a\\x0b\\x00\\x7f]", address.toString().replaceFirst(
101           "^.*?\\[", "["));
102     }
103     assertTrue(address.isInAbstractNamespace());
104     assertFalse(address.hasFilename());
105     assertThrows(FileNotFoundException.class, () -> address.getFile());
106   }

113   @Test
114   public void testByteConstructor() throws Exception {
115     assertEquals("@", AFUNIXSocketAddress.of(new byte[] {0}).getPath());
116     assertEquals("@..", AFUNIXSocketAddress.of(new byte[] {0, (byte) 128, (byte) 255}).getPath());
117 
118     assertEquals("abc.123", AFUNIXSocketAddress.of("abc.123".getBytes(AFUNIXSocketAddress
119         .addressCharset())).getPath());
120     assertEquals(new File("abc.123"), AFUNIXSocketAddress.of(new File("abc.123")).getFile());
121     assertEquals("abc.123", AFUNIXSocketAddress.of(new File("abc.123")).getPath());
122   }

124   @Test
125   public void testInetAddress() throws Exception {
126     AFUNIXSocketAddress addr1 = AFUNIXSocketAddress.of(new File("/tmp/testing"));
127     assertTrue(AFUNIXSocketAddress.isSupportedAddress(addr1));
128     assertNull(addr1.getAddress()); // sadly
129     assertTrue(AFUNIXSocketAddress.isSupportedAddress(addr1.wrapAddress()));
130     AFUNIXSocketAddress addr2 = AFUNIXSocketAddress.unwrap(addr1.wrapAddress(), 0);
131     assertEquals(addr1, addr2);
132     assertTrue(AFUNIXSocketAddress.isSupportedAddress(addr2));
133     assertTrue(AFUNIXSocketAddress.isSupportedAddress(addr2.wrapAddress()));
134     assertNull(addr2.getAddress()); // sadly
135   }
