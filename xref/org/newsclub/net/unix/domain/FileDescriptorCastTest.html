65    @Test
66    @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR)
67    public void testSocketPair() throws Exception {
68      AFUNIXSocketPair<AFUNIXSocketChannel> socketPair = AFUNIXSocketPair.open();
69      AFUNIXSocketChannel sock1chan = socketPair.getSocket1();
70      FileDescriptor sock2fd = socketPair.getSocket2().getFileDescriptor();
71  
72      assertTrue(sock1chan.isConnected());
73  
74      FileDescriptorCast fdc = FileDescriptorCast.using(sock2fd);
75  
76      // Limitation: When we emulate socket pairs through non AF_UNIX sockets, we currently do not
77      // support casting to Socket.class
78      assertEquals(AFUNIXSocket.supports(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR), fdc
79          .isAvailable(Socket.class));
80  
81      if (!AFUNIXSocket.supports(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR)) {
82        FileDescriptorCast.using(sock2fd).as(FileChannel.class);
83      } else {
84        Socket sock2 = FileDescriptorCast.using(sock2fd).as(Socket.class);
85        assertEquals(AFUNIXSocket.class, sock2.getClass());
86        assertTrue(sock2.isConnected());
87        assertEquals(sock2fd, ((AFUNIXSocket) sock2).getFileDescriptor());
88      }
89    }

115   @Test
116   public void testUnconnectedServerAsSocket() throws Exception {
117     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(AFUNIXSocketAddress.ofNewTempFile(),
118         true);
119         AFUNIXServerSocket ass2 = AFUNIXServerSocket.bindOn(AFUNIXSocketAddress.ofNewTempFile(),
120             true)) {
121       FileDescriptor fd = ass.getFileDescriptor();
122       assertTrue(fd.valid());
123 
124       AFUNIXSocket sock = FileDescriptorCast.using(fd).as(AFUNIXSocket.class);
125       assertTrue(sock.isBound());
126       assertFalse(sock.isConnected());
127 
128       // We can cast a server socket (which is listening by default) to a socket, but we can't
129       // connect
130 
131       // NOTE: OpenBSD blocks on such a non-permissible cast, so we set a timeout
132       // This will cause a SocketTimeoutException, which we change to a SocketException
133       // so our tests can still pass
134       sock.setSoTimeout(1);
135 
136       assertThrows(SocketException.class, () -> {
137         try {
138           sock.connect(ass2.getLocalSocketAddress());
139         } catch (SocketTimeoutException e) {
140           throw new SocketException().initCause(e);
141         }
142       });
143       assertThrows(SocketException.class, () -> {
144         try {
145           sock.connect(ass.getLocalSocketAddress());
146         } catch (SocketTimeoutException e) {
147           throw new SocketException().initCause(e);
148         }
149       });
150       assertThrows(SocketException.class, () -> {
151         try {
152           sock.getInputStream().read();
153         } catch (SocketTimeoutException e) {
154           throw new SocketException().initCause(e);
155         }
156       });
157 
158       assertNull(sock.getRemoteSocketAddress());
159       assertFalse(sock.isConnected());
160     }
161   }

163   @Test
164   public void testServer() throws Exception {
165     AFUNIXSocketAddress serverAddress = AFUNIXSocketAddress.ofNewTempFile();
166     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(serverAddress, true);
167         AFUNIXSocket sock = AFUNIXSocket.connectTo(serverAddress)) {
168       AFUNIXSocket sock1 = FileDescriptorCast.using(sock.getFileDescriptor()).as(
169           AFUNIXSocket.class);
170       assertEquals(sock.isBound(), sock1.isBound());
171       assertEquals(sock.isConnected(), sock1.isConnected());
172       assertTrue(sock1.isConnected());
173       assertEquals(sock.getRemoteSocketAddress(), sock1.getRemoteSocketAddress());
174       assertEquals(ass.getLocalSocketAddress(), sock.getRemoteSocketAddress());
175 
176       AFUNIXServerSocket serverSock = FileDescriptorCast.using(ass.getFileDescriptor()).as(
177           AFUNIXServerSocket.class);
178       AFUNIXSocket sock0 = serverSock.accept();
179       assertEquals(ass.getLocalSocketAddress(), sock0.getLocalSocketAddress());
180 
181       sock0.getOutputStream().write(23);
182       assertEquals(23, sock1.getInputStream().read());
183     }

186   @Test
187   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
188   public void testDatagramSocket() throws Exception {
189     AFUNIXSocketAddress addr1 = AFUNIXSocketAddress.ofNewTempFile();
190     AFUNIXSocketAddress addr2 = AFUNIXSocketAddress.ofNewTempFile();
191     try (AFUNIXDatagramChannel dc1 = AFUNIXDatagramChannel.open();
192         AFUNIXDatagramChannel dc2 = AFUNIXDatagramChannel.open()) {
193       dc1.setDeleteOnClose(true);
194       dc2.setDeleteOnClose(true);
195       ByteBuffer bb = ByteBuffer.allocate(32);
196       bb.putInt(1234);
197       bb.flip();
198 
199       dc1.getOption(StandardSocketOptions.SO_SNDBUF);
200 
201       AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor()).as(
202           AFUNIXDatagramChannel.class);
203       AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor()).as(
204           AFUNIXDatagramChannel.class);
205 
206       assertFalse(dc1.isBound());
207       assertFalse(dc1c.isBound());
208       assertFalse(dc2.isBound());
209       assertFalse(dc2c.isBound());
210       dc1.bind(addr1);
211       dc2.bind(addr2);
212       assertTrue(dc1.isBound());
213       assertTrue(dc1c.isBound());
214       assertTrue(dc2.isBound());
215       assertTrue(dc2c.isBound());
216 
217       assertEquals(dc1.getLocalAddress(), dc1c.getLocalAddress());
218       assertEquals(dc2.getLocalAddress(), dc2c.getLocalAddress());
219 
220       dc1c.connect(addr2);
221       dc2c.connect(addr1);
222       assertTrue(dc1c.isConnected());
223       assertTrue(dc1.isConnected());
224 
225       assertEquals(dc1.getLocalAddress(), dc2c.getRemoteAddress());
226       assertEquals(dc1.getRemoteAddress(), dc2c.getLocalAddress());
227       assertEquals(dc1c.getLocalAddress(), dc2.getRemoteAddress());
228       assertEquals(dc1c.getRemoteAddress(), dc2.getLocalAddress());
229 
230       assertEquals(4, dc1c.write(bb));
231       bb.clear();
232       assertEquals(4, dc2c.read(bb));
233       bb.flip();
234       assertEquals(1234, bb.getInt());
235     }
236   }

238   @Test
239   public void testSocketPorts() throws Exception {
240     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempPath(123);
241     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(addr)) {
242       assertEquals(123, ass.getLocalPort());
243 
244       AFUNIXServerSocket ass1 = FileDescriptorCast.using(ass.getFileDescriptor()).withLocalPort(123)
245           .as(AFUNIXServerSocket.class);
246       assertEquals(123, ass.getLocalPort());
247 
248       AFUNIXSocket socket = AFUNIXSocket.connectTo(addr);
249       assertEquals(123, socket.getRemoteSocketAddress().getPort());
250 
251       AFUNIXSocket socket1 = FileDescriptorCast.using(socket.getFileDescriptor()).withRemotePort(
252           123).as(AFUNIXSocket.class);
253       assertEquals(socket.getRemoteSocketAddress(), socket1.getRemoteSocketAddress());
254       assertEquals(123, socket1.getRemoteSocketAddress().getPort());
255 
256       AFUNIXSocket ss = ass1.accept();
257       assertEquals(123, ss.getLocalPort());
258       assertEquals(123, ss.getLocalSocketAddress().getPort());
259     }
260   }

266       return addr.getPort();

270   @Test
271   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
272   public void testDatagramPorts() throws Exception {
273     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempPath(123);
274     assertEquals(123, addr.getPort());
275     try (AFUNIXDatagramChannel dc1 = AFUNIXDatagramChannel.open();
276         AFUNIXDatagramChannel dc2 = AFUNIXDatagramChannel.open()) {
277       dc1.setDeleteOnClose(true);
278       dc2.setDeleteOnClose(true);
279 
280       assertNull(dc1.getLocalAddress());
281       assertNull(dc1.getRemoteAddress());
282       assertNull(dc2.getLocalAddress());
283       assertNull(dc2.getRemoteAddress());
284 
285       dc1.bind(addr);
286       dc2.connect(addr);
287 
288       assertEquals(dc1.getLocalAddress(), dc2.getRemoteAddress());
289 
290       assertEquals(123, getPort(dc1.getLocalAddress()));
291 
292       // if we don't specify a port, we'll lose that information when casting (only stored in Java)
293       {
294         AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor()).as(
295             AFUNIXDatagramChannel.class);
296         AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor()).as(
297             AFUNIXDatagramChannel.class);
298 
299         assertEquals(0, getPort(dc1c.getLocalAddress()));
300         assertEquals(0, getPort(dc2c.getRemoteAddress()));
301       }
302 
303       // we need to specify local and/or remote ports when working directly with file descriptors
304       {
305         AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor())
306             .withLocalPort(123).as(AFUNIXDatagramChannel.class);
307         AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor())
308             .withRemotePort(123).as(AFUNIXDatagramChannel.class);
309 
310         assertEquals(123, getPort(dc1c.getLocalAddress()));
311         assertEquals(123, getPort(dc2c.getRemoteAddress()));
312       }
313     }

346   /**
347    * Passes a socket to a newly created Java process as "standard input". The new process sends back
348    * "Hello world" over that socket.
349    *
350    * @throws Exception on failure.
351    * @see StdinSocketApp
352    */
353   @Test
354   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_FD_AS_REDIRECT)
355   @ProcessUtilRequirement(canGetJavaCommandArguments = true)
356   public void testForkedVMRedirectStdin() throws Exception {
357     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
358     try (AFUNIXServerSocket serverSocket = AFUNIXServerSocket.bindOn(addr);
359         AFUNIXSocket clientConn = AFUNIXSocket.connectTo(addr);
360         AFUNIXSocket serverConn = serverSocket.accept()) {
361       assertTrue(addr.getFile().delete());
362 
363       ForkedVM vm = new ForkedVM(StdinSocketApp.class);
364       vm.setRedirectInput(FileDescriptorCast.using(clientConn.getFileDescriptor()).as(
365           Redirect.class));
366       vm.setRedirectError(Redirect.INHERIT);
367       // vm.setRedirectOutput(Redirect.INHERIT);
368       Process p = vm.fork();
369       try {
370         try (BufferedReader br = new BufferedReader(new InputStreamReader(serverConn
371             .getInputStream(), StandardCharsets.UTF_8))) {
372           assertEquals("Hello world", br.readLine());
373           String l;
374           if ((l = br.readLine()) != null) {
375             fail("Unexpected output: " + l);
376           }
377         }
378         assertTrue(p.waitFor(30, TimeUnit.SECONDS));
379         assertEquals(0, p.exitValue());
380       } finally {
381         p.destroyForcibly();
382       }
383     }
