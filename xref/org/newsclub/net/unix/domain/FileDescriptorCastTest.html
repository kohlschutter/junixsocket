74    @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR)
75    public void testSocketPair() throws Exception {
76      try (AFUNIXSocketPair<AFUNIXSocketChannel> socketPair = AFUNIXSocketPair.open()) {
77        AFUNIXSocketChannel sock1chan = socketPair.getSocket1();
78        FileDescriptor sock2fd = socketPair.getSocket2().getFileDescriptor();
79  
80        assertTrue(sock1chan.isConnected());
81  
82        FileDescriptorCast fdc = FileDescriptorCast.using(sock2fd);
83  
84        // Limitation: When we emulate socket pairs through non AF_UNIX sockets, we currently do not
85        // support casting to Socket.class
86        assertEquals(AFUNIXSocket.supports(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR), fdc
87            .isAvailable(Socket.class));
88  
89        if (!AFUNIXSocket.supports(AFSocketCapability.CAPABILITY_NATIVE_SOCKETPAIR)) {
90          FileDescriptorCast.using(sock2fd).as(FileChannel.class);
91        } else {
92          Socket sock2 = FileDescriptorCast.using(sock2fd).as(Socket.class);
93          assertEquals(AFUNIXSocket.class, sock2.getClass());
94  
95          assertEquals(sock2fd, ((AFUNIXSocket) sock2).getFileDescriptor());
96          try {
97            assertTrue(sock2.isConnected());
98          } catch (AssertionFailedError e) {
99            if (TestUtil.isHaikuOS()) {
100             throw TestUtil.haikuBug18534(e);
101           } else {
102             throw e;
103           }
104         }
105       }
106     }
107   }

144   @Test
145   public void testUnconnectedServerAsSocket() throws Exception {
146     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(AFUNIXSocketAddress.ofNewTempFile(),
147         true);
148         AFUNIXServerSocket ass2 = AFUNIXServerSocket.bindOn(AFUNIXSocketAddress.ofNewTempFile(),
149             true)) {
150       FileDescriptor fd = ass.getFileDescriptor();
151       assertTrue(fd.valid());
152 
153       AFUNIXSocket sock = FileDescriptorCast.using(fd).as(AFUNIXSocket.class);
154       assertTrue(sock.isBound());
155       assertFalse(sock.isConnected());
156 
157       // We can cast a server socket (which is listening by default) to a socket, but we can't
158       // connect
159 
160       // NOTE: OpenBSD blocks on such a non-permissible cast, so we set a timeout
161       // This will cause a SocketTimeoutException, which we change to a SocketException
162       // so our tests can still pass
163       sock.setSoTimeout(1);
164 
165       assertThrows(SocketException.class, () -> {
166         try {
167           sock.connect(ass2.getLocalSocketAddress());
168         } catch (SocketTimeoutException e) {
169           throw new SocketException().initCause(e);
170         }
171       });
172       assertThrows(SocketException.class, () -> {
173         try {
174           sock.connect(ass.getLocalSocketAddress());
175         } catch (SocketTimeoutException e) {
176           throw new SocketException().initCause(e);
177         }
178       });
179       assertThrows(SocketException.class, () -> {
180         try {
181           sock.getInputStream().read();
182         } catch (SocketTimeoutException e) {
183           throw new SocketException().initCause(e);
184         }
185       });
186 
187       assertNull(sock.getRemoteSocketAddress());
188       assertFalse(sock.isConnected());
189     }
190   }

194     AFUNIXSocketAddress serverAddress = AFUNIXSocketAddress.ofNewTempFile();
195     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(serverAddress, true);
196         AFUNIXSocket sock = AFUNIXSocket.connectTo(serverAddress)) {

202         assertTrue(sock1.isConnected());
203       } catch (AssertionFailedError e) {
204         if (TestUtil.isHaikuOS()) {
205           throw TestUtil.haikuBug18534(e);
206         } else {
207           throw e;
208         }
209       }
210       assertEquals(sock.getRemoteSocketAddress(), sock1.getRemoteSocketAddress());
211       assertEquals(ass.getLocalSocketAddress(), sock.getRemoteSocketAddress());
212 
213       AFUNIXServerSocket serverSock = FileDescriptorCast.using(ass.getFileDescriptor()).as(
214           AFUNIXServerSocket.class);
215       AFUNIXSocket sock0 = serverSock.accept();
216       assertEquals(ass.getLocalSocketAddress(), sock0.getLocalSocketAddress());
217 
218       sock0.getOutputStream().write(23);
219       assertEquals(23, sock1.getInputStream().read());
220     }

223   @Test
224   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
225   public void testDatagramSocket() throws Exception {
226     AFUNIXSocketAddress addr1 = AFUNIXSocketAddress.ofNewTempFile();
227     AFUNIXSocketAddress addr2 = AFUNIXSocketAddress.ofNewTempFile();
228     try (AFUNIXDatagramChannel dc1 = AFUNIXDatagramChannel.open();
229         AFUNIXDatagramChannel dc2 = AFUNIXDatagramChannel.open()) {
230       dc1.setDeleteOnClose(true);
231       dc2.setDeleteOnClose(true);
232       ByteBuffer bb = ByteBuffer.allocate(32);
233       bb.putInt(1234);
234       bb.flip();
235 
236       dc1.getOption(StandardSocketOptions.SO_SNDBUF);
237 
238       AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor()).as(
239           AFUNIXDatagramChannel.class);
240       AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor()).as(
241           AFUNIXDatagramChannel.class);
242 
243       assertFalse(dc1.isBound());
244       assertFalse(dc1c.isBound());
245       assertFalse(dc2.isBound());
246       assertFalse(dc2c.isBound());
247       dc1.bind(addr1);
248       dc2.bind(addr2);
249       assertTrue(dc1.isBound());
250       assertTrue(dc1c.isBound());
251       assertTrue(dc2.isBound());
252       assertTrue(dc2c.isBound());
253 
254       assertEquals(dc1.getLocalAddress(), dc1c.getLocalAddress());
255       assertEquals(dc2.getLocalAddress(), dc2c.getLocalAddress());
256 
257       dc1c.connect(addr2);
258       dc2c.connect(addr1);
259       assertTrue(dc1c.isConnected());
260 
261       assertEquals(4, dc1c.write(bb));
262       bb.clear();
263       assertEquals(4, dc2c.read(bb));
264       bb.flip();
265       assertEquals(1234, bb.getInt());
266 
267       try {
268         assertTrue(dc1.isConnected());
269         assertEquals(dc1.getLocalAddress(), dc2c.getRemoteAddress());
270         assertEquals(dc1.getRemoteAddress(), dc2c.getLocalAddress());
271         assertEquals(dc1c.getLocalAddress(), dc2.getRemoteAddress());
272         assertEquals(dc1c.getRemoteAddress(), dc2.getLocalAddress());
273       } catch (AssertionFailedError e) {
274         if (TestUtil.isHaikuOS()) {
275           throw TestUtil.haikuBug18534(e);
276         } else {
277           throw e;
278         }
279       }
280     }

285   public void testSocketPorts() throws Exception {
286     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempPath(123);
287     try (AFUNIXServerSocket ass = AFUNIXServerSocket.bindOn(addr)) {
288       assertEquals(123, ass.getLocalPort());
289 
290       @SuppressWarnings("checkstyle:VariableDeclarationUsageDistance")
291       AFUNIXServerSocket ass1 = FileDescriptorCast.using(ass.getFileDescriptor()).withLocalPort(123)
292           .as(AFUNIXServerSocket.class);
293       assertEquals(123, ass.getLocalPort());
294 
295       try (AFUNIXSocket socket = AFUNIXSocket.connectTo(addr)) {
296         AFUNIXSocketAddress rsa = socket.getRemoteSocketAddress();
297 
298         try {
299           Objects.requireNonNull(rsa);
300         } catch (NullPointerException e) {
301           if (TestUtil.isHaikuOS()) {
302             throw TestUtil.haikuBug18534(e);
303           } else {
304             throw e; // NOPMD.AvoidThrowingNullPointerException
305           }
306         }
307 
308         assertEquals(123, rsa.getPort());
309 
310         AFUNIXSocket socket1 = FileDescriptorCast.using(socket.getFileDescriptor()).withRemotePort(
311             123).as(AFUNIXSocket.class);
312         assertEquals(socket.getRemoteSocketAddress(), socket1.getRemoteSocketAddress());
313         assertEquals(123, socket1.getRemoteSocketAddress().getPort());
314 
315         AFUNIXSocket ss = ass1.accept();
316         assertEquals(123, ss.getLocalPort());
317         assertEquals(123, ss.getLocalSocketAddress().getPort());
318       }

330   @Test
331   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
332   public void testDatagramPorts() throws Exception {
333     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempPath(123);
334     assertEquals(123, addr.getPort());
335     try (AFUNIXDatagramChannel dc1 = AFUNIXDatagramChannel.open();
336         AFUNIXDatagramChannel dc2 = AFUNIXDatagramChannel.open()) {
337       dc1.setDeleteOnClose(true);
338       dc2.setDeleteOnClose(true);
339 
340       assertNull(dc1.getLocalAddress());
341       assertNull(dc1.getRemoteAddress());
342       assertNull(dc2.getLocalAddress());
343       assertNull(dc2.getRemoteAddress());
344 
345       dc1.bind(addr);
346       dc2.connect(addr);
347 
348       try {
349         assertEquals(dc1.getLocalAddress(), dc2.getRemoteAddress());
350       } catch (AssertionFailedError e) {
351         if (TestUtil.isHaikuOS() && dc2.getRemoteAddress() == null) {
352           throw TestUtil.haikuBug18534(e);
353         } else {
354           throw e;
355         }
356       }
357 
358       assertEquals(123, getPort(dc1.getLocalAddress()));
359 
360       // if we don't specify a port, we'll lose that information when casting (only stored in Java)
361       {
362         AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor()).as(
363             AFUNIXDatagramChannel.class);
364         AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor()).as(
365             AFUNIXDatagramChannel.class);
366 
367         assertEquals(0, getPort(dc1c.getLocalAddress()));
368         assertEquals(0, getPort(dc2c.getRemoteAddress()));
369       }
370 
371       // we need to specify local and/or remote ports when working directly with file descriptors
372       {
373         AFUNIXDatagramChannel dc1c = FileDescriptorCast.using(dc1.getFileDescriptor())
374             .withLocalPort(123).as(AFUNIXDatagramChannel.class);
375         AFUNIXDatagramChannel dc2c = FileDescriptorCast.using(dc2.getFileDescriptor())
376             .withRemotePort(123).as(AFUNIXDatagramChannel.class);
377 
378         if (dc1c.getLocalAddress() != null) { // address may be null on IBM i
379           assertEquals(123, getPort(dc1c.getLocalAddress()));
380         }
381         int remotePort = getPort(dc2c.getRemoteAddress());
382         if (remotePort == -1) {
383           // that's acceptable, too (seen on z/OS)
384           assertNull(dc2c.getRemoteAddress());
385         } else {
386           assertEquals(123, remotePort);
387         }
388       }
389     }

422   /**
423    * Passes a socket to a newly created Java process as "standard input". The new process sends back
424    * "Hello world" over that socket.
425    *
426    * @throws Exception on failure.
427    * @see StdinSocketApp
428    */
429   @Test
430   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_FD_AS_REDIRECT)
431   @ProcessUtilRequirement(canGetJavaCommandArguments = true)
432   @ForkedVMRequirement(forkSupported = true)
433   public void testForkedVMRedirectStdin() throws Exception {
434     AFUNIXSocketAddress addr = AFUNIXSocketAddress.ofNewTempFile();
435     try (AFUNIXServerSocket serverSocket = AFUNIXServerSocket.bindOn(addr);
436         AFUNIXSocket clientConn = AFUNIXSocket.connectTo(addr);
437         AFUNIXSocket serverConn = serverSocket.accept()) {
438       assertTrue(addr.getFile().delete());
439 
440       ForkedVM vm = new ForkedVM(StdinSocketApp.class);
441       vm.setRedirectInput(FileDescriptorCast.using(clientConn.getFileDescriptor()).as(
442           Redirect.class));
443       vm.setRedirectError(Redirect.INHERIT);
444       // vm.setRedirectOutput(Redirect.INHERIT);
445 
446       CompletableFuture<Object> cf = CompletableFuture.supplyAsync(() -> {
447         try {
448           return vm.fork();
449         } catch (UnsupportedOperationException | IOException e) {
450           return e;
451         }
452       });
453 
454       Object processObject;
455       try {
456         processObject = cf.get(10, TimeUnit.SECONDS);
457       } catch (Exception e) {
458         throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_INFORMATIONAL,
459             "Environment may not support forking new processes, which one test requires", e);
460       }
461       if (!(processObject instanceof Process)) {
462         if (processObject instanceof Exception) {
463           throw (Exception) processObject;
464         } else if (processObject instanceof Throwable) {
465           throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_INFORMATIONAL,
466               "Environment may not support forking new processes, which one test requires",
467               (Throwable) processObject);
468         } else {
469           throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_INFORMATIONAL,
470               "Environment may not support forking new processes, which one test requires");
471         }
472       }
473 
474       Process p = (Process) processObject;
475       try {
476         try (BufferedReader br = new BufferedReader(new InputStreamReader(serverConn
477             .getInputStream(), StandardCharsets.UTF_8))) {
478           assertEquals("Hello world", br.readLine());
479           String l;
480           if ((l = br.readLine()) != null) {
481             fail("Unexpected output: " + l);
482           }
483         }
484 
485         long time = System.currentTimeMillis();
486         long elapsed;
487         while ((elapsed = (System.currentTimeMillis() - time)) < 30 * 1000L) {
488           if (p.waitFor(1, TimeUnit.SECONDS)) {
489             break;
490           }
491           if (elapsed > 5000) {
492             System.out.println("Still waiting for process to terminate: " + p);
493           }
494         }
495 
496         assertEquals(0, p.exitValue());
497       } finally {
498         p.destroyForcibly();
499       }
500     }
