51  @AFSocketCapabilityRequirement({
52      AFSocketCapability.CAPABILITY_UNIX_DOMAIN, AFSocketCapability.CAPABILITY_PEER_CREDENTIALS})
53  @SuppressFBWarnings({
54      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
55  public final class PeerCredentialsTest extends SocketTestBase<AFUNIXSocketAddress> {
56    private static AFUNIXSocketCredentials credsSockets = null;
57    private static AFUNIXSocketCredentials credsDatagramSockets = null;
58  
59    public PeerCredentialsTest() {
60      super(AFUNIXAddressSpecifics.INSTANCE);
61    }
62  
63    @Test
64    public void testSocketsSameProcess() throws Exception {
65      assertTimeoutPreemptively(Duration.ofSeconds(10), () -> {
66        final CompletableFuture<AFUNIXSocketCredentials> clientCredsFuture =
67            new CompletableFuture<>();
68  
69        Semaphore sema = new Semaphore(0);
70        try (AFUNIXServerThread serverThread = new AFUNIXServerThread() {
71  
72          @Override
73          protected void handleConnection(final AFUNIXSocket socket) throws IOException {
74            AFUNIXSocketCredentials clientCreds = socket.getPeerCredentials();
75            clientCredsFuture.complete(clientCreds);
76            try {
77              sema.acquire();
78            } catch (InterruptedException e) {
79              // ignore
80            }
81          }
82        }; AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress())) {
83          try (InputStream in = socket.getInputStream()) {
84            AFUNIXSocketCredentials serverCreds = socket.getPeerCredentials();
85            AFUNIXSocketCredentials clientCreds = clientCredsFuture.get();
86  
87            sema.release();
88  
89            assertEquals(clientCreds, serverCreds,
90                "Since our tests run in the same process, the peer credentials must be identical");
91            assertEquals(clientCreds.toString(), serverCreds.toString(),
92                "Since our tests run in the same process, the peer credentials must be identical");
93            assertEquals(clientCreds.getGid(), serverCreds.getGid(),
94                "Since our tests run in the same process, the peer credentials must be identical");
95            assertArrayEquals(clientCreds.getGids(), serverCreds.getGids(),
96                "Since our tests run in the same process, the peer credentials must be identical");
97            assertEquals(clientCreds.getUid(), serverCreds.getUid(),
98                "Since our tests run in the same process, the peer credentials must be identical");
99            assertEquals(clientCreds.getPid(), serverCreds.getPid(),
100               "Since our tests run in the same process, the peer credentials must be identical");
101           assertEquals(clientCreds.getUUID(), serverCreds.getUUID(),
102               "Since our tests run in the same process, the peer credentials must be identical");
103 
104           if (clientCreds.getPid() == -1) {
105             // PID information is unvailable on this platform
106           } else {
107             assertEquals(ProcessUtil.getPid(), clientCreds.getPid(),
108                 "The returned PID must be the one of our process");
109           }
110 
111           setCredsSockets(clientCreds);
112         }
113       }
114     });
115   }

117   private static void checkCredentialFeatures(StringBuilder sbYes, StringBuilder sbNo, String key,
118       boolean supported) {
119     ((supported) ? sbYes : sbNo).append(' ').append(key);
120   }
121 
122   private static void checkCredentialFeatures(AFUNIXSocketCredentials creds) {
123     StringBuilder sbYes = new StringBuilder(32);
124     StringBuilder sbNo = new StringBuilder(32);
125 
126     checkCredentialFeatures(sbYes, sbNo, "pid", (creds.getPid() > 0));
127     checkCredentialFeatures(sbYes, sbNo, "uid", (creds.getUid() != -1));
128     checkCredentialFeatures(sbYes, sbNo, "gid", (creds.getGid() != -1));
129     checkCredentialFeatures(sbYes, sbNo, "additional_gids", (creds.getGids() != null && creds
130         .getGids().length > 0));
131     checkCredentialFeatures(sbYes, sbNo, "uuid", (creds.getUUID() != null));
132 
133     System.out.print("Supported credentials:  ");
134     if (sbYes.length() == 0) {
135       System.out.println(" (none)");
136     } else {
137       System.out.println(sbYes);
138     }
139     System.out.print("Unsupported credentials:");
140     if (sbNo.length() == 0) {
141       System.out.println(" (none)");
142     } else {
143       System.out.println(sbNo);
144     }
145   }

147   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
148   @Test
149   public void testDatagramSocket() throws Exception {
150     AFUNIXSocketAddress ds1Addr = AFUNIXSocketAddress.of(newTempFile());
151     AFUNIXSocketAddress ds2Addr = AFUNIXSocketAddress.of(newTempFile());
152 
153     try (AFUNIXDatagramSocket ds1 = AFUNIXDatagramSocket.newInstance();
154         AFUNIXDatagramSocket ds2 = AFUNIXDatagramSocket.newInstance();) {
155 
156       ds1.bind(ds1Addr);
157       ds2.bind(ds2Addr);
158       ds1.connect(ds2Addr);
159       ds2.connect(ds1Addr);
160 
161       DatagramPacket dp = AFDatagramUtil.datagramWithCapacityAndPayload("Hello".getBytes(
162           StandardCharsets.UTF_8));
163       ds2.send(dp);
164       dp = AFDatagramUtil.datagramWithCapacity(1024);
165 
166       ds1.receive(dp);
167       AFUNIXSocketCredentials pc1 = ds1.getPeerCredentials();
168 
169       AFUNIXSocketCredentials pc2 = ds2.getPeerCredentials();
170 
171       assertEquals(pc1, pc2);
172       setCredsDatagramSockets(pc1);
173     }
174   }
175 
176   private static void setCredsSockets(AFUNIXSocketCredentials creds) {
177     credsSockets = creds;
178   }
179 
180   private static void setCredsDatagramSockets(AFUNIXSocketCredentials creds) {
181     credsDatagramSockets = creds;
182   }
183 
184   @AfterAll
185   public static void ensureSameCreds() {
186     if (credsSockets != null) {
187       checkCredentialFeatures(credsSockets);
188       if (credsDatagramSockets != null) {
189         if (credsDatagramSockets.isEmpty() && !credsSockets.isEmpty()) {
190           System.out.println("WARNING: No peer credentials for datagram sockets");
191         } else {
192           assertEquals(credsSockets, credsDatagramSockets,
193               "The credentials received via Socket and via DatagramSocket should be the same");
194         }
195       }
196     }
197   }
