71    @Test
72    public void testSendRecvFileDescriptors() throws Exception {
73      assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
74        try (AFUNIXServerThread serverThread = new AFUNIXServerThread() {
75          @Override
76          protected void handleConnection(final AFUNIXSocket socket) throws IOException {
77            socket.setOutboundFileDescriptors(FileDescriptor.in, FileDescriptor.err);
78            assertTrue(socket.hasOutboundFileDescriptors());
79            try (OutputStream outputStream = socket.getOutputStream()) {
80              outputStream.write("HELLO".getBytes(UTF_8));
81            }
82            assertFalse(socket.hasOutboundFileDescriptors());
83  
84            stopAcceptingConnections();
85          }
86        };
87            AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress()); //
88            InputStream in = socket.getInputStream()) {
89          socket.setAncillaryReceiveBufferSize(1024);
90  
91          byte[] buf = new byte[64];
92          FileDescriptor[] fds;
93          int numRead;
94          fds = socket.getReceivedFileDescriptors();
95          assertArrayEquals(new FileDescriptor[0], fds, "Initially, there are no file descriptors");
96  
97          numRead = in.read(buf);
98          assertEquals(5, numRead, "'HELLO' is five bytes long");
99          assertEquals("HELLO", new String(buf, 0, numRead, UTF_8));
100 
101         fds = socket.getReceivedFileDescriptors();
102         assertEquals(2, fds.length, "Now, we should have two file descriptors");
103 
104         fds = socket.getReceivedFileDescriptors();
105         assertArrayEquals(new FileDescriptor[0], fds,
106             "If we ask again, these new file descriptors should be gone");
107 
108         try (InputStream is = socket.getInputStream()) {
109           numRead = is.read(buf);
110           assertEquals(-1, numRead, "There shouldn't be anything left to read");
111           fds = socket.getReceivedFileDescriptors();
112           assertArrayEquals(new FileDescriptor[0], fds,
113               "There shouldn't be any new file descriptors");
114         }
115       }
116     });
117   }

202     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
203       try (AFUNIXServerThread serverThread = new AFUNIXServerThread() {
204 
205         @Override
206         protected void handleConnection(final AFUNIXSocket socket) throws IOException {
207           socket.setOutboundFileDescriptors(FileDescriptor.in, FileDescriptor.err);
208 
209           // NOTE: send an arbitrary byte — we can't send fds without any in-band data
210           try (OutputStream outputStream = socket.getOutputStream()) {
211             outputStream.write(123);
212           }
213 
214           stopAcceptingConnections();
215         }
216       }; //
217           AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress()); //
218           InputStream inputStream = socket.getInputStream()) {
219         // NOTE: we haven't set the ancillary receive buffer size
220 
221         try {
222           assertEquals(123, inputStream.read());
223         } catch (SocketException e) {
224           // on Linux, a SocketException may be thrown (an ancillary message was sent, but not read)
225         }
226         assertArrayEquals(new FileDescriptor[0], socket.getReceivedFileDescriptors());
227         assertEquals(0, socket.getAncillaryReceiveBufferSize());
228       }

232   @Test
233   public void testAncillaryReceiveBufferTooSmall() throws Exception {
234     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
235       try (AFUNIXServerThread serverThread = new AFUNIXServerThread() {
236 
237         @Override
238         protected void handleConnection(final AFUNIXSocket socket) throws IOException {
239           socket.setOutboundFileDescriptors(FileDescriptor.in, FileDescriptor.err);
240           assertTrue(socket.hasOutboundFileDescriptors());
241 
242           // NOTE: send an arbitrary byte — we can't send fds without any in-band data
243           try (OutputStream outputStream = socket.getOutputStream()) {
244             try {
245               outputStream.write(123);
246             } catch (SocketException e) {
247               if (!socket.isConnected()) {
248                 // permit "Broken pipe" exception, which may be thrown in some OSes
249                 // seen with Linux 5.4.0-1094-azure / Microsoft CBL-Mariner 2.0
250               } else {
251                 throw e;
252               }
253             }
254           }
255 
256           assertFalse(socket.hasOutboundFileDescriptors());
257 
258           stopAcceptingConnections();
259         }
260       };
261           AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress());
262           InputStream inputStream = socket.getInputStream()) {
263         // using this call directly to get a really small buffer
264         ImplUtil.setAncillaryReceiveBufferSize(socket, 13);
265         try {
266           assertEquals(123, inputStream.read());
267           FileDescriptor[] fds = socket.getReceivedFileDescriptors();
268           if (fds.length != 0) {
269             if (fds.length == 2) {
270               // space was sufficient
271             } else if (fds.length == 1) {
272               // Not all operating systems throw a "No buffer space available" message
273               System.err.println("WARNING: Not all file descriptors were received");
274             } else {
275               assertEquals(2, fds.length, "Received wrong number of file descriptors");
276             }
277           } else {
278             Assertions.fail("Expected a \"No buffer space available\" SocketException");
279           }
280         } catch (SocketException e) {
281           // expected
282         }
283         assertArrayEquals(new FileDescriptor[0], socket.getReceivedFileDescriptors());
284       }
285     });
286   }

291 
292     assertTimeoutPreemptively(Duration.ofSeconds(2), () -> {
293       try (AFUNIXServerThread serverThread = new AFUNIXServerThread() {
294

297           try (FileOutputStream fos = new FileOutputStream(tmpFile)) {
298             fos.write("WORLD!".getBytes(UTF_8));
299           }
300           try (FileInputStream fin = new FileInputStream(tmpFile)) {
301             socket.setOutboundFileDescriptors(fin.getFD());
302             try (OutputStream outputStream = socket.getOutputStream()) {
303               outputStream.write("HELLO".getBytes(UTF_8));
304             }
305           }
306 
307           stopAcceptingConnections();
308         }
309       }; //
310           AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress()); //
311           InputStream in = socket.getInputStream()) {
312         socket.setAncillaryReceiveBufferSize(1024);
313 
314         byte[] buf = new byte[64];
315         FileDescriptor[] fds;
316         int numRead;
317 
318         numRead = in.read(buf);
319         assertEquals(5, numRead, "'HELLO' is five bytes long");
320         assertEquals("HELLO", new String(buf, 0, numRead, UTF_8));
321 
322         fds = socket.getReceivedFileDescriptors();
323         assertEquals(1, fds.length, "Now, we should have two file descriptors");
324         FileDescriptor fdesc = fds[0];
325 
326         try (FileInputStream fin = new FileInputStream(fdesc)) {
327           numRead = fin.read(buf);
328           assertEquals(6, numRead, "'WORLD!' is six bytes long");
329           assertEquals("WORLD!", new String(buf, 0, numRead, UTF_8));
330         }
331       } finally {
332         Files.deleteIfExists(tmpFile.toPath());
333       }
334     });

350             assertEquals('W', fin.read());
351 
352             // We send the file descriptor of fin, from which we already consumed one byte.
353             socket.setOutboundFileDescriptors(fin.getFD());
354             try (OutputStream outputStream = socket.getOutputStream()) {
355               outputStream.write("HELLO".getBytes(UTF_8));
356             }
357           }
358 
359           stopAcceptingConnections();
360         }
361       }; //
362           AFUNIXSocket socket = (AFUNIXSocket) connectTo(serverThread.getServerAddress()); //
363           InputStream in = socket.getInputStream()) {
364         socket.setAncillaryReceiveBufferSize(1024);
365 
366         byte[] buf = new byte[64];
367         FileDescriptor[] fds;
368         int numRead;
369 
370         numRead = in.read(buf);
371         assertEquals(5, numRead, "'HELLO' is five bytes long");
372         assertEquals("HELLO", new String(buf, 0, numRead, UTF_8));
373 
374         fds = socket.getReceivedFileDescriptors();
375         assertEquals(1, fds.length, "Now, we should have two file descriptors");
376         FileDescriptor fdesc = fds[0];

389   @AFSocketCapabilityRequirement(AFSocketCapability.CAPABILITY_UNIX_DATAGRAMS)
390   @Test
391   public void testDatagramSocket() throws Exception {
392     assertTimeoutPreemptively(Duration.ofSeconds(30), () -> {
393       AFUNIXSocketAddress ds1Addr = AFUNIXSocketAddress.of(newTempFile());
394       AFUNIXSocketAddress ds2Addr = AFUNIXSocketAddress.of(newTempFile());
395 
396       try (AFUNIXDatagramSocket ds1 = AFUNIXDatagramSocket.newInstance();
397           AFUNIXDatagramSocket ds2 = AFUNIXDatagramSocket.newInstance();) {
398         ds1.setAncillaryReceiveBufferSize(1024);
399         ds2.setAncillaryReceiveBufferSize(1024);
400 
401         ds1.bind(ds1Addr);
402         ds2.bind(ds2Addr);
403         ds1.connect(ds2Addr);
404         ds2.connect(ds1Addr);
405 
406         File tmpOut = newTempFile();
407         try (FileOutputStream fos = new FileOutputStream(tmpOut)) {
408           ds1.setOutboundFileDescriptors(fos.getFD());
409           DatagramPacket dp = AFDatagramUtil.datagramWithCapacity(64);
410           assertTrue(ds1.hasOutboundFileDescriptors());
411           ds1.send(dp);
412           assertFalse(ds1.hasOutboundFileDescriptors());
413           ds2.receive(dp);
414           FileDescriptor[] fds = ds2.getReceivedFileDescriptors();
415           assertEquals(1, fds.length);
416 
417           try (FileOutputStream fos2 = new FileOutputStream(fds[0])) {
418             fos2.write("Hello".getBytes(UTF_8));
419             // closing the received file descriptor will not close the original one ...
420           }
421 
422           // ... which is why we can append the data here
423           fos.write("World".getBytes(UTF_8));
424         }
425 
426         try (FileInputStream fin = new FileInputStream(tmpOut)) {
427           String text = new String(IOUtil.readAllBytes(fin), UTF_8);
428           // ... and the final output will contain both parts
429           assertEquals("HelloWorld", text);
430         }
431       }
432     });
