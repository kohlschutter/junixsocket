18  package org.newsclub.net.unix;
19  
20  import java.net.Socket;
21  import java.nio.charset.StandardCharsets;
22  import java.util.Objects;
23  
24  import com.kohlschutter.annotations.compiletime.SuppressFBWarnings;
25  
26  /**
27   * A potentially file descriptor-leaking client.
28   * 
29   * See <a href="https://github.com/kohlschutter/junixsocket/pull/29">issue 29</a> for details.
30   * 
31   * @see FinalizeTest
32   * @author Christian Kohlschütter
33   */
34  @SuppressFBWarnings({
35      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
36  public class FinalizeTestClient {
37    @SuppressFBWarnings({"RV_RETURN_VALUE_IGNORED"})
38    public static void main(String[] args) throws Exception {
39      String socketType = System.getProperty("test.junixsocket.socket.type", "");
40      String socketName = System.getProperty("test.junixsocket.socket", "");
41      if (socketType.isEmpty() || socketName.isEmpty()) {
42        throw new IllegalArgumentException(
43            "Both test.junixsocket.socket and test.junixsocket.socket.type must be specified as system properties");
44      }
45  
46      Socket socket;
47      AFSocketAddress addr;
48      if ("UNIX".equals(socketType)) {
49        addr = AFUNIXSocketAddress.unwrap(socketName, 0);
50      } else if ("TIPC".equals(socketType)) {
51        addr = AFTIPCSocketAddress.unwrap(socketName, 0);
52      } else {
53        throw new IllegalArgumentException("Unsupported socket type: " + socketType);
54      }
55      socket = AFSocket.connectTo(Objects.requireNonNull(addr));
56      socket.getInputStream().read();
57      while (true) {
58        // create some pressure on GC
59        new String("junixsocket".getBytes(StandardCharsets.UTF_8), StandardCharsets.UTF_8);
60      }
61    }
62  }
