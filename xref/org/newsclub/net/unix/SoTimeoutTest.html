41  /**
42   * Tests {@link Socket#setSoTimeout(int)} behavior.
43   *
44   * @see <a href="http://code.google.com/p/junixsocket/issues/detail?id=14">Issue 14</a>
45   */
46  @SuppressFBWarnings({
47      "THROWS_METHOD_THROWS_CLAUSE_THROWABLE", "THROWS_METHOD_THROWS_CLAUSE_BASIC_EXCEPTION"})
48  public abstract class SoTimeoutTest<A extends SocketAddress> extends SocketTestBase<A> {
49    protected SoTimeoutTest(AddressSpecifics<A> asp) {
50      super(asp);
51    }
52  
53    /**
54     * Triggers a case where {@link Socket#setSoTimeout(int)} fails on some platforms: when the socket
55     * is closed.
56     *
57     * @throws IOException on error.
58     */
59    @Test
60    public void issue14Fail() throws Exception {
61      Semaphore sema = new Semaphore(0);
62      try (ServerThread serverThread = new ServerThread() {
63  
64        @Override
65        protected void handleConnection(final Socket socket) throws IOException {
66          try {
67            socket.close();
68          } finally {
69            sema.release();
70          }
71        }
72      }; Socket sock = connectTo(serverThread.getServerAddress())) {
73        sema.acquire();
74  
75        try {
76          sock.setSoTimeout((int) TimeUnit.SECONDS.toMillis(12));
77          // Socket#setSoTimeout(int) did not throw a SocketException. This is OK.
78        } catch (final SocketException e) {
79          // expected, as the socket is actually closed
80        }
81      }
82    }
83  
84    /**
85     * Triggers a regular case where {@link Socket#setSoTimeout(int)} should work.
86     *
87     * @throws IOException on error.
88     */
89    @Test
90    public void issue14Pass() throws Exception {
91      Semaphore sema = new Semaphore(0);
92      try (ServerThread serverThread = new ServerThread() {
93  
94        @Override
95        protected void handleConnection(final Socket sock) throws IOException {
96          // Let's wait some time for a byte that never gets sent by the
97          // client
98          try (InputStream inputStream = sock.getInputStream()) {
99            inputStream.read();
100           sema.acquire();
101         } catch (InterruptedException | IOException e) {
102           // ignore
103         }
104       }
105     }; Socket sock = connectTo(serverThread.getServerAddress())) {
106       assertTrue(sock.isConnected());
107       assertFalse(sock.isClosed());
108       try {
109         sock.setSoTimeout((int) TimeUnit.SECONDS.toMillis(12));
110       } catch (IOException e) {
111         e.printStackTrace();
112         throw e;
113       } finally {
114         sema.release();
115       }
116     }
117   }

119   @Test
120   public void testSocketTimeoutExceptionRead() throws Exception {
121     try (CloseablePair<? extends Socket> pair = newInterconnectedSockets();) {
122       Socket socket = pair.getSecond();
123       socket.setSoTimeout(500);
124       byte[] buf = new byte[socket.getReceiveBufferSize()];
125       InputStream in = socket.getInputStream();
126 
127       assertThrows(SocketTimeoutException.class, () -> {
128         AssertUtil.ignoreValue(in.read(buf));
129       });
130     }
131   }
132 
133   @Test
134   public void testSocketTimeoutExceptionWrite() throws Exception {
135     try (CloseablePair<? extends Socket> pair = newInterconnectedSockets();) {
136       Socket socket = pair.getSecond();
137       socket.setSoTimeout(500);
138       if (socket.getSoTimeout() == 0) {
139         boolean isZOS = "z/OS".equals(System.getProperty("os.name", ""));
140 
141         String afs = addressFamilyString();
142 
143         if (isZOS && "AF_UNIX".equals(afs)) {
144           // https://www.ibm.com/docs/en/zos/2.5.0?topic=csd-getsockopt-setsockopt-bpx1opt-bpx4opt-get-set-options-associated-socket
145           throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_WITH_ISSUES,
146               "Could not set socket timeout via Socket.setSoTimeout (" + addressFamilyString()
147                   + "); this is a known limitation of z/OS");
148         } else {
149           throw new TestAbortedWithImportantMessageException(MessageType.TEST_ABORTED_WITH_ISSUES,
150               "Could not set socket timeout via Socket.setSoTimeout (" + addressFamilyString()
151                   + ")");
152         }
153       }
154       byte[] buf = new byte[socket.getSendBufferSize()];
155       OutputStream out = socket.getOutputStream();
156 
157       try {
158         out.write(buf);
159         out.write(buf);
160       } catch (SocketTimeoutException e) {
161         // expected but not guaranteed
162       }
163     }
164   }
