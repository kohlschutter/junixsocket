47  /**
48   * A wrapper that allows a {@link FileDescriptor} be sent via RMI over AF_UNIX sockets.
49   * 
50   * @author Christian Kohlschütter
51   * @see RemoteFileInput subclass for sending {@link FileInputStream}s.
52   * @see RemoteFileOutput subclass for sending {@link FileOutputStream}s.
53   */
54  public abstract class RemoteFileDescriptorBase<T> implements Externalizable, Closeable,
55      FileDescriptorAccess {
56    private static final String PROP_SERVER_TIMEOUT =
57        "org.newsclub.net.unix.rmi.rfd-server-timeout-millis";
58    private static final String PROP_CONNECT_TIMEOUT =
59        "org.newsclub.net.unix.rmi.rfd-connect-timeout-millis";
60  
61    private static final int SERVER_TIMEOUT = //
62        parseTimeoutMillis(System.getProperty(PROP_SERVER_TIMEOUT, "10000"), false);
63    private static final int CONNECT_TIMEOUT = //
64        parseTimeoutMillis(System.getProperty(PROP_CONNECT_TIMEOUT, "1000"), true);
65  
66    static final int MAGIC_VALUE_MASK = 0x00FD0000;
67    static final int BIT_READABLE = 1 << 0;
68    static final int BIT_WRITABLE = 1 << 1;
69  
70    private static final long serialVersionUID = 1L;
71  
72    private final AtomicReference<DataInputStream> remoteConnection = new AtomicReference<>();
73    private final AtomicReference<AFUNIXSocket> remoteServer = new AtomicReference<>();
74  
75    /**
76     * An optional, closeable resource that is related to this instance. If the reference is non-null,
77     * this will be closed upon {@link #close()}.
78     * 
79     * For unidirectional implementations, this could be the corresponding input/output stream. For
80     * bidirectional implementations (e.g., a Socket, Pipe, etc.), this should close both directions.
81     */
82    protected final transient AtomicReference<T> resource = new AtomicReference<>();
83  
84    private int magicValue;
85    private FileDescriptor fd;
86    private AFUNIXRMISocketFactory socketFactory;
87  
88    /**
89     * Creates an uninitialized instance; used for externalization.
90     * 
91     * @see #readExternal(ObjectInput)
92     */
93    RemoteFileDescriptorBase() {
94    }
95  
96    RemoteFileDescriptorBase(AFUNIXRMISocketFactory socketFactory, T stream, FileDescriptor fd,
97        int magicValue) {
98      this.resource.set(stream);
99      this.socketFactory = socketFactory;
100     this.fd = fd;
101     this.magicValue = magicValue;
102   }
103 
104   @Override
105   @SuppressWarnings("PMD.ExceptionAsFlowControl")
106   public final void writeExternal(ObjectOutput objOut) throws IOException {
107     if (fd == null || !fd.valid()) {
108       throw new IOException("No or invalid file descriptor");
109     }
110     final int randomValue = ThreadLocalRandom.current().nextInt();
111 
112     int localPort;
113     try {
114       AFServerSocket<?> serverSocket = (AFServerSocket<?>) socketFactory.createServerSocket(0);
115       localPort = serverSocket.getLocalPort();
116 
117       AFSocketServer<?> server = new AFSocketServer<AFSocketAddress>(serverSocket) {
118         @Override
119         protected void doServeSocket(AFSocket<?> socket) throws IOException {
120           AFUNIXSocket unixSocket = (AFUNIXSocket) socket;
121           try (DataOutputStream out = new DataOutputStream(socket.getOutputStream());
122               InputStream in = socket.getInputStream();) {
123             unixSocket.setOutboundFileDescriptors(fd);
124             out.writeInt(randomValue);
125 
126             try {
127               socket.setSoTimeout(CONNECT_TIMEOUT);
128             } catch (IOException e) {
129               // ignore
130             }
131 
132             // This call blocks until the remote is done with the file descriptor, or we time out.
133             int response = in.read();
134             if (response != 1) {
135               if (response == -1) {
136                 // EOF, remote terminated
137               } else {
138                 throw new IOException("Unexpected response: " + response);
139               }
140             }
141           } finally {
142             stop();
143           }
144         }
145 
146         @Override
147         protected void onServerStopped(AFServerSocket<?> socket) {
148           try {
149             serverSocket.close();
150           } catch (IOException e) {
151             // ignore
152           }
153         }
154 
155       };
156       server.startThenStopAfter(SERVER_TIMEOUT, TimeUnit.MILLISECONDS);
157     } catch (IOException e) {
158       objOut.writeObject(e);
159       throw e;
160     }
161 
162     objOut.writeObject(socketFactory);
163     objOut.writeInt(magicValue);
164     objOut.writeInt(randomValue);
165     objOut.writeInt(localPort);
166     objOut.flush();
167   }
168 
169   @SuppressWarnings("resource")
170   @Override
171   public final void readExternal(ObjectInput objIn) throws IOException, ClassNotFoundException {
172     DataInputStream in1 = remoteConnection.getAndSet(null);
173     if (in1 != null) {
174       in1.close();
175     }
176 
177     Object obj = objIn.readObject();
178     if (obj instanceof IOException) {
179       IOException e = new IOException("Could not read RemoteFileDescriptor");
180       e.addSuppressed((IOException) obj);
181       throw e;
182     }
183     this.socketFactory = (AFUNIXRMISocketFactory) obj;
184 
185     // Since ancillary messages can only be read in combination with real data, we read and verify a
186     // magic value
187     this.magicValue = objIn.readInt();
188     if ((magicValue & MAGIC_VALUE_MASK) != MAGIC_VALUE_MASK) {
189       throw new IOException("Unexpected magic value: " + Integer.toHexString(magicValue));
190     }
191     final int randomValue = objIn.readInt();
192     int port = objIn.readInt();
193 
194     AFUNIXSocket socket = (AFUNIXSocket) socketFactory.createSocket("", port);
195     if (remoteServer.getAndSet(socket) != null) {
196       throw new IllegalStateException("remoteServer was not null");
197     }
198 
199     try {
200       socket.setSoTimeout(CONNECT_TIMEOUT);
201     } catch (IOException e) {
202       // ignore
203     }
204 
205     in1 = new DataInputStream(socket.getInputStream());
206     this.remoteConnection.set(in1);
207     socket.ensureAncillaryReceiveBufferSize(128);
208 
209     int random = in1.readInt();
210 
211     if (random != randomValue) {
212       throw new IOException("Invalid socket connection");
213     }
214     FileDescriptor[] descriptors = socket.getReceivedFileDescriptors();
215 
216     if (descriptors == null || descriptors.length != 1) {
217       throw new IOException("Did not receive exactly 1 file descriptor but " + (descriptors == null
218           ? 0 : descriptors.length));
219     }
220 
221     this.fd = descriptors[0];
222   }
223 
224   /**
225    * Returns the file descriptor.
226    * 
227    * This is either the original one that was specified in the constructor or a copy that was sent
228    * via RMI over an AF_UNIX connection as part of an ancillary message.
229    * 
230    * @return The file descriptor.
231    */
232   @Override
233   @SuppressFBWarnings("EI_EXPOSE_REP")
234   public final FileDescriptor getFileDescriptor() {
235     return fd;
236   }
237 
238   /**
239    * Returns the "magic value" for this type of file descriptor.
240    * 
241    * The magic value consists of an indicator ("this is a file descriptor") as well as its
242    * capabilities (read/write). It is used to prevent, for example, converting an output stream to
243    * an input stream.
244    * 
245    * @return The magic value.
246    */
247   protected final int getMagicValue() {
248     return magicValue;
249   }
250 
251   @SuppressWarnings("resource")
252   @Override
253   public void close() throws IOException {
254     DataInputStream in1 = remoteConnection.getAndSet(null);
255     if (in1 != null) {
256       try {
257         in1.close();
258       } catch (SocketException e) {
259         // ignore
260       }
261     }
262 
263     AFUNIXSocket remoteSocket = remoteServer.getAndSet(null);
264     if (remoteSocket != null) {
265       try (OutputStream out = remoteSocket.getOutputStream()) {
266         out.write(1);
267       } catch (SocketException e) {
268         // ignore
269       }
270       remoteSocket.close();
271     }
272 
273     @SuppressWarnings("null")
274     T c = this.resource.getAndSet(null);
275     if (c != null) {
276       if (c instanceof Closeable) {
277         ((Closeable) c).close();
278       }
279     }
280   }
281 
282   private static int parseTimeoutMillis(String s, boolean zeroPermitted) {
283     Objects.requireNonNull(s);
284     int duration;
285     try {
286       duration = Integer.parseInt(s);
287     } catch (Exception e) {
288       throw new IllegalArgumentException("Illegal timeout value: " + s, e);
289     }
290     if (duration < 0 || (duration == 0 && !zeroPermitted)) {
291       throw new IllegalArgumentException("Illegal timeout value: " + s);
292     }
293     return duration;
294   }
