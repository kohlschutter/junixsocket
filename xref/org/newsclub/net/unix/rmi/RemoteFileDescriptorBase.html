208   /**
209    * Returns the "magic value" for this type of file descriptor.

221   @Override
222   public synchronized void close() throws IOException {
223     T c = this.resource;
224     if (c != null) {
225       this.resource = null;
226       if (c instanceof Closeable) {
227         ((Closeable) c).close();
228       }
229     }

232   private static int parseTimeoutMillis(String s, boolean zeroPermitted) {
233     Objects.requireNonNull(s);
234     int duration;
235     try {
236       duration = Integer.parseInt(s);
237     } catch (Exception e) {
238       throw new IllegalArgumentException("Illegal timeout value: " + s, e);
239     }
240     if (duration < 0 || (duration == 0 && !zeroPermitted)) {
241       throw new IllegalArgumentException("Illegal timeout value: " + s);
242     }
243     return duration;
244   }
