20  import java.io.UnsupportedEncodingException;
21  import java.net.DatagramPacket;
22  import java.net.Inet4Address;

24  import java.net.InetSocketAddress;

27  import java.net.URLDecoder;

29  import java.net.UnknownHostException;
30  import java.nio.charset.StandardCharsets;
31  import java.util.Locale;
32  import java.util.Objects;

34  /**
35   * A workaround to create an {@link InetAddress} for an {@link AFSocketAddress}.
36   *
37   * {@link DatagramPacket} internally requires InetAddress compatibility. Even if it pretends to
38   * accept {@link SocketAddress}es, it refuses anything other than {@link InetSocketAddress}
39   * <em>and</em> then even stores host and port separately.
40   *
41   * This implementation deserializes a specially crafted {@link InetAddress} with a hostname that
42   * encodes the raw bytes of an {@link AFSocketAddress}. We do this because the deserialization code
43   * path does not attempt DNS resolution (which would fail one way or another).
44   *
45   * The hostnames we use end with ".junixsocket", to distinguish them from regular hostnames.
46   *
47   * @author Christian Kohlschütter
48   */
49  class AFInetAddress {
50    private static final byte[] LOCAL_AF = {0x7f, 0, 0, (byte) 0xaf};
51  
52    private static final char PREFIX = '[';
53    private static final String MARKER_HEX_ENCODING = "%%";
54    static final String INETADDR_SUFFIX = ".junixsocket";
55  
56    /**
57     * Encodes a junixsocket socketAddress into a string that is (somewhat) guaranteed to not be
58     * resolved by java.net code.
59     *
60     * Implementation detail: The "[" prefix (with the corresponding "]" suffix missing from the
61     * input) should cause an early {@link UnknownHostException} be thrown, which is caught within
62     * {@link InetSocketAddress#InetSocketAddress(String, int)}, causing the hostname be marked as
63     * "unresolved" (without an address set).
64     *
65     * @param socketAddress The socket address.
66     * @return A string, to be used when calling
67     *         {@link InetSocketAddress#InetSocketAddress(String, int)}, etc.
68     */
69    static final String createUnresolvedHostname(byte[] socketAddress, AFAddressFamily<?> af) {
70      StringBuilder sb = new StringBuilder(1 + socketAddress.length + INETADDR_SUFFIX.length() + 8);
71      sb.append(PREFIX);
72      try {
73        sb.append(URLEncoder.encode(new String(socketAddress, StandardCharsets.ISO_8859_1),
74            StandardCharsets.ISO_8859_1.toString()));
75      } catch (UnsupportedEncodingException e) {
76        throw new IllegalStateException(e);
77      }
78      sb.append('.');
79      sb.append(af.getJuxString());
80      sb.append(INETADDR_SUFFIX);
81  
82      String str = sb.toString();
83  
84      if (str.length() < 64 || str.getBytes(StandardCharsets.UTF_8).length <= 255) {
85        return str;
86      }
87  
88      sb.setLength(0);
89      sb.append(PREFIX);
90      sb.append(MARKER_HEX_ENCODING);
91      for (int i = 0, n = socketAddress.length; i < n; i++) {
92        sb.append(String.format(Locale.ENGLISH, "%02x", socketAddress[i]));
93      }
94  
95      sb.append('.');
96      sb.append(af.getJuxString());
97      sb.append(INETADDR_SUFFIX);
98      return sb.toString();
99    }
100 
101   /**
102    * Creates an InetAddress that is considered "resolved" internally (using a static loopback
103    * address), without actually having to resolve the address via DNS, thus still carrying the
104    * "hostname" field containing a hostname as returned by
105    * {@link #createUnresolvedHostname(byte[],AFAddressFamily)}.
106    *
107    * @param socketAddress The socket address.
108    * @return The {@link InetAddress}.
109    */
110   static final InetAddress wrapAddress(byte[] socketAddress, AFAddressFamily<?> af) {
111     Objects.requireNonNull(af);
112     if (socketAddress == null || socketAddress.length == 0) {
113       return null;
114     }
115 
116     String hostname = createUnresolvedHostname(socketAddress, af);
117     byte[] bytes = hostname.getBytes(StandardCharsets.UTF_8);
118     if (bytes.length > 255) {
119       throw new IllegalStateException("junixsocket address is too long to wrap as InetAddress");
120     }
121 
122     try {
123       return InetAddress.getByAddress(hostname, LOCAL_AF);
124     } catch (UnknownHostException e) {
125       throw new IllegalStateException(e);
126     }
127   }
128 
129   static final byte[] unwrapAddress(InetAddress addr, AFAddressFamily<?> af)
130       throws SocketException {
131     Objects.requireNonNull(addr);
132 
133     if (!isSupportedAddress(addr, af)) {
134       throw new SocketException("Unsupported address");
135     }
136 
137     String hostname = addr.getHostName();
138     try {
139       return unwrapAddress(hostname, af);
140     } catch (IllegalArgumentException e) {
141       throw (SocketException) new SocketException("Unsupported address").initCause(e);
142     }
143   }
144 
145   static final byte[] unwrapAddress(String hostname, AFAddressFamily<?> af) throws SocketException {
146     Objects.requireNonNull(hostname);
147     if (!hostname.endsWith(INETADDR_SUFFIX)) {
148       throw new SocketException("Unsupported address");
149     }
150 
151     final int end = hostname.length() - INETADDR_SUFFIX.length();
152     char c;
153     int domDot = -1;
154     for (int i = end - 1; i >= 0; i--) {
155       c = hostname.charAt(i);
156       if (c == '.') {
157         domDot = i;
158         break;
159       }
160     }
161 
162     String juxString = hostname.substring(domDot + 1, end);
163     if (AFAddressFamily.getAddressFamily(juxString) != af) { // NOPMD
164       throw new SocketException("Incompatible address");
165     }
166 
167     String encodedHostname = hostname.substring(1, domDot);
168     if (encodedHostname.startsWith(MARKER_HEX_ENCODING)) {
169       // Hex-only encoding
170       int len = encodedHostname.length();
171       if ((len & 1) == 1) {
172         throw new IllegalStateException("Length of hex-encoded wrapping must be even");
173       }
174       byte[] unwrapped = new byte[(len - 2) / 2];
175       for (int i = 2, n = encodedHostname.length(), o = 0; i < n; i += 2, o++) {
176         int v = Integer.parseInt(encodedHostname.substring(i, i + 2), 16);
177         unwrapped[o] = (byte) (v & 0xFF);
178       }
179       return unwrapped;
180     } else {
181       // URL-encoding
182       try {
183         return URLDecoder.decode(encodedHostname, StandardCharsets.ISO_8859_1.toString()).getBytes(
184             StandardCharsets.ISO_8859_1);
185       } catch (UnsupportedEncodingException e) {
186         throw new IllegalStateException(e);
187       }
188     }
189   }
190 
191   static boolean isSupportedAddress(InetAddress addr, AFAddressFamily<?> af) {
192     if (addr instanceof Inet4Address && addr.isLoopbackAddress()) {
193       String hostname = addr.getHostName();
194       return hostname.endsWith(af.getJuxInetAddressSuffix());
195     }
196     return false;
197   }
