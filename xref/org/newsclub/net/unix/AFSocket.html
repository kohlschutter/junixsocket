98      /**
99       * Constructs a new {@link AFSocket} subclass instance.
100      * 
101      * @param fdObj The file descriptor.
102      * @param factory The socket factory instance.
103      * @return The instance.
104      * @throws SocketException on error.
105      */
106     @NonNull
107     AFSocket<A> newInstance(FileDescriptor fdObj, AFSocketFactory<A> factory)
108         throws SocketException;
109   }

111   static <A extends AFSocketAddress> AFSocket<A> newInstance(Constructor<A> constr,
112       AFSocketFactory<A> sf, FileDescriptor fdObj, int localPort, int remotePort)
113       throws IOException {
114     if (!fdObj.valid()) {
115       throw new SocketException("Invalid file descriptor");
116     }
117     int status = NativeUnixSocket.socketStatus(fdObj);
118     if (status == NativeUnixSocket.SOCKETSTATUS_INVALID) {
119       throw new SocketException("Not a valid socket");
120     }
121 
122     AFSocket<A> socket = newInstance0(constr, fdObj, sf);
123     socket.getAFImpl().updatePorts(localPort, remotePort);
124 
125     switch (status) {
126       case NativeUnixSocket.SOCKETSTATUS_CONNECTED:
127         socket.internalDummyConnect();
128         break;
129       case NativeUnixSocket.SOCKETSTATUS_BOUND:
130         socket.internalDummyBind();
131         break;
132       case NativeUnixSocket.SOCKETSTATUS_UNKNOWN:
133         break;
134       default:
135         throw new IllegalStateException("Invalid socketStatus response: " + status);
136     }
137     socket.getAFImpl().setSocketAddress(socket.getLocalSocketAddress());
138 
139     return socket;
140   }

204   @Override
205   public final void bind(SocketAddress bindpoint) throws IOException {
206     if (bindpoint == null) {
207       throw new IllegalArgumentException();
208     }
209     if (isClosed()) {
210       throw new SocketException("Socket is closed");
211     }
212     if (isBound()) {
213       throw new SocketException("Already bound");
214     }
215     preprocessSocketAddress(bindpoint);
216     throw new SocketException("Use AF*ServerSocket#bind or #bindOn");
217   }

239   private AFSocketAddress preprocessSocketAddress(SocketAddress endpoint) throws SocketException {
240     if (endpoint == null) {
241       throw new IllegalArgumentException("endpoint is null");
242     } else if (endpoint instanceof SentinelSocketAddress) {
243       return (AFSocketAddress) endpoint;
244     } else {
245       return AFSocketAddress.preprocessSocketAddress(socketAddressClass(), endpoint, afh);
246     }
247   }
248 
249   final boolean connect0(SocketAddress endpoint, int timeout) throws IOException {
250     if (timeout < 0) {
251       throw new IllegalArgumentException("connect: timeout can't be negative");
252     }
253     if (isClosed()) {
254       throw new SocketException("Socket is closed");
255     }
256 
257     if (connectFilter != null) {
258       endpoint = connectFilter.apply(endpoint);
259     }
260 
261     AFSocketAddress address = preprocessSocketAddress(endpoint);
262 
263     if (!isBound()) {
264       internalDummyBind();
265     }
266 
267     boolean success = getAFImpl().connect0(address, timeout);
268     if (success) {
269       int port = address.getPort();
270       if (port > 0) {
271         getAFImpl().updatePorts(getLocalPort(), port);
272       }
273     }
274     internalDummyConnect();
275     return success;
276   }

278   final void internalDummyConnect() throws IOException {
279     if (!isConnected()) {
280       super.connect(AFSocketAddress.INTERNAL_DUMMY_CONNECT, 0);
281     }
282   }

374   private static int initCapabilities() {
375     if (!isSupported()) {
376       return 0;
377     } else {
378       int v = NativeUnixSocket.capabilities();
379 
380       if (System.getProperty("osv.version") != null) {
381         // no fork, no redirect...
382         v &= ~(AFSocketCapability.CAPABILITY_FD_AS_REDIRECT.getBitmask());
383       }
384 
385       for (AFSocketCapability cap : AFSocketCapability.values()) {
386         if (isCapDisabled(cap)) {
387           v &= ~(cap.getBitmask());
388         }
389       }
390 
391       return v;
392     }
393   }
394 
395   private static boolean isCapDisabled(AFSocketCapability cap) {
396     return Boolean.valueOf(System.getProperty(PROP_LIBRARY_DISABLE_CAPABILITY_PREFIX + cap.name(),
397         "false"));
398   }

553   /**
554    * Probes the status of the socket connection.
555    * 
556    * This usually involves checking for {@link #isConnected()}, and if assumed connected, also
557    * sending a zero-length message to the remote.
558    * 
559    * @return {@code true} if the connection is known to be closed, {@code false} if the connection
560    *         is open/not closed or the condition is unknown.
561    * @throws IOException on an unexpected error.
562    */
563   public boolean checkConnectionClosed() throws IOException {
564     if (!isConnected()) {
565       return true;
566     }
567     try {
568       if (!AFSocket.supports(AFSocketCapability.CAPABILITY_ZERO_LENGTH_SEND)) {
569         return false;
570       }
571       getOutputStream().write(ZERO_BYTES);
572       return false;
573     } catch (SocketClosedException e) {
574       return true;
575     } catch (IOException e) {
576       if (!isConnected()) {
577         return true;
578       } else {
579         throw e;
580       }
581     }
