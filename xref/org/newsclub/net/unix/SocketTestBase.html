44    public SocketTestBase() throws IOException {
45      this.serverAddress = new AFUNIXSocketAddress(SOCKET_FILE);
46    }

48    @BeforeEach
49    public void ensureSocketFileIsDeleted() throws IOException {
50      Files.deleteIfExists(SOCKET_FILE.toPath());
51    }

53    @AfterAll
54    public static void tearDownClass() throws IOException {
55      Files.deleteIfExists(SOCKET_FILE.toPath());
56    }

59      return serverAddress;

62    static File initSocketFile() {
63      File f;
64      String explicitFile = System.getProperty("org.newsclub.net.unix.testsocket");
65      if (explicitFile != null) {
66        f = new File(explicitFile);
67      } else {
68        try {
69          f = File.createTempFile("jutest", ".sock");
70          f.deleteOnExit(); // always delete on exit to clean-up sockets created under that name
71        } catch (IOException e) {
72          throw new IllegalStateException("Can't create temporary file", e);
73        }
74      }
75      if (!f.delete()) {
76        throw new IllegalStateException("Could not delete temporary file that we just created: " + f);
77      }
78      return f;
79    }
80  
81    protected File getSocketFile() {
82      return SOCKET_FILE;
83    }
84  
85    protected AFUNIXServerSocket startServer() throws IOException {
86      caller = new Exception();
87      final AFUNIXServerSocket server = AFUNIXServerSocket.newInstance();
88      server.bind(serverAddress);
89      return server;
90    }
91  
92    protected AFUNIXSocket connectToServer() throws IOException {
93      return AFUNIXSocket.connectTo(serverAddress);
94    }

101   protected abstract class ServerThread extends Thread {
102     private final AFUNIXServerSocket serverSocket;
103     private volatile Exception exception = null;
104     private final AtomicBoolean loop = new AtomicBoolean(true);
105     private final Semaphore sema = new Semaphore(0);
106 
107     @SuppressFBWarnings("SC_START_IN_CTOR")
108     protected ServerThread() throws IOException {
109       super();
110       serverSocket = startServer(); // NOPMD
111       setDaemon(true);
112 
113       start();
114     }
115 
116     protected AFUNIXServerSocket startServer() throws IOException {
117       return SocketTestBase.this.startServer();
118     }
119 
120     /**
121      * Stops the server.
122      * 
123      * @throws IOException on error.
124      */
125     public void shutdown() throws IOException {
126       stopAcceptingConnections();
127       if (serverSocket != null) {
128         serverSocket.close();
129       }
130     }
131 
132     /**
133      * Callback used to handle a connection call.
134      * 
135      * Use {@link #stopAcceptingConnections()} to stop accepting new calls.
136      * 
137      * @param sock The socket to handle.
138      * @throws IOException upon error.
139      */
140     protected abstract void handleConnection(final Socket sock) throws IOException;
141 
142     /**
143      * Called from within {@link #handleConnection(Socket)} to tell the server to no longer accept
144      * new calls and to terminate the server thread.
145      */
146     protected void stopAcceptingConnections() {
147       loop.set(false);
148     }
149 
150     protected void onServerSocketClose() {
151     }
152 
153     public ServerSocket getServerSocket() {
154       return serverSocket;
155     }
156 
157     protected void handleException(Exception e) {
158       e.printStackTrace();
159     }
160 
161     protected void acceptAndHandleConnection() throws IOException {
162       try (Socket sock = serverSocket.accept()) {
163         handleConnection(sock);
164       }
165     }
166 
167     @Override
168     public final void run() {
169       try {
170         loop.set(true);
171         try {
172           while (loop.get()) {
173             acceptAndHandleConnection();
174           }
175         } finally {
176           onServerSocketClose();
177           if (serverSocket != null) {
178             serverSocket.close();
179           }
180         }
181       } catch (IOException e) {
182         if (!loop.get() && (serverSocket == null || serverSocket.isClosed())) {
183           // ignore
184         } else {
185           e.addSuppressed(caller);
186           handleException(e);
187           exception = e;
188         }
189       }
190       sema.release();
191     }
192 
193     /**
194      * Checks if there were any exceptions thrown during the lifetime of this ServerThread.
195      * 
196      * NOTE: This call blocks until the Thread actually terminates.
197      * 
198      * @throws Exception upon error.
199      */
200     public void checkException() throws Exception {
201       sema.acquire();
202       if (exception != null) {
203         throw exception;
204       }
205     }
206   }

208   /**
209    * Sleeps for the given amount of milliseconds.
210    * 
211    * @param ms The duration in milliseconds.
212    * @throws InterruptedIOException when interrupted.
213    */
214   protected void sleepFor(final int ms) throws InterruptedIOException {
215     try {
216       Thread.sleep(ms);
217     } catch (InterruptedException e) {
218       Thread.currentThread().interrupt();
219       throw (InterruptedIOException) new InterruptedIOException("sleep interrupted").initCause(e);
220     }
221   }
