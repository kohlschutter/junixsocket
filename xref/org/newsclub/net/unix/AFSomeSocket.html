18  package org.newsclub.net.unix;
19  
20  import java.io.Closeable;
21  import java.net.DatagramSocket;
22  import java.net.Socket;
23  import java.net.SocketAddress;
24  import java.nio.channels.DatagramChannel;
25  import java.nio.channels.SocketChannel;
26  
27  import org.eclipse.jdt.annotation.Nullable;
28  
29  /**
30   * Marker interface that combines junixsocket-based {@link SocketChannel}s, {@link Socket}s,
31   * {@link DatagramChannel}s and {@link DatagramSocket}s.
32   *
33   * @author Christian Kohlschütter
34   * @see AFSocketPair
35   * @see AFSocket
36   * @see AFSocketChannel
37   * @see AFDatagramSocket
38   * @see AFDatagramChannel
39   */
40  public interface AFSomeSocket extends Closeable, FileDescriptorAccess {
41    /**
42     * Returns the socket's local socket address, or {@code null} if unavailable or if there was a
43     * problem retrieving it.
44     *
45     * @return The local socket address, or {@code null}.
46     */
47    @Nullable
48    SocketAddress getLocalSocketAddress();
49  
50    /**
51     * Returns the socket's remote socket address, or {@code null} if unavailable/not connected, or if
52     * there was a problem retrieving it.
53     *
54     * @return The remote socket address, or {@code null}.
55     */
56    @Nullable
57    SocketAddress getRemoteSocketAddress();
58  }
