18  package org.newsclub.net.mysql.demo;
19  
20  import java.sql.Connection;
21  import java.sql.DriverManager;
22  import java.sql.ResultSet;
23  import java.sql.Statement;
24  import java.util.Properties;
25  
26  import org.newsclub.net.mysql.AFUNIXDatabaseSocketFactory;
27  
28  /**
29   * Demonstrates how to connect to a local MySQL server.
30   * 
31   * @author Christian Kohlschuetter
32   */
33  public class AFUNIXDatabaseSocketFactoryDemo {
34    public static void main(String[] args) throws Exception {
35      Class.forName("com.mysql.jdbc.Driver");
36      Properties props = new Properties();
37      props.put("socketFactory", AFUNIXDatabaseSocketFactory.class.getName());
38  
39      // props.put("user", "test");
40      // props.put("password", "test");
41      // props.put("junixsocket.file", "/var/lib/mysql.sock");
42      props.put("user", "root");
43      props.put("password", "");
44      props.put("junixsocket.file", "/tmp/mysql.sock");
45  
46      // SHOW DATABASES three times
47      for (int i = 0; i < 3; i++) {
48        try (Connection conn = DriverManager.getConnection("jdbc:mysql://", props)) {
49          System.out.println("Connection: " + conn);
50  
51          System.out.println("SHOW DATABASES:");
52          try (Statement stmt = conn.createStatement();
53              ResultSet rs = stmt.executeQuery("SHOW DATABASES")) {
54            while (rs.next()) {
55              System.out.println("\t" + rs.getString(1));
56            }
57          }
58        }
59      }
60    }
61  }
